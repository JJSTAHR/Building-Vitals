// FOLIO DICT STRUCTURE
// ============================================================================

class FolioDict {
  constructor(data = {}) {
    // Core identity
    this.id = data.id || this.generateId(data);
    this.dis = data.dis || data.name || this.id;

    // Marker tags (m: notation)
    this.markerTags = new Set();

    // Value tags
    this.valueTags = {};

    // Reference tags
    this.refs = {};

    // Parse incoming data
    this.parseDict(data);
  }

  generateId(data) {
    // Generate hierarchical ref ID
    const parts = [];
    if (data.site) parts.push(data.site);
    if (data.equip) parts.push(data.equip);
    if (data.name) parts.push(data.name.replace(/[:\s]/g, '_'));
    return `r:${parts.join('.')}`;
  }

  parseDict(data) {
    for (const [key, value] of Object.entries(data)) {
      if (value === 'm:') {
        // Marker tag
        this.markerTags.add(key);
      } else if (typeof value === 'string' && value.startsWith('r:')) {
        // Reference tag
        this.refs[key] = value;
      } else if (typeof value === 'string' && value.startsWith('n:')) {
        // Number tag
        this.valueTags[key] = parseFloat(value.slice(2));
      } else if (key !== 'id' && key !== 'dis') {
        // Regular value tag
        this.valueTags[key] = value;
      }
    }
  }

  toJSON() {
    const json = {
      id: this.id,
      dis: this.dis
    };

    // Add marker tags
    for (const tag of this.markerTags) {
      json[tag] = 'm:';
    }

    // Add value tags
    for (const [key, value] of Object.entries(this.valueTags)) {
      if (typeof value === 'number') {
        json[key] = `n:${value}`;
      } else {
        json[key] = value;
      }
    }

    // Add refs
    for (const [key, ref] of Object.entries(this.refs)) {
      json[key] = ref;
    }

    return json;
  }

  hasTag(tag) {
    return this.markerTags.has(tag);
  }

  getTag(tag) {
    return this.valueTags[tag] || this.refs[tag];
  }

  matches(filter) {
    for (const [key, value] of Object.entries(filter)) {
      if (value === 'm:' && !this.hasTag(key)) return false;
      if (value !== 'm:' && this.getTag(key) !== value) return false;
    }
    return true;
  }
}

class EquipmentRelationshipEngine {
  constructor() {
    this.relationships = new Map();
    this.hierarchy = new Map();
  }

  parsePointName(name) {
    // Enhanced pattern matching for equipment detection
    const patterns = {
      ahu: /^AHU[-_:]?(\d+)/i,
      vav: /^VAV[-_:]?(\d+[-_]?\d*)/i,
      chiller: /^(CH|CHILLER)[-_:]?(\d*)/i,
      boiler: /^(BLR|BOILER|HW)[-_:]?(\d*)/i,
      pump: /^(PUMP|P)[-_:]?(\d+)[-_:]?(CHW|HW|CW)?/i,
      fan: /^(FAN|EF|SF|RF)[-_:]?(\d*)/i,
      meter: /^(METER|MTR)[-_:]?(\d*)[-_:]?(ELEC|GAS|WATER)?/i,
      rtu: /^RTU[-_:]?(\d+)/i,
      fcu: /^FCU[-_:]?(\d+)/i,
      ct: /^CT[-_:]?(\d+)/i  // Cooling Tower
    };

    for (const [type, pattern] of Object.entries(patterns)) {
      const match = name.match(pattern);
      if (match) {
        return {
          type,
          id: match[1] || match[2] || '1',
          subType: match[3] || null,
          raw: match[0]
        };
      }
    }
    return null;
  }

  buildRelationships(points, siteName) {
    const equipment = new Map();
    const spaces = new Map();

    // First pass: identify equipment and spaces
    for (const point of points) {
      const parsed = this.parsePointName(point.name);
      if (parsed) {
        const equipId = `r:${siteName}.${parsed.type}${parsed.id}`;

        if (!equipment.has(equipId)) {
          equipment.set(equipId, {
            id: equipId,
            dis: `${parsed.type.toUpperCase()} ${parsed.id}`,
            equip: 'm:',
            [parsed.type]: 'm:',
            siteRef: `r:${siteName}`,
            points: []
          });
        }

        equipment.get(equipId).points.push(point.id || point.name);

        // Detect space from VAV or zone indicators
        if (parsed.type === 'vav' || point.name.includes('ZN')) {
          const zoneMatch = point.name.match(/(\d+)[-_](\d+)/);
          if (zoneMatch) {
            const floor = zoneMatch[1];
            const zone = zoneMatch[2];
            const spaceId = `r:${siteName}.floor${floor}.zone${zone}`;

            if (!spaces.has(spaceId)) {
              spaces.set(spaceId, {
                id: spaceId,
                dis: `Floor ${floor} - Zone ${zone}`,
                space: 'm:',
                zone: 'm:',
                siteRef: `r:${siteName}`,
                floorRef: `r:${siteName}.floor${floor}`
              });
            }
          }
        }
      }
    }

    // Second pass: detect relationships
    this.detectFlowRelationships(equipment);
    this.detectHierarchy(equipment);

    return { equipment, spaces };
  }

  detectFlowRelationships(equipment) {
    // Detect chilled water loop relationships
    const chillers = [];
    const pumps = [];
    const ahus = [];

    for (const [id, equip] of equipment) {
      if (equip.chiller) chillers.push(equip);
      if (equip.pump && equip.dis.includes('CHW')) pumps.push(equip);
      if (equip.ahu) ahus.push(equip);
    }

    // Link chillers to AHUs via chilled water
    for (const chiller of chillers) {
      chiller.chilledWaterLoop = 'm:';
      chiller.serves = ahus.map(a => a.id);
    }

    // Link pumps to equipment
    for (const pump of pumps) {
      pump.chilledWaterLoop = 'm:';
      pump.serves = [...chillers.map(c => c.id), ...ahus.map(a => a.id)];
    }
  }

  detectHierarchy(equipment) {
    // Build equipment hierarchy (site -> plant -> distribution -> terminal)
    const hierarchy = {
      plant: [],      // Chillers, Boilers
      distribution: [], // AHUs, Pumps
      terminal: []    // VAVs, FCUs
    };

    for (const [id, equip] of equipment) {
      if (equip.chiller || equip.boiler) {
        hierarchy.plant.push(equip);
        equip.level = 'plant';
      } else if (equip.ahu || equip.pump) {
        hierarchy.distribution.push(equip);
        equip.level = 'distribution';
      } else if (equip.vav || equip.fcu) {
        hierarchy.terminal.push(equip);
        equip.level = 'terminal';
      }
    }

    return hierarchy;
  }
}

class AIInsightsEngine {
  generateInsights(point, equipment, historicalData = null) {
    const insights = {
      patterns: {},
      recommendations: [],
      anomalies: [],
      optimization: [],
      compliance: []
    };

    // Point type specific insights
    this.addPointTypeInsights(point, insights);

    // Equipment specific insights
    this.addEquipmentInsights(point, equipment, insights);

    // Historical analysis if available
    if (historicalData) {
      this.addHistoricalInsights(historicalData, insights);
    }

    // Energy optimization insights
    this.addEnergyInsights(point, insights);

    // Compliance checks
    this.addComplianceChecks(point, insights);

    return insights;
  }

  addPointTypeInsights(point, insights) {
    const value = parseFloat(point.curVal || point.value);

    // Temperature points
    if (point.temp === 'm:' || point.unit?.includes('°')) {
      if (point.zone === 'm:') {
        insights.recommendations.push('Monitor for ASHRAE 55 comfort compliance (68-76°F)');
        if (value < 68) insights.anomalies.push('Zone temperature below comfort range');
        if (value > 76) insights.anomalies.push('Zone temperature above comfort range');
      }

      if (point.supply === 'm:' || point.dis?.includes('Supply')) {
        insights.recommendations.push('Typical supply air: 55-60°F cooling, 90-110°F heating');
        if (value > 65 && value < 85) {
          insights.anomalies.push('Supply temperature in dead band - check mode');
        }
      }
    }

    // CO2 points
    if (point.co2 === 'm:' || point.unit === 'ppm') {
      insights.recommendations.push('ASHRAE 62.1 recommends < 1000 ppm for acceptable IAQ');
      if (value > 1000) insights.anomalies.push('CO2 levels exceed recommended maximum');
      if (value > 1500) insights.anomalies.push('⚠️ High CO2 - increase ventilation immediately');
    }

    // Power/Energy points
    if (point.power === 'm:' || point.unit === 'kW') {
      insights.patterns.energy_monitoring = true;
      insights.optimization.push('Track for demand response opportunities');
      insights.optimization.push('Consider load shifting during peak hours');
    }

    // Damper/Valve positions
    if (point.damper === 'm:' || point.valve === 'm:') {
      if (value > 90) {
        insights.anomalies.push('Near full open - equipment may be undersized');
        insights.optimization.push('Review capacity requirements');
      }
      if (value < 10) {
        insights.anomalies.push('Near closed - verify minimum position requirements');
      }
    }
  }

  addEquipmentInsights(point, equipment, insights) {
    if (!equipment) return;

    // AHU-specific
    if (equipment.ahu === 'm:') {
      if (point.filter === 'm:' && point.pressure === 'm:') {
        insights.recommendations.push('Monitor filter ΔP for maintenance scheduling');
        const value = parseFloat(point.curVal || point.value);
        if (value > 1.5) insights.anomalies.push('High filter pressure drop - schedule replacement');
      }

      if (point.economizer === 'm:' || point.outside === 'm:') {
        insights.optimization.push('Enable economizer when OAT < 60°F for free cooling');
      }
    }

    // Chiller-specific
    if (equipment.chiller === 'm:') {
      if (point.temp === 'm:' && point.leaving === 'm:') {
        insights.recommendations.push('Optimal CHWST: 44-46°F for efficiency');
        insights.optimization.push('Consider reset schedule based on load');
      }
    }

    // Pump-specific
    if (equipment.pump === 'm:') {
      if (point.vfd === 'm:' || point.speed === 'm:') {
        const value = parseFloat(point.curVal || point.value);
        if (value < 30) {
          insights.optimization.push('Low speed operation - consider cycling strategy');
        }
      }
    }
  }

  addHistoricalInsights(data, insights) {
    // Pattern detection
    if (data.daily_cycle) {
      insights.patterns.daily_cycle = true;
      insights.patterns.occupied_hours = data.occupied_hours;
    }

    // Trend analysis
    if (data.trend === 'increasing') {
      insights.patterns.trend = 'increasing';
      insights.anomalies.push('Upward trend detected - investigate cause');
    }

    // Baseline deviation
    if (data.baseline && data.current) {
      const deviation = Math.abs(data.current - data.baseline) / data.baseline;
      if (deviation > 0.15) {
        insights.anomalies.push(`15%+ deviation from baseline (${data.baseline.toFixed(1)})`);
      }
    }
  }

  addEnergyInsights(point, insights) {
    if (point.energy === 'm:' || point.power === 'm:') {
      insights.optimization.push('Implement demand limiting during peak periods');
      insights.optimization.push('Consider power factor correction if < 0.95');
      insights.recommendations.push('Track against energy baseline for savings verification');
    }
  }

  addComplianceChecks(point, insights) {
    // Ventilation compliance
    if (point.outside === 'm:' && point.air === 'm:' && point.flow === 'm:') {
      insights.compliance.push('ASHRAE 62.1 minimum OA requirements');
    }

    // Temperature compliance
    if (point.zone === 'm:' && point.temp === 'm:') {
      insights.compliance.push('ASHRAE 55 thermal comfort');
      insights.compliance.push('Local code requirements');
    }

    // Humidity compliance
    if (point.humidity === 'm:') {
      insights.compliance.push('ASHRAE 55: 30-60% RH recommended');
    }
  }
}

export default {
  async fetch(request, env, ctx) {
    const url = new URL(request.url);

    // CORS headers
    const corsHeaders = {
      'Access-Control-Allow-Origin': '*',
      'Access-Control-Allow-Methods': 'GET, POST, PUT, DELETE, OPTIONS',
      'Access-Control-Allow-Headers': 'Content-Type, X-ACE-Token',
      'Access-Control-Expose-Headers': 'X-Cache-Status, X-Processing-Time'
    };

    // Handle preflight
    if (request.method === 'OPTIONS') {
      return new Response(null, { headers: corsHeaders });
    }

    // Extract token
    const token = request.headers.get('X-ACE-Token');
    if (!token) {
      return new Response(JSON.stringify({ error: 'No token provided' }), {
        status: 401,
        headers: { ...corsHeaders, 'Content-Type': 'application/json' }
      });
    }

    const startTime = Date.now();

    try {
      // Route handling
      const response = await this.routeRequest(url, request, env, token);

      // Add processing time header
      const processingTime = Date.now() - startTime;
      response.headers.set('X-Processing-Time', `${processingTime}ms`);

      // Add CORS headers
      for (const [key, value] of Object.entries(corsHeaders)) {
        response.headers.set(key, value);
      }

      return response;

    } catch (error) {
      console.error('Worker error:', error);
      return new Response(JSON.stringify({
        error: error.message,
        stack: error.stack
      }), {
        status: 500,
        headers: { ...corsHeaders, 'Content-Type': 'application/json' }
      });
    }
  },

  async routeRequest(url, request, env, token) {
    const path = url.pathname;

    // Sites endpoint
    if (path === '/api/sites') {
      return await this.handleSites(env, token);
    }

    // Enhanced points endpoint with Folio structure
    const pointsMatch = path.match(/^\/api\/sites\/([^\/]+)\/configured_points$/);
    if (pointsMatch) {
      const siteName = pointsMatch[1];
      return await this.handleEnhancedPoints(env, siteName, token);
    }

    // Equipment endpoint (new)
    const equipMatch = path.match(/^\/api\/sites\/([^\/]+)\/equipment$/);
    if (equipMatch) {
      const siteName = equipMatch[1];
      return await this.handleEquipment(env, siteName, token);
    }

    // Query endpoint for tag-based searches (new)
    if (path === '/api/query') {
      const filter = await request.json();
      return await this.handleQuery(env, filter, token);
    }

    // Timeseries endpoint
    const timeseriesMatch = path.match(/^\/api\/points\/(.+)\/timeseries$/);
    if (timeseriesMatch) {
      const pointName = decodeURIComponent(timeseriesMatch[1]);
      return await this.handleTimeseries(env, pointName, url.searchParams, token);
    }

    // Batch endpoint for multiple points (new)
    if (path === '/api/batch/timeseries' && request.method === 'POST') {
      const { points, start_time, end_time } = await request.json();
      return await this.handleBatchTimeseries(env, points, start_time, end_time, token);
    }

    return new Response('Not found', { status: 404 });
  },

  async handleSites(env, token) {
    const cacheKey = 'sites:all';
    const cached = await env.POINTS_KV.get(cacheKey, { type: 'json' });

    if (cached && cached.timestamp > Date.now() - 3600000) {
      return new Response(JSON.stringify({
        items: cached.sites,
        fromCache: true
      }), {
        headers: {
          'Content-Type': 'application/json',
          'X-Cache-Status': 'HIT'
        }
      });
    }

    const response = await fetch(`${env.ACE_API_URL}/sites`, {
      headers: { 'Authorization': `Bearer ${token}` }
    });

    const data = await response.json();
    const sites = data.items || data;

    // Enhance sites with Folio structure
    const enhancedSites = sites.map(site => {
      const folioSite = new FolioDict({
        id: `r:${site.name || site.id}`,
        dis: site.name || `Site ${site.id}`,
        site: 'm:',
        area: site.area || null,
        tz: site.timezone || 'America/Chicago'
      });
      return folioSite.toJSON();
    });

    // Cache enhanced sites
    await env.POINTS_KV.put(cacheKey, JSON.stringify({
      sites: enhancedSites,
      timestamp: Date.now()
    }), {
      expirationTtl: 86400
    });

    return new Response(JSON.stringify({
      items: enhancedSites,
      fromCache: false
    }), {
      headers: {
        'Content-Type': 'application/json',
        'X-Cache-Status': 'MISS'
      }
    });
  },

  async handleEnhancedPoints(env, siteName, token) {
    const cacheKey = `points:${siteName}:v3`;
    const cached = await env.POINTS_KV.get(cacheKey, { type: 'json' });

    if (cached && cached.timestamp > Date.now() - 3600000) {
      return new Response(JSON.stringify({
        items: cached.points,
        equipment: cached.equipment,
        spaces: cached.spaces,
        fromCache: true,
        version: '3.0.0'
      }), {
        headers: {
          'Content-Type': 'application/json',
          'X-Cache-Status': 'HIT'
        }
      });
    }

    // Fetch raw points
    const response = await fetch(
      `${env.ACE_API_URL}/sites/${siteName}/configured_points`,
      {
        headers: { 'Authorization': `Bearer ${token}` }
      }
    );

    if (!response.ok) {
      throw new Error(`ACE API error: ${response.status}`);
    }

    const data = await response.json();
    const rawPoints = data.items || data;

    // Build relationships
    const relationshipEngine = new EquipmentRelationshipEngine();
    const { equipment, spaces } = relationshipEngine.buildRelationships(rawPoints, siteName);

    // Enhance points with Folio structure
    const insightsEngine = new AIInsightsEngine();
    const enhancedPoints = rawPoints.map(point => {
      const enhanced = this.enhancePointWithFolio(
        point,
        siteName,
        equipment,
        spaces,
        insightsEngine
      );
      return enhanced.toJSON();
    });

    // Convert equipment and spaces to arrays
    const equipmentArray = Array.from(equipment.values());
    const spacesArray = Array.from(spaces.values());

    // Cache everything
    await env.POINTS_KV.put(cacheKey, JSON.stringify({
      points: enhancedPoints,
      equipment: equipmentArray,
      spaces: spacesArray,
      timestamp: Date.now()
    }), {
      expirationTtl: 86400
    });

    return new Response(JSON.stringify({
      items: enhancedPoints,
      equipment: equipmentArray,
      spaces: spacesArray,
      fromCache: false,
      version: '3.0.0'
    }), {
      headers: {
        'Content-Type': 'application/json',
        'X-Cache-Status': 'MISS'
      }
    });
  },

  enhancePointWithFolio(rawPoint, siteName, equipment, spaces, insightsEngine) {
    const point = new FolioDict();

    // Parse equipment from name
    const relationshipEngine = new EquipmentRelationshipEngine();
    const parsed = relationshipEngine.parsePointName(rawPoint.name);

    // Build ID
    const idParts = [siteName];
    if (parsed) {
      idParts.push(`${parsed.type}${parsed.id}`);
    }
    idParts.push(rawPoint.name.replace(/[:\s]/g, '_').toLowerCase());
    point.id = `r:${idParts.join('.')}`;

    // Display name
    point.dis = this.generateDisplayName(rawPoint.name, parsed);

    // Add marker tags
    point.markerTags.add('point');
    point.markerTags.add('his');

    // Detect point type and add specific tags
    this.detectPointType(rawPoint.name, point);

    // Add refs
    point.refs.siteRef = `r:${siteName}`;

    if (parsed) {
      const equipId = `r:${siteName}.${parsed.type}${parsed.id}`;
      point.refs.equipRef = equipId;

      // Inherit equipment tags
      const equip = equipment.get(equipId);
      if (equip) {
        for (const [key, value] of Object.entries(equip)) {
          if (value === 'm:' && key !== 'equip') {
            point.markerTags.add(key);
          }
        }
      }
    }

    // Add unit
    const unit = this.detectUnit(rawPoint.name, point);
    if (unit) {
      point.valueTags.unit = unit;
    }

    // Add current value
    if (rawPoint.value !== undefined) {
      point.valueTags.curVal = parseFloat(rawPoint.value) || rawPoint.value;
      point.valueTags.curTime = new Date().toISOString();
    }

    // Generate insights
    const equip = parsed ? equipment.get(`r:${siteName}.${parsed.type}${parsed.id}`) : null;
    const insights = insightsEngine.generateInsights(point.toJSON(), equip);
    if (insights && Object.keys(insights).length > 0) {
      point.valueTags.aiInsights = insights;
    }

    // Add metadata
    point.valueTags.qualityScore = this.calculateQualityScore(point);
    point.valueTags._originId = rawPoint.name;
    point.valueTags._originPath = rawPoint.id || rawPoint.path;

    return point;
  },

  generateDisplayName(name, parsed) {
    let displayName = '';

    // Add equipment prefix
    if (parsed) {
      displayName = `${parsed.type.toUpperCase()}-${parsed.id} `;
    }

    // Parse the rest of the name
    const parts = name.split(/[-_:]/);
    const cleaned = parts
      .map(p => {
        // Expand common abbreviations
        const expansions = {
          'T': 'Temperature',
          'SP': 'Setpoint',
          'P': 'Pressure',
          'F': 'Flow',
          'H': 'Humidity',
          'VLV': 'Valve',
          'DMP': 'Damper',
          'POS': 'Position',
          'STS': 'Status',
          'CMD': 'Command',
          'ZN': 'Zone',
          'OA': 'Outside Air',
          'RA': 'Return Air',
          'SA': 'Supply Air',
          'CHW': 'Chilled Water',
          'HW': 'Hot Water',
          'CW': 'Condenser Water'
        };
        return expansions[p.toUpperCase()] || p;
      })
      .filter(p => p && p !== parsed?.raw);

    displayName += cleaned.join(' ');
    return displayName.trim();
  },

  detectPointType(name, point) {
    const patterns = {
      temp: [/[-_:]T[-_]?$/i, /TEMP/i],
      pressure: [/[-_:]P[-_]?$/i, /PRESS/i],
      flow: [/FLOW/i, /[-_:]F[-_]?$/i],
      humidity: [/RH/i, /HUM/i],
      co2: [/CO2/i],
      power: [/KW$/i, /POWER/i],
      energy: [/KWH/i, /ENERGY/i],
      damper: [/DMP/i, /DAMPER/i],
      valve: [/VLV/i, /VALVE/i],
      fan: [/FAN/i],
      status: [/STS/i, /STATUS/i, /RUN/i],
      cmd: [/CMD/i, /COMMAND/i],
      sp: [/SP$/i, /SETPOINT/i]
    };

    for (const [tag, regexes] of Object.entries(patterns)) {
      for (const regex of regexes) {
        if (regex.test(name)) {
          point.markerTags.add(tag);

          // Add point function
          if (tag === 'cmd') point.valueTags.pointFunction = 'cmd';
          else if (tag === 'sp') point.valueTags.pointFunction = 'sp';
          else point.valueTags.pointFunction = 'sensor';

          break;
        }
      }
    }

    // Air stream detection
    if (/OA[-_:]/i.test(name)) point.markerTags.add('outside');
    if (/RA[-_:]/i.test(name)) point.markerTags.add('return');
    if (/SA[-_:]/i.test(name)) point.markerTags.add('supply');
    if (/ZN[-_:]/i.test(name)) point.markerTags.add('zone');

    // Water loop detection
    if (/CHW/i.test(name)) {
      point.markerTags.add('chilled');
      point.markerTags.add('water');
    }
    if (/HW/i.test(name)) {
      point.markerTags.add('hot');
      point.markerTags.add('water');
    }
  },

  detectUnit(name, point) {
    if (point.hasTag('temp')) return '°F';
    if (point.hasTag('pressure')) {
      if (point.hasTag('water')) return 'psi';
      return 'in.w.c.';
    }
    if (point.hasTag('flow')) {
      if (point.hasTag('water')) return 'GPM';
      return 'CFM';
    }
    if (point.hasTag('humidity')) return '%RH';
    if (point.hasTag('co2')) return 'ppm';
    if (point.hasTag('power')) return 'kW';
    if (point.hasTag('energy')) return 'kWh';
    if (point.hasTag('damper') || point.hasTag('valve')) return '%';
    if (point.hasTag('status')) return 'bool';
    return null;
  },

  calculateQualityScore(point) {
    let score = 0;

    // Has display name (20 points)
    if (point.dis && point.dis !== point.id) score += 20;

    // Has unit (20 points)
    if (point.valueTags.unit) score += 20;

    // Has equipment ref (20 points)
    if (point.refs.equipRef) score += 20;

    // Has multiple marker tags (20 points)
    if (point.markerTags.size > 5) score += 20;

    // Has AI insights (20 points)
    if (point.valueTags.aiInsights) score += 20;

    return score;
  },

  async handleEquipment(env, siteName, token) {
    const cacheKey = `equipment:${siteName}`;
    const cached = await env.POINTS_KV.get(cacheKey, { type: 'json' });

    if (cached) {
      return new Response(JSON.stringify({
        items: cached,
        fromCache: true
      }), {
        headers: {
          'Content-Type': 'application/json',
          'X-Cache-Status': 'HIT'
        }
      });
    }

    // Get points to build equipment
    const pointsResponse = await this.handleEnhancedPoints(env, siteName, token);
    const pointsData = await pointsResponse.json();

    return new Response(JSON.stringify({
      items: pointsData.equipment || [],
      fromCache: false
    }), {
      headers: {
        'Content-Type': 'application/json',
        'X-Cache-Status': 'MISS'
      }
    });
  },

  async handleQuery(env, filter, token) {
    // Implement tag-based query across all sites
    const sites = await this.getSites(env, token);
    const results = [];

    for (const site of sites) {
      const cacheKey = `points:${site.name}:v3`;
      const cached = await env.POINTS_KV.get(cacheKey, { type: 'json' });

      if (cached) {
        const matches = cached.points.filter(p => {
          const point = new FolioDict(p);
          return point.matches(filter);
        });
        results.push(...matches);
      }
    }

    return new Response(JSON.stringify({
      items: results,
      filter,
      count: results.length
    }), {
      headers: { 'Content-Type': 'application/json' }
    });
  },

  async handleTimeseries(env, pointName, params, token) {
    const queryString = params.toString();
    const response = await fetch(
      `${env.ACE_API_URL}/points/${encodeURIComponent(pointName)}/timeseries?${queryString}`,
      {
        headers: { 'Authorization': `Bearer ${token}` }
      }
    );

    const data = await response.json();

    // Transform to tuple format
    const transformed = data.map(item => [
      new Date(item.time).getTime(),
      parseFloat(item.value) || 0
    ]);

    return new Response(JSON.stringify(transformed), {
      headers: { 'Content-Type': 'application/json' }
    });
  },

  async handleBatchTimeseries(env, points, startTime, endTime, token) {
    const promises = points.map(async (pointName) => {
      const response = await fetch(
        `${env.ACE_API_URL}/points/${encodeURIComponent(pointName)}/timeseries?start_time=${startTime}&end_time=${endTime}`,
        {
          headers: { 'Authorization': `Bearer ${token}` }
        }
      );

      const data = await response.json();
      const transformed = data.map(item => [
        new Date(item.time).getTime(),
        parseFloat(item.value) || 0
      ]);

      return {
        point: pointName,
        data: transformed
      };
    });

    const results = await Promise.all(promises);

    return new Response(JSON.stringify({
      results,
      points: points.length,
      startTime,
      endTime
    }), {
      headers: { 'Content-Type': 'application/json' }
    });
  },

  async getSites(env, token) {
    const response = await this.handleSites(env, token);
    const data = await response.json();
    return data.items || [];
  }
