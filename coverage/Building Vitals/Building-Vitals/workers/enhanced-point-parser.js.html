// EQUIPMENT PATTERN DEFINITIONS
// ============================================================================

const EQUIPMENT_PATTERNS = [
  // VAV patterns (Variable Air Volume) - most common
  {
    regex: /\b(Vav)[-_]?(\d+)(?:[-_](\d+))?\b/i,
    type: 'vav',
    format: (match) => ({
      type: 'vav',
      id: match[2],
      subId: match[3],
      display: `VAV ${match[2]}${match[3] ? `-${match[3]}` : ''}`
    })
  },

  // RTU patterns (Rooftop Unit)
  {
    regex: /\b(Rtu|RTU)[-_]?(\d+)(?:[-_](\d+))?\b/i,
    type: 'rtu',
    format: (match) => ({
      type: 'rtu',
      id: match[2],
      subId: match[3],
      display: `RTU ${match[2]}${match[3] ? `-${match[3]}` : ''}`
    })
  },

  // AHU patterns (Air Handling Unit)
  {
    regex: /\b(Ahu|AHU)[-_]?(\d+)\b/i,
    type: 'ahu',
    format: (match) => ({
      type: 'ahu',
      id: match[2],
      display: `AHU ${match[2]}`
    })
  },

  // Chiller patterns
  {
    regex: /\b(CH|Chiller|CHLR)[-_]?(\d+)?\b/i,
    type: 'chiller',
    format: (match) => ({
      type: 'chiller',
      id: match[2] || '1',
      display: `Chiller ${match[2] || '1'}`
    })
  },

  // Boiler patterns
  {
    regex: /\b(BLR|Boiler|HWB)[-_]?(\d+)?\b/i,
    type: 'boiler',
    format: (match) => ({
      type: 'boiler',
      id: match[2] || '1',
      display: `Boiler ${match[2] || '1'}`
    })
  },

  // Pump patterns
  {
    regex: /\b(Pump|P|CHWP|HWP|CWP)[-_]?(\d+)\b/i,
    type: 'pump',
    format: (match) => ({
      type: 'pump',
      id: match[2],
      pumpType: match[1].toUpperCase(),
      display: `${match[1].toUpperCase() === 'CHWP' ? 'CHW ' : match[1].toUpperCase() === 'HWP' ? 'HW ' : match[1].toUpperCase() === 'CWP' ? 'CW ' : ''}Pump ${match[2]}`
    })
  },

  // FCU patterns (Fan Coil Unit)
  {
    regex: /\b(FCU|Fcu)[-_]?(\d+)\b/i,
    type: 'fcu',
    format: (match) => ({
      type: 'fcu',
      id: match[2],
      display: `FCU ${match[2]}`
    })
  },

  // Cooling Tower patterns
  {
    regex: /\b(CT|CoolingTower|Tower)[-_]?(\d+)\b/i,
    type: 'ct',
    format: (match) => ({
      type: 'ct',
      id: match[2],
      display: `CT ${match[2]}`
    })
  }
];

/**
 * Extract equipment and point information from a BACnet path
 * @param {string} fullPath - The complete point path
 * @returns {Object} Parsed equipment and point information
 */
function extractEquipmentFromPath(fullPath) {
  if (!fullPath || typeof fullPath !== 'string') {
    return {
      equipment: null,
      pointName: '',
      displayName: fullPath || '',
      unit: null,
      category: null
    };
  }

  // Step 1: Extract equipment using patterns
  let equipment = null;
  for (const pattern of EQUIPMENT_PATTERNS) {
    const match = fullPath.match(pattern.regex);
    if (match) {
      equipment = pattern.format(match);
      break;
    }
  }

  // Step 2: Extract the point name (the part after .points. or the last segment)
  let pointName = '';
  const pointsMatch = fullPath.match(/\.points\.([^.]+)$/i);
  if (pointsMatch) {
    pointName = pointsMatch[1];
  } else {
    // Fallback: take the last meaningful segment
    const segments = fullPath.split(/[./]/);
    // Skip common prefixes and take the last 1-2 segments
    const filtered = segments.filter(s =>
      s &&
      !s.match(/^(ses|S|FallsCity|CMC|C|Drivers|BacnetNetwork|Net\d+|points)$/i)
    );
    pointName = filtered[filtered.length - 1] || '';
  }

  // Step 3: Detect point type and unit
  let pointType = null;
  let unit = null;
  let category = null;

  for (const [typeName, config] of Object.entries(POINT_TYPES)) {
    for (const pattern of config.patterns) {
      if (pattern.test(pointName)) {
        pointType = typeName;
        unit = typeof config.unit === 'function'
          ? config.unit(pointName)
          : config.unit;
        category = config.category;
        break;
      }
    }
    if (pointType) break;
  }

  // Step 4: Generate display name
  const displayName = generateDisplayName(pointName, equipment);

  return {
    // Equipment info
    equipment: equipment?.type || null,
    equipmentId: equipment?.id || null,
    equipmentDisplay: equipment?.display || null,

    // Point info
    pointName,
    displayName,
    unit,
    category,
    pointType,

    // Full parsed object
    parsed: equipment,

    // Quality indicators
    hasEquipment: !!equipment,
    hasUnit: !!unit,
    confidence: calculateConfidence(equipment, pointType, displayName)
  };
}

// DISPLAY NAME GENERATION
// ============================================================================

/**
 * Generate a human-readable display name
 * @param {string} pointName - The extracted point name
 * @param {Object} equipment - The equipment object
 * @returns {string} Human-readable display name
 */
function generateDisplayName(pointName, equipment) {
  let name = '';

  // Add equipment prefix if available
  if (equipment?.display) {
    name = equipment.display + ' ';
  }

  // Process the point name
  let processedName = pointName;

  // Split by common delimiters but preserve known abbreviations
  const parts = processedName.split(/[-_.]/);

  const expandedParts = parts
    .filter(p => p && p.length > 0)
    .map(part => {
      // Check for exact abbreviation match
      if (ABBREVIATIONS[part]) {
        return ABBREVIATIONS[part];
      }

      // Check for abbreviation at start of part
      for (const [abbr, expansion] of Object.entries(ABBREVIATIONS)) {
        if (part.startsWith(abbr) && part.length > abbr.length) {
          // e.g., "SaTemp" -> "Supply Air Temp"
          const remainder = part.substring(abbr.length);
          return expansion + ' ' + expandWord(remainder);
        }
      }

      // No abbreviation found, just expand the word
      return expandWord(part);
    });

  name += expandedParts.join(' ');

  // Clean up extra spaces and return
  return name.replace(/\s+/g, ' ').trim();
}

/**
 * Expand a word (handle camelCase, add spaces before capitals)
 * @param {string} word - The word to expand
 * @returns {string} Expanded word
 */
function expandWord(word) {
  if (!word) return '';

  // Check if it's an abbreviation
  if (ABBREVIATIONS[word]) {
    return ABBREVIATIONS[word];
  }

  // Add space before capital letters (camelCase)
  let expanded = word.replace(/([a-z])([A-Z])/g, '$1 $2');

  // Add space before numbers
  expanded = expanded.replace(/([a-zA-Z])(\d)/g, '$1 $2');

  // Capitalize first letter
  expanded = expanded.charAt(0).toUpperCase() + expanded.slice(1);

  return expanded;
}

/**
 * Calculate confidence score for the parsing result
 * @param {Object} equipment - Equipment object
 * @param {string} pointType - Detected point type
 * @param {string} displayName - Generated display name
 * @returns {number} Confidence score (0-100)
 */
function calculateConfidence(equipment, pointType, displayName) {
  let score = 0;

  // Equipment detected
  if (equipment) {
    score += 40;
    if (equipment.id) score += 10;
  }

  // Point type detected
  if (pointType) {
    score += 30;
  }

  // Display name quality
  if (displayName && displayName.length > 10) {
    score += 10;
  }

  // Display name has multiple words (indicates good parsing)
  const wordCount = displayName.split(/\s+/).length;
  if (wordCount >= 3) {
    score += 10;
  }

  return Math.min(score, 100);
}

// ============================================================================
// BATCH PROCESSING
// ============================================================================

/**
 * Process multiple points in batch
 * @param {Array} points - Array of point objects with Name property
 * @returns {Array} Array of enhanced point objects
 */
function enhancePointsBatch(points) {
  if (!Array.isArray(points)) {
    return [];
  }

  return points.map(point => {
    const pointName = point.Name || point.name || '';
    const parsed = extractEquipmentFromPath(pointName);

    return {
      ...point,
      display_name: parsed.displayName,
      unit: parsed.unit,
      equipment: parsed.equipment,
      equipmentId: parsed.equipmentId,
      equipmentDisplay: parsed.equipmentDisplay,
      category: parsed.category,
      pointType: parsed.pointType,
      confidence: parsed.confidence,
      _enhanced: true
    };
  });
}
