// Force Update Script - Clears ALL caches and forces fresh load
// This is a more aggressive version to ensure users get the latest code

(function() {
  const FORCE_VERSION = '2.2.1-emergency';
  const FORCE_KEY = 'building-vitals-force-update';
  
  console.log('[Force Update] Checking for updates...');
  
  const lastForceVersion = localStorage.getItem(FORCE_KEY);
  
  if (lastForceVersion !== FORCE_VERSION) {
    console.log(`[Force Update] New version detected: ${FORCE_VERSION} (was ${lastForceVersion})`);
    
    // 1. Unregister ALL service workers
    if ('serviceWorker' in navigator) {
      navigator.serviceWorker.getRegistrations().then(function(registrations) {
        console.log(`[Force Update] Found ${registrations.length} service workers`);
        registrations.forEach(function(registration) {
          registration.unregister().then(function(success) {
            console.log('[Force Update] Service worker unregistered:', success);
          });
        });
      });
    }
    
    // 2. Clear ALL caches
    if ('caches' in window) {
      caches.keys().then(function(cacheNames) {
        console.log(`[Force Update] Found ${cacheNames.length} caches`);
        return Promise.all(
          cacheNames.map(function(cacheName) {
            return caches.delete(cacheName).then(function() {
              console.log('[Force Update] Cache cleared:', cacheName);
            });
          })
        );
      });
    }
    
    // 3. Clear IndexedDB
    if ('indexedDB' in window) {
      // Clear known IndexedDB databases
      const dbNames = [
        'building-vitals-cache',
        'building-vitals-api-cache',
        'firebaseLocalStorageDb',
        'firebase-heartbeat-database',
        'firebase-installations-database'
      ];
      
      dbNames.forEach(function(dbName) {
        try {
          const deleteReq = indexedDB.deleteDatabase(dbName);
          deleteReq.onsuccess = function() {
            console.log('[Force Update] IndexedDB cleared:', dbName);
          };
          deleteReq.onerror = function() {
            console.log('[Force Update] Failed to clear IndexedDB:', dbName);
          };
        } catch (e) {
          console.error('[Force Update] Error clearing IndexedDB:', e);
        }
      });
    }
    
    // 4. Clear sessionStorage
    try {
      sessionStorage.clear();
      console.log('[Force Update] Session storage cleared');
    } catch (e) {
      console.error('[Force Update] Failed to clear session storage:', e);
    }
    
    // 5. Clear localStorage (preserve auth)
    try {
      const authKeys = [];
      const firebaseAuthPattern = /^firebase:authUser:/;
      const firebaseUIPattern = /^firebaseui:/;
      
      // Save auth data
      for (let i = 0; i < localStorage.length; i++) {
        const key = localStorage.key(i);
        if (key && (firebaseAuthPattern.test(key) || firebaseUIPattern.test(key))) {
          const value = localStorage.getItem(key);
          if (value) {
            authKeys.push({ key, value });
          }
        }
      }
      
      // Clear all
      localStorage.clear();
      
      // Restore auth
      authKeys.forEach(function(item) {
        localStorage.setItem(item.key, item.value);
      });
      
      console.log('[Force Update] Local storage cleared (auth preserved)');
    } catch (e) {
      console.error('[Force Update] Failed to clear local storage:', e);
    }
    
    // 6. Update version
    localStorage.setItem(FORCE_KEY, FORCE_VERSION);
    
    // 7. Force hard reload after a short delay (but prevent loops)
    if (!window.location.search.includes('_force_cleared')) {
      setTimeout(function() {
        console.log('[Force Update] Forcing hard reload...');
        const url = new URL(window.location.href);
        url.searchParams.set('_force_cleared', '1');
        url.searchParams.set('_v', FORCE_VERSION);
        url.searchParams.set('_t', Date.now());
        window.location.href = url.toString();
      }, 100);
    } else {
      console.log('[Force Update] Already reloaded, skipping to prevent loop');
    }
  } else {
    console.log('[Force Update] Already on latest version:', FORCE_VERSION);
  }
})();
