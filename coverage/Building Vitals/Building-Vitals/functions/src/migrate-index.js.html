import { createLogger } from '../../src/utils/logger';

const logger = createLogger('MigrateIndex');

const fs = require('fs');
const path = require('path');

// Read the original index.ts
const originalPath = path.join(__dirname, 'index.ts');
const refactoredPath = path.join(__dirname, 'index-refactored.ts');
const outputPath = path.join(__dirname, 'index-new.ts');

const original = fs.readFileSync(originalPath, 'utf8');
const refactored = fs.readFileSync(refactoredPath, 'utf8');

// Extract all onCall functions and other exports from original
const onCallRegex = /export const \w+ = onCall\([^}]+\}\);/gs;
const onCallFunctions = original.match(onCallRegex) || [];

// Extract scheduled functions
const scheduledRegex = /export const \w+ = onSchedule\([^}]+\}\);/gs;
const scheduledFunctions = original.match(scheduledRegex) || [];

// Extract helper functions like ensureAdmin
const helperRegex = /const ensureAdmin[^}]+\};/gs;
const helperFunctions = original.match(helperRegex) || [];

// Extract interfaces
const interfaceRegex = /interface \w+[^}]+\}/gs;
const interfaces = original.match(interfaceRegex) || [];

// Build the new index.ts
let newIndex = refactored;

// Find the insertion point (after the apiProxy export)
const insertPoint = newIndex.indexOf('// Keep all existing callable functions');

if (insertPoint > -1) {
  const beforeInsert = newIndex.substring(0, insertPoint);
  const afterInsert = newIndex.substring(insertPoint);
  
  // Build the functions section
  let functionsSection = '\n// Helper functions\n';
  functionsSection += helperFunctions.join('\n\n');
  functionsSection += '\n\n// Interfaces\n';
  functionsSection += interfaces.join('\n\n');
  functionsSection += '\n\n// Callable functions\n';
  functionsSection += onCallFunctions.join('\n\n');
  functionsSection += '\n\n// Scheduled functions\n';
  functionsSection += scheduledFunctions.join('\n\n');
  
  newIndex = beforeInsert + functionsSection + '\n\n' + afterInsert;
}

// Write the new index file
fs.writeFileSync(outputPath, newIndex);

logger.info('Migration complete! New index created at:', outputPath);
logger.info('Found:', {
  onCallFunctions: onCallFunctions.length,
  scheduledFunctions: scheduledFunctions.length,
  helperFunctions: helperFunctions.length,
  interfaces: interfaces.length
});
