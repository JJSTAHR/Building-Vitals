/**
 * Binary Data Transfer Endpoints for Timeseries
 * Reduces payload size by 80% for high-resolution data
 */

import * as functions from 'firebase-functions/v1';
import { z } from 'zod';
import * as pako from 'pako';

// ============================================================================
// Types & Schemas
// ============================================================================

const TimeseriesRequestSchema = z.object({
  pointIds: z.array(z.string()).min(1).max(100),
  startTime: z.string().datetime(),
  endTime: z.string().datetime(),
  format: z.enum(['json', 'binary', 'compressed']).optional().default('json'),
  resolution: z.enum(['raw', '1min', '5min', '15min', '1hour']).optional().default('raw'),
});

interface DataPoint {
  timestamp: number;
  value: number | null;
  quality: 'good' | 'uncertain' | 'bad';
}

interface TimeseriesData {
  pointId: string;
  unit?: string;
  data: DataPoint[];
}

// ============================================================================
// Binary Format Helpers
// ============================================================================

/**
 * Convert timeseries data to binary format
 * Format: [header][point_data][point_data]...
 * Header: version(1) + pointCount(4)
 * Point data: pointIdLength(2) + pointId + hasUnit(1) + [unitLength(1) + unit] + sampleCount(4) + samples
 * Sample: timestamp(8) + value(4) + quality(1)
 */
function serializeTimeseriesToBinary(data: TimeseriesData[]): Buffer {
  const buffers: Buffer[] = [];
  
  // Write header
  const header = Buffer.allocUnsafe(5);
  header.writeUInt8(1, 0); // version
  header.writeUInt32LE(data.length, 1); // point count
  buffers.push(header);
  
  // Write each series
  for (const series of data) {
    const pointIdBuffer = Buffer.from(series.pointId, 'utf-8');
    
    // Point ID length and data
    const pointIdHeader = Buffer.allocUnsafe(2);
    pointIdHeader.writeUInt16LE(pointIdBuffer.length, 0);
    buffers.push(pointIdHeader);
    buffers.push(pointIdBuffer);
    
    // Unit (optional)
    if (series.unit) {
      const unitBuffer = Buffer.from(series.unit, 'utf-8');
      const unitHeader = Buffer.allocUnsafe(2);
      unitHeader.writeUInt8(1, 0); // has unit
      unitHeader.writeUInt8(unitBuffer.length, 1);
      buffers.push(unitHeader);
      buffers.push(unitBuffer);
    } else {
      buffers.push(Buffer.from([0])); // no unit
    }
    
    // Sample count
    const sampleCountBuffer = Buffer.allocUnsafe(4);
    sampleCountBuffer.writeUInt32LE(series.data.length, 0);
    buffers.push(sampleCountBuffer);
    
    // Samples
    const samplesBuffer = Buffer.allocUnsafe(series.data.length * 13);
    let offset = 0;
    
    for (const point of series.data) {
      // Timestamp as float64
      samplesBuffer.writeDoubleLE(point.timestamp, offset);
      offset += 8;
      
      // Value as float32 (NaN for null)
      samplesBuffer.writeFloatLE(point.value ?? NaN, offset);
      offset += 4;
      
      // Quality as uint8
      const qualityByte = point.quality === 'good' ? 0 : 
                         point.quality === 'uncertain' ? 1 : 2;
      samplesBuffer.writeUInt8(qualityByte, offset);
      offset += 1;
    }
    
    buffers.push(samplesBuffer);
  }
  
  return Buffer.concat(buffers);
}

/**
 * Compress binary data using gzip
 */
function compressBinary(data: Buffer): Buffer {
  return Buffer.from(pako.gzip(data));
}

// ============================================================================
// Aggregation Helpers
// ============================================================================

function aggregateTimeseries(
  data: DataPoint[],
  resolution: string
): DataPoint[] {
  if (resolution === 'raw' || data.length === 0) return data;
  
  const bucketSize = {
    '1min': 60 * 1000,
    '5min': 5 * 60 * 1000,
    '15min': 15 * 60 * 1000,
    '1hour': 60 * 60 * 1000,
  }[resolution] || 60 * 1000;
  
  const buckets = new Map<number, { values: number[]; qualities: string[] }>();
  
  for (const point of data) {
    if (point.value === null) continue;
    
    const bucketKey = Math.floor(point.timestamp / bucketSize) * bucketSize;
    
    if (!buckets.has(bucketKey)) {
      buckets.set(bucketKey, { values: [], qualities: [] });
    }
    
    const bucket = buckets.get(bucketKey)!;
    bucket.values.push(point.value);
    bucket.qualities.push(point.quality);
  }
  
  return Array.from(buckets.entries())
    .map(([timestamp, bucket]) => ({
      timestamp,
      value: bucket.values.reduce((a, b) => a + b, 0) / bucket.values.length,
      quality: bucket.qualities.includes('bad') ? 'bad' as const :
               bucket.qualities.includes('uncertain') ? 'uncertain' as const : 
               'good' as const,
    }))
    .sort((a, b) => a.timestamp - b.timestamp);
}

export const getTimeseriesBinary = functions
  .runWith({
    memory: '2GB',
    timeoutSeconds: 300,
  })
  .https.onRequest(async (req: any, res: any) => {
    // Enable CORS
    res.set('Access-Control-Allow-Origin', '*');
    res.set('Access-Control-Allow-Methods', 'GET, POST, OPTIONS');
    res.set('Access-Control-Allow-Headers', 'Content-Type, Authorization');
    
    if (req.method === 'OPTIONS') {
      res.status(204).send('');
      return;
    }
    
    try {
      // Parse and validate request
      const params = req.method === 'POST' ? req.body : req.query;
      const validated = TimeseriesRequestSchema.parse(params);
      
      // Fetch data from Firestore or your data source
      const timeseriesData: TimeseriesData[] = [];
      
      for (const pointId of validated.pointIds) {
        // TODO: Replace with actual data fetching
        const mockData: DataPoint[] = [];
        const startTime = new Date(validated.startTime).getTime();
        const endTime = new Date(validated.endTime).getTime();
        const interval = 60000; // 1 minute
        
        for (let t = startTime; t <= endTime; t += interval) {
          mockData.push({
            timestamp: t,
            value: Math.random() * 100,
            quality: Math.random() > 0.95 ? 'uncertain' : 'good',
          });
        }
        
        // Apply aggregation if requested
        const aggregatedData = aggregateTimeseries(mockData, validated.resolution);
        
        timeseriesData.push({
          pointId,
          unit: 'kW', // Mock unit
          data: aggregatedData,
        });
      }
      
      // Format response based on requested format
      switch (validated.format) {
        case 'binary': {
          const binaryData = serializeTimeseriesToBinary(timeseriesData);
          res.set('Content-Type', 'application/octet-stream');
          res.set('Content-Length', String(binaryData.length));
          res.set('X-Data-Points', String(timeseriesData.reduce((sum, ts) => sum + ts.data.length, 0)));
          res.send(binaryData);
          break;
        }
        
        case 'compressed': {
          const binaryData = serializeTimeseriesToBinary(timeseriesData);
          const compressed = compressBinary(binaryData);
          res.set('Content-Type', 'application/octet-stream');
          res.set('Content-Encoding', 'gzip');
          res.set('Content-Length', String(compressed.length));
          res.set('X-Original-Size', String(binaryData.length));
          res.set('X-Compression-Ratio', String((1 - compressed.length / binaryData.length) * 100).slice(0, 5) + '%');
          res.set('X-Data-Points', String(timeseriesData.reduce((sum, ts) => sum + ts.data.length, 0)));
          res.send(compressed);
          break;
        }
        
        default: {
          // JSON format (fallback)
          res.json({
            format: 'json',
            pointCount: timeseriesData.length,
            totalDataPoints: timeseriesData.reduce((sum, ts) => sum + ts.data.length, 0),
            data: timeseriesData,
          });
        }
      }
    } catch (error) {
      console.error('Binary endpoint error:', error);
      res.status(400).json({
        error: error instanceof Error ? error.message : 'Unknown error',
      });
    }
  });

export const getTimeseriesChunked = functions
  .runWith({
    memory: '2GB',
    timeoutSeconds: 540, // 9 minutes
  })
  .https.onRequest(async (req: any, res: any) => {
    res.set('Access-Control-Allow-Origin', '*');
    res.set('Access-Control-Allow-Methods', 'GET, POST, OPTIONS');
    res.set('Access-Control-Allow-Headers', 'Content-Type, Authorization');
    
    if (req.method === 'OPTIONS') {
      res.status(204).send('');
      return;
    }
    
    try {
      const params = req.method === 'POST' ? req.body : req.query;
      const validated = TimeseriesRequestSchema.parse(params);
      
      // Set up SSE (Server-Sent Events) for streaming
      res.set('Content-Type', 'text/event-stream');
      res.set('Cache-Control', 'no-cache');
      res.set('Connection', 'keep-alive');
      res.set('X-Accel-Buffering', 'no'); // Disable Nginx buffering
      
      // Send initial metadata
      res.write(`event: metadata\n`);
      res.write(`data: ${JSON.stringify({
        format: validated.format,
        pointCount: validated.pointIds.length,
        resolution: validated.resolution,
      })}\n\n`);
      
      // Process each point and stream results
      for (let i = 0; i < validated.pointIds.length; i++) {
        const pointId = validated.pointIds[i];
        
        // Fetch data for this point
        const mockData: DataPoint[] = [];
        const startTime = new Date(validated.startTime).getTime();
        const endTime = new Date(validated.endTime).getTime();
        const interval = 60000; // 1 minute
        
        for (let t = startTime; t <= endTime; t += interval) {
          mockData.push({
            timestamp: t,
            value: Math.random() * 100,
            quality: Math.random() > 0.95 ? 'uncertain' : 'good',
          });
        }
        
        const aggregatedData = aggregateTimeseries(mockData, validated.resolution);
        
        // Send progress update
        res.write(`event: progress\n`);
        res.write(`data: ${JSON.stringify({
          current: i + 1,
          total: validated.pointIds.length,
          percentage: ((i + 1) / validated.pointIds.length) * 100,
        })}\n\n`);
        
        // Send data chunk
        if (validated.format === 'binary' || validated.format === 'compressed') {
          const chunkData = serializeTimeseriesToBinary([{
            pointId,
            unit: 'kW',
            data: aggregatedData,
          }]);
          
          const finalData = validated.format === 'compressed' 
            ? compressBinary(chunkData) 
            : chunkData;
          
          res.write(`event: data\n`);
          res.write(`data: ${JSON.stringify({
            pointId,
            format: validated.format,
            binary: finalData.toString('base64'),
            dataPoints: aggregatedData.length,
          })}\n\n`);
        } else {
          res.write(`event: data\n`);
          res.write(`data: ${JSON.stringify({
            pointId,
            unit: 'kW',
            data: aggregatedData,
          })}\n\n`);
        }
        
        // Prevent timeout by yielding control
        await new Promise(resolve => setTimeout(resolve, 10));
      }
      
      // Send completion event
      res.write(`event: complete\n`);
      res.write(`data: ${JSON.stringify({ status: 'success' })}\n\n`);
      
      res.end();
    } catch (error) {
      console.error('Chunked endpoint error:', error);
      res.write(`event: error\n`);
      res.write(`data: ${JSON.stringify({
        error: error instanceof Error ? error.message : 'Unknown error',
      })}\n\n`);
      res.end();
    }
  });
