// Configuration interface
interface Config {
  // API URLs
  aceApiUrl: string;
  appUrl: string;

  // Service authentication
  serviceAuthSecret?: string;

  // Request settings
  requestTimeout: number;
  
  // CORS settings
  allowedOrigins: string[];

  // Rate limiting
  rateLimitWindow: number;
  rateLimitMaxRequests: number;

  // Environment
  environment: "development" | "production" | "staging";
  
  // Firebase settings
  firebaseProjectId: string;

  // Email settings (if needed)
  emailFrom?: string;
  emailApiKey?: string;

  // Feature flags
  enableDetailedErrors: boolean;
  enableMetrics: boolean;
  
  // Redis settings
  redisUrl?: string;
  enableRedisCache: boolean;
}

// Load and validate configuration
const loadConfig = (): Config => {
  // Get environment
  const environment = process.env.NODE_ENV === "production" ? "production" : 
                     process.env.NODE_ENV === "staging" ? "staging" : "development";

  // Build configuration object
  const config: Config = {
    // API URLs
    aceApiUrl: process.env.ACE_API_URL || "https://manager.aceiot.cloud/api",
    
    
    appUrl: process.env.APP_URL || "https://building-vitals.web.app",

    // Service authentication
    serviceAuthSecret: process.env.SERVICE_AUTH_SECRET,

    // Request settings
    requestTimeout: parseInt(process.env.REQUEST_TIMEOUT || "30000"),

    // CORS settings
    allowedOrigins: [
      "https://building-vitals.web.app",
      "https://building-vitals.firebaseapp.com",
      ...(environment !== "production" ? [
        "http://localhost:3000",
        "http://localhost:5173",
        "http://localhost:4173",
        "http://localhost:4174"
      ] : [])
    ],

    // Rate limiting
    rateLimitWindow: parseInt(process.env.RATE_LIMIT_WINDOW || "300000"), // 5 minutes
    rateLimitMaxRequests: parseInt(process.env.RATE_LIMIT_MAX_REQUESTS || "100"),

    // Environment
    environment,

    // Firebase settings
    firebaseProjectId: process.env.GCLOUD_PROJECT || 
                      process.env.GCP_PROJECT || 
                      "building-vitals",

    // Email settings
    emailFrom: process.env.EMAIL_FROM || "noreply@building-vitals.com",
    
    emailApiKey: process.env.EMAIL_API_KEY,

    // Feature flags
    enableDetailedErrors: environment === "development",
    enableMetrics: process.env.ENABLE_METRICS === "true" || environment === "production",
    
    // Redis settings
    redisUrl: process.env.REDIS_URL,
    enableRedisCache: process.env.ENABLE_REDIS_CACHE !== "false" // Default to true if Redis URL is provided
  };

  // Validate required configuration
  validateConfig(config);

  return config;
};

// Validate configuration
const validateConfig = (config: Config): void => {
  const requiredFields: (keyof Config)[] = [
    "aceApiUrl",
    "firebaseProjectId"
  ];

  const missingFields = requiredFields.filter(field => !config[field]);

  if (missingFields.length > 0) {
    throw new Error(`Missing required configuration fields: ${missingFields.join(", ")}`);
  }

  // Validate URLs
  try {
    new URL(config.aceApiUrl);
  } catch (error) {
    throw new Error("Invalid URL in configuration");
  }

  // Validate numeric values
  if (config.requestTimeout < 1000 || config.requestTimeout > 300000) {
    throw new Error("Request timeout must be between 1 and 300 seconds");
  }

  if (config.rateLimitMaxRequests < 1 || config.rateLimitMaxRequests > 10000) {
    throw new Error("Rate limit max requests must be between 1 and 10000");
  }
};

// Export singleton configuration
export const config = loadConfig();

// Helper function to get configuration value with fallback
export const getConfigValue = <K extends keyof Config>(
  key: K,
  fallback?: Config[K]
): Config[K] => {
  const value = config[key];
  if (value !== undefined) {
    return value;
  }
  if (fallback !== undefined) {
    return fallback;
  }
  throw new Error(`Configuration value for '${String(key)}' is not defined and no fallback provided`);
};

// Helper to check if in production
export const isProduction = (): boolean => {
  return config.environment === "production";
};

// Helper to check if in development
export const isDevelopment = (): boolean => {
  return config.environment === "development";
};

// Export configuration type for use in other modules
export type { Config };
