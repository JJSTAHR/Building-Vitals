/**
 * Get runtime options for v1 functions
 */
export function getRuntimeOptions(functionName: string): RuntimeOptions {
  const profile = FUNCTION_PROFILES[functionName] || "medium";
  
  return {
    memory: MEMORY_SIZES[profile] as any,
    timeoutSeconds: TIMEOUT_SIZES[profile],
    maxInstances: MAX_INSTANCES[profile]
  };
}

/**
 * Get schedule options for v2 scheduled functions
 */
export function getScheduleOptions(
  functionName: string, 
  schedule: string,
  timeZone: string = "UTC"
): ScheduleOptions {
  const profile = FUNCTION_PROFILES[functionName] || "medium";
  
  return {
    schedule,
    timeZone,
    memory: MEMORY_SIZES[profile] as any,
    timeoutSeconds: TIMEOUT_SIZES[profile],
    maxInstances: MAX_INSTANCES[profile]
  };
}

/**
 * Get HTTPS options for v2 HTTP functions
 */
export function getHttpsOptions(functionName: string): HttpsOptions {
  const profile = FUNCTION_PROFILES[functionName] || "medium";
  
  return {
    memory: MEMORY_SIZES[profile] as any,
    timeoutSeconds: TIMEOUT_SIZES[profile],
    maxInstances: MAX_INSTANCES[profile],
    // Additional v2 options
    concurrency: profile === "minimal" || profile === "light" ? 80 : 40,
    cpu: profile === "intensive" ? 2 : 1
  };
}

/**
 * Lazy load heavy dependencies
 */
export class DependencyLoader {
  private static instances: Map<string, any> = new Map();
  
  /**
   * Load a dependency lazily with caching
   */
  static async load<T>(name: string, loader: () => Promise<T>): Promise<T> {
    if (!this.instances.has(name)) {
      const instance = await loader();
      this.instances.set(name, instance);
    }
    return this.instances.get(name) as T;
  }
  
  /**
   * Clear cached dependencies (for testing)
   */
  static clear(): void {
    this.instances.clear();
  }
}

/**
 * Memory monitoring utilities
 */
export class MemoryMonitor {
  
  /**
   * Log current memory usage
   */
  static logMemoryUsage(context: string): void {
    const usage = process.memoryUsage();
    const heapUsedMB = Math.round(usage.heapUsed / 1024 / 1024);
    const heapTotalMB = Math.round(usage.heapTotal / 1024 / 1024);
    const rssMB = Math.round(usage.rss / 1024 / 1024);
    
    console.log(`[${context}] Memory: Heap ${heapUsedMB}/${heapTotalMB}MB, RSS: ${rssMB}MB`);
  }
  
  /**
   * Check if memory usage is high
   */
  static isMemoryHigh(threshold: number = 0.8): boolean {
    const usage = process.memoryUsage();
    return usage.heapUsed / usage.heapTotal > threshold;
  }
  
  /**
   * Force garbage collection if available
   */
  static forceGC(): void {
    if (global.gc) {
      global.gc();
      this.logMemoryUsage("After GC");
    }
  }
}

/**
 * Function initialization optimizations
 */
export class FunctionOptimizer {
  /**
   * Optimize function cold starts
   */
  static optimizeColdStart(): void {
    // Pre-warm common dependencies
    require("firebase-admin");
    require("firebase-functions");
    
    // Set process priorities
    if (process.platform !== "win32") {
      try {
        (process as any).nice(10); // Lower priority for background tasks
      } catch (e) {
        // Ignore on platforms that don't support nice
      }
    }
  }
  
  /**
   * Clean up resources before function termination
   */
  static async cleanup(): Promise<void> {
    // Clear any intervals or timeouts
    // @ts-ignore
    const activeHandles = process._getActiveHandles();
    activeHandles.forEach((handle: any) => {
      if (handle.constructor.name === "Timer") {
        clearTimeout(handle);
      }
    });
    
    // Force garbage collection
    MemoryMonitor.forceGC();
  }
}

/**
 * Dynamic memory adjustment based on historical usage
 */
export class MemoryAutoAdjuster {
  private static readonly ADJUSTMENT_THRESHOLD = 0.8; // 80% usage triggers adjustment
  private static readonly MIN_SAMPLES = 10; // Minimum samples before adjusting
  
  /**
   * Record memory usage for a function
   */
  static async recordUsage(functionName: string, usagePercent: number): Promise<void> {
    try {
      const doc = admin.firestore()
        .collection("function_metrics")
        .doc(functionName);
      
      await doc.update({
        usageHistory: admin.firestore.FieldValue.arrayUnion({
          timestamp: Date.now(),
          usage: usagePercent,
          memory: MEMORY_SIZES[FUNCTION_PROFILES[functionName] || "medium"]
        }),
        lastUpdated: admin.firestore.FieldValue.serverTimestamp()
      });
    } catch (error) {
      logger.warn(`Failed to record memory usage for ${functionName}:`, error);
    }
  }
  
  /**
   * Get recommended memory size based on usage history
   */
  static async getRecommendedMemory(functionName: string): Promise<MemoryProfile | null> {
    try {
      const doc = await admin.firestore()
        .collection("function_metrics")
        .doc(functionName)
        .get();
      
      if (!doc.exists) return null;
      
      const data = doc.data();
      const history = data?.usageHistory || [];
      
      if (history.length < this.MIN_SAMPLES) return null;
      
      // Get recent samples (last 20)
      const recentSamples = history.slice(-20);
      const avgUsage = recentSamples.reduce((sum: number, sample: any) => sum + sample.usage, 0) / recentSamples.length;
      
      const currentProfile = FUNCTION_PROFILES[functionName] || "medium";
      
      // Recommend adjustment if consistently over/under utilized
      if (avgUsage > this.ADJUSTMENT_THRESHOLD) {
        // Need more memory
        return this.getNextHigherProfile(currentProfile);
      } else if (avgUsage < 0.3) {
        // Can use less memory
        return this.getNextLowerProfile(currentProfile);
      }
      
      return null;
    } catch (error) {
      logger.warn(`Failed to get memory recommendation for ${functionName}:`, error);
      return null;
    }
  }
  
  private static getNextHigherProfile(current: MemoryProfile): MemoryProfile | null {
    const profiles: MemoryProfile[] = ["minimal", "light", "medium", "heavy", "intensive"];
    const currentIndex = profiles.indexOf(current);
    return currentIndex < profiles.length - 1 ? profiles[currentIndex + 1] : null;
  }
  
  private static getNextLowerProfile(current: MemoryProfile): MemoryProfile | null {
    const profiles: MemoryProfile[] = ["minimal", "light", "medium", "heavy", "intensive"];
    const currentIndex = profiles.indexOf(current);
    return currentIndex > 0 ? profiles[currentIndex - 1] : null;
  }
}

/**

export class FunctionPreWarmer {
  private static readonly PREWARM_FUNCTIONS = [
    "getUserProfileOnLogin",
    "verifyTokenAndGetUser",
    "getTimeSeriesData"

  /**
   * Pre-warm critical functions

  static async prewarmCriticalFunctions(): Promise<void> {
    for (const functionName of this.PREWARM_FUNCTIONS) {
      try {
        // Make a lightweight health check call to warm the function
        logger.info(`Pre-warming function: ${functionName}`);
      } catch (error) {
        logger.warn(`Failed to pre-warm ${functionName}:`, error);
      }
    }
  }
}
