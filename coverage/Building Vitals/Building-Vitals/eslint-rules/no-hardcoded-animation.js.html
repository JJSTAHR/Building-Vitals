/**
 * ESLint Rule: no-hardcoded-animation
 *
 * Flags hardcoded animation durations
 *
 * Purpose: Ensure performance-based animation modes are used
 * Detection: animation: true, animationDuration: 1000 without using ANIMATION_MODES
 * Fix: Suggest getAnimationConfig() from useBaseChartOptions
 *
 * @example
 * // ❌ Hardcoded
 * animation: true, animationDuration: 1000
 *
 * // ✅ Use tokens
 * ...baseOptions.getAnimationConfig(dataPoints)
 */

module.exports = {
  meta: {
    type: 'problem',
    docs: {
      description: 'Prevent hardcoded animation configurations',
      category: 'Performance',
      recommended: true,
    },
    fixable: null,
    schema: [],
    messages: {
      hardcodedAnimation: 'Use CHART_DESIGN_TOKENS.animation or getAnimationConfig() instead of hardcoded animation values',
      suggestAnimationMode: 'Use CHART_DESIGN_TOKENS.animation.{{mode}} or baseOptions.getAnimationConfig(dataPointCount)',
    },
  },

  create(context) {
    const filename = context.getFilename();

    // Only check chart files
    if (!filename.includes('src/components/charts') && !filename.includes('src\\components\\charts')) {
      return {};
    }

    // Skip test files and design tokens file
    if (
      filename.includes('.test.') ||
      filename.includes('.spec.') ||
      filename.includes('chartDesignTokens') ||
      filename.includes('useBaseChartOptions')
    ) {
      return {};
    }

    // Animation-related properties
    const animationProperties = new Set([
      'animation',
      'animationDuration',
      'animationEasing',
      'animationDelay',
      'progressive',
      'progressiveThreshold',
    ]);

    /**
     * Check if object has multiple animation properties (indicates manual config)
     */
    function hasMultipleAnimationProps(objectNode) {
      let animPropCount = 0;

      if (objectNode.type !== 'ObjectExpression') {
        return false;
      }

      objectNode.properties.forEach((prop) => {
        if (prop.type === 'Property') {
          const key = prop.key.name || prop.key.value;
          if (animationProperties.has(key)) {
            animPropCount++;
          }
        }
      });

      return animPropCount >= 2;
    }

    /**
     * Check if code uses design token animation modes
     */
    function usesDesignTokens(node) {
      const sourceCode = context.getSourceCode();
      const text = sourceCode.getText(node);

      return (
        text.includes('CHART_DESIGN_TOKENS.animation') ||
        text.includes('ANIMATION_MODES') ||
        text.includes('getAnimationConfig')
      );
    }

    return {
      Property(node) {
        const key = node.key.name || node.key.value;

        // Only check animation properties
        if (!animationProperties.has(key)) {
          return;
        }

        // Skip if already using design tokens
        if (usesDesignTokens(node)) {
          return;
        }

        // Get parent object to check for multiple animation props
        let parentObj = node.parent;
        while (parentObj && parentObj.type !== 'ObjectExpression') {
          parentObj = parentObj.parent;
        }

        if (!parentObj) {
          return;
        }

        // Flag if multiple animation properties are being set manually
        if (hasMultipleAnimationProps(parentObj)) {
          // Determine suggested mode based on value
          let suggestedMode = 'quality';

          if (key === 'animation' && node.value.type === 'Literal') {
            suggestedMode = node.value.value ? 'quality' : 'performance';
          }

          if (key === 'animationDuration' && node.value.type === 'Literal') {
            const duration = node.value.value;
            if (duration === 0) suggestedMode = 'performance';
            else if (duration < 500) suggestedMode = 'balanced';
            else suggestedMode = 'quality';
          }

          context.report({
            node: parentObj,
            messageId: 'hardcodedAnimation',
            // Note: Consider using CHART_DESIGN_TOKENS.animation.{{suggestedMode}}
          });
        }
      },

      // Check for animation config in option objects
      ObjectExpression(node) {
        // Must be within a chart context
        let isChartOption = false;
        let parent = node.parent;
        let depth = 0;

        while (parent && depth < 5) {
          if (parent.type === 'VariableDeclarator' && parent.id) {
            const varName = parent.id.name;
            if (varName && (varName === 'option' || varName === 'baseOptions' || varName.includes('Option'))) {
              isChartOption = true;
              break;
            }
          }
          parent = parent.parent;
          depth++;
        }

        if (!isChartOption) {
          return;
        }

        // Check if this object has animation config
        if (hasMultipleAnimationProps(node) && !usesDesignTokens(node)) {
          context.report({
            node,
            messageId: 'hardcodedAnimation',
            // Note: Use animation mode based on data size (quality/balanced/performance)
          });
        }
      },
    };
  },
};
