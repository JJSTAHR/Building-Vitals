import type { Middleware } from '@reduxjs/toolkit';
import type { RootState } from '../index';
import { PERSISTENCE_CONFIG } from '../api/cacheConfig';
import { persistCacheEntry, loadPersistedCacheEntry, clearPersistedCache } from '../api/cacheUtils';

// Middleware for cache persistence
export const cachePersistenceMiddleware: Middleware<{}, RootState> =
  (store) => (next) => (action) => {
    const result = next(action);

    // Handle fulfilled queries
    if (action.type.endsWith('/fulfilled') && action.meta?.arg?.endpointName) {
      const { endpointName, originalArgs } = action.meta.arg;

      // Check if this endpoint should be persisted
      if (PERSISTENCE_CONFIG.PERSIST_ENDPOINTS.includes(endpointName)) {
        const cacheKey = `${PERSISTENCE_CONFIG.KEY_PREFIX}${endpointName}_${JSON.stringify(originalArgs)}`;

        // Check blacklist patterns
        const isBlacklisted = PERSISTENCE_CONFIG.BLACKLIST_PATTERNS.some((pattern) =>
          pattern.test(cacheKey)
        );

        if (!isBlacklisted) {
          persistCacheEntry(cacheKey, action.payload, Date.now());
        }
      }
    }

    // Handle cache cleanup on logout
    if (action.type === 'auth/logout') {
      clearPersistedCache();
    }

    return result;
  };

// Middleware for cache warming on app start
export const cacheWarmingMiddleware: Middleware<{}, RootState> = (store) => (next) => (action) => {
  const result = next(action);

  // Warm cache when app initializes
  if (action.type === 'auth/loginSuccess' || action.type === 'app/initialized') {
    setTimeout(() => {
      warmCacheFromPersistence(store.dispatch);
    }, 100);
  }

  return result;
};

// Function to warm cache from persisted data
async function warmCacheFromPersistence(dispatch: any) {
  const keys = Object.keys(localStorage).filter((key) =>
    key.startsWith(PERSISTENCE_CONFIG.KEY_PREFIX)
  );

  for (const key of keys) {
    const cached = loadPersistedCacheEntry(key);
    if (cached) {
      // Check if data is still valid
      const age = Date.now() - cached.timestamp;
      if (age < PERSISTENCE_CONFIG.MAX_AGE) {
        // Parse endpoint and args from key
        const keyParts = key.replace(PERSISTENCE_CONFIG.KEY_PREFIX, '').split('_');
        const endpointName = keyParts[0];
        const argsJson = keyParts.slice(1).join('_');

        try {
          const args = JSON.parse(argsJson);

          // Dispatch action to populate cache
          dispatch({
            type: `api/queries/${endpointName}/fulfilled`,
            payload: cached.data,
            meta: {
              arg: {
                endpointName,
                originalArgs: args,
              },
              requestId: `persisted-${Date.now()}`,
              requestStatus: 'fulfilled',
            },
          });
        } catch (error) {
          console.error('Failed to restore cache entry:', error);
        }
      } else {
        // Remove stale entry
        localStorage.removeItem(key);
      }
    }
  }
}

// Middleware for cache size management
export const cacheSizeManagementMiddleware: Middleware<{}, RootState> =
  (store) => (next) => (action) => {
    const result = next(action);

    // Periodically check cache size
    if (action.type.endsWith('/fulfilled')) {
      checkAndCleanCache();
    }

    return result;
  };

// Function to check and clean cache if needed
let lastCacheCheck = 0;
const CACHE_CHECK_INTERVAL = 60000; // 1 minute

function checkAndCleanCache() {
  const now = Date.now();
  if (now - lastCacheCheck < CACHE_CHECK_INTERVAL) {
    return;
  }

  lastCacheCheck = now;

  // Calculate total cache size
  let totalSize = 0;
  const cacheKeys: Array<{ key: string; size: number; timestamp: number }> = [];

  Object.keys(localStorage).forEach((key) => {
    if (key.startsWith(PERSISTENCE_CONFIG.KEY_PREFIX)) {
      const value = localStorage.getItem(key) || '';
      const size = new Blob([value]).size;
      totalSize += size;

      try {
        const cached = JSON.parse(value);
        cacheKeys.push({ key, size, timestamp: cached.timestamp || 0 });
      } catch {
        // Invalid entry, mark for removal
        cacheKeys.push({ key, size, timestamp: 0 });
      }
    }
  });

  // Check if we need to clean up (e.g., > 10MB)
  const MAX_CACHE_SIZE = 10 * 1024 * 1024; // 10MB
  if (totalSize > MAX_CACHE_SIZE) {
    // Sort by timestamp (oldest first)
    cacheKeys.sort((a, b) => a.timestamp - b.timestamp);

    // Remove oldest entries until we're under the limit
    let removedSize = 0;
    for (const { key, size } of cacheKeys) {
      if (totalSize - removedSize <= MAX_CACHE_SIZE * 0.8) {
        break; // Keep 20% buffer
      }

      localStorage.removeItem(key);
      removedSize += size;
    }

    console.log(`Cache cleanup: removed ${removedSize} bytes`);
  }
}
