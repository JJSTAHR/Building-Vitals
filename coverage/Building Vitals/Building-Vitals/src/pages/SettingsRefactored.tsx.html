import React, { useState, useCallback, useMemo } from 'react';
import { Box, Typography, Paper, Alert, Snackbar } from '@mui/material';
import { NetworkCheck as NetworkCheckIcon } from '@mui/icons-material';
import { DashboardMigrationTool } from '../components/dashboard/DashboardMigrationTool';
import { ApiDiagnosticsPanel } from '../components/common/ApiDiagnosticsPanel';
import { useAppSelector, useAppDispatch } from '../store/hooks';
import { setDarkMode, setAutoRefresh } from '../store/slices/uiSlice';

// Import the new components
import { ApiTokenSection } from '../components/settings/ApiTokenSection';
import { UiSettingsSection } from '../components/settings/UiSettingsSection';
import { DashboardMaintenanceSection } from '../components/settings/DashboardMaintenanceSection';
import { AboutSection } from '../components/settings/AboutSection';

// Import the custom hook
import { useTokenManagement } from '../hooks/useTokenManagement';

interface SnackbarState {
  open: boolean;
  message: string;
  severity: 'success' | 'error';
}

const SettingsRefactored: React.FC = () => {
  const dispatch = useAppDispatch();
  const { darkMode, autoRefresh } = useAppSelector((state) => state.ui);
  const { aceToken } = useAppSelector((state) => state.auth);

  // Snackbar state
  const [snackbar, setSnackbar] = useState<SnackbarState>({
    open: false,
    message: '',
    severity: 'success',
  });

  // Migration tool state
  const [showMigrationTool, setShowMigrationTool] = useState(false);

  // Snackbar handlers
  const handleSnackbarOpen = useCallback((message: string, severity: 'success' | 'error') => {
    setSnackbar({ open: true, message, severity });
  }, []);

  const handleSnackbarClose = useCallback(() => {
    setSnackbar((prev) => ({ ...prev, open: false }));
  }, []);

  // Token management with custom hook
  const tokenManagement = useTokenManagement(
    (message: string) => handleSnackbarOpen(message, 'success'),
    (message: string) => handleSnackbarOpen(message, 'error')
  );

  // UI Settings handlers
  const handleDarkModeToggle = useCallback(() => {
    dispatch(setDarkMode(!darkMode));
  }, [dispatch, darkMode]);

  const handleAutoRefreshToggle = useCallback(() => {
    dispatch(setAutoRefresh(!autoRefresh));
  }, [dispatch, autoRefresh]);

  // Migration tool handlers
  const handleOpenMigrationTool = useCallback(() => {
    setShowMigrationTool(true);
  }, []);

  const handleCloseMigrationTool = useCallback(() => {
    setShowMigrationTool(false);
  }, []);

  // Check if using direct API mode
  const isDirect = useMemo(() => import.meta.env.VITE_USE_DIRECT_API === 'true', []);

  return (
    <Box className="settings-page-container page-container" sx={{ maxWidth: 800, mx: 'auto' }}>
      <Typography gutterBottom component="h1" variant="h4">
        Settings
      </Typography>

      {/* API Token Settings */}
      <ApiTokenSection
        aceToken={aceToken}
        isDirect={isDirect}
        isInitialLoading={tokenManagement.isInitialLoading}
        isTokenLoading={tokenManagement.isTokenLoading}
        tokenError={tokenManagement.tokenError || ''}
        tokenInput={tokenManagement.tokenInput}
        onTokenChange={tokenManagement.handleTokenInputChange}
        onTokenClear={tokenManagement.handleTokenClear}
        onTokenSave={tokenManagement.handleSaveToken}
      />

      {/* API Diagnostics */}
      <Paper sx={{ p: 3, mb: 3 }}>
        <Typography
          gutterBottom
          sx={{ display: 'flex', alignItems: 'center', gap: 1 }}
          variant="h6"
        >
          <NetworkCheckIcon /> API Connection Status
        </Typography>
        <Typography paragraph color="text.secondary" variant="body2">
          Check your API connection status and diagnose common issues.
        </Typography>
        <ApiDiagnosticsPanel autoRun={!!aceToken} compact={false} />
      </Paper>

      {/* UI Settings */}
      <UiSettingsSection
        autoRefresh={autoRefresh}
        darkMode={darkMode}
        onAutoRefreshToggle={handleAutoRefreshToggle}
        onDarkModeToggle={handleDarkModeToggle}
      />

      {/* Dashboard Maintenance */}
      <DashboardMaintenanceSection onOpenMigrationTool={handleOpenMigrationTool} />

      {/* About Section */}
      <AboutSection />

      {/* Snackbar for notifications */}
      <Snackbar
        anchorOrigin={{ vertical: 'bottom', horizontal: 'center' }}
        autoHideDuration={6000}
        open={snackbar.open}
        onClose={handleSnackbarClose}
      >
        <Alert severity={snackbar.severity} sx={{ width: '100%' }} onClose={handleSnackbarClose}>
          {snackbar.message}
        </Alert>
      </Snackbar>

      {/* Dashboard Migration Tool */}
      <DashboardMigrationTool open={showMigrationTool} onClose={handleCloseMigrationTool} />
    </Box>
  );
};

export default SettingsRefactored;
