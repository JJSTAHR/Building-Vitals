import { messaging } from './firebase';
import type { Messaging } from 'firebase/messaging';
import { getToken, onMessage } from 'firebase/messaging';
import { logAnalyticsEvent } from './analytics';

// Your Firebase Cloud Messaging Vapid Key (get from Firebase Console)
const VAPID_KEY = import.meta.env.VITE_FCM_VAPID_KEY || '';

export interface NotificationPayload {
  title: string;
  body: string;
  icon?: string;
  badge?: string;
  data?: Record<string, any>;
}

class CloudMessagingService {
  private messaging: Messaging | null = null;
  private isInitialized = false;
  private notificationHandlers: ((payload: NotificationPayload) => void)[] = [];

  /**
   * Initialize Cloud Messaging
   */
  async initialize(): Promise<void> {
    if (this.isInitialized || !messaging) {
      return;
    }

    try {
      this.messaging = messaging;

      // Handle foreground messages
      onMessage(this.messaging!, (payload) => {
        const notification: NotificationPayload = {
          title: payload.notification?.title || 'New Alert',
          body: payload.notification?.body || '',
          icon: payload.notification?.icon,
          data: payload.data,
        };

        // Call all registered handlers
        this.notificationHandlers.forEach((handler) => handler(notification));

        // Track notification received
        logAnalyticsEvent('notification_received', {
          title: notification.title,
          type: 'foreground',
          data_type: payload.data?.type || 'unknown',
        });
      });

      this.isInitialized = true;
    } catch (error) {
      throw error;
    }
  }

  /**
   * Request notification permission and get FCM token
   */
  async requestPermissionAndGetToken(): Promise<string | null> {
    try {
      // Request notification permission
      const permission = await Notification.requestPermission();

      if (permission !== 'granted') {
        return null;
      }

      // Get FCM token
      if (!this.messaging) {
        await this.initialize();
      }

      if (!this.messaging) {
        throw new Error('Messaging not initialized');
      }

      const token = await getToken(this.messaging, {
        vapidKey: VAPID_KEY,
      });

      if (token) {
        // Track permission granted
        logAnalyticsEvent('notification_permission_granted');

        return token;
      } else {
        return null;
      }
    } catch (error) {
      // Track permission error
      logAnalyticsEvent('notification_permission_error', {
        error: (error as Error).message,
      });

      throw error;
    }
  }

  /**
   * Register a handler for incoming notifications
   */
  onNotification(handler: (payload: NotificationPayload) => void): () => void {
    this.notificationHandlers.push(handler);

    // Return unsubscribe function
    return () => {
      const index = this.notificationHandlers.indexOf(handler);
      if (index > -1) {
        this.notificationHandlers.splice(index, 1);
      }
    };
  }

  /**
   * Show a notification (useful for testing or local notifications)
   */
  async showNotification(notification: NotificationPayload): Promise<void> {
    if ('Notification' in window && Notification.permission === 'granted') {
      const notif = new Notification(notification.title, {
        body: notification.body,
        icon: notification.icon || '/icon-192x192.png',
        badge: notification.badge || '/icon-192x192.png',
        data: notification.data,
      });

      // Handle notification click
      notif.onclick = () => {
        window.focus();
        notif.close();

        // Track notification click
        logAnalyticsEvent('notification_clicked', {
          title: notification.title,
        });

        // Handle specific actions based on data
        if (notification.data?.dashboardId) {
          window.location.href = `/dashboard/${notification.data.dashboardId}`;
        }
      };
    }
  }

  /**
   * Subscribe to alert notifications for a specific dashboard
   */
  async subscribeToAlerts(dashboardId: string, userId: string): Promise<void> {
    try {
      const token = await this.requestPermissionAndGetToken();

      if (!token) {
        throw new Error('No FCM token available');
      }

      // Save the token to Firestore for the user
      // This would be used by backend functions to send notifications
      const { doc, setDoc } = await import('firebase/firestore');
      const { db } = await import('./firebase');

      await setDoc(
        doc(db, 'fcm_tokens', userId),
        {
          token,
          dashboards: [dashboardId],
          updatedAt: new Date(),
        },
        { merge: true }
      );

      // Track subscription
      logAnalyticsEvent('alert_subscription', {
        dashboard_id: dashboardId,
      });
    } catch (error) {
      throw error;
    }
  }

  /**
   * Unsubscribe from alert notifications
   */
  async unsubscribeFromAlerts(dashboardId: string, userId: string): Promise<void> {
    try {
      const { doc, updateDoc, arrayRemove } = await import('firebase/firestore');
      const { db } = await import('./firebase');

      await updateDoc(doc(db, 'fcm_tokens', userId), {
        dashboards: arrayRemove(dashboardId),
      });

      // Track unsubscription
      logAnalyticsEvent('alert_unsubscription', {
        dashboard_id: dashboardId,
      });
    } catch (error) {
      throw error;
    }
  }
}

// Export singleton instance
export const cloudMessaging = new CloudMessagingService();

// React hook for using cloud messaging
export const useCloudMessaging = () => {
  const [isSupported, setIsSupported] = useState(false);
  const [permission, setPermission] = useState<NotificationPermission>('default');
  const [token, setToken] = useState<string | null>(null);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    // Check if notifications are supported
    const checkSupport = async () => {
      const supported = 'Notification' in window && messaging !== null;
      setIsSupported(supported);

      if (supported) {
        setPermission(Notification.permission);

        // Initialize messaging
        try {
          await cloudMessaging.initialize();
        } catch (error) {}
      }

      setLoading(false);
    };

    checkSupport();
  }, []);

  const requestPermission = async () => {
    if (!isSupported) {
      throw new Error('Notifications not supported');
    }

    setLoading(true);
    try {
      const fcmToken = await cloudMessaging.requestPermissionAndGetToken();
      setToken(fcmToken);
      setPermission(Notification.permission);
      return fcmToken;
    } finally {
      setLoading(false);
    }
  };

  const subscribeToAlerts = async (dashboardId: string, userId: string) => {
    if (!isSupported) {
      throw new Error('Notifications not supported');
    }

    await cloudMessaging.subscribeToAlerts(dashboardId, userId);
  };

  const unsubscribeFromAlerts = async (dashboardId: string, userId: string) => {
    if (!isSupported) {
      throw new Error('Notifications not supported');
    }

    await cloudMessaging.unsubscribeFromAlerts(dashboardId, userId);
  };

  return {
    isSupported,
    permission,
    token,
    loading,
    requestPermission,
    subscribeToAlerts,
    unsubscribeFromAlerts,
    onNotification: cloudMessaging?.onNotification ? cloudMessaging.onNotification.bind(cloudMessaging) : () => {},
    showNotification: cloudMessaging?.showNotification ? cloudMessaging.showNotification.bind(cloudMessaging) : () => {},
  };
};
