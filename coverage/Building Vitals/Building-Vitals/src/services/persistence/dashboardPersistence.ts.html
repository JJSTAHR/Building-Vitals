/**
 * Comprehensive dashboard save function that tries Firestore first,
 * then falls back to localStorage if Firestore fails
 */
export const saveDashboardConfiguration = async (
  userId: string | null,
  dashboardId: string,
  layouts: Layouts,
  widgets: VisualizationConfig[], // Corrected type
  dashboardName?: string,
  // Parameter can be either ApiTimeRange or DashboardTimeRange from different call paths
  timeRangeInput?: ApiTimeRange | DashboardTimeRange,
  customStartDate?: string | null,
  customEndDate?: string | null
): Promise<void> => {
  // Prepare ApiTimeRange for saveDashboardToLocalStorage
  let apiTimeRangeForStorage: ApiTimeRange | undefined = undefined;
  if (timeRangeInput) {
    if (typeof timeRangeInput === 'string') {
      // Convert DashboardTimeRange strings to ApiTimeRange format
      const validApiTimeRanges = ['24h', '3d', '7d', '30d', '90d', '180d', '365d', 'custom'];
      if (validApiTimeRanges.includes(timeRangeInput)) {
        apiTimeRangeForStorage = timeRangeInput as ApiTimeRange;
      } else {
        // For other time range strings (1h, 3h, etc.), treat as custom
        apiTimeRangeForStorage = 'custom';
      }
    } else if ('startTime' in timeRangeInput) {
      // It's an ApiTimeRangeObject
      apiTimeRangeForStorage = timeRangeInput;
    } else if ('start' in (timeRangeInput as any)) {
      // It's a DashboardTimeRangeObject, convert to ApiTimeRangeObject
      const rangeObj = timeRangeInput as any;
      apiTimeRangeForStorage = {
        startTime: new Date(rangeObj.start).toISOString(),
        endTime: new Date(rangeObj.end).toISOString(),
      };
    }
  }

  saveDashboardToLocalStorage(
    userId,
    dashboardId,
    layouts,
    widgets,
    dashboardName,
    apiTimeRangeForStorage, // Pass the explicitly formatted ApiTimeRange
    customStartDate,
    customEndDate
  );

  // Then try to save to Firestore if user is logged in
  if (userId) {
    // Prepare DashboardTimeRange for Firestore
    let firestoreTimeRange: DashboardTimeRange | undefined = undefined;
    if (timeRangeInput) {
      if (typeof timeRangeInput === 'string') {
        firestoreTimeRange = timeRangeInput as DashboardTimeRange;
      } else if ('startTime' in timeRangeInput || 'start' in (timeRangeInput as any)) {
        // For object time ranges, use 'custom' as the time range string
        firestoreTimeRange = 'custom';
      }
    }

    const configForFirestore: Partial<DashboardConfiguration> = {
      layouts,
      widgets,
      name: dashboardName,
      timeRange: firestoreTimeRange, // Pass DashboardTimeRange
      customStartDate,
      customEndDate,
    };
    await saveDashboardToFirestore(userId, dashboardId, configForFirestore);
  }
};

/**
 * Get all dashboards for a user
 */
export const getDashboards = async (userId: string | null): Promise<DashboardMetadata[]> => {
  let dashboards: DashboardMetadata[] = [];

  // Try to get from Firestore first if user is logged in
  if (userId) {
    dashboards = await getDashboardsFromFirestore(userId);
  }

  // Fall back to localStorage if Firestore fails or user is not logged in
  if (dashboards.length === 0) {
    dashboards = getDashboardsFromLocalStorage(userId);
  }

  // If still no dashboards, create a default one
  if (dashboards.length === 0) {
    const now = new Date().toISOString();
    dashboards = [
      {
        id: DEFAULT_DASHBOARD_ID,
        name: DEFAULT_DASHBOARD_NAME,
        createdAt: now,
        lastModified: now,
      },
    ];
  }

  return dashboards;
};

/**
 * Comprehensive dashboard load function that tries Firestore first,
 * then falls back to localStorage if Firestore fails
 */
export const loadDashboardConfiguration = async (
  userId: string | null,
  dashboardId?: string
): Promise<{
  layouts: Layouts;
  widgets: VisualizationConfig[];
  name: string;
  dashboardId: string;
  timeRange?: DashboardTimeRange; // Return type should be DashboardTimeRange for internal consistency
  customStartDate?: string | null;
  customEndDate?: string | null;
} | null> => {
  // Get the dashboard ID (use provided ID or get active dashboard ID)
  const activeDashboardId = dashboardId || getActiveDashboardId(userId);

  let result: DashboardConfiguration | null = null;

  // Try to load from Firestore first if user is logged in
  if (userId) {
    result = await loadDashboardFromFirestore(userId, activeDashboardId);
  }

  // Fall back to localStorage if Firestore fails or user is not logged in
  if (!result) {
    result = loadDashboardFromLocalStorage(userId, activeDashboardId);
  }

  // If still no result, create an empty dashboard
  if (!result) {
    const owner = userId || 'anonymous'; // Determine owner ID
    const now = new Date().toISOString();
    result = {
      dashboardId: activeDashboardId,
      layouts: {},
      widgets: [],
      name: DEFAULT_DASHBOARD_NAME,
      ownerId: owner,
      permissions: {},
      timeRange: '24h' as DashboardTimeRange,
      createdAt: now,
      lastModified: now,
    } as DashboardConfiguration;

    // Save the empty dashboard
    await saveDashboardConfiguration(
      userId,
      activeDashboardId,
      result.layouts,
      result.widgets,
      result.name
    );
  }

  // Add null check before logging
  if (result) {
    // Result exists, proceed
  }

  // Construct the return object explicitly based on the function's return type
  if (result) {
    // result.timeRange is now DashboardTimeRange, which matches the new return type
    return {
      layouts: result.layouts || {},
      widgets: result.widgets || [],
      name: result.name || DEFAULT_DASHBOARD_NAME,
      dashboardId: activeDashboardId,
      timeRange: result.timeRange, // Return DashboardTimeRange directly
      customStartDate: result.customStartDate || null,
      customEndDate: result.customEndDate || null,
    };
  } else {
    // Return null if no configuration could be loaded or created
    return null;
  }
};

/**
 * Create a new dashboard
 */
export const createDashboard = async (userId: string | null, name: string): Promise<string> => {
  // Generate a unique ID
  const dashboardId = `dashboard-${Date.now()}-${Math.floor(Math.random() * 1000)}`;

  // Create an empty dashboard
  await saveDashboardConfiguration(userId, dashboardId, {}, [], name);

  return dashboardId;
};

/**
 * Rename a dashboard
 */
export const renameDashboard = async (
  userId: string | null,
  dashboardId: string,
  newName: string
): Promise<boolean> => {
  try {
    // // Update in Firestore if user is logged in
    if (userId) {
      try {
        await renameDashboardInFirestore(userId, dashboardId, newName);
      } catch (_firestoreError) {
        // Continue with localStorage even if Firestore fails
      }
    }

    // Always update in localStorage
    const _localStorageResult = renameDashboardInLocalStorage(userId, dashboardId, newName);
    // After renaming, reload the dashboard to ensure the name is updated
    const _updatedConfig = await loadDashboardFromLocalStorage(userId, dashboardId);
    return true;
  } catch (_error) {
    return false;
  }
};

// Get dashboards shared with a user
export const getSharedDashboards = async (
  userId: string | null
): Promise<{ ownerId: string; dashboardId: string; name: string }[]> => {
  try {
    if (!userId) {
      return [];
    }

    // db is already imported as a constant
    if (!db) {
      return [];
    }

    // Use the dashboardShares collection for shared dashboards
    const sharesCollection = collection(db, 'dashboardShares');
    const sharesQuery = query(sharesCollection, where('userId', '==', userId));

    const querySnapshot = await getDocs(sharesQuery);
    const sharedDashboards: { ownerId: string; dashboardId: string; name: string }[] = [];

    querySnapshot.forEach((doc) => {
      const data = doc.data();
      sharedDashboards.push({
        ownerId: data.ownerId || 'unknown',
        dashboardId: data.dashboardId || 'unknown',
        name: data.dashboardName || 'Unnamed Dashboard',
      });
    });

    return sharedDashboards;
  } catch (_error) {
    return [];
  }
};

// Helper function to determine user role for a dashboard
export const getUserRoleForDashboard = (
  userId: string | null,
  dashboardConfig: DashboardConfiguration | null,
  userEmail: string | null,
  adminEmails: string[]
): UserRole | 'owner' | null => {
  // Default to null if no user or config
  if (!userId || !dashboardConfig) {
    return null;
  }

  // Check if user is admin by email
  const isAdminByEmail = userEmail ? adminEmails.includes(userEmail) : false;
  if (isAdminByEmail) {
    return 'owner';
  }

  // Check if user is the dashboard owner
  if (userId === dashboardConfig.ownerId) {
    return 'owner';
  }

  // Check permissions map
  const permissions = dashboardConfig.permissions || {};
  if (permissions[userId]) {
    const role = permissions[userId];
    return role;
  }

  // If nothing else, user is probably a viewer of a shared dashboard
  return 'viewer';
};
