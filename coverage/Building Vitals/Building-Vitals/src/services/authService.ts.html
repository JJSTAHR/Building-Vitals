import type {
  User,
  AuthProvider,
  UserCredential} from 'firebase/auth';
import {
  signInWithPopup,
  signInWithRedirect,
  getRedirectResult,
  signInWithEmailAndPassword,
  createUserWithEmailAndPassword,
  sendPasswordResetEmail
} from 'firebase/auth';
import { doc, setDoc, getDoc, serverTimestamp } from 'firebase/firestore';
import { auth, db, googleProvider, microsoftProvider } from './firebase';
// Use console directly to avoid circular dependency
const logger = {
  info: (message: string, ...args: any[]) => console.log(`[AuthService] ${message}`, ...args),
  warn: (message: string, ...args: any[]) => console.warn(`[AuthService] ${message}`, ...args),
  error: (message: string, ...args: any[]) => console.error(`[AuthService] ${message}`, ...args),
  debug: (message: string, ...args: any[]) => console.debug(`[AuthService] ${message}`, ...args),
};

export interface UserData {
  uid: string;
  email: string | null;
  displayName: string | null;
  photoURL: string | null;
  provider: 'password' | 'google.com' | 'microsoft.com';
  role: 'user' | 'admin';
  createdAt: any;
  lastLoginAt: any;
  requiresPasswordChange?: boolean;
}

/**
 * Handle SSO sign in with popup or redirect
 */
export const signInWithSSO = async (
  provider: 'google' | 'microsoft',
  useRedirect = false
): Promise<UserCredential> => {
  const authProvider = provider === 'google' ? googleProvider : microsoftProvider;

  try {
    let result: UserCredential;

    if (useRedirect) {
      await signInWithRedirect(auth, authProvider);
      // For redirect flow, we need to handle the result separately
      result = (await getRedirectResult(auth)) as UserCredential;
      if (!result) {
        throw new Error('No redirect result available');
      }
    } else {
      result = await signInWithPopup(auth, authProvider);
    }

    // Handle first-time SSO user
    await handleSSOUser(result.user, authProvider);

    return result;
  } catch (error: any) {
    // Handle specific OAuth errors
    if (error.code === 'auth/popup-blocked') {
      throw new Error('Popup was blocked. Please allow popups for this site or use redirect mode.');
    } else if (error.code === 'auth/popup-closed-by-user') {
      throw new Error('Sign in was cancelled.');
    } else if (error.code === 'auth/unauthorized-domain') {
      throw new Error('This domain is not authorized for OAuth operations.');
    } else if (error.code === 'auth/account-exists-with-different-credential') {
      throw new Error(
        'An account already exists with the same email address but different sign-in credentials.'
      );
    }

    throw error;
  }
};

/**
 * Handle SSO user creation/update in Firestore
 */
const handleSSOUser = async (user: User, provider: AuthProvider): Promise<void> => {
  const userRef = doc(db, 'users', user.uid);

  try {
    const userDoc = await getDoc(userRef);

    if (!userDoc.exists()) {
      // First-time SSO user - create user document
      // Determine initial role based on admin emails configuration
      const adminEmails =
        import.meta.env.VITE_ADMIN_EMAILS?.split(',').map((email: string) => email.trim()) || [];
      const isAdmin = user.email && adminEmails.includes(user.email);

      const userData: UserData = {
        uid: user.uid,
        email: user.email,
        displayName: user.displayName,
        photoURL: user.photoURL,
        provider: provider.providerId as 'google.com' | 'microsoft.com',
        role: isAdmin ? 'admin' : 'user',
        createdAt: serverTimestamp(),
        lastLoginAt: serverTimestamp(),
      };

      await setDoc(userRef, userData);
    } else {
      // Existing user - update last login
      await setDoc(
        userRef,
        {
          lastLoginAt: serverTimestamp(),
          displayName: user.displayName,
          photoURL: user.photoURL,
        },
        { merge: true }
      );
    }
  } catch (error) {
    throw new Error('Failed to complete SSO sign in. Please try again.');
  }
};

/**
 * Check if redirect result is pending (for handling redirect flow)
 */
export const checkRedirectResult = async (): Promise<UserCredential | null> => {
  try {
    const result = await getRedirectResult(auth);
    if (result) {
      await handleSSOUser(result.user, result.providerId as unknown as AuthProvider);
    }
    return result;
  } catch (error) {
    return null;
  }
};

/**
 * Traditional email/password sign in
 */
export const signInWithEmail = async (email: string, password: string): Promise<UserCredential> => {
  try {
    const result = await signInWithEmailAndPassword(auth, email, password);

    // Update last login for email users
    const userRef = doc(db, 'users', result.user.uid);
    await setDoc(
      userRef,
      {
        lastLoginAt: serverTimestamp(),
      },
      { merge: true }
    );

    return result;
  } catch (error: any) {
    throw error;
  }
};

/**
 * Create new email/password user
 */
export const createEmailUser = async (
  email: string,
  password: string,
  displayName?: string
): Promise<UserCredential> => {
  try {
    const result = await createUserWithEmailAndPassword(auth, email, password);

    // Create user document
    const adminEmails =
      import.meta.env.VITE_ADMIN_EMAILS?.split(',').map((email: string) => email.trim()) || [];
    const isAdmin = adminEmails.includes(email);

    const userData: UserData = {
      uid: result.user.uid,
      email: result.user.email,
      displayName: displayName || null,
      photoURL: null,
      provider: 'password',
      role: isAdmin ? 'admin' : 'user',
      createdAt: serverTimestamp(),
      lastLoginAt: serverTimestamp(),
      requiresPasswordChange: true, // New email users should change password
    };

    await setDoc(doc(db, 'users', result.user.uid), userData);

    return result;
  } catch (error: any) {
    throw error;
  }
};

/**
 * Send password reset email
 */
export const resetPassword = async (email: string): Promise<void> => {
  try {
    await sendPasswordResetEmail(auth, email);
  } catch (error) {
    throw error;
  }
};

/**
 * Get current user data from Firestore
 */
export const getCurrentUserData = async (uid: string): Promise<UserData | null> => {
  try {
    const userDoc = await getDoc(doc(db, 'users', uid));
    return userDoc.exists() ? (userDoc.data() as UserData) : null;
  } catch (error) {
    return null;
  }
};
