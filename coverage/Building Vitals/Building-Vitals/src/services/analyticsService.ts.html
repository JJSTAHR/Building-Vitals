/**
 * Analytics Service
 *
 * Fetches monitoring metrics from Cloudflare Analytics Engine API
 * Provides real-time and historical data for the monitoring dashboard
 */

import type {
  WorkerMetrics,
  HistoricalMetrics,
  ErrorLog,
  CacheAnalytics,
  SystemStatus,
  TimeRange,
  AnalyticsApiResponse,
} from '../types/monitoring';

// Cloudflare Analytics Engine endpoint
const ANALYTICS_API_BASE = import.meta.env.VITE_CLOUDFLARE_ANALYTICS_API ||
  'https://api.cloudflare.com/client/v4/accounts';

const ACCOUNT_ID = import.meta.env.VITE_CLOUDFLARE_ACCOUNT_ID;
const API_TOKEN = import.meta.env.VITE_CLOUDFLARE_API_TOKEN;

// Fallback to mock data in development if credentials not configured
const USE_MOCK_DATA = import.meta.env.DEV && (!ACCOUNT_ID || !API_TOKEN);

/**
 * Convert time range to seconds for API queries
 */
function timeRangeToSeconds(range: TimeRange): number {
  switch (range) {
    case '1h':
      return 3600;
    case '24h':
      return 86400;
    case '7d':
      return 604800;
    case '30d':
      return 2592000;
    default:
      return 86400;
  }
}

/**
 * Generate mock historical data
 */
function generateMockHistorical(timeRange: TimeRange): HistoricalMetrics {
  const points = timeRange === '1h' ? 12 : timeRange === '24h' ? 24 : 30;
  const now = Date.now();
  const interval = timeRangeToSeconds(timeRange) * 1000 / points;

  const generateSeries = (base: number, variance: number) =>
    Array.from({ length: points }, (_, i) => ({
      timestamp: now - (points - i) * interval,
      value: base + Math.random() * variance - variance / 2,
    }));

  return {
    latency: {
      p50: generateSeries(35, 10),
      p95: generateSeries(120, 30),
      p99: generateSeries(280, 50),
    },
    cacheHitRate: generateSeries(0.87, 0.06),
    errorRate: generateSeries(0.002, 0.003),
    requestVolume: generateSeries(1000, 300),
  };
}

/**
 * Fetch metrics from Cloudflare Analytics Engine
 */
export async function fetchMetrics(timeRange: TimeRange = '24h'): Promise<WorkerMetrics> {
  if (USE_MOCK_DATA) {
    // Return mock data in development
    await new Promise((resolve) => setTimeout(resolve, 500)); // Simulate network delay
    return generateMockMetrics(timeRange);
  }

  try {
    const since = Math.floor(Date.now() / 1000) - timeRangeToSeconds(timeRange);

    const response = await fetch(
      `${ANALYTICS_API_BASE}/${ACCOUNT_ID}/analytics_engine/sql`,
      {
        method: 'POST',
        headers: {
          'Authorization': `Bearer ${API_TOKEN}`,
          'Content-Type': 'application/json',
        },
        body: JSON.stringify({
          query: `
            SELECT
              COUNT(*) as requestCount,
              AVG(latency) as avgLatency,
              QUANTILE(latency, 0.5) as p50Latency,
              QUANTILE(latency, 0.95) as p95Latency,
              QUANTILE(latency, 0.99) as p99Latency,
              SUM(CASE WHEN cacheStatus = 'HIT' THEN 1 ELSE 0 END) as cacheHits,
              SUM(CASE WHEN cacheStatus = 'MISS' THEN 1 ELSE 0 END) as cacheMisses,
              SUM(CASE WHEN status >= 400 THEN 1 ELSE 0 END) as errorCount
            FROM ace_worker_analytics
            WHERE timestamp >= ${since}
          `,
        }),
      }
    );

    if (!response.ok) {
      throw new Error(`Analytics API error: ${response.statusText}`);
    }

    const data = await response.json();
    const row = data.data[0];

    const totalRequests = row.requestCount || 0;
    const cacheHits = row.cacheHits || 0;
    const cacheMisses = row.cacheMisses || 0;
    const errorCount = row.errorCount || 0;

    return {
      requestCount: totalRequests,
      avgLatency: row.avgLatency || 0,
      p50Latency: row.p50Latency || 0,
      p95Latency: row.p95Latency || 0,
      p99Latency: row.p99Latency || 0,
      cacheHitRate: totalRequests > 0 ? cacheHits / totalRequests : 0,
      cacheHits,
      cacheMisses,
      errorRate: totalRequests > 0 ? errorCount / totalRequests : 0,
      errorCount,
      apiCallReduction: totalRequests > 0 ? cacheHits / totalRequests : 0,
      bandwidthSaved: (cacheHits * 2.5) / 1024, // Estimate 2.5KB per cached request
      timeRange,
      lastUpdated: new Date(),
    };
  } catch (error) {
    console.error('Failed to fetch metrics:', error);
    // Fallback to mock data on error
    return generateMockMetrics(timeRange);
  }
}

/**
 * Fetch historical metrics for charts
 */
export async function fetchHistoricalMetrics(timeRange: TimeRange = '24h'): Promise<HistoricalMetrics> {
  if (USE_MOCK_DATA) {
    await new Promise((resolve) => setTimeout(resolve, 300));
    return generateMockHistorical(timeRange);
  }

  try {
    const since = Math.floor(Date.now() / 1000) - timeRangeToSeconds(timeRange);
    const buckets = timeRange === '1h' ? 12 : timeRange === '24h' ? 24 : 30;
    const bucketSize = timeRangeToSeconds(timeRange) / buckets;

    const response = await fetch(
      `${ANALYTICS_API_BASE}/${ACCOUNT_ID}/analytics_engine/sql`,
      {
        method: 'POST',
        headers: {
          'Authorization': `Bearer ${API_TOKEN}`,
          'Content-Type': 'application/json',
        },
        body: JSON.stringify({
          query: `
            SELECT
              intDiv(timestamp, ${bucketSize}) * ${bucketSize} as bucket,
              QUANTILE(latency, 0.5) as p50,
              QUANTILE(latency, 0.95) as p95,
              QUANTILE(latency, 0.99) as p99,
              AVG(CASE WHEN cacheStatus = 'HIT' THEN 1 ELSE 0 END) as hitRate,
              AVG(CASE WHEN status >= 400 THEN 1 ELSE 0 END) as errorRate,
              COUNT(*) as requests
            FROM ace_worker_analytics
            WHERE timestamp >= ${since}
            GROUP BY bucket
            ORDER BY bucket
          `,
        }),
      }
    );

    if (!response.ok) {
      throw new Error(`Analytics API error: ${response.statusText}`);
    }

    const data = await response.json();

    return {
      latency: {
        p50: data.data.map((row: any) => ({ timestamp: row.bucket * 1000, value: row.p50 })),
        p95: data.data.map((row: any) => ({ timestamp: row.bucket * 1000, value: row.p95 })),
        p99: data.data.map((row: any) => ({ timestamp: row.bucket * 1000, value: row.p99 })),
      },
      cacheHitRate: data.data.map((row: any) => ({ timestamp: row.bucket * 1000, value: row.hitRate })),
      errorRate: data.data.map((row: any) => ({ timestamp: row.bucket * 1000, value: row.errorRate })),
      requestVolume: data.data.map((row: any) => ({ timestamp: row.bucket * 1000, value: row.requests })),
    };
  } catch (error) {
    console.error('Failed to fetch historical metrics:', error);
    return generateMockHistorical(timeRange);
  }
}

/**
 * Subscribe to real-time metrics updates via WebSocket or polling
 */
export function subscribeToMetrics(
  callback: (metrics: WorkerMetrics) => void,
  interval: number = 5000
): () => void {
  const poll = async () => {
    const metrics = await fetchMetrics('1h');
    callback(metrics);
  };

  // Initial fetch
  void poll();

  // Set up polling interval
  const intervalId = setInterval(poll, interval);

  // Return cleanup function
  return () => clearInterval(intervalId);
}
