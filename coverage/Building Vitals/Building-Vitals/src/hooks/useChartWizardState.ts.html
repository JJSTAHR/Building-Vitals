import { useWizardState } from './useWizardState';
import type { MonitorTemplate } from '../data/monitorTemplates';

/**
 * Specialized wizard state hook for the Chart Wizard
 * Builds on top of useWizardState with chart-specific logic
 */

export interface ChartWizardData {
  // Step 0: Template/Chart Selection
  selectedTemplate: MonitorTemplate | null;
  chartType: string;

  // Step 1: Site Selection
  siteId: string;
  siteName: string;

  // Step 2: Point Selection
  selectedPoints: Array<{
    id: string;
    name: string;
    unit: string;
  }>;

  // Step 3: Configuration
  title: string;
  timeRange: string;
  config: {
    aggregation?: string;
    showLegend?: boolean;
    showGrid?: boolean;
    yAxisLabel?: string;
    refreshInterval?: number;
    mixedAirSetpoint?: number;
    minimumOAPercentage?: number;
    economizerHighLimit?: number;
    returnAirTemp?: number;
    temperatureUnit?: string;
  };

  // Additional state
  customStartDate?: string;
  customEndDate?: string;
}

const initialChartWizardData: ChartWizardData = {
  selectedTemplate: null,
  chartType: '',
  siteId: '',
  siteName: '',
  selectedPoints: [],
  title: '',
  timeRange: '24h',
  config: {
    aggregation: 'mean',
    showLegend: true,
    showGrid: true,
    refreshInterval: 300000, // 5 minutes
  },
};

/**
 * Custom hook for managing chart wizard state
 *
 * @returns Wizard state and actions
 *
 * @example
 * const chartWizard = useChartWizardState();
 *
 * // Update chart type
 * chartWizard.actions.updateData({ chartType: 'TimeSeries' });
 *
 * // Move to next step
 * if (chartWizard.actions.validateCurrentStep()) {
 *   chartWizard.actions.nextStep();
 * }
 */
export function useChartWizardState() {
  const wizard = useWizardState<ChartWizardData>({
    initialData: initialChartWizardData,
    totalSteps: 4,
    validateStep: (step, data) => {
      switch (step) {
        case 0: // Template/Chart selection
          if (!data.chartType && !data.selectedTemplate) {
            return 'Please select a chart type or template';
          }
          return true;

        case 1: // Site selection
          if (!data.siteId) {
            return 'Please select a site';
          }
          return true;

        case 2: // Point selection
          if (!data.selectedPoints || data.selectedPoints.length === 0) {
            return 'Please select at least one data point';
          }

          // Chart-specific validations
          const minPoints = getMinPointsForChart(data.chartType);
          if (data.selectedPoints.length < minPoints) {
            return `Please select at least ${minPoints} data points for this chart type`;
          }

          const maxPoints = getMaxPointsForChart(data.chartType);
          if (data.selectedPoints.length > maxPoints) {
            return `This chart type supports a maximum of ${maxPoints} data points`;
          }

          return true;

        case 3: // Configuration
          if (!data.title || data.title.trim() === '') {
            return 'Please enter a title for the monitor';
          }
          return true;

        default:
          return true;
      }
    },
  });

  // Helper function to get minimum points for chart type
  function getMinPointsForChart(chartType: string): number {
    switch (chartType) {
      case 'Gauge':
      case 'GaugeChart':
        return 1;
      case 'Scatter':
      case 'ScatterPlot':
      case 'Psychrometric':
      case 'PerfectEconomizer':
        return 2;
      case 'ParallelCoordinates':
        return 3;
      default:
        return 1;
    }
  }

  // Helper function to get maximum points for chart type
  function getMaxPointsForChart(chartType: string): number {
    switch (chartType) {
      case 'Gauge':
      case 'GaugeChart':
        return 1;
      case 'Psychrometric':
        return 2;
      case 'PerfectEconomizer':
        return 4;
      case 'ParallelCoordinates':
        return 10;
      default:
        return 20; // General limit
    }
  }

  // Enhanced actions with chart-specific logic
  const selectTemplate = (template: MonitorTemplate) => {
    wizard.actions.updateData({
      selectedTemplate: template,
      chartType: template.chartType,
      title: template.name,
      config: {
        ...wizard.state.data.config,
        ...template.defaultConfig,
      },
    });
  };

  const selectChartType = (chartType: string) => {
    wizard.actions.updateData({
      chartType,
      selectedTemplate: null, // Clear template when manually selecting chart
    });
  };

  const selectSite = (siteId: string, siteName: string) => {
    wizard.actions.updateData({
      siteId,
      siteName,
      selectedPoints: [], // Clear points when site changes
    });
  };

  const updatePoints = (points: ChartWizardData['selectedPoints']) => {
    wizard.actions.updateData({
      selectedPoints: points,
    });
  };

  const updateConfig = (config: Partial<ChartWizardData['config']>) => {
    wizard.actions.updateData({
      config: {
        ...wizard.state.data.config,
        ...config,
      },
    });
  };

  const setTimeRange = (timeRange: string, customStart?: string, customEnd?: string) => {
    wizard.actions.updateData({
      timeRange,
      customStartDate: customStart,
      customEndDate: customEnd,
    });
  };

  // Build final monitor configuration
  const buildMonitorConfig = () => {
    const { data } = wizard.state;
    return {
      title: data.title,
      chartType: data.chartType,
      siteId: data.siteId,
      siteName: data.siteName,
      selectedPoints: data.selectedPoints,
      timeRange: data.timeRange,
      config: data.config,
      template: data.selectedTemplate,
      customDateRange:
        data.timeRange === 'custom'
          ? { start: data.customStartDate, end: data.customEndDate }
          : undefined,
    };
  };

  return {
    ...wizard,
    actions: {
      ...wizard.actions,
      selectTemplate,
      selectChartType,
      selectSite,
      updatePoints,
      updateConfig,
      setTimeRange,
      buildMonitorConfig,
    },
  };
}
