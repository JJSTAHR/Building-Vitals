/**
 * React Hook for fetching chart data via Cloudflare Worker
 * Replaces existing hooks with optimized edge-based data fetching
 */

import { useQuery, useQueries, UseQueryResult } from '@tanstack/react-query';
import { cloudflareWorkerClient } from '../services/api/cloudflareWorkerClient';
import { useMemo } from 'react';

interface ChartDataParams {
  siteId: string;
  pointIds: string[];
  timeRange: {
    start: Date;
    end: Date;
  };
  chartType: 'line' | 'bar' | 'scatter' | 'heatmap' | 'gauge' | 'area' | '3d';
  enabled?: boolean;
  refreshInterval?: number;
}

interface ChartDataResult {
  data: any;
  isLoading: boolean;
  error: any;
  isFetching: boolean;
  refetch: () => void;
  dataUpdatedAt: number;
  cacheStatus?: string;
}

/**
 * Main hook for fetching chart data
 */
export function useChartDataWorker(params: ChartDataParams): ChartDataResult {
  const { 
    siteId, 
    pointIds, 
    timeRange, 
    chartType, 
    enabled = true,
    refreshInterval
  } = params;

  const query = useQuery({
    queryKey: ['chartData', 'worker', siteId, pointIds, timeRange, chartType],
    queryFn: async () => {
      const data = await cloudflareWorkerClient.getTimeseries(
        pointIds,
        timeRange.start,
        timeRange.end,
        {
          format: pointIds.length > 10 ? 'binary' : 'json',
          compression: 'auto',
          maxPoints: 1000000  // 1M points for full resolution
        }
      );

      // Transform data for specific chart types
      return transformForChart(data, chartType);
    },
    enabled: enabled && pointIds.length > 0,
    staleTime: 5 * 60 * 1000,  // 5 minutes for raw data
    cacheTime: 30 * 60 * 1000,  // 30 minutes cache
    refetchInterval: refreshInterval,
    retry: 3,
    retryDelay: attemptIndex => Math.min(1000 * 2 ** attemptIndex, 30000),
  });

  return {
    data: query.data,
    isLoading: query.isLoading,
    error: query.error,
    isFetching: query.isFetching,
    refetch: query.refetch,
    dataUpdatedAt: query.dataUpdatedAt,
    cacheStatus: query.isStale ? 'stale' : 'fresh'
  };
}

/**
 * Hook for fetching multiple chart data in parallel
 */
export function useMultipleChartData(
  charts: ChartDataParams[]
): UseQueryResult<any>[] {
  return useQueries({
    queries: charts.map(chart => ({
      queryKey: ['chartData', 'worker', chart.siteId, chart.pointIds, chart.timeRange, chart.chartType],
      queryFn: () => cloudflareWorkerClient.fetchChartData(
        chart.siteId,
        chart.pointIds,
        chart.timeRange,
        chart.chartType
      ),
      enabled: chart.enabled !== false && chart.pointIds.length > 0,
      staleTime: 5 * 60 * 1000,
      cacheTime: 30 * 60 * 1000,
    }))
  });
}

/**
 * Hook for fetching sites
 */
export function useSites(enabled = true) {
  return useQuery({
    queryKey: ['sites', 'worker'],
    queryFn: () => cloudflareWorkerClient.getSites(),
    enabled,
    staleTime: 60 * 60 * 1000, // 1 hour
    cacheTime: 2 * 60 * 60 * 1000, // 2 hours
  });
}

/**
 * Hook for fetching points
 */
export function usePoints(siteId: string, collectEnabled = true, enabled = true) {
  return useQuery({
    queryKey: ['points', 'worker', siteId, collectEnabled],
    queryFn: () => cloudflareWorkerClient.getPoints(siteId, collectEnabled),
    enabled: enabled && !!siteId,
    staleTime: 30 * 60 * 1000, // 30 minutes
    cacheTime: 60 * 60 * 1000, // 1 hour
  });
}

/**
 * Hook for batch fetching
 */
export function useBatchData(requests: any[], enabled = true) {
  return useQuery({
    queryKey: ['batch', 'worker', requests],
    queryFn: () => cloudflareWorkerClient.batch(requests),
    enabled: enabled && requests.length > 0,
    staleTime: 5 * 60 * 1000,
    cacheTime: 15 * 60 * 1000,
  });
}

/**
 * Hook for worker metrics
 */
export function useWorkerMetrics(refreshInterval = 30000) {
  return useQuery({
    queryKey: ['metrics', 'worker'],
    queryFn: () => cloudflareWorkerClient.getMetrics(),
    refetchInterval: refreshInterval,
    staleTime: 0, // Always fresh
  });
}

// Helper functions

function transformForChart(data: any, chartType: string): any {
  // Basic transformation - the worker does most of the work
  switch (chartType) {
    case 'line':
    case 'area':
      return transformForLineChart(data);
    case 'bar':
      return transformForBarChart(data);
    case 'heatmap':
      return transformForHeatmap(data);
    case 'scatter':
      return transformForScatterChart(data);
    case 'gauge':
      return transformForGaugeChart(data);
    case '3d':
      return transformFor3DChart(data);
    default:
      return data;
  }
}

function transformForLineChart(data: any) {
  if (!data.points) return { series: [] };

  const series = Object.entries(data.points).map(([pointId, pointData]: any) => ({
    name: pointId,
    type: 'line',
    data: pointData.values?.map((v: any) => [
      new Date(v.timestamp).getTime(),
      v.value
    ]) || [],
    smooth: false,  // Disable smoothing to show exact data points
    // No sampling - show all data points
    large: true,  // Enable large data optimization
    largeThreshold: 2000,  // Lower threshold for optimization
    symbol: 'none',
    lineStyle: {
      width: 2
    }
  }));

  return {
    xAxis: {
      type: 'time',
      boundaryGap: false
    },
    yAxis: {
      type: 'value',
      scale: true
    },
    series,
    dataZoom: [
      {
        type: 'inside',
        start: 0,
        end: 100
      },
      {
        type: 'slider',
        show: true,
        start: 0,
        end: 100
      }
    ],
    tooltip: {
      trigger: 'axis',
      axisPointer: {
        type: 'cross'
      }
    },
    legend: {
      data: Object.keys(data.points),
      type: 'scroll',
      bottom: 0
    }
  };
}

function transformForBarChart(data: any) {
  if (!data.points) return { series: [] };

  // Group by time periods for bar chart
  const timeGroups = new Map();
  
  Object.entries(data.points).forEach(([pointId, pointData]: any) => {
    pointData.values?.forEach((v: any) => {
      const time = new Date(v.timestamp).toISOString().split('T')[0];
      if (!timeGroups.has(time)) {
        timeGroups.set(time, {});
      }
      timeGroups.get(time)[pointId] = v.value;
    });
  });

  const categories = Array.from(timeGroups.keys()).sort();
  const series = Object.keys(data.points).map(pointId => ({
    name: pointId,
    type: 'bar',
    data: categories.map(cat => timeGroups.get(cat)?.[pointId] || 0)
  }));

  return {
    xAxis: {
      type: 'category',
      data: categories
    },
    yAxis: {
      type: 'value'
    },
    series,
    tooltip: {
      trigger: 'axis',
      axisPointer: {
        type: 'shadow'
      }
    }
  };
}

function transformForHeatmap(data: any) {
  if (!data.points) return { series: [] };

  const heatmapData: any[] = [];
  const xAxisData: string[] = [];
  const yAxisData = Object.keys(data.points);

  // Collect all unique timestamps
  const allTimestamps = new Set<string>();
  Object.values(data.points).forEach((pointData: any) => {
    pointData.values?.forEach((v: any) => {
      allTimestamps.add(v.timestamp);
    });
  });

  const sortedTimestamps = Array.from(allTimestamps).sort();
  xAxisData.push(...sortedTimestamps.map(t => 
    new Date(t).toLocaleString('en-US', { 
      hour: 'numeric', 
      minute: '2-digit',
      hour12: true 
    })
  ));

  // Build heatmap data [x, y, value]
  yAxisData.forEach((pointId, yIndex) => {
    const pointData = data.points[pointId];
    pointData.values?.forEach((v: any) => {
      const xIndex = sortedTimestamps.indexOf(v.timestamp);
      if (xIndex !== -1) {
        heatmapData.push([xIndex, yIndex, v.value]);
      }
    });
  });

  const values = heatmapData.map(d => d[2]);
  const min = Math.min(...values);
  const max = Math.max(...values);

  return {
    xAxis: {
      type: 'category',
      data: xAxisData,
      splitArea: { show: true }
    },
    yAxis: {
      type: 'category',
      data: yAxisData,
      splitArea: { show: true }
    },
    visualMap: {
      min,
      max,
      calculable: true,
      orient: 'horizontal',
      left: 'center',
      bottom: '15%',
      inRange: {
        color: ['#313695', '#4575b4', '#74add1', '#abd9e9', '#e0f3f8', '#ffffbf', '#fee090', '#fdae61', '#f46d43', '#d73027', '#a50026']
      }
    },
    series: [{
      name: 'Heatmap',
      type: 'heatmap',
      data: heatmapData,
      label: { show: false },
      emphasis: {
        itemStyle: {
          shadowBlur: 10,
          shadowColor: 'rgba(0, 0, 0, 0.5)'
        }
      }
    }]
  };
}

function transformForScatterChart(data: any) {
  if (!data.points) return { series: [] };

  const series = Object.entries(data.points).map(([pointId, pointData]: any) => ({
    name: pointId,
    type: 'scatter',
    data: pointData.values?.map((v: any, index: number) => [
      new Date(v.timestamp).getTime(),
      v.value
    ]) || [],
    symbolSize: 5
  }));

  return {
    xAxis: {
      type: 'time',
      scale: true
    },
    yAxis: {
      type: 'value',
      scale: true
    },
    series,
    tooltip: {
      trigger: 'item',
      formatter: (params: any) => {
        const date = new Date(params.value[0]);
        return `${params.seriesName}<br/>${date.toLocaleString()}: ${params.value[1]}`;
      }
    }
  };
}

function transformForGaugeChart(data: any) {
  if (!data.points) return { series: [] };

  // Get latest value from first point
  const firstPoint = Object.values(data.points)[0] as any;
  const latestValue = firstPoint?.values?.[firstPoint.values.length - 1]?.value || 0;
  const metadata = firstPoint?.metadata || {};

  return {
    series: [{
      type: 'gauge',
      min: metadata.min || 0,
      max: metadata.max || 100,
      detail: { 
        formatter: '{value}',
        fontSize: 24
      },
      data: [{ 
        value: latestValue, 
        name: metadata.unit || 'Value' 
      }],
      axisLine: {
        lineStyle: {
          width: 30,
          color: [
            [0.3, '#67e0e3'],
            [0.7, '#37a2da'],
            [1, '#fd666d']
          ]
        }
      }
    }]
  };
}

function transformFor3DChart(data: any) {
  if (!data.points) return { series: [] };

  // Transform for 3D surface chart
  const pointIds = Object.keys(data.points);
  const timestamps = new Set<string>();
  
  Object.values(data.points).forEach((pointData: any) => {
    pointData.values?.forEach((v: any) => {
      timestamps.add(v.timestamp);
    });
  });

  const sortedTimestamps = Array.from(timestamps).sort();
  const data3D: any[] = [];

  pointIds.forEach((pointId, yIndex) => {
    const pointData = data.points[pointId];
    sortedTimestamps.forEach((timestamp, xIndex) => {
      const value = pointData.values?.find((v: any) => v.timestamp === timestamp);
      data3D.push([xIndex, yIndex, value?.value || 0]);
    });
  });

  return {
    xAxis3D: {
      type: 'category',
      data: sortedTimestamps.map(t => new Date(t).toLocaleTimeString('en-US', { hour: 'numeric', minute: '2-digit', hour12: true }))
    },
    yAxis3D: {
      type: 'category',
      data: pointIds
    },
    zAxis3D: {
      type: 'value'
    },
    grid3D: {
      boxWidth: 200,
      boxDepth: 80,
      viewControl: {
        projection: 'orthographic'
      }
    },
    series: [{
      type: 'surface',
      data: data3D,
      shading: 'color',
      colorMaterial: {
        detailTexture: '#fff'
      }
    }]
  };
}
