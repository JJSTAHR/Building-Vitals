import { useEffect, useState } from 'react';
import { doc, onSnapshot } from 'firebase/firestore';
import { db } from '../services/firebase';
import type { SystemRealTimeSettings } from '../components/admin/SystemSettings';
import { debugService } from '../services/debugService';

const DEFAULT_SETTINGS: SystemRealTimeSettings = {
  enableRealTimeFeatures: true,
  enableSSEStreaming: false,
  enableAutoRefresh: true,
  minRefreshInterval: 30,
  maxLiveChartsPerUser: 5,
  maxPointsPerLiveChart: 10,
  maxSSEConnectionsPerUser: 3,
  defaultRefreshMode: 'interval',
  defaultRefreshInterval: 300,
  allowUserOverride: true,
  requireAdminApprovalForLive: false,
  autoDisableOnCostThreshold: true,
  monthlyApiCallLimit: 1000000,
  dailyApiCallLimit: 50000,
  costWarningThreshold: 100,
};

interface UseSystemSettingsReturn {
  settings: SystemRealTimeSettings;
  loading: boolean;
  canUseRealTime: boolean;
  canUseSSE: boolean;
  canUseAutoRefresh: boolean;
  getEffectiveRefreshInterval: (requested: number) => number;
  isWithinLimits: (chartCount: number, pointCount: number) => boolean;
}

export function useSystemSettings(): UseSystemSettingsReturn {
  const [settings, setSettings] = useState<SystemRealTimeSettings>(DEFAULT_SETTINGS);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    // Set up real-time listener for settings changes
    const unsubscribe = onSnapshot(
      doc(db, 'systemSettings', 'realTimeConfig'),
      (snapshot) => {
        if (snapshot.exists()) {
          const data = snapshot.data() as SystemRealTimeSettings;
          setSettings({ ...DEFAULT_SETTINGS, ...data });
          debugService.info('SystemSettings', 'Settings updated', data);
        } else {
          // No settings doc yet, use defaults
          setSettings(DEFAULT_SETTINGS);
        }
        setLoading(false);
      },
      (error) => {
        debugService.error('SystemSettings', 'Failed to load settings', error);
        setLoading(false);
      }
    );

    return () => unsubscribe();
  }, []);

  // Computed properties
  const canUseRealTime = settings.enableRealTimeFeatures;
  const canUseSSE = settings.enableRealTimeFeatures && settings.enableSSEStreaming;
  const canUseAutoRefresh = settings.enableRealTimeFeatures && settings.enableAutoRefresh;

  // Helper function to get effective refresh interval
  const getEffectiveRefreshInterval = (requested: number): number => {
    if (!canUseAutoRefresh) {
      return 0; // No refresh
    }
    // Enforce minimum interval
    return Math.max(requested, settings.minRefreshInterval);
  };

  // Check if user is within limits
  const isWithinLimits = (chartCount: number, pointCount: number): boolean => {
    if (!canUseRealTime) {
      return false;
    }

    if (chartCount > settings.maxLiveChartsPerUser) {
      debugService.warn(
        'SystemSettings',
        `Chart limit exceeded: ${chartCount} > ${settings.maxLiveChartsPerUser}`
      );
      return false;
    }

    if (pointCount > settings.maxPointsPerLiveChart) {
      debugService.warn(
        'SystemSettings',
        `Point limit exceeded: ${pointCount} > ${settings.maxPointsPerLiveChart}`
      );
      return false;
    }

    return true;
  };

  return {
    settings,
    loading,
    canUseRealTime,
    canUseSSE,
    canUseAutoRefresh,
    getEffectiveRefreshInterval,
    isWithinLimits,
  };
}

// Hook to check if a specific refresh mode is allowed
export function useCanUseRefreshMode(mode: 'none' | 'interval' | 'live' | 'onChange'): boolean {
  const { settings, canUseRealTime, canUseSSE, canUseAutoRefresh } = useSystemSettings();

  switch (mode) {
    case 'none':
      return true; // Always allowed
    case 'interval':
      return canUseAutoRefresh;
    case 'live':
      return canUseSSE;
    case 'onChange':
      return canUseRealTime && settings.allowUserOverride;
    default:
      return false;
  }
}
