/**
 * Ensures timeseries data is in the correct format for ECharts
 * Handles various data formats and transforms them to the expected array format
 */

import { ChartDataDiagnostics } from './chartDataDiagnostics';

interface InputDataPoint {
  timestamp?: number;
  time?: number | string;
  value?: number | string;
  val?: number;
  v?: number;
}

interface OutputSeries {
  name: string;
  data: Array<[number, number]>; // [timestamp, value] tuples
  unit?: string;
  markerTag?: string;
  markerTags?: string[];
}

/**
 * Transform various data formats to ECharts format
 */
export function ensureTimeseriesFormat(data: any): OutputSeries[] {
  console.log('[ensureTimeseriesFormat] Input data:', data);

  // Handle null/undefined
  if (!data) {
    console.warn('[ensureTimeseriesFormat] No data provided');
    return [];
  }

  // If data has a series property, use that
  const rawData = data.series || data;

  // Ensure we have an array
  if (!Array.isArray(rawData)) {
    console.warn('[ensureTimeseriesFormat] Data is not an array:', typeof rawData);
    return [];
  }

  const result: OutputSeries[] = rawData.map((series: any, index: number) => {
    // Extract series metadata
    const name = series.name || series.pointName || `Series ${index + 1}`;
    const unit = series.unit || '';
    const markerTag = series.markerTag;
    const markerTags = series.markerTags || (markerTag ? [markerTag] : []);

    // Transform data points
    let transformedData: Array<[number, number]> = [];

    if (Array.isArray(series.data)) {
      transformedData = series.data
        .map((point: any) => {
          // Handle array format [timestamp, value]
          if (Array.isArray(point) && point.length >= 2) {
            const timestamp = Number(point[0]);
            const value = Number(point[1]);

            if (!isNaN(timestamp) && !isNaN(value)) {
              return [timestamp, value] as [number, number];
            }
          }

          // Handle object format { timestamp, value }
          if (typeof point === 'object' && point !== null) {
            const timestamp = extractTimestamp(point);
            const value = extractValue(point);

            if (timestamp !== null && value !== null) {
              return [timestamp, value] as [number, number];
            }
          }

          return null;
        })
        .filter((point: any): point is [number, number] => point !== null);
    }

    console.log(`[ensureTimeseriesFormat] Transformed series ${index}:`, {
      name,
      originalDataLength: series.data?.length || 0,
      transformedDataLength: transformedData.length,
      samplePoints: transformedData.slice(0, 3),
    });

    return {
      name,
      data: transformedData,
      unit,
      markerTag,
      markerTags,
    };
  });

  // Log diagnostic summary
  const summary = ChartDataDiagnostics.createSummary('ensureTimeseriesFormat', result);
  console.log(summary);

  return result;
}

/**
 * Extract timestamp from various formats
 */
function extractTimestamp(point: InputDataPoint): number | null {
  // Try different timestamp fields
  const timeValue = point.timestamp || point.time;

  if (timeValue === undefined || timeValue === null) {
    return null;
  }

  // If already a number, use it
  if (typeof timeValue === 'number') {
    return timeValue;
  }

  // If string, parse it
  if (typeof timeValue === 'string') {
    const parsed = Date.parse(timeValue);
    return isNaN(parsed) ? null : parsed;
  }

  return null;
}

/**
 * Extract value from various formats
 */
function extractValue(point: InputDataPoint): number | null {
  // Try different value fields
  const rawValue = point.value ?? point.val ?? point.v;

  if (rawValue === undefined || rawValue === null) {
    return null;
  }

  // Convert to number
  const numValue = Number(rawValue);

  if (isNaN(numValue)) {
    // Handle boolean strings
    if (typeof rawValue === 'string') {
      const lower = rawValue.toLowerCase().trim();
      if (lower === 'true' || lower === 'on') {return 1;}
      if (lower === 'false' || lower === 'off') {return 0;}
    }
    return null;
  }

  return numValue;
}

/**
 * Validate that data is properly formatted for charts
 */
export function validateTimeseriesFormat(data: OutputSeries[]): boolean {
  if (!Array.isArray(data)) {return false;}

  return data.every((series) => {
    if (!series.name || !Array.isArray(series.data)) {return false;}

    return series.data.every((point) => {
      if (!Array.isArray(point) || point.length !== 2) {return false;}
      if (typeof point[0] !== 'number' || typeof point[1] !== 'number') {return false;}
      if (isNaN(point[0]) || isNaN(point[1])) {return false;}
      return true;
    });
  });
}
