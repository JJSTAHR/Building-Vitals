/**
 * Debug chart options
 */
export function debugChartOptions(option: any, label: string = 'ChartOptions'): DebugChartOptions {
  const debug: DebugChartOptions = {
    hasOption: !!option,
    optionKeys: [],
    hasSeries: false,
    seriesCount: 0,
    hasXAxis: false,
    hasYAxis: false,
    seriesTypes: [],
  };

  if (!option) {
    return debug;
  }

  debug.optionKeys = Object.keys(option);
  debug.hasXAxis = !!option.xAxis;
  debug.hasYAxis = !!option.yAxis;

  if (option.series) {
    debug.hasSeries = Array.isArray(option.series) && option.series.length > 0;
    debug.seriesCount = Array.isArray(option.series) ? option.series.length : 0;
    debug.seriesTypes = Array.isArray(option.series)
      ? option.series.map((s: any) => s.type || 'unknown')
      : [];
  }

  if (option.series && Array.isArray(option.series)) {
    option.series.forEach((series: any, index: number) => {});
  }

  return debug;
}

/**

 */
export function debugChartPipeline(
  widget: any,
  rawData: any,
  preparedData: DebugChartData,
  chartOptions: any
): void {
  debugWidget(widget, 'Widget Config');
  debugChartData(rawData, 'Raw API Data');
  debugPreparedData(preparedData, 'Prepared Data');
  debugChartOptions(chartOptions, 'Final Chart Options');

  // Summary analysis
  const issues: string[] = [];

  if (!widget.selectedPoints || widget.selectedPoints.length === 0) {
    issues.push('No points selected');

/**
 * Enable comprehensive chart debugging
 */
export function enableChartDebugMode(): void {
  // Add global debugging for ECharts
  if (typeof window !== 'undefined') {
    (window as any).chartDebugEnabled = true;
    (window as any).enableChartDebug = enableChartDebugMode;
    (window as any).disableChartDebug = disableChartDebugMode;
  }
}

/**
 * Disable chart debugging
 */
export function disableChartDebugMode(): void {
  if (typeof window !== 'undefined') {
    (window as any).chartDebugEnabled = false;
  }
}
