// CORS Helper Utility
// Helps diagnose and provide solutions for CORS-related API errors

export interface CorsErrorInfo {
  isCorsProblem: boolean;
  origin?: string;
  requestedUrl?: string;
  suggestions: string[];
  fixCommands?: string[];
}

export function analyzeCorsError(error: any): CorsErrorInfo {
  const errorMessage = error?.message?.toLowerCase() || '';
  const errorString = error?.toString?.()?.toLowerCase() || '';

  // Check if this is likely a CORS error
  const corsIndicators = [
    'cors',
    'cross-origin',
    'access-control-allow-origin',
    'preflight',
    'blocked by cors policy',
    "no 'access-control-allow-origin' header",
  ];

  const isCorsProblem = corsIndicators.some(
    (indicator) => errorMessage.includes(indicator) || errorString.includes(indicator)
  );

  if (!isCorsProblem) {
    return {
      isCorsProblem: false,
      suggestions: [],
    };
  }

  const origin = window.location.origin;
  const requestedUrl = error?.config?.url || error?.request?.url || 'Unknown URL';

  const suggestions: string[] = [];
  const fixCommands: string[] = [];

  // Analyze the specific situation
  if (origin.includes('localhost')) {
    suggestions.push(
      'You are running from localhost. The Python service should handle CORS for local development.'
    );
    suggestions.push('Ensure your Python service is deployed and running.');
    fixCommands.push(
      'Check Python service at: https://building-vitals-api-763711142996.us-central1.run.app/health'
    );
  } else if (
    origin.includes('building-vitals.web.app') ||
    origin.includes('building-vitals.firebaseapp.com')
  ) {
    suggestions.push('Production domain detected. CORS should be configured correctly.');
    suggestions.push('The issue might be with Firebase Functions configuration.');
    fixCommands.push('firebase deploy --only functions');
  } else {
    suggestions.push(`Your current origin (${origin}) may not be in the CORS allowed list.`);
    suggestions.push('Add your domain to the ALLOWED_ORIGINS in functions/src/index.ts');
  }

  // General suggestions
  suggestions.push(
    'Check if Python service is responding: curl -I ' +
      (import.meta.env.VITE_API_BASE_URL ||
        'https://building-vitals-api-763711142996.us-central1.run.app/health')
  );
  suggestions.push('Verify you are using the Python service URL, not direct ACE API URL.');

  return {
    isCorsProblem: true,
    origin,
    requestedUrl,
    suggestions,
    fixCommands,
  };
}

export function getCorsFixInstructions(): string {
  return `
CORS Error Fix Instructions:

1. **For Local Development:**
   - Ensure you're pointing to the correct Python service
   - Your .env should have:
     VITE_USE_DIRECT_API=false
     VITE_API_BASE_URL=https://building-vitals-api-763711142996.us-central1.run.app

2. **Check Python Service Health:**
   \`\`\`bash
   curl https://building-vitals-api-763711142996.us-central1.run.app/health
   \`\`\`

3. **Check Cloud Run Logs:**
   \`\`\`bash
   gcloud run logs read --service=building-vitals-api
   \`\`\`

4. **Verify CORS Headers:**
   The Python service should include these headers:
   - Access-Control-Allow-Origin: [your-origin]
   - Access-Control-Allow-Methods: GET, POST, PUT, DELETE, OPTIONS
   - Access-Control-Allow-Headers: Content-Type, Authorization

5. **If Still Having Issues:**
   - Clear browser cache and cookies
   - Try in an incognito window
   - Check browser console for specific error details
   - Run API diagnostics: runApiDiagnostics() in console
`;
}

// Helper to check if we're in proxy mode
export function isUsingProxy(): boolean {
  return import.meta.env.VITE_USE_DIRECT_API !== 'true';
}

// Helper to get the correct API URL
export function getApiUrl(): string {
  if (isUsingProxy()) {
    return (
      import.meta.env.VITE_API_BASE_URL ||
      'https://building-vitals-api-763711142996.us-central1.run.app'
    );
  }
  return import.meta.env.VITE_ACE_API_URL || '';
}

}
