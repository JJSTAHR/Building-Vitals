// Transform API data to BoxPlot format
export function transformToBoxPlot(apiData: APIData[]) {
  return apiData.map((pointData) => {
    const values = pointData.data.map((d) => d.value);

    if (values.length === 0) {
      return {
        name: pointData.pointName,
        min: 0,
        q1: 0,
        median: 0,
        q3: 0,
        max: 0,
        outliers: [],
      };
    }

    // Calculate quartiles
    const q1 = quantile(values, 0.25);
    const q3 = quantile(values, 0.75);
    const iqr = q3 - q1;
    const lowerBound = q1 - 1.5 * iqr;
    const upperBound = q3 + 1.5 * iqr;

    // Find outliers
    const outliers = values.filter((v) => v < lowerBound || v > upperBound);

    // Filter values without outliers for min/max
    const filteredValues = values.filter((v) => v >= lowerBound && v <= upperBound);

    return {
      name: pointData.pointName,
      min: filteredValues.length > 0 ? min(filteredValues) : min(values),
      q1: q1,
      median: quantile(values, 0.5),
      q3: q3,
      max: filteredValues.length > 0 ? max(filteredValues) : max(values),
      outliers: outliers,
    };
  });
}

// Transform API data to Candlestick format (for time periods)
export function transformToCandlestick(apiData: APIData, period: 'hourly' | 'daily' = 'hourly') {
  if (!apiData.data || apiData.data.length === 0) {
    return [];
  }

  const sortedData = [...apiData.data].sort((a, b) => a.timestamp - b.timestamp);
  const startDate = new Date(sortedData[0].timestamp);
  const endDate = new Date(sortedData[sortedData.length - 1].timestamp);

  const intervals =
    period === 'hourly'
      ? eachHourOfInterval({ start: startDate, end: endDate })
      : eachDayOfInterval({ start: startDate, end: endDate });

  return intervals
    .map((interval) => {
      const nextInterval = new Date(
        interval.getTime() + (period === 'hourly' ? 3600000 : 86400000)
      );

      const periodData = sortedData.filter((d) => {
        const date = new Date(d.timestamp);
        return date >= interval && date < nextInterval;
      });

      if (periodData.length === 0) {
        return null;
      }

      const values = periodData.map((d) => d.value);

      return {
        timestamp: interval.getTime(),
        open: periodData[0].value,
        close: periodData[periodData.length - 1].value,
        high: max(values),
        low: min(values),
        volume: periodData.length, // Using count as volume
      };
    })
    .filter((d) => d !== null);
}

/**
 * Enhanced transformer for Radar chart data
 * Accepts series data in standard format and transforms to radar format
 * Supports different aggregation types and auto-calculated indicator ranges
 * 
 * @param data - Series data in standard format: [{name, data: [[timestamp, value]], unit, markerTags}]
 * @param options - Configuration options for radar transformation
 * @returns Radar chart data with indicators and series
 */
export function transformToRadar(
  data: any,
  options: {
    aggregation?: 'latest' | 'average' | 'max' | 'min';
    indicators?: RadarIndicator[];
    showMarkerTags?: boolean;
  } = {}
): RadarChartData {
  const { aggregation = 'latest', indicators, showMarkerTags = true } = options;
  
  // Handle different input formats
  let seriesData: Array<{
    name: string;
    data: Array<[number, number]>;
    unit?: string;
    markerTags?: string[];
    pointMetadata?: PointMetadata;
  }> = [];
  
  // Convert legacy APIData format to standard format
  if (data && Array.isArray(data) && data[0] && 'pointId' in data[0]) {
    // Legacy API format
    seriesData = data.map((apiData: APIData) => ({
      name: apiData.pointName,
      data: apiData.data.map(d => [d.timestamp, d.value] as [number, number]),
      unit: undefined,
      markerTags: undefined
    }));
  } else if (data && Array.isArray(data)) {
    // Standard series format
    seriesData = data.filter(series => 
      series && 
      typeof series.name === 'string' &&
      Array.isArray(series.data)
    );
  } else {
    return { indicators: [], series: [] };
  }
  
  if (seriesData.length === 0) {
    return { indicators: [], series: [] };
  }
  
  // Generate indicators (metrics) from data if not provided
  const radarIndicators: RadarIndicator[] = indicators || seriesData.map((series) => {
    const values = series.data.map(([_, value]) => value).filter(v => typeof v === 'number' && !isNaN(v));
    
    if (values.length === 0) {
      return { name: series.name, max: 100, min: 0, unit: series.unit };
    }
    
    const minVal = Math.min(...values);
    const maxVal = Math.max(...values);
    const padding = Math.abs(maxVal - minVal) * 0.1; // 10% padding
    
    return {
      name: series.name,
      max: maxVal + padding,
      min: Math.max(0, minVal - padding), // Don't go below 0 unless data is negative
      unit: series.unit
    };
  });
  
  // Calculate aggregated values for each series based on aggregation type
  const calculateAggregatedValue = (data: Array<[number, number]>, type: 'latest' | 'average' | 'max' | 'min'): number => {
    if (!data || data.length === 0) return 0;
    
    const values = data.map(([_, value]) => value).filter(v => typeof v === 'number' && !isNaN(v));
    if (values.length === 0) return 0;
    
    switch (type) {
      case 'latest':
        return values[values.length - 1];
      case 'average':
        return values.reduce((sum, val) => sum + val, 0) / values.length;
      case 'max':
        return Math.max(...values);
      case 'min':
        return Math.min(...values);
      default:
        return values[values.length - 1];
    }
  };
  
  // Create radar series data
  const radarSeries: RadarSeriesData[] = [];
  
  // For multiple aggregation types, create separate series
  if (aggregation === 'latest' || !aggregation) {
    // Create current/latest values series
    radarSeries.push({
      name: 'Current',
      value: seriesData.map(series => calculateAggregatedValue(series.data, 'latest')),
      markerTags: showMarkerTags ? seriesData.flatMap(s => s.markerTags || []) : undefined
    });
  } else {
    // Single aggregation type
    radarSeries.push({
      name: aggregation.charAt(0).toUpperCase() + aggregation.slice(1),
      value: seriesData.map(series => calculateAggregatedValue(series.data, aggregation)),
      markerTags: showMarkerTags ? seriesData.flatMap(s => s.markerTags || []) : undefined
    });
  }
  
  // Optionally add comparison series (average and peak for 'latest')
  if (aggregation === 'latest') {
    radarSeries.push(
      {
        name: 'Average',
        value: seriesData.map(series => calculateAggregatedValue(series.data, 'average'))
      },
      {
        name: 'Peak',
        value: seriesData.map(series => calculateAggregatedValue(series.data, 'max'))
      }
    );
  }
  
  return {
    indicators: radarIndicators,
    series: radarSeries
  };
}

// Transform API data to Calendar Heatmap format
export function transformToCalendarHeatmap(apiData: APIData) {
  if (!apiData.data || apiData.data.length === 0) {
    return [];
  }

  // Group data by day and calculate daily average
  const dailyData = new Map<string, number[]>();

  apiData.data.forEach((point) => {
    const date = format(new Date(point.timestamp), 'yyyy-MM-dd');
    if (!dailyData.has(date)) {
      dailyData.set(date, []);
    }
    dailyData.get(date)!.push(point.value);
  });

  // Calculate daily averages
  return Array.from(dailyData.entries()).map(([date, values]) => ({
    date,
    value: mean(values),
  }));
}

// Aggregate data by time period
export function aggregateByPeriod(
  apiData: APIData,
  period: 'hourly' | 'daily' | 'weekly' | 'monthly'
): { categories: string[]; values: number[] } {
  if (!apiData.data || apiData.data.length === 0) {
    return { categories: [], values: [] };
  }

  const aggregatedData = new Map<string, number[]>();

  apiData.data.forEach((point) => {
    let key: string;
    const date = new Date(point.timestamp);

    switch (period) {
      case 'hourly':
        key = format(startOfHour(date), 'yyyy-MM-dd HH:00');
        break;
      case 'daily':
        key = format(startOfDay(date), 'yyyy-MM-dd');
        break;
      case 'weekly':
        key = format(startOfDay(date), "yyyy-'W'ww");
        break;
      case 'monthly':
        key = format(date, 'yyyy-MM');
        break;
    }

    if (!aggregatedData.has(key)) {
      aggregatedData.set(key, []);
    }
    aggregatedData.get(key)!.push(point.value);
  });

  // Sort by date and calculate averages
  const sortedEntries = Array.from(aggregatedData.entries()).sort((a, b) =>
    a[0].localeCompare(b[0])
  );

  return {
    categories: sortedEntries.map(([key]) => key),
    values: sortedEntries.map(([, values]) => mean(values)),
  };
}

// Calculate correlation between two data series
export function calculateCorrelation(data1: APIDataPoint[], data2: APIDataPoint[]): number {
  // Align data by timestamp
  const timestamps1 = new Set(data1.map((d) => d.timestamp));
  const timestamps2 = new Set(data2.map((d) => d.timestamp));
  const commonTimestamps = Array.from(timestamps1).filter((t) => timestamps2.has(t));

  if (commonTimestamps.length < 2) {
    return 0;
  }

  const values1 = commonTimestamps.map((t) => data1.find((d) => d.timestamp === t)!.value);
  const values2 = commonTimestamps.map((t) => data2.find((d) => d.timestamp === t)!.value);

  // Calculate Pearson correlation
  const n = values1.length;
  const sum1 = values1.reduce((a, b) => a + b, 0);
  const sum2 = values2.reduce((a, b) => a + b, 0);
  const sum1Sq = values1.reduce((a, b) => a + b * b, 0);
  const sum2Sq = values2.reduce((a, b) => a + b * b, 0);
  const pSum = values1.reduce((a, b, i) => a + b * values2[i], 0);

  const num = pSum - (sum1 * sum2) / n;
  const den = Math.sqrt((sum1Sq - (sum1 * sum1) / n) * (sum2Sq - (sum2 * sum2) / n));

  return den === 0 ? 0 : num / den;
}

/**
 * Transform data for time series charts (line, area, stacked area)
 */
export function transformToTimeSeries(
  data: any,
  options?: {
    type?: 'line' | 'area' | 'stackedArea';
    showMarkerTags?: boolean;
  }
): TimeSeriesData[] {
  if (!data || !Array.isArray(data)) {return [];}

  const { type = 'line', showMarkerTags = true } = options || {};

  return data.map((series) => {
    const displayName =
      showMarkerTags && series.markerTags
        ? formatPointWithTags(
            { id: series.name, name: series.name, markerTags: series.markerTags },
            { maxTags: 2, includeUnit: false }
          )
        : series.name;

    const result: TimeSeriesData = {
      name: displayName,
      type: 'line',
      data: series.data || [],
      unit: series.unit,
      color: series.color,
      markerTags: series.markerTags,
      pointMetadata: series.pointMetadata,
    };

    // Apply area styling
    if (type === 'area' || type === 'stackedArea') {
      result.areaStyle = {};
    }

    // Apply stacking
    if (type === 'stackedArea') {
      result.stack = 'Total';
    }

    return result;
  });
}

/**
 * Transform data for bar charts
 * Extracts the LATEST value from each time series for bar display
 */
export function transformToBar(data: any): Array<{ name: string; value: number; unit?: string }> {
  if (!data || !Array.isArray(data)) {return [];}

  return data.map((series) => {
    let latestValue = 0;
    
    // Handle time series data format: { name, data: [[timestamp, value], ...], unit }
    if (series.data && Array.isArray(series.data) && series.data.length > 0) {
      // Get the latest (last) value from time series data
      const lastDataPoint = series.data[series.data.length - 1];
      if (Array.isArray(lastDataPoint) && lastDataPoint.length >= 2) {
        latestValue = lastDataPoint[1]; // Extract value from [timestamp, value] pair
      } else if (typeof lastDataPoint === 'number') {
        latestValue = lastDataPoint;
      }
    }
    // Handle simple value format: { name, value, unit }
    else if (typeof series.value === 'number') {
      latestValue = series.value;
    }
    
    return {
      name: series.name || 'Unknown',
      value: latestValue,
      unit: series.unit,
    };
  });
}

 */

  }

  // Handle single value format
  if (typeof data === 'number') {

  if (typeof data === 'object' && 'value' in data) {

      unit: data.unit,
      markerTags: data.markerTags,
    };
  }

  return { type: 'gauge', data: [{ value: 0, name: 'Invalid Data' }] };
}

/**
 * Transform data for enhanced box plots
 * Transforms series data into BoxPlotData format with calculated statistics
 * Returns data in ECharts boxplot format: [min, Q1, median, Q3, max]
 */
export function transformToEnhancedBoxPlot(data: any): BoxPlotData {
  if (!data || !Array.isArray(data)) {
    return { type: 'boxplot', data: [] };
  }

  const boxplotData = data
    .map((series) => {
      let values: number[] = [];
      
      // Handle time series data format: { name, data: [[timestamp, value], ...] }
      if (series && series.data && Array.isArray(series.data)) {
        values = series.data
          .map((d: any) => {
            // Handle [timestamp, value] pairs
            if (Array.isArray(d) && d.length >= 2) {
              return d[1];
            }
            // Handle direct numbers
            if (typeof d === 'number') {
              return d;
            }
            return null;
          })
          .filter((v: any) => typeof v === 'number' && !isNaN(v));
      }
      // Handle direct array of values
      else if (Array.isArray(series) && typeof series[0] === 'number') {
        values = series.filter((v: any) => typeof v === 'number' && !isNaN(v));
      }
      
      if (values.length === 0) {
        return null;
      }

      // Sort values for statistical calculations
      const sortedValues = [...values].sort((a: number, b: number) => a - b);

      // Calculate statistics
      const minVal = sortedValues[0];
      const maxVal = sortedValues[sortedValues.length - 1];
      const q1 = percentile(sortedValues, 0.25);
      const median = percentile(sortedValues, 0.5);
      const q3 = percentile(sortedValues, 0.75);
      
      // Calculate outliers using IQR method
      const iqr = q3 - q1;
      const lowerBound = q1 - 1.5 * iqr;
      const upperBound = q3 + 1.5 * iqr;
      
      // Adjust min/max to exclude outliers for whiskers
      const filteredValues = sortedValues.filter(v => v >= lowerBound && v <= upperBound);
      const whiskerMin = filteredValues.length > 0 ? Math.min(...filteredValues) : minVal;
      const whiskerMax = filteredValues.length > 0 ? Math.max(...filteredValues) : maxVal;

      // Return in ECharts boxplot format: [min, Q1, median, Q3, max]
      return [whiskerMin, q1, median, q3, whiskerMax];
    })
    .filter(Boolean);

  return { type: 'boxplot', data: boxplotData };
}

/**
 * Transform data for enhanced candlestick charts
 * Aggregates time series data into OHLC format
 */
export function transformToEnhancedCandlestick(
  data: any,
  interval: 'hourly' | 'daily' | 'weekly' = 'hourly'

    return { type: 'candlestick', data: [] };
  }

  const candlestickData: Array<[number, number, number, number]> = [];

  data.forEach((series) => {
    if (!series.data || !Array.isArray(series.data)) {return;}

    // Group data by interval
    const grouped = groupByInterval(series.data, interval);

    // Convert each group to OHLC
    Object.values(grouped).forEach((group: any) => {
      if (group.length === 0) {return;}

      const values = group.map((d: [number, number]) => d[1]);
      const open = values[0];
      const close = values[values.length - 1];
      const lowest = Math.min(...values);
      const highest = Math.max(...values);

      candlestickData.push([open, close, lowest, highest]);
    });
  });

  return { type: 'candlestick', data: candlestickData };
}

/**
 * Transform timeseries data to device deviation heatmap format
 * Converts multiple timeseries into a grid showing deviations from targets
 */
export function transformToDeviationHeatmap(
  seriesData: Array<{
    name: string;
    data: Array<[number, number]> | Array<{timestamp: number, value: number}>;
    unit?: string;
    markerTags?: string[];
  }>,
  configs?: Array<{
    pointName: string;
    target?: number;
    minRange?: number;
    maxRange?: number;
    toleranceMode?: 'percentage' | 'absolute';
    tolerance?: number;
    criticalThreshold?: number;
  }>,
  timeInterval: 'daily' | 'hourly' | '15min' | '5min' | 'none' = 'hourly'
): any {
  if (!seriesData || seriesData.length === 0) {
    return { data: [], xAxis: [], yAxis: [], configs: [] };
  }

  // Normalize data to [timestamp, value] format
  const normalizedSeries = seriesData.map(series => ({
    ...series,
    data: series.data.map(point => 
      Array.isArray(point) ? point : [point.timestamp, point.value]
    ) as Array<[number, number]>
  }));

  // Find time range across all series
  let minTime = Infinity;
  let maxTime = -Infinity;
  normalizedSeries.forEach(series => {
    series.data.forEach(([timestamp]) => {
      minTime = Math.min(minTime, timestamp);
      maxTime = Math.max(maxTime, timestamp);
    });
  });

  // Create time buckets based on interval
  let timeBuckets: number[] = [];
  
  if (timeInterval === 'none') {
    // For 'none' interval, collect ALL unique timestamps from all series
    const uniqueTimestamps = new Set<number>();
    normalizedSeries.forEach(series => {
      series.data.forEach(([timestamp]) => {
        uniqueTimestamps.add(timestamp);
      });
    });
    timeBuckets = Array.from(uniqueTimestamps).sort((a, b) => a - b);
  } else {
    // Original bucketing logic with daily support
    const intervalMs = timeInterval === 'daily' ? 86400000 :  // 24 hours
                       timeInterval === 'hourly' ? 3600000 :   // 1 hour
                       timeInterval === '15min' ? 900000 :      // 15 minutes
                       timeInterval === '5min' ? 300000 : 3600000; // 5 minutes or default to hourly
    
    // Align to interval boundaries for cleaner buckets
    const alignedMinTime = Math.floor(minTime / intervalMs) * intervalMs;
    const alignedMaxTime = Math.ceil(maxTime / intervalMs) * intervalMs;
    
    for (let t = alignedMinTime; t <= alignedMaxTime; t += intervalMs) {
      timeBuckets.push(t);
    }
  }

  // Create heatmap data grid
  const heatmapData: Array<{
    value: [number, number, number];
    actualValue: number;
    deviation: number;
    status: 'normal' | 'warning' | 'critical';
    timestamp: number;
    pointName: string;
    unit?: string;
    config?: any;
  }> = [];

  normalizedSeries.forEach((series, yIndex) => {
    // Find config for this series
    let config = configs?.find(c => 
      c.pointName === series.name || 
      c.pointName.includes(series.name) ||
      series.name.includes(c.pointName)
    );

    // If no config provided, generate intelligent defaults instead of disabling deviations
    if (!config && series.data.length > 0) {
      console.log('[transformToDeviationHeatmap] No config for series:', series.name, 
        '- generating intelligent default config');
      
      // Generate intelligent config based on point name and unit
      const generatedConfig = generateDeviationConfig({
        name: series.name,
        unit: series.unit,
        markerTags: series.markerTags
      });
      
      // Convert to transformer format
      config = {
        pointName: series.name,
        target: generatedConfig.mode === 'target' ? generatedConfig.targetValue : undefined,
        minRange: generatedConfig.mode === 'range' ? generatedConfig.minValue : undefined,
        maxRange: generatedConfig.mode === 'range' ? generatedConfig.maxValue : undefined,
        toleranceMode: generatedConfig.toleranceMode,
        tolerance: generatedConfig.tolerance,
        criticalThreshold: generatedConfig.criticalThreshold
      };
    }

    if (timeInterval === 'none') {
      // For 'none' interval, process each data point individually
      // Create a map for quick timestamp lookup
      const dataMap = new Map<number, number>();
      series.data.forEach(([timestamp, value]) => {
        dataMap.set(timestamp, value);
      });

      // Process each unique timestamp
      timeBuckets.forEach((timestamp, xIndex) => {
        const value = dataMap.get(timestamp);
        
        // Only create a heatmap point if this series has data at this timestamp
        if (value !== undefined) {
          // Calculate deviation for this individual point
          let deviation = 0;
          let normalizedDeviation = 0;
          let status: 'normal' | 'warning' | 'critical' = 'normal';
          
          if (config) {
            if (config.target !== undefined) {
              // Target-based deviation
              deviation = value - config.target;
              
              // Use user-configured tolerance if provided
              let tolerance: number;
              let criticalTolerance: number;
              
              if (config.tolerance !== undefined && config.criticalThreshold !== undefined) {
                if (config.toleranceMode === 'percentage') {
                  tolerance = Math.abs(config.target * config.tolerance / 100);
                  criticalTolerance = Math.abs(config.target * config.criticalThreshold / 100);
                } else {
                  tolerance = config.tolerance;
                  criticalTolerance = config.criticalThreshold;
                }
              } else {
                // Smart tolerance calculation
                const targetMagnitude = Math.abs(config.target);
                if (targetMagnitude < 0.1) {
                  criticalTolerance = Math.max(0.002, targetMagnitude * 0.1);
                } else if (targetMagnitude < 1) {
                  criticalTolerance = targetMagnitude * 0.15;
                } else if (targetMagnitude < 10) {
                  criticalTolerance = targetMagnitude * 0.1;
                } else {
                  criticalTolerance = targetMagnitude * 0.05;
                }
                tolerance = criticalTolerance * 0.3;
              }
              
              // Normalize deviation
              const absDeviation = Math.abs(deviation);
              if (absDeviation <= tolerance) {
                normalizedDeviation = (absDeviation / tolerance) * 0.5 * Math.sign(deviation);
              } else if (absDeviation <= criticalTolerance) {
                const warningRange = criticalTolerance - tolerance;
                const warningProgress = (absDeviation - tolerance) / warningRange;
                normalizedDeviation = (0.5 + warningProgress * 0.5) * Math.sign(deviation);
              } else {
                normalizedDeviation = (absDeviation / criticalTolerance) * Math.sign(deviation);
              }
              
              const absNorm = Math.abs(normalizedDeviation);
              status = absNorm > 1 ? 'critical' : absNorm > 0.5 ? 'warning' : 'normal';
              
            } else if (config.minRange !== undefined && config.maxRange !== undefined) {
              // Range-based compliance
              if (value >= config.minRange && value <= config.maxRange) {
                deviation = 0;
                normalizedDeviation = 0;
                status = 'normal';
              } else {
                if (value < config.minRange) {
                  deviation = value - config.minRange;
                } else {
                  deviation = value - config.maxRange;
                }
                
                // Calculate normalized deviation for ranges
                let tolerance: number;
                let criticalTolerance: number;
                
                if (config.tolerance !== undefined && config.criticalThreshold !== undefined) {
                  if (config.toleranceMode === 'percentage') {
                    const rangeSpan = config.maxRange - config.minRange;
                    tolerance = rangeSpan * config.tolerance / 100;
                    criticalTolerance = rangeSpan * config.criticalThreshold / 100;
                  } else {
                    tolerance = config.tolerance;
                    criticalTolerance = config.criticalThreshold;
                  }
                } else {
                  const rangeSpan = config.maxRange - config.minRange;
                  if (rangeSpan < 1) {
                    criticalTolerance = rangeSpan * 0.2;
                  } else if (rangeSpan < 10) {
                    criticalTolerance = rangeSpan * 0.15;
                  } else {
                    criticalTolerance = rangeSpan * 0.1;
                  }
                  tolerance = criticalTolerance * 0.3;
                }
                
                const absDeviation = Math.abs(deviation);
                if (absDeviation <= tolerance) {
                  normalizedDeviation = (absDeviation / tolerance) * 0.5 * Math.sign(deviation);
                } else if (absDeviation <= criticalTolerance) {
                  const warningRange = criticalTolerance - tolerance;
                  const warningProgress = (absDeviation - tolerance) / warningRange;
                  normalizedDeviation = (0.5 + warningProgress * 0.5) * Math.sign(deviation);
                } else {
                  normalizedDeviation = (absDeviation / criticalTolerance) * Math.sign(deviation);
                }
                
                const absNorm = Math.abs(normalizedDeviation);
                status = absNorm > 1 ? 'critical' : 'warning';
              }
            }
          }
          
          // Smart clamping with logarithmic scaling for extreme values
          // This ensures gradients are visible even with extreme deviations
          if (Math.abs(normalizedDeviation) > 1) {
            // Apply logarithmic scaling for values beyond ±1
            const sign = Math.sign(normalizedDeviation);
            const absValue = Math.abs(normalizedDeviation);
            // Map values > 1 to range [0.9, 1] using logarithmic scale
            normalizedDeviation = sign * (0.9 + 0.1 * Math.tanh(Math.log(absValue)));
          } else {
            // Keep values within ±1 as-is, but scale to use more of the range
            normalizedDeviation = normalizedDeviation * 0.9;
          }
          
          // Add the heatmap point - use xIndex for categorical axis (heatmaps need indices)
          heatmapData.push({
            value: [xIndex, yIndex, normalizedDeviation],
            actualValue: value,
            deviation,
            status,
            timestamp,
            pointName: series.name,
            unit: series.unit,
            markerTags: series.markerTags,
            config,
            isCarriedForward: false,
            lastDataTimestamp: timestamp
          });
        }
      });
    } else {
      // Original bucketing logic
      const intervalMs = timeInterval === 'daily' ? 86400000 :  // 24 hours
                         timeInterval === 'hourly' ? 3600000 :   // 1 hour
                         timeInterval === '15min' ? 900000 :      // 15 minutes
                         timeInterval === '5min' ? 300000 : 3600000; // 5 minutes or default to hourly
      
      // Create time-bucketed data
      const bucketedData = new Map<number, number[]>();
      series.data.forEach(([timestamp, value]) => {
        const bucketTime = Math.floor(timestamp / intervalMs) * intervalMs;
        if (!bucketedData.has(bucketTime)) {
          bucketedData.set(bucketTime, []);
        }
        bucketedData.get(bucketTime)!.push(value);
      });

      // Track the last known value for carry-forward
      let lastKnownValue: number | null = null;
      let lastKnownTimestamp: number | null = null;

      // Calculate deviation for each time bucket
      timeBuckets.forEach((bucketTime, xIndex) => {
        const values = bucketedData.get(bucketTime);
      
      let avgValue: number | null = null;
      let isCarriedForward = false;
      let isMissing = false;
      
      // If no data in this bucket, check if we should mark as missing or carry forward
      if (!values || values.length === 0) {
        // Calculate time since last known data
        const timeSinceLastData = lastKnownTimestamp ? bucketTime - lastKnownTimestamp : Infinity;
        const maxGapToCarryForward = intervalMs * 3; // Only carry forward for up to 3 intervals
        
        if (lastKnownValue !== null && timeSinceLastData <= maxGapToCarryForward) {
          // Carry forward if gap is small
          avgValue = lastKnownValue;
          isCarriedForward = true;
        } else {
          // Mark as missing data for large gaps
          isMissing = true;
          avgValue = null;
        }
      } else {
        // Average values in bucket and update last known value
        avgValue = values.reduce((a, b) => a + b, 0) / values.length;
        lastKnownValue = avgValue;
        lastKnownTimestamp = bucketTime;
      }
      
      // Handle missing data
      if (isMissing || avgValue === null) {
        // Add a special entry for missing data - use NaN which ECharts handles specially
        heatmapData.push({
          value: [xIndex, yIndex, NaN], // Use xIndex for categorical axis; NaN will be rendered with a special color
          actualValue: null as any,
          deviation: 0,
          status: 'missing' as any,
          timestamp: bucketTime,
          pointName: series.name,
          unit: series.unit,
          markerTags: series.markerTags,
          config,
          isCarriedForward: false,
          isMissing: true,
          lastDataTimestamp: lastKnownTimestamp || bucketTime
        });
        return; // Skip to next bucket
      }
      
      // Calculate deviation with proper compliance visualization
      let deviation = 0;
      let normalizedDeviation = 0;
      let status: 'normal' | 'warning' | 'critical' = 'normal';
      
      if (config) {
        if (config.target !== undefined) {
          // Target-based deviation (for things like pressure setpoints)
          deviation = avgValue - config.target;
          
          // Use user-configured tolerance if provided
          let tolerance: number;
          let criticalTolerance: number;
          
          if (config.tolerance !== undefined && config.criticalThreshold !== undefined) {
            // User has configured custom tolerance
            if (config.toleranceMode === 'percentage') {
              // Calculate absolute values from percentages
              tolerance = Math.abs(config.target * config.tolerance / 100);
              criticalTolerance = Math.abs(config.target * config.criticalThreshold / 100);
            } else {
              // Use absolute values directly
              tolerance = config.tolerance;
              criticalTolerance = config.criticalThreshold;
            }
          } else {
            // Fallback to smart tolerance calculation based on magnitude
            const targetMagnitude = Math.abs(config.target);
            
            if (targetMagnitude < 0.1) {
              // For tiny values like 0.02" pressure
              // Use more reasonable tolerances that show gradients properly
              // For 0.02 target: tolerance=0.005 (25%), critical=0.02 (100%)
              criticalTolerance = Math.max(targetMagnitude, 0.02); // At least 0.02 or 100% of target
              tolerance = criticalTolerance * 0.25; // Start gradient at 25% of critical
            } else if (targetMagnitude < 1) {
              // For small values (0.1 to 1)
              criticalTolerance = targetMagnitude * 0.3; // 30% of target
              tolerance = criticalTolerance * 0.33; // Start gradient at 1/3 of critical
            } else if (targetMagnitude < 10) {
              // For medium values (1 to 10)
              criticalTolerance = targetMagnitude * 0.2; // 20% of target
              tolerance = criticalTolerance * 0.4; // Start gradient at 40% of critical
            } else {
              // For larger values (temperature, humidity percentages)
              criticalTolerance = targetMagnitude * 0.1; // 10% of target
              tolerance = criticalTolerance * 0.5; // Start gradient at 50% of critical
            }
          }
          
          // Normalize with gradient zones - use full color range for better visibility
          const absDeviation = Math.abs(deviation);
          if (absDeviation <= tolerance) {
            // Within tolerance zone: map to 0 to 0.3 (green zone)
            normalizedDeviation = (absDeviation / tolerance) * 0.3 * Math.sign(deviation);
          } else if (absDeviation <= criticalTolerance) {
            // Warning zone: map to 0.3 to 1.0 (yellow to orange)
            const warningRange = criticalTolerance - tolerance;
            const warningProgress = (absDeviation - tolerance) / warningRange;
            normalizedDeviation = (0.3 + warningProgress * 0.7) * Math.sign(deviation);
          } else {
            // Beyond critical: > 1.0 (red zone)
            // Cap at 2.0 to prevent extreme outliers from washing out the gradient
            normalizedDeviation = Math.min(2.0, absDeviation / criticalTolerance) * Math.sign(deviation);
          }
          
          // Status based on normalized deviation
          const absNorm = Math.abs(normalizedDeviation);
          status = absNorm > 1 ? 'critical' : 
                   absNorm > 0.5 ? 'warning' : 'normal';
                   
        } else if (config.minRange !== undefined && config.maxRange !== undefined) {
          // Range-based compliance (for things like humidity 30-60%)
          if (avgValue >= config.minRange && avgValue <= config.maxRange) {
            // COMPLIANT - solid green
            deviation = 0;
            normalizedDeviation = 0;
            status = 'normal';
          } else {
            // NON-COMPLIANT - gradient based on distance
            if (avgValue < config.minRange) {
              deviation = avgValue - config.minRange; // Negative
            } else {
              deviation = avgValue - config.maxRange; // Positive
            }
            
            // Use user-configured tolerance if provided, otherwise calculate
            let tolerance: number;
            let criticalTolerance: number;
            
            if (config.tolerance !== undefined && config.criticalThreshold !== undefined) {
              // User has configured custom tolerance for range
              if (config.toleranceMode === 'percentage') {
                // Calculate based on range span
                const rangeSpan = config.maxRange - config.minRange;
                tolerance = rangeSpan * config.tolerance / 100;
                criticalTolerance = rangeSpan * config.criticalThreshold / 100;
              } else {
                // Use absolute values directly
                tolerance = config.tolerance;
                criticalTolerance = config.criticalThreshold;
              }
            } else {
              // Fallback to smart tolerance calculation
              const rangeSpan = config.maxRange - config.minRange;
              
              if (rangeSpan < 1) {
                // Tight range (like 0-0.05 pressure)
                criticalTolerance = rangeSpan * 0.2; // 20% of range
              } else if (rangeSpan < 10) {
                // Medium range
                criticalTolerance = rangeSpan * 0.15;
              } else {
                // Wide range (like 30-60% humidity)
                // Use 10% of range span for full red
                criticalTolerance = rangeSpan * 0.1;
              }
              tolerance = criticalTolerance * 0.3; // Start gradient at 30%
            }
            
            // Normalize with gradient zones
            const absDeviation = Math.abs(deviation);
            if (absDeviation <= tolerance) {
              // Within tolerance zone: map to 0 to 0.5
              normalizedDeviation = (absDeviation / tolerance) * 0.5 * Math.sign(deviation);
            } else if (absDeviation <= criticalTolerance) {
              // Warning zone: map to 0.5 to 1.0
              const warningRange = criticalTolerance - tolerance;
              const warningProgress = (absDeviation - tolerance) / warningRange;
              normalizedDeviation = (0.5 + warningProgress * 0.5) * Math.sign(deviation);
            } else {
              // Beyond critical: > 1.0
              normalizedDeviation = (absDeviation / criticalTolerance) * Math.sign(deviation);
            }
            
            // Status based on how far out of compliance
            const absNorm = Math.abs(normalizedDeviation);
            status = absNorm > 1 ? 'critical' : 'warning';
          }
        }
      } else {
        // No config - just show the values without deviation coloring
        normalizedDeviation = 0;
        status = 'normal';
      }

      // Clamp to -1 to 1 for consistent color mapping
      normalizedDeviation = Math.max(-1, Math.min(1, normalizedDeviation));

      heatmapData.push({
        value: [xIndex, yIndex, normalizedDeviation], // Use xIndex for categorical axis
        actualValue: avgValue,
        deviation,
        status,
        timestamp: bucketTime,
        pointName: series.name,
        unit: series.unit,
        markerTags: series.markerTags,
        config,
        isCarriedForward,
        lastDataTimestamp: isCarriedForward ? lastKnownTimestamp : bucketTime,
        // Add subtle visual indicator for carried-forward values
        itemStyle: isCarriedForward ? {
          opacity: 0.85, // Slightly transparent to indicate carried-forward
          borderWidth: 0.5,
          borderColor: 'rgba(255, 255, 255, 0.2)'
        } : undefined
      });
    });
    }
  });

  // Debug yAxis formatting
  const yAxisLabels = normalizedSeries.map(s => {
    const hasMarkerTags = s.markerTags && s.markerTags.length > 0;
    let formattedLabel = s.name;
    
    if (hasMarkerTags) {
      formattedLabel = formatPointWithTags(
        { id: s.name, name: s.name, markerTags: s.markerTags },
        { maxTags: 2, includeUnit: false }
      );
    }
    
    console.log('[transformToDeviationHeatmap] Y-axis label formatting:', {
      originalName: s.name,
      hasMarkerTags,
      markerTags: s.markerTags,
      formattedLabel
    });
    
    return formattedLabel;
  });

  // Create formatted time strings for xAxis labels (categorical axis needs string labels)
  const xAxisLabels = timeBuckets.map(timestamp => {
    const date = new Date(timestamp);
    // Format based on the time interval for readability
    if (timeInterval === 'daily') {
      return date.toLocaleDateString('en-US', { month: 'short', day: 'numeric' });
    } else if (timeInterval === 'hourly') {
      return date.toLocaleTimeString('en-US', { hour: 'numeric', hour12: true });
    } else if (timeInterval === '15min' || timeInterval === '5min') {
      return date.toLocaleTimeString('en-US', { hour: 'numeric', minute: '2-digit', hour12: true });
    } else {
      // For 'none' interval, show more detailed time
      const now = Date.now();
      const daysDiff = (now - timestamp) / (24 * 60 * 60 * 1000);
      
      if (daysDiff > 1) {
        return date.toLocaleDateString('en-US', { month: 'short', day: 'numeric' }) + 
               '\n' + date.toLocaleTimeString('en-US', { hour: 'numeric', minute: '2-digit', hour12: true });
      } else {
        return date.toLocaleTimeString('en-US', { hour: 'numeric', minute: '2-digit', hour12: true });
      }
    }
  });

  return {
    data: heatmapData,
    xAxis: xAxisLabels, // Return formatted time strings for categorical axis
    yAxis: yAxisLabels,
    configs: configs || []
  };
}

/**
 * Transform data for calendar heatmap
 */
export function transformToCalendar(data: any): HeatmapData {
  if (!data || !Array.isArray(data)) {
    return { type: 'heatmap', coordinateSystem: 'calendar', data: [] };
  }

  const calendarData: Array<[string, number]> = [];

  // Aggregate data by date
  const dateMap = new Map<string, number[]>();

  data.forEach((series) => {
    if (!series.data || !Array.isArray(series.data)) {return;}

    series.data.forEach((point: [number, number]) => {
      const date = new Date(point[0]).toISOString().split('T')[0];
      if (!dateMap.has(date)) {
        dateMap.set(date, []);
      }
      dateMap.get(date)!.push(point[1]);
    });
  });

  // Calculate average for each date
  dateMap.forEach((values, date) => {
    const avg = values.reduce((a, b) => a + b, 0) / values.length;
    calendarData.push([date, avg]);
  });

  return {
    type: 'heatmap',
    coordinateSystem: 'calendar',
    data: calendarData as any,
  };
}

/**
 * Transform data for parallel coordinates
 */
export function transformToParallel(data: any): ParallelData {
  if (!data || !Array.isArray(data)) {
    return { axes: [], data: [] };
  }

  // Extract axes from first data point
  const axes: ParallelData['axes'] = [];
  const parallelData: Array<Array<number | string>> = [];

  if (data.length > 0 && data[0].data && data[0].data.length > 0) {
    // Create axes based on series names
    data.forEach((series, index) => {
      axes.push({
        dim: index,
        name: series.name,
        type: 'value',
        min: Math.min(...series.data.map((d: [number, number]) => d[1])),
        max: Math.max(...series.data.map((d: [number, number]) => d[1])),
      });
    });

    // Transform data points
    const timestamps = new Set<number>();
    data.forEach((series) => {
      series.data.forEach((d: [number, number]) => timestamps.add(d[0]));
    });

    Array.from(timestamps).forEach((timestamp) => {
      const point: (number | null)[] = [];
      data.forEach((series) => {
        const value = series.data.find((d: [number, number]) => d[0] === timestamp);
        point.push(value ? value[1] : null);
      });
      parallelData.push(point.filter((v): v is number => v !== null));
    });
  }

  return { axes, data: parallelData };
}

    showMarkerTags?: boolean;

  // Case 1: Already in psychrometric format
  if (Array.isArray(data) && data.length > 0 && 
      typeof data[0] === 'object' && 'temperature' in data[0] && 'humidity' in data[0]) {
    points = data.map((point, index) => ({
      temperature: typeof point.temperature === 'number' ? point.temperature : 0,
      humidity: typeof point.humidity === 'number' ? point.humidity : 0,
      timestamp: point.timestamp || Date.now() + index,
      name: point.name || `Point ${index + 1}`,
      dewPoint: point.dewPoint,
      wetBulb: point.wetBulb,
      enthalpy: point.enthalpy,
      specificVolume: point.specificVolume,
      markerTags: showMarkerTags ? point.markerTags : undefined,
      pointMetadata: point.pointMetadata
    }));
  }
  // Case 2: Series format - need temperature/humidity correlation
  else if (Array.isArray(data) && data.length > 0 && 
           typeof data[0] === 'object' && 'name' in data[0] && 'data' in data[0]) {
    points = correlateTempHumidityFromSeries(data, { correlationMethod, maxCorrelationGapMs, showMarkerTags });
  }
  // Case 3: Pre-transformed data with points array
  else if (typeof data === 'object' && 'data' in data && Array.isArray(data.data)) {
    points = correlateTempHumidityFromSeries(data.data, { correlationMethod, maxCorrelationGapMs, showMarkerTags });
  }
  
  // Filter out invalid points
  points = points.filter(point => 
    typeof point.temperature === 'number' && 
    typeof point.humidity === 'number' &&
    !isNaN(point.temperature) &&
    !isNaN(point.humidity) &&
    point.humidity >= 0 && point.humidity <= 100
  );
  
  return {
    points,
    temperatureUnit,
    correlationMethod
  };
}

/**
 * Correlate temperature and humidity data from separate series
 * Implements sophisticated correlation logic that was previously in the component
 */
function correlateTempHumidityFromSeries(
  seriesData: Array<{
    name: string;
    data: Array<[number, number]>;
    unit?: string;
    markerTags?: string[];
    pointMetadata?: PointMetadata;
  }>,
  options: {
    correlationMethod: 'latest' | 'average' | 'paired' | 'interpolated';
    maxCorrelationGapMs: number;
    showMarkerTags: boolean;
  }
): PsychrometricDataPoint[] {
  if (!seriesData || seriesData.length === 0) {
    return [];
  }
  
  // Identify temperature and humidity series using advanced pattern matching
  const tempSeries = findTemperatureSeries(seriesData);
  const humiditySeries = findHumiditySeries(seriesData);
  
  if (tempSeries.length === 0 || humiditySeries.length === 0) {
    // Fallback: try to use first two series if we can't identify temp/humidity
    if (seriesData.length >= 2) {
      return correlateTwoSeries(
        seriesData[0], 
        seriesData[1], 
        'temperature', 
        'humidity', 
        options
      );
    }
    return [];
  }
  
  const points: PsychrometricDataPoint[] = [];
  
  // Correlate each temperature series with best matching humidity series
  tempSeries.forEach(tempS => {
    // Find best matching humidity series (by name similarity or use first available)
    const matchingHumidity = findBestHumidityMatch(tempS, humiditySeries) || humiditySeries[0];
    
    if (matchingHumidity) {
      const correlatedPoints = correlateTwoSeries(tempS, matchingHumidity, 'temperature', 'humidity', options);
      points.push(...correlatedPoints);
    }
  });
  
  return points;
}

/**
 * Find temperature series using multiple identification strategies
 */
function findTemperatureSeries(seriesData: Array<{ name: string; unit?: string }>): typeof seriesData {
  return seriesData.filter(s => {
    const name = s.name.toLowerCase();
    const unit = s.unit?.toLowerCase() || '';
    
    // Check for temperature keywords in name
    const tempKeywords = ['temp', 'temperature', 'dry', 'bulb', 'ambient', 'air temp', 'db'];
    const hasTemperatureKeyword = tempKeywords.some(keyword => name.includes(keyword));
    
    // Check for temperature units
    const tempUnits = ['°c', '°f', 'celsius', 'fahrenheit', 'deg'];
    const hasTemperatureUnit = tempUnits.some(unitKeyword => unit.includes(unitKeyword));
    
    // Exclude humidity-related temperature measurements
    const humidityExclusions = ['humid', 'rh', 'wet bulb', 'wb', 'dew'];
    const isHumidityRelated = humidityExclusions.some(exclusion => name.includes(exclusion));
    
    return (hasTemperatureKeyword || hasTemperatureUnit) && !isHumidityRelated;
  });
}

/**
 * Find humidity series using multiple identification strategies
 */
function findHumiditySeries(seriesData: Array<{ name: string; unit?: string }>): typeof seriesData {
  return seriesData.filter(s => {
    const name = s.name.toLowerCase();
    const unit = s.unit?.toLowerCase() || '';
    
    // Check for humidity keywords in name
    const humidityKeywords = ['humid', 'rh', 'relative humidity', 'moisture', 'water vapor'];
    const hasHumidityKeyword = humidityKeywords.some(keyword => name.includes(keyword));
    
    // Check for humidity units (percentage)
    const hasPercentageUnit = unit.includes('%') || unit.includes('percent');
    
    // Check if name suggests percentage even without explicit unit
    const nameHasPercent = name.includes('%') || name.includes('percent');
    
    return hasHumidityKeyword || (hasPercentageUnit && !name.includes('temp')) || nameHasPercent;
  });
}

/**
 * Find best matching humidity series for a given temperature series
 */
function findBestHumidityMatch(
  tempSeries: { name: string; unit?: string },
  humiditySeries: Array<{ name: string; unit?: string }>
): typeof humiditySeries[0] | null {
  if (humiditySeries.length === 0) return null;
  if (humiditySeries.length === 1) return humiditySeries[0];
  
  const tempName = tempSeries.name.toLowerCase();
  
  // Look for series with similar names (same zone, device, etc.)
  const nameMatches = humiditySeries.filter(h => {
    const humidName = h.name.toLowerCase();
    
    // Extract common identifiers (numbers, zone names, device IDs)
    const tempTokens = tempName.split(/[\s-_\.]/)
      .filter(token => token.length > 0 && (token.match(/\d/) || token.length > 3));
    const humidTokens = humidName.split(/[\s-_\.]/)
      .filter(token => token.length > 0 && (token.match(/\d/) || token.length > 3));
    
    // Check for common tokens
    return tempTokens.some(tempToken => 
      humidTokens.some(humidToken => 
        tempToken.includes(humidToken) || humidToken.includes(tempToken)
      )
    );
  });
  
  return nameMatches.length > 0 ? nameMatches[0] : humiditySeries[0];
}

/**
 * Correlate two series into psychrometric data points
 */
function correlateTwoSeries(
  series1: { name: string; data: Array<[number, number]>; markerTags?: string[]; pointMetadata?: PointMetadata },
  series2: { name: string; data: Array<[number, number]>; markerTags?: string[]; pointMetadata?: PointMetadata },
  series1Type: 'temperature' | 'humidity',
  series2Type: 'temperature' | 'humidity',
  options: {
    correlationMethod: 'latest' | 'average' | 'paired' | 'interpolated';
    maxCorrelationGapMs: number;
    showMarkerTags: boolean;
  }
): PsychrometricDataPoint[] {
  const { correlationMethod, maxCorrelationGapMs, showMarkerTags } = options;
  
  if (!series1.data?.length || !series2.data?.length) {
    return [];
  }
  
  const points: PsychrometricDataPoint[] = [];
  const combinedName = `${series1.name} / ${series2.name}`;
  const combinedMarkerTags = showMarkerTags ? [
    ...(series1.markerTags || []),
    ...(series2.markerTags || [])
  ] : undefined;
  
  switch (correlationMethod) {
    case 'latest': {
      // Use the most recent values from each series
      const latest1 = series1.data[series1.data.length - 1];
      const latest2 = series2.data[series2.data.length - 1];
      
      if (latest1 && latest2) {
        const point: PsychrometricDataPoint = {
          temperature: series1Type === 'temperature' ? latest1[1] : latest2[1],
          humidity: series1Type === 'humidity' ? latest1[1] : latest2[1],
          timestamp: Math.max(latest1[0], latest2[0]),
          name: combinedName,
          markerTags: combinedMarkerTags
        };
        points.push(point);
      }
      break;
    }
    
    case 'average': {
      // Calculate average values from each series
      const avg1 = series1.data.reduce((sum, [_, val]) => sum + val, 0) / series1.data.length;
      const avg2 = series2.data.reduce((sum, [_, val]) => sum + val, 0) / series2.data.length;
      
      const point: PsychrometricDataPoint = {
        temperature: series1Type === 'temperature' ? avg1 : avg2,
        humidity: series1Type === 'humidity' ? avg1 : avg2,
        timestamp: Date.now(),
        name: `${combinedName} (Average)`,
        markerTags: combinedMarkerTags
      };
      points.push(point);
      break;
    }
    
    case 'paired':
    case 'interpolated': {
      // Create paired data points based on timestamps
      const useInterpolation = correlationMethod === 'interpolated';
      const pairedPoints = createPairedData(series1.data, series2.data, maxCorrelationGapMs, useInterpolation);
      
      pairedPoints.forEach((pair, index) => {
        const point: PsychrometricDataPoint = {
          temperature: series1Type === 'temperature' ? pair.value1 : pair.value2,
          humidity: series1Type === 'humidity' ? pair.value1 : pair.value2,
          timestamp: pair.timestamp,
          name: `${combinedName} ${index + 1}`,
          markerTags: combinedMarkerTags
        };
        points.push(point);
      });
      break;
    }
  }
  
  return points;
}

/**
 * Create paired data points from two time series
 */
function createPairedData(
  data1: Array<[number, number]>,
  data2: Array<[number, number]>,
  maxGapMs: number,
  useInterpolation: boolean
): Array<{ timestamp: number; value1: number; value2: number }> {
  const result: Array<{ timestamp: number; value1: number; value2: number }> = [];
  
  if (useInterpolation) {
    // Get all unique timestamps from both series
    const allTimestamps = [...new Set([
      ...data1.map(([t]) => t),
      ...data2.map(([t]) => t)
    ])].sort((a, b) => a - b);
    
    allTimestamps.forEach(timestamp => {
      const value1 = interpolateValueAtTimestamp(data1, timestamp, maxGapMs);
      const value2 = interpolateValueAtTimestamp(data2, timestamp, maxGapMs);
      
      if (value1 !== null && value2 !== null) {
        result.push({ timestamp, value1, value2 });
      }
    });
  } else {
    // Simple pairing: match closest timestamps
    data1.forEach(([timestamp1, value1]) => {
      const closest = findClosestDataPoint(data2, timestamp1);
      
      if (closest && Math.abs(closest.timestamp - timestamp1) <= maxGapMs) {
        result.push({
          timestamp: timestamp1,
          value1,
          value2: closest.value
        });
      }
    });
  }
  
  return result;
}

/**
 * Interpolate value at specific timestamp
 */
function interpolateValueAtTimestamp(
  data: Array<[number, number]>,
  timestamp: number,
  maxGapMs: number
): number | null {
  if (!data.length) return null;
  
  // Find exact match
  const exact = data.find(([t]) => t === timestamp);
  if (exact) return exact[1];
  
  // Find surrounding points
  let before: [number, number] | null = null;
  let after: [number, number] | null = null;
  
  for (const [t, v] of data) {
    if (t < timestamp) {
      before = [t, v];
    } else if (t > timestamp && !after) {
      after = [t, v];
      break;
    }
  }
  
  // Use interpolation if we have both points and gap is reasonable
  if (before && after && (after[0] - before[0]) <= maxGapMs) {
    const ratio = (timestamp - before[0]) / (after[0] - before[0]);
    return before[1] + ratio * (after[1] - before[1]);
  }
  
  // Use closest point if within acceptable gap
  if (before && Math.abs(timestamp - before[0]) <= maxGapMs) {
    return before[1];
  }
  if (after && Math.abs(after[0] - timestamp) <= maxGapMs) {
    return after[1];
  }
  
  return null;
}

/**
 * Find closest data point to a given timestamp
 */
function findClosestDataPoint(
  data: Array<[number, number]>,
  targetTimestamp: number
): { timestamp: number; value: number } | null {
  if (!data.length) return null;
  
  let closest = data[0];
  let minDiff = Math.abs(closest[0] - targetTimestamp);
  
  for (const [timestamp, value] of data) {
    const diff = Math.abs(timestamp - targetTimestamp);
    if (diff < minDiff) {
      minDiff = diff;
      closest = [timestamp, value];
    }
  }
  
  return { timestamp: closest[0], value: closest[1] };
}

/**
 * Transform data for economizer chart
 */
export function transformToEconomizer(data: any): any[] {
  // Similar to psychrometric, needs outdoor and return air data
  if (!data || !Array.isArray(data)) {return [];}

  const outdoorTemp = data.find(
    (s) => s.name.toLowerCase().includes('outdoor') && s.name.toLowerCase().includes('temp')
  );
  const returnTemp = data.find(
    (s) => s.name.toLowerCase().includes('return') && s.name.toLowerCase().includes('temp')
  );

  if (!outdoorTemp || !returnTemp) {return data;}

  // Combine data points
  const economizerData: any[] = [];
  const timestamps = new Set<number>();

  outdoorTemp.data.forEach((d: [number, number]) => timestamps.add(d[0]));

  Array.from(timestamps).forEach((timestamp) => {
    const outdoor = outdoorTemp.data.find((d: [number, number]) => d[0] === timestamp);
    const returnAir = returnTemp.data.find((d: [number, number]) => d[0] === timestamp);

    if (outdoor && returnAir) {
      economizerData.push({
        timestamp,
        outdoorTemp: outdoor[1],
        returnTemp: returnAir[1],
      });
    }
  });

  return economizerData;
}

// Helper functions

function percentile(values: number[], p: number): number {
  const index = (values.length - 1) * p;
  const lower = Math.floor(index);
  const upper = Math.ceil(index);
  const weight = index % 1;

  if (lower === upper) {
    return values[lower];
  }

  return values[lower] * (1 - weight) + values[upper] * weight;
}

function groupByInterval(
  data: Array<[number, number]>,
  interval: 'hourly' | 'daily' | 'weekly'
): Record<string, Array<[number, number]>> {
  const grouped: Record<string, Array<[number, number]>> = {};

  data.forEach((point) => {
    const date = new Date(point[0]);
    let key: string;

    switch (interval) {
      case 'hourly':
        key = `${date.getFullYear()}-${date.getMonth()}-${date.getDate()}-${date.getHours()}`;
        break;
      case 'daily':
        key = `${date.getFullYear()}-${date.getMonth()}-${date.getDate()}`;
        break;
      case 'weekly':
        const week = Math.floor(date.getDate() / 7);
        key = `${date.getFullYear()}-${date.getMonth()}-W${week}`;
        break;
    }

    if (!grouped[key]) {
      grouped[key] = [];
    }
    grouped[key].push(point);
  });

  return grouped;
}
