/**
 * Main fault detection engine for HVAC system analysis
 * Implements statistical anomaly detection, cross-point correlation, and weather-based analysis
 */

import type {
  FaultDetectionResult,
  TimeSeriesData,
  AnalysisContext,
  FaultDetectionConfig,
  StatisticalAnalysisResult,
  CorrelationResult,
  StatisticalThresholds,
  CorrelationAnalysisConfig,
  DataPoint,
  FaultSeverity,
  VAVZoneData,
  EconomizerData,
  VAVDiagnosticConfig,
  EconomizerDiagnosticConfig} from './types';
import {
  FaultType
} from './types';
import { detectVAVFaults } from './vavDiagnostics';
import { detectEconomizerFaults } from './economzerFaults';

/**
 * Default configuration for fault detection algorithms
 */
const DEFAULT_CONFIG: FaultDetectionConfig = {
  statistical: {
    standardDeviationMultiplier: 2.5,
    movingAverageWindow: 24, // 24 hours
    minimumDataPoints: 10,
    confidenceLevel: 0.95,
  },
  vav: {
    stuckDamperThreshold: {
      maxVariancePercent: 5,
      minimumTimeHours: 4,
    },
    huntingThreshold: {
      oscillationFrequency: 6, // 6 oscillations per hour
      amplitudeThreshold: 10, // 10% of full range
      minimumDuration: 30, // 30 minutes
    },
    sensorDriftThreshold: {
      maxDeviationFromPeers: 3, // standard deviations
      minimumSampleSize: 5,
      consistencyPeriod: 8, // 8 hours
    },
    simultaneousHeatingCooling: {
      heatingThreshold: 10, // 10% output
      coolingThreshold: 10, // 10% output
      simultaneousDuration: 15, // 15 minutes
    },
  },
  economizer: {
    freecooling: {
      temperatureDifferentialThreshold: 5, // 5F
      damperClosedThreshold: 20, // 20% open
      minimumOpportunityDuration: 60, // 60 minutes
    },
    stuckDamper: {
      maxVariancePercent: 5,
      minimumTimeHours: 4,
    },
    lockoutConfig: {
      highTempLockout: 75, // 75F
      lowTempLockout: 55, // 55F
      enthalpyLockout: 28, // 28 BTU/lb
    },
  },
  weather: {
    mildWeatherRange: {
      minTemp: 65, // 65F
      maxTemp: 75, // 75F
    },
    energyBaselineMultiplier: 1.5,
    temperatureTrackingThreshold: 5, // 5F deviation
  },
  correlation: {
    minimumCorrelationWindow: 4, // 4 hours
    peerDeviationThreshold: 2.5, // 2.5 standard deviations
    minimumPeerCount: 3,
    similarityThreshold: 0.7, // 0.7 correlation coefficient
  },
  enabledAnalyses: {
    statistical: true,
    vav: true,
    economizer: true,
    weatherBased: true,
    crossPointCorrelation: true,
  },
};

/**
 * Main fault detection engine class
 */
export class FaultDetectionEngine {
  private config: FaultDetectionConfig;

  constructor(config?: Partial<FaultDetectionConfig>) {
    this.config = { ...DEFAULT_CONFIG, ...config };
  }

  /**
   * Main entry point for fault detection analysis
   */
  async detectFaults(
    data: TimeSeriesData[],
    context: AnalysisContext,
    config?: Partial<FaultDetectionConfig>
  ): Promise<FaultDetectionResult[]> {
    const analysisConfig = { ...this.config, ...config };
    const faults: FaultDetectionResult[] = [];

    console.log(`[FaultDetection] Starting analysis for ${data.length} time series`);

    try {
      // Statistical anomaly detection
      if (analysisConfig.enabledAnalyses.statistical) {
        const statisticalFaults = await this.detectStatisticalAnomalies(
          data,
          context,
          analysisConfig.statistical
        );
        faults.push(...statisticalFaults);
      }

      // Cross-point correlation analysis
      if (analysisConfig.enabledAnalyses.crossPointCorrelation) {
        const correlationFaults = await this.detectCorrelationFaults(
          data,
          context,
          analysisConfig.correlation
        );
        faults.push(...correlationFaults);
      }

      // Weather-based fault detection
      if (analysisConfig.enabledAnalyses.weatherBased && context.weatherData) {
        const weatherFaults = await this.detectWeatherBasedFaults(
          data,
          context,
          analysisConfig.weather
        );
        faults.push(...weatherFaults);
      }

      console.log(`[FaultDetection] Detected ${faults.length} potential faults`);
      return this.rankAndFilterFaults(faults);
    } catch (error) {
      console.error('[FaultDetection] Error during fault detection:', error);
      return [];
    }
  }

  /**
   * Detect VAV-specific faults
   */
  async detectVAVFaults(
    vavData: VAVZoneData[],
    context: AnalysisContext,
    config?: VAVDiagnosticConfig
  ): Promise<FaultDetectionResult[]> {
    if (!this.config.enabledAnalyses.vav) {return [];}

    const vavConfig = { ...this.config.vav, ...config };
    return detectVAVFaults(vavData, context, vavConfig);
  }

  /**
   * Detect economizer-specific faults
   */
  async detectEconomizerFaults(
    economizerData: EconomizerData,
    context: AnalysisContext,
    config?: EconomizerDiagnosticConfig
  ): Promise<FaultDetectionResult[]> {
    if (!this.config.enabledAnalyses.economizer) {return [];}

    const economizerConfig = { ...this.config.economizer, ...config };
    return detectEconomizerFaults(economizerData, context, economizerConfig);
  }

  /**
   * Statistical anomaly detection using standard deviation and moving averages
   */
  private async detectStatisticalAnomalies(
    data: TimeSeriesData[],
    context: AnalysisContext,
    config: StatisticalThresholds
  ): Promise<FaultDetectionResult[]> {
    const faults: FaultDetectionResult[] = [];

    for (const series of data) {
      if (series.data.length < config.minimumDataPoints) {continue;}

      const stats = this.analyzeStatistics(series, config);

      // Detect outliers
      if (stats.outliers.length > 0) {
        const outlierRatio = stats.outliers.length / series.data.length;

        if (outlierRatio > 0.05) {
          // More than 5% outliers
          faults.push({
            id: `statistical_${series.pointName}_${Date.now()}`,
            type: 'statistical_anomaly',
            severity: this.calculateSeverity(outlierRatio, 0.05, 0.15, 0.25),
            timestamp: context.timeRange.end.getTime(),
            detectedAt: new Date(),
            affectedPoints: [series.pointName],
            description: `Statistical anomaly detected: ${stats.outliers.length} outliers (${(outlierRatio * 100).toFixed(1)}%) found in ${series.pointName}`,
            recommendedAction:
              'Investigate sensor calibration and data quality. Check for measurement errors or equipment malfunctions.',
            potentialImpact: {
              comfort: 'Unreliable readings may affect control accuracy',
              maintenance: 'May indicate sensor degradation requiring calibration or replacement',
            },
            confidence: Math.min(0.95, outlierRatio * 10),
            metadata: {
              algorithm: 'statistical_outlier_detection',
              parameters: { threshold: config.standardDeviationMultiplier },
              thresholds: { outlierRatio, maxAcceptableRatio: 0.05 },
            },
          });
        }
      }

      // Detect trend anomalies
      if (stats.trend !== 'stable') {
        const trendStrength = this.calculateTrendStrength(series.data);
        if (trendStrength > 0.7) {
          faults.push({
            id: `trend_${series.pointName}_${Date.now()}`,
            type: 'statistical_anomaly',
            severity: this.calculateSeverity(trendStrength, 0.7, 0.8, 0.9),
            timestamp: context.timeRange.end.getTime(),
            detectedAt: new Date(),
            affectedPoints: [series.pointName],
            description: `Strong ${stats.trend} trend detected in ${series.pointName} (strength: ${(trendStrength * 100).toFixed(1)}%)`,
            recommendedAction:
              stats.trend === 'increasing'
                ? 'Monitor for equipment degradation or calibration drift'
                : 'Check for equipment efficiency issues or control problems',
            potentialImpact: {
              energy:
                stats.trend === 'increasing'
                  ? 'Potential energy waste from degrading performance'
                  : 'May indicate underperformance',
              maintenance: 'Trending behavior may indicate need for preventive maintenance',
            },
            confidence: trendStrength,
            metadata: {
              algorithm: 'trend_analysis',
              parameters: { trendDirection: stats.trend, trendStrength },
              thresholds: { minTrendStrength: 0.7 },
            },
          });
        }
      }
    }

    return faults;
  }

  /**
   * Cross-point correlation analysis to detect deviating equipment
   */
  private async detectCorrelationFaults(
    data: TimeSeriesData[],
    context: AnalysisContext,
    config: CorrelationAnalysisConfig
  ): Promise<FaultDetectionResult[]> {
    const faults: FaultDetectionResult[] = [];

    // Group similar points (by unit or naming pattern)
    const groupedPoints = this.groupSimilarPoints(data);

    for (const [group, points] of groupedPoints.entries()) {
      if (points.length < config.minimumPeerCount) {continue;}

      const correlations = this.analyzeCorrelations(points, config);

      // Find points that don't correlate well with their peers
      for (const point of points) {
        const pointCorrelations = correlations.filter(
          (c) => c.pointA === point.pointName || c.pointB === point.pointName
        );

        const avgCorrelation =
          pointCorrelations.reduce((sum, c) => sum + Math.abs(c.coefficient), 0) /
          pointCorrelations.length;

        if (avgCorrelation < config.similarityThreshold) {
          faults.push({
            id: `correlation_${point.pointName}_${Date.now()}`,
            type: 'deviation_from_peers',
            severity: this.calculateSeverity(1 - avgCorrelation, 0.3, 0.5, 0.7),
            timestamp: context.timeRange.end.getTime(),
            detectedAt: new Date(),
            affectedPoints: [point.pointName],
            description: `${point.pointName} shows poor correlation with peer points (avg correlation: ${(avgCorrelation * 100).toFixed(1)}%)`,
            recommendedAction:
              'Check sensor calibration, wiring, and local conditions. Compare with peer equipment.',
            potentialImpact: {
              comfort: 'Deviating performance may indicate local comfort issues',
              maintenance: 'May indicate equipment requiring attention or calibration',
            },
            confidence: 1 - avgCorrelation,
            metadata: {
              algorithm: 'peer_correlation_analysis',
              parameters: { groupName: group, peerCount: points.length },
              thresholds: { minCorrelation: config.similarityThreshold, avgCorrelation },
            },
          });
        }
      }
    }

    return faults;
  }

  /**
   * Weather-based fault detection
   */
  private async detectWeatherBasedFaults(
    data: TimeSeriesData[],
    context: AnalysisContext,
    config: any
  ): Promise<FaultDetectionResult[]> {
    const faults: FaultDetectionResult[] = [];

    if (!context.weatherData?.outdoorTemp) {return faults;}

    const outdoorTemp = context.weatherData.outdoorTemp;

    // Find periods of mild weather
    const mildWeatherPeriods = this.findMildWeatherPeriods(outdoorTemp, config.mildWeatherRange);

    // Analyze energy consumption during mild weather
    const energyPoints = data.filter(
      (series) =>
        series.pointName.toLowerCase().includes('power') ||
        series.pointName.toLowerCase().includes('energy') ||
        series.pointName.toLowerCase().includes('kw')
    );

    for (const energyPoint of energyPoints) {
      for (const period of mildWeatherPeriods) {
        const periodData = this.getDataForPeriod(energyPoint.data, period.start, period.end);
        if (periodData.length === 0) {continue;}

        const avgConsumption =
          periodData.reduce((sum, dp) => sum + dp.value, 0) / periodData.length;
        const baseline = this.calculateEnergyBaseline(energyPoint.data);

        if (avgConsumption > baseline * config.energyBaselineMultiplier) {
          faults.push({
            id: `excessive_energy_${energyPoint.pointName}_${period.start}`,
            type: 'excessive_energy_use',
            severity: this.calculateSeverity(avgConsumption / baseline, 1.5, 2.0, 3.0),
            timestamp: period.end,
            detectedAt: new Date(),
            affectedPoints: [energyPoint.pointName],
            description: `Excessive energy use during mild weather: ${avgConsumption.toFixed(1)} ${energyPoint.unit || 'units'} (${((avgConsumption / baseline - 1) * 100).toFixed(1)}% above baseline)`,
            recommendedAction:
              'Review HVAC scheduling, setpoints, and economizer operation during mild weather conditions.',
            potentialImpact: {
              energy: `Estimated ${((avgConsumption - baseline) * period.durationHours).toFixed(1)} ${energyPoint.unit || 'units'} of excess consumption`,
            },
            confidence: Math.min(0.9, (avgConsumption / baseline - 1) * 2),
            metadata: {
              algorithm: 'weather_based_energy_analysis',
              parameters: {
                mildWeatherRange: config.mildWeatherRange,
                periodDuration: period.durationHours,
              },
              thresholds: { baseline, multiplier: config.energyBaselineMultiplier },
            },
          });
        }
      }
    }

    return faults;
  }

  /**
   * Analyze statistics for a time series
   */
  analyzeStatistics(
    data: TimeSeriesData,
    config?: StatisticalThresholds
  ): StatisticalAnalysisResult {
    const cfg = { ...this.config.statistical, ...config };
    const values = data.data.map((dp) => dp.value);

    const mean = values.reduce((sum, val) => sum + val, 0) / values.length;
    const variance = values.reduce((sum, val) => sum + Math.pow(val - mean, 2), 0) / values.length;
    const standardDeviation = Math.sqrt(variance);

    // Calculate moving average
    const movingAverage: number[] = [];
    for (let i = 0; i < values.length; i++) {
      const windowStart = Math.max(0, i - Math.floor(cfg.movingAverageWindow / 2));
      const windowEnd = Math.min(values.length, i + Math.floor(cfg.movingAverageWindow / 2) + 1);
      const windowValues = values.slice(windowStart, windowEnd);
      const windowMean = windowValues.reduce((sum, val) => sum + val, 0) / windowValues.length;
      movingAverage.push(windowMean);
    }

    // Identify outliers
    const outliers: DataPoint[] = [];
    const threshold = cfg.standardDeviationMultiplier * standardDeviation;

    data.data.forEach((dp, index) => {
      if (Math.abs(dp.value - mean) > threshold) {
        outliers.push(dp);
      }
    });

    // Calculate trend
    const trend = this.calculateTrend(values);

    return {
      mean,
      standardDeviation,
      variance,
      movingAverage,
      outliers,
      trend: trend > 0.1 ? 'increasing' : trend < -0.1 ? 'decreasing' : 'stable',
      seasonality: {
        detected: false, // Simplified for now
      },
    };
  }

  /**
   * Analyze correlations between multiple time series
   */
  analyzeCorrelations(
    datasets: TimeSeriesData[],
    config?: CorrelationAnalysisConfig
  ): CorrelationResult[] {
    const cfg = { ...this.config.correlation, ...config };
    const correlations: CorrelationResult[] = [];

    for (let i = 0; i < datasets.length; i++) {
      for (let j = i + 1; j < datasets.length; j++) {
        const correlation = this.calculateCorrelation(datasets[i].data, datasets[j].data);

        correlations.push({
          pointA: datasets[i].pointName,
          pointB: datasets[j].pointName,
          coefficient: correlation.coefficient,
          significance: correlation.significance,
          relationship: this.categorizeCorrelation(Math.abs(correlation.coefficient)),
        });
      }
    }

    return correlations;
  }

  // Helper methods

  private calculateSeverity(
    value: number,
    lowThreshold: number,
    mediumThreshold: number,
    highThreshold: number
  ): FaultSeverity {
    if (value >= highThreshold) {return 'critical';}
    if (value >= mediumThreshold) {return 'high';}
    if (value >= lowThreshold) {return 'medium';}
    return 'low';
  }

  private calculateTrendStrength(data: DataPoint[]): number {
    if (data.length < 2) {return 0;}

    // Simple linear regression to determine trend strength
    const n = data.length;
    const x = data.map((_, index) => index);
    const y = data.map((dp) => dp.value);

    const sumX = x.reduce((sum, val) => sum + val, 0);
    const sumY = y.reduce((sum, val) => sum + val, 0);
    const sumXY = x.reduce((sum, val, index) => sum + val * y[index], 0);
    const sumXX = x.reduce((sum, val) => sum + val * val, 0);

    const slope = (n * sumXY - sumX * sumY) / (n * sumXX - sumX * sumX);

    // Normalize slope by data range
    const yRange = Math.max(...y) - Math.min(...y);
    return Math.abs((slope * n) / yRange);
  }

  private calculateTrend(values: number[]): number {
    if (values.length < 2) {return 0;}

    const n = values.length;
    const x = Array.from({ length: n }, (_, i) => i);
    const sumX = x.reduce((sum, val) => sum + val, 0);
    const sumY = values.reduce((sum, val) => sum + val, 0);
    const sumXY = x.reduce((sum, val, index) => sum + val * values[index], 0);
    const sumXX = x.reduce((sum, val) => sum + val * val, 0);

    return (n * sumXY - sumX * sumY) / (n * sumXX - sumX * sumX);
  }

  private groupSimilarPoints(data: TimeSeriesData[]): Map<string, TimeSeriesData[]> {
    const groups = new Map<string, TimeSeriesData[]>();

    data.forEach((series) => {
      // Group by unit or common naming patterns
      let groupKey = series.unit || 'unknown';

      // Additional grouping by point name patterns
      if (series.pointName.toLowerCase().includes('temp')) {
        groupKey += '_temperature';
      } else if (series.pointName.toLowerCase().includes('pressure')) {
        groupKey += '_pressure';
      } else if (series.pointName.toLowerCase().includes('flow')) {
        groupKey += '_flow';
      }

      if (!groups.has(groupKey)) {
        groups.set(groupKey, []);
      }
      groups.get(groupKey)!.push(series);
    });

    return groups;
  }

  private calculateCorrelation(
    dataA: DataPoint[],
    dataB: DataPoint[]
  ): { coefficient: number; significance: number } {
    // Align data points by timestamp
    const alignedData = this.alignTimeSeriesData(dataA, dataB);
    if (alignedData.length < 3) {return { coefficient: 0, significance: 0 };}

    const valuesA = alignedData.map((pair) => pair[0]);
    const valuesB = alignedData.map((pair) => pair[1]);

    const meanA = valuesA.reduce((sum, val) => sum + val, 0) / valuesA.length;
    const meanB = valuesB.reduce((sum, val) => sum + val, 0) / valuesB.length;

    let numerator = 0;
    let denominatorA = 0;
    let denominatorB = 0;

    for (let i = 0; i < valuesA.length; i++) {
      const diffA = valuesA[i] - meanA;
      const diffB = valuesB[i] - meanB;

      numerator += diffA * diffB;
      denominatorA += diffA * diffA;
      denominatorB += diffB * diffB;
    }

    const coefficient = numerator / Math.sqrt(denominatorA * denominatorB) || 0;
    const significance =
      (Math.abs(coefficient) * Math.sqrt(alignedData.length - 2)) /
        Math.sqrt(1 - coefficient * coefficient) || 0;

    return { coefficient, significance };
  }

  private alignTimeSeriesData(dataA: DataPoint[], dataB: DataPoint[]): Array<[number, number]> {
    const aligned: Array<[number, number]> = [];
    let indexA = 0;
    let indexB = 0;

    while (indexA < dataA.length && indexB < dataB.length) {
      const timestampA = dataA[indexA].timestamp;
      const timestampB = dataB[indexB].timestamp;

      if (Math.abs(timestampA - timestampB) <= 60000) {
        // Within 1 minute
        aligned.push([dataA[indexA].value, dataB[indexB].value]);
        indexA++;
        indexB++;
      } else if (timestampA < timestampB) {
        indexA++;
      } else {
        indexB++;
      }
    }

    return aligned;
  }

  private categorizeCorrelation(coefficient: number): 'strong' | 'moderate' | 'weak' | 'none' {
    if (coefficient >= 0.7) {return 'strong';}
    if (coefficient >= 0.3) {return 'moderate';}
    if (coefficient >= 0.1) {return 'weak';}
    return 'none';
  }

  private findMildWeatherPeriods(
    outdoorTemp: TimeSeriesData,
    range: { minTemp: number; maxTemp: number }
  ) {
    const periods: Array<{ start: number; end: number; durationHours: number }> = [];
    let currentPeriodStart: number | null = null;

    for (const dataPoint of outdoorTemp.data) {
      const temp = dataPoint.value;

      if (temp >= range.minTemp && temp <= range.maxTemp) {
        if (currentPeriodStart === null) {
          currentPeriodStart = dataPoint.timestamp;
        }
      } else {
        if (currentPeriodStart !== null) {
          const durationHours = (dataPoint.timestamp - currentPeriodStart) / (1000 * 60 * 60);
          if (durationHours >= 2) {
            // At least 2 hours of mild weather
            periods.push({
              start: currentPeriodStart,
              end: dataPoint.timestamp,
              durationHours,
            });
          }
          currentPeriodStart = null;
        }
      }
    }

    return periods;
  }

  private getDataForPeriod(data: DataPoint[], start: number, end: number): DataPoint[] {
    return data.filter((dp) => dp.timestamp >= start && dp.timestamp <= end);
  }

  private calculateEnergyBaseline(data: DataPoint[]): number {
    // Simple baseline calculation - could be enhanced with more sophisticated methods
    const values = data.map((dp) => dp.value).sort((a, b) => a - b);
    const percentile25 = values[Math.floor(values.length * 0.25)];
    return percentile25 || 0;
  }

  private rankAndFilterFaults(faults: FaultDetectionResult[]): FaultDetectionResult[] {
    return faults
      .filter((fault) => fault.confidence > 0.3) // Filter low confidence faults
      .sort((a, b) => {
        // Sort by severity first, then by confidence
        const severityOrder = { critical: 4, high: 3, medium: 2, low: 1 };
        const severityDiff = severityOrder[b.severity] - severityOrder[a.severity];
        if (severityDiff !== 0) {return severityDiff;}
        return b.confidence - a.confidence;
      });
  }
