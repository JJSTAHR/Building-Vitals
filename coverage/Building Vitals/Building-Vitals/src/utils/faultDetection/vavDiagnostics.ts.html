/**
 * VAV-specific fault detection algorithms
 * Detects stuck dampers, hunting, sensor drift, and simultaneous heating/cooling
 */

import type {
  FaultDetectionResult,
  VAVZoneData,
  AnalysisContext,
  VAVDiagnosticConfig,
  DataPoint,
  FaultSeverity,
} from './types';

/**
 * Detect VAV-specific faults across multiple zones
 */
export async function detectVAVFaults(
  vavData: VAVZoneData[],
  context: AnalysisContext,
  config: VAVDiagnosticConfig
): Promise<FaultDetectionResult[]> {
  const faults: FaultDetectionResult[] = [];

  console.log(`[VAVDiagnostics] Analyzing ${vavData.length} VAV zones`);

  for (const zone of vavData) {
    try {
      // Detect stuck damper
      const stuckDamperFaults = await detectStuckDamper(zone, context, config);
      faults.push(...stuckDamperFaults);

      // Detect hunting/oscillation
      const huntingFaults = await detectHunting(zone, context, config);
      faults.push(...huntingFaults);

      // Detect simultaneous heating and cooling
      const simultaneousFaults = await detectSimultaneousHeatingCooling(zone, context, config);
      faults.push(...simultaneousFaults);
    } catch (error) {
      console.error(`[VAVDiagnostics] Error analyzing zone ${zone.zoneName}:`, error);
    }
  }

  // Detect sensor drift across all zones
  const sensorDriftFaults = await detectSensorDrift(vavData, context, config);
  faults.push(...sensorDriftFaults);

  console.log(`[VAVDiagnostics] Detected ${faults.length} VAV-related faults`);
  return faults;
}

/**
 * Detect stuck damper - damper position shows minimal variation over extended period
 */
async function detectStuckDamper(
  zone: VAVZoneData,
  context: AnalysisContext,
  config: VAVDiagnosticConfig
): Promise<FaultDetectionResult[]> {
  const faults: FaultDetectionResult[] = [];

  if (!zone.damperPosition || zone.damperPosition.data.length < 10) {
    return faults;
  }

  const damperData = zone.damperPosition.data;
  const timeSpanHours =
    (damperData[damperData.length - 1].timestamp - damperData[0].timestamp) / (1000 * 60 * 60);

  if (timeSpanHours < config.stuckDamperThreshold.minimumTimeHours) {
    return faults;
  }

  // Calculate variance in damper position
  const positions = damperData.map((dp) => dp.value);
  const mean = positions.reduce((sum, pos) => sum + pos, 0) / positions.length;
  const variance =
    positions.reduce((sum, pos) => sum + Math.pow(pos - mean, 2), 0) / positions.length;
  const standardDeviation = Math.sqrt(variance);
  const coefficientOfVariation = (standardDeviation / mean) * 100;

  if (coefficientOfVariation < config.stuckDamperThreshold.maxVariancePercent) {
    // Check if zone temperature is deviating from setpoint
    const tempDeviation = calculateTemperatureSetpointDeviation(zone);

    const severity = calculateStuckDamperSeverity(coefficientOfVariation, tempDeviation, mean);

    faults.push({
      id: `stuck_damper_${zone.zoneName}_${Date.now()}`,
      type: 'stuck_damper',
      severity,
      timestamp: context.timeRange.end.getTime(),
      detectedAt: new Date(),
      affectedPoints: [zone.damperPosition.pointName],
      description: `Stuck damper detected in ${zone.zoneName}: Position variance is ${coefficientOfVariation.toFixed(2)}% (mean position: ${mean.toFixed(1)}%)`,
      recommendedAction:
        'Inspect damper actuator, linkage, and mechanical components. Check for obstructions or actuator failure.',
      potentialImpact: {
        comfort:
          tempDeviation > 2
            ? `Zone temperature deviating ${tempDeviation.toFixed(1)}F from setpoint`
            : 'Potential comfort issues if damper cannot modulate',
        energy:
          'Fixed damper position prevents optimal airflow control, reducing energy efficiency',
        maintenance: 'Immediate inspection required to prevent equipment damage',
      },
      confidence: Math.min(
        0.95,
        (config.stuckDamperThreshold.maxVariancePercent - coefficientOfVariation) /
          config.stuckDamperThreshold.maxVariancePercent +
          0.3
      ),
      metadata: {
        algorithm: 'stuck_damper_detection',
        parameters: {
          timeSpanHours,
          meanPosition: mean,
          variance,
          standardDeviation,
          coefficientOfVariation,
        },
        thresholds: {
          maxVariancePercent: config.stuckDamperThreshold.maxVariancePercent,
          minimumTimeHours: config.stuckDamperThreshold.minimumTimeHours,
        },
      },
    });
  }

  return faults;
}

/**
 * Detect hunting/oscillation - rapid oscillations in damper position or zone temperature
 */
async function detectHunting(
  zone: VAVZoneData,
  context: AnalysisContext,
  config: VAVDiagnosticConfig
): Promise<FaultDetectionResult[]> {
  const faults: FaultDetectionResult[] = [];

  // Check damper position for oscillations
  if (zone.damperPosition && zone.damperPosition.data.length > 20) {
    const damperOscillations = detectOscillations(
      zone.damperPosition.data,
      config.huntingThreshold
    );

    if (damperOscillations.detected) {
      faults.push({
        id: `hunting_damper_${zone.zoneName}_${Date.now()}`,
        type: 'hunting_oscillation',
        severity: calculateHuntingSeverity(
          damperOscillations.frequency,
          damperOscillations.amplitude
        ),
        timestamp: context.timeRange.end.getTime(),
        detectedAt: new Date(),
        affectedPoints: [zone.damperPosition.pointName],
        description: `Damper hunting detected in ${zone.zoneName}: ${damperOscillations.frequency.toFixed(1)} oscillations/hour with ${damperOscillations.amplitude.toFixed(1)}% amplitude`,
        recommendedAction:
          'Tune PID control parameters. Check for sensor noise, control loop instability, or inappropriate control gains.',
        potentialImpact: {
          comfort: 'Oscillating damper position causes temperature swings and occupant discomfort',
          energy: 'Hunting increases wear on actuators and reduces energy efficiency',
          maintenance: 'Excessive actuator cycling shortens equipment lifespan',
        },
        confidence: Math.min(0.9, damperOscillations.frequency / 10),
        metadata: {
          algorithm: 'hunting_oscillation_detection',
          parameters: {
            frequency: damperOscillations.frequency,
            amplitude: damperOscillations.amplitude,
            duration: damperOscillations.duration,
          },
          thresholds: config.huntingThreshold,
        },
      });
    }
  }

  // Check zone temperature for oscillations
  if (zone.zoneTemp && zone.zoneTemp.data.length > 20) {
    const tempOscillations = detectOscillations(zone.zoneTemp.data, config.huntingThreshold);

    if (tempOscillations.detected) {
      faults.push({
        id: `hunting_temp_${zone.zoneName}_${Date.now()}`,
        type: 'hunting_oscillation',
        severity: calculateHuntingSeverity(tempOscillations.frequency, tempOscillations.amplitude),
        timestamp: context.timeRange.end.getTime(),
        detectedAt: new Date(),
        affectedPoints: [zone.zoneTemp.pointName],
        description: `Temperature hunting detected in ${zone.zoneName}: ${tempOscillations.frequency.toFixed(1)} oscillations/hour with ${tempOscillations.amplitude.toFixed(1)}F amplitude`,
        recommendedAction:
          'Tune control parameters and check damper response time. Verify sensor calibration and control loop stability.',
        potentialImpact: {
          comfort: 'Temperature oscillations cause significant occupant discomfort',
          energy: 'Temperature hunting indicates poor control efficiency',
        },
        confidence: Math.min(0.9, tempOscillations.frequency / 8),
        metadata: {
          algorithm: 'temperature_hunting_detection',
          parameters: {
            frequency: tempOscillations.frequency,
            amplitude: tempOscillations.amplitude,
            duration: tempOscillations.duration,
          },
          thresholds: config.huntingThreshold,
        },
      });
    }
  }

  return faults;
}

/**
 * Detect simultaneous heating and cooling
 */
async function detectSimultaneousHeatingCooling(
  zone: VAVZoneData,
  context: AnalysisContext,
  config: VAVDiagnosticConfig
): Promise<FaultDetectionResult[]> {
  const faults: FaultDetectionResult[] = [];

  if (!zone.heatingOutput || !zone.coolingOutput) {
    return faults;
  }

  const heatingData = zone.heatingOutput.data;
  const coolingData = zone.coolingOutput.data;

  // Align data points by timestamp
  const alignedData = alignTimeSeriesData(heatingData, coolingData);

  let simultaneousCount = 0;
  let maxSimultaneousValue = 0;
  const simultaneousPeriods: Array<{
    start: number;
    end: number;
    avgHeating: number;
    avgCooling: number;
  }> = [];

  for (let i = 0; i < alignedData.length; i++) {
    const [heating, cooling, timestamp] = alignedData[i];

    if (
      heating > config.simultaneousHeatingCooling.heatingThreshold &&
      cooling > config.simultaneousHeatingCooling.coolingThreshold
    ) {
      simultaneousCount++;
      maxSimultaneousValue = Math.max(maxSimultaneousValue, Math.min(heating, cooling));

      // Track periods of simultaneous operation
      if (simultaneousCount === 1) {
        simultaneousPeriods.push({
          start: timestamp,
          end: timestamp,
          avgHeating: heating,
          avgCooling: cooling,
        });
      } else if (simultaneousPeriods.length > 0) {
        const currentPeriod = simultaneousPeriods[simultaneousPeriods.length - 1];
        currentPeriod.end = timestamp;
        currentPeriod.avgHeating = (currentPeriod.avgHeating + heating) / 2;
        currentPeriod.avgCooling = (currentPeriod.avgCooling + cooling) / 2;
      }
    } else {
      simultaneousCount = 0;
    }
  }

  // Filter periods that meet minimum duration requirement
  const significantPeriods = simultaneousPeriods.filter(
    (period) =>
      period.end - period.start >=
      config.simultaneousHeatingCooling.simultaneousDuration * 60 * 1000
  );

  if (significantPeriods.length > 0) {
    const totalSimultaneousTime =
      significantPeriods.reduce((sum, period) => sum + (period.end - period.start), 0) /
      (1000 * 60); // Convert to minutes

    faults.push({
      id: `simultaneous_heating_cooling_${zone.zoneName}_${Date.now()}`,
      type: 'simultaneous_heating_cooling',
      severity: calculateSimultaneousSeverity(totalSimultaneousTime, maxSimultaneousValue),
      timestamp: context.timeRange.end.getTime(),
      detectedAt: new Date(),
      affectedPoints: [zone.heatingOutput.pointName, zone.coolingOutput.pointName],
      description: `Simultaneous heating and cooling detected in ${zone.zoneName}: ${significantPeriods.length} periods totaling ${totalSimultaneousTime.toFixed(1)} minutes`,
      recommendedAction:
        'Check control sequence logic, review heating and cooling lockout settings, and verify control system programming.',
      potentialImpact: {
        energy: `Estimated ${((maxSimultaneousValue * totalSimultaneousTime) / 60).toFixed(1)} units of wasted energy`,
        maintenance: 'Simultaneous operation increases wear on both heating and cooling equipment',
      },
      confidence: Math.min(
        0.95,
        significantPeriods.length * 0.2 + (totalSimultaneousTime / 60) * 0.1
      ),
      metadata: {
        algorithm: 'simultaneous_heating_cooling_detection',
        parameters: {
          periodsCount: significantPeriods.length,
          totalTimeMinutes: totalSimultaneousTime,
          maxSimultaneousValue,
          periods: significantPeriods.map((p) => ({
            durationMinutes: (p.end - p.start) / (1000 * 60),
            avgHeating: p.avgHeating,
            avgCooling: p.avgCooling,
          })),
        },
        thresholds: config.simultaneousHeatingCooling,
      },
    });
  }

  return faults;
}

/**
 * Detect sensor drift by comparing zone temperatures across similar zones
 */
async function detectSensorDrift(
  vavData: VAVZoneData[],
  context: AnalysisContext,
  config: VAVDiagnosticConfig
): Promise<FaultDetectionResult[]> {
  const faults: FaultDetectionResult[] = [];

  if (vavData.length < config.sensorDriftThreshold.minimumSampleSize) {
    return faults;
  }

  // Calculate average temperature for each zone
  const zoneAverages = vavData
    .map((zone) => {
      if (!zone.zoneTemp || zone.zoneTemp.data.length === 0) {return null;}

      const avgTemp =
        zone.zoneTemp.data.reduce((sum, dp) => sum + dp.value, 0) / zone.zoneTemp.data.length;
      return {
        zoneName: zone.zoneName,
        pointName: zone.zoneTemp.pointName,
        avgTemp,
        data: zone.zoneTemp.data,
      };
    })
    .filter((zone) => zone !== null);

  if (zoneAverages.length < config.sensorDriftThreshold.minimumSampleSize) {
    return faults;
  }

  // Calculate overall average and standard deviation
  const overallAvg =
    zoneAverages.reduce((sum, zone) => sum + zone.avgTemp, 0) / zoneAverages.length;
  const variance =
    zoneAverages.reduce((sum, zone) => sum + Math.pow(zone.avgTemp - overallAvg, 2), 0) /
    zoneAverages.length;
  const standardDeviation = Math.sqrt(variance);

  // Identify zones that deviate significantly from peers
  for (const zone of zoneAverages) {
    if (!zone) {continue;}

    const deviationFromMean = Math.abs(zone.avgTemp - overallAvg);
    const standardizedDeviation = deviationFromMean / standardDeviation;

    if (standardizedDeviation > config.sensorDriftThreshold.maxDeviationFromPeers) {
      // Verify consistency over time
      const isConsistent = checkConsistentDeviation(
        zone.data,
        zoneAverages.filter((z) => z && z.zoneName !== zone.zoneName),
        config.sensorDriftThreshold.consistencyPeriod
      );

      if (isConsistent) {
        faults.push({
          id: `sensor_drift_${zone.zoneName}_${Date.now()}`,
          type: 'sensor_drift',
          severity: calculateSensorDriftSeverity(standardizedDeviation),
          timestamp: context.timeRange.end.getTime(),
          detectedAt: new Date(),
          affectedPoints: [zone.pointName],
          description: `Temperature sensor drift detected in ${zone.zoneName}: ${deviationFromMean.toFixed(1)}F deviation from peer average (${standardizedDeviation.toFixed(1)} std deviations)`,
          recommendedAction:
            'Calibrate temperature sensor or replace if drift is excessive. Compare readings with calibrated reference instrument.',
          potentialImpact: {
            comfort: 'Inaccurate temperature readings affect zone comfort control',
            energy:
              'Sensor drift can lead to over-heating or over-cooling, increasing energy consumption',
            maintenance: 'Sensor requires calibration or replacement',
          },
          confidence: Math.min(
            0.95,
            (standardizedDeviation - config.sensorDriftThreshold.maxDeviationFromPeers) * 0.2 + 0.5
          ),
          metadata: {
            algorithm: 'peer_comparison_sensor_drift',
            parameters: {
              avgTemp: zone.avgTemp,
              overallAvg,
              deviationFromMean,
              standardizedDeviation,
              peerCount: zoneAverages.length - 1,
            },
            thresholds: config.sensorDriftThreshold,
          },
        });
      }
    }
  }

  return faults;
}

// Helper functions

function calculateTemperatureSetpointDeviation(zone: VAVZoneData): number {
  if (
    !zone.zoneTemp ||
    !zone.zoneSetpoint ||
    zone.zoneTemp.data.length === 0 ||
    zone.zoneSetpoint.data.length === 0
  ) {
    return 0;
  }

  const alignedData = alignTimeSeriesData(zone.zoneTemp.data, zone.zoneSetpoint.data);
  if (alignedData.length === 0) {return 0;}

  const deviations = alignedData.map(([temp, setpoint]) => Math.abs(temp - setpoint));
  return deviations.reduce((sum, dev) => sum + dev, 0) / deviations.length;
}

function calculateStuckDamperSeverity(
  variance: number,
  tempDeviation: number,
  position: number
): FaultSeverity {
  let score = 0;

  // Lower variance = higher severity
  if (variance < 1) {score += 3;}
  else if (variance < 2) {score += 2;}
  else if (variance < 3) {score += 1;}

  // Higher temperature deviation = higher severity
  if (tempDeviation > 5) {score += 2;}
  else if (tempDeviation > 3) {score += 1;}

  // Extreme positions (fully open/closed) are more concerning
  if (position < 10 || position > 90) {score += 1;}

  if (score >= 5) {return 'critical';}
  if (score >= 3) {return 'high';}
  if (score >= 2) {return 'medium';}
  return 'low';
}

function calculateHuntingSeverity(frequency: number, amplitude: number): FaultSeverity {
  let score = 0;

  // Higher frequency = higher severity
  if (frequency > 12) {score += 3;}
  else if (frequency > 8) {score += 2;}
  else if (frequency > 6) {score += 1;}

  // Higher amplitude = higher severity
  if (amplitude > 30) {score += 2;}
  else if (amplitude > 20) {score += 1;}

  if (score >= 4) {return 'critical';}
  if (score >= 3) {return 'high';}
  if (score >= 2) {return 'medium';}
  return 'low';
}

function calculateSimultaneousSeverity(totalTime: number, maxValue: number): FaultSeverity {
  let score = 0;

  // Longer duration = higher severity
  if (totalTime > 120)
    {score += 3;} // 2 hours
  else if (totalTime > 60)
    {score += 2;} // 1 hour
  else if (totalTime > 30) {score += 1;} // 30 minutes

  // Higher simultaneous values = higher severity
  if (maxValue > 50) {score += 2;}
  else if (maxValue > 25) {score += 1;}

  if (score >= 4) {return 'critical';}
  if (score >= 3) {return 'high';}
  if (score >= 2) {return 'medium';}
  return 'low';
}

function calculateSensorDriftSeverity(standardizedDeviation: number): FaultSeverity {
  if (standardizedDeviation > 6) {return 'critical';}
  if (standardizedDeviation > 4) {return 'high';}
  if (standardizedDeviation > 3) {return 'medium';}
  return 'low';
}

function detectOscillations(
  data: DataPoint[],
  config: any
): { detected: boolean; frequency: number; amplitude: number; duration: number } {
  if (data.length < 10) {return { detected: false, frequency: 0, amplitude: 0, duration: 0 };}

  const values = data.map((dp) => dp.value);
  const timeSpan = (data[data.length - 1].timestamp - data[0].timestamp) / (1000 * 60 * 60); // hours

  // Simple peak detection
  let peaks = 0;
  let valleys = 0;
  const amplitudes: number[] = [];

  for (let i = 1; i < values.length - 1; i++) {
    if (values[i] > values[i - 1] && values[i] > values[i + 1]) {
      peaks++;
    } else if (values[i] < values[i - 1] && values[i] < values[i + 1]) {
      valleys++;
    }
  }

  const oscillations = Math.min(peaks, valleys);
  const frequency = oscillations / timeSpan;

  // Calculate amplitude (range of oscillation)
  const maxValue = Math.max(...values);
  const minValue = Math.min(...values);
  const amplitude = maxValue - minValue;

  const detected = frequency > config.oscillationFrequency && amplitude > config.amplitudeThreshold;

  return {
    detected,
    frequency,
    amplitude,
    duration: timeSpan * 60, // minutes
  };
}

function alignTimeSeriesData(
  dataA: DataPoint[],
  dataB: DataPoint[]
): Array<[number, number, number]> {
  const aligned: Array<[number, number, number]> = [];
  let indexA = 0;
  let indexB = 0;

  while (indexA < dataA.length && indexB < dataB.length) {
    const timestampA = dataA[indexA].timestamp;
    const timestampB = dataB[indexB].timestamp;

    if (Math.abs(timestampA - timestampB) <= 60000) {
      // Within 1 minute
      aligned.push([dataA[indexA].value, dataB[indexB].value, timestampA]);
      indexA++;
      indexB++;
    } else if (timestampA < timestampB) {
      indexA++;
    } else {
      indexB++;
    }
  }

  return aligned;
}

function checkConsistentDeviation(
  zoneData: DataPoint[],
  peerZones: Array<{ data: DataPoint[] }>,
  consistencyPeriodHours: number
): boolean {
  // Simplified consistency check - could be enhanced with more sophisticated analysis
  const timeWindow = consistencyPeriodHours * 60 * 60 * 1000; // Convert to milliseconds
  const endTime = Math.max(...zoneData.map((dp) => dp.timestamp));
  const startTime = endTime - timeWindow;

  const recentZoneData = zoneData.filter((dp) => dp.timestamp >= startTime);
  if (recentZoneData.length < 5) {return false;}

  const zoneAvg = recentZoneData.reduce((sum, dp) => sum + dp.value, 0) / recentZoneData.length;

  // Calculate peer average for the same time period
  let peerSum = 0;
  let peerCount = 0;

  for (const peer of peerZones) {
    const recentPeerData = peer.data.filter((dp) => dp.timestamp >= startTime);
    if (recentPeerData.length > 0) {
      peerSum += recentPeerData.reduce((sum, dp) => sum + dp.value, 0) / recentPeerData.length;
      peerCount++;
    }
  }

  if (peerCount === 0) {return false;}

  const peerAvg = peerSum / peerCount;
  const deviation = Math.abs(zoneAvg - peerAvg);

  // Consider it consistent if deviation is significant (>2F)
  return deviation > 2;
}
