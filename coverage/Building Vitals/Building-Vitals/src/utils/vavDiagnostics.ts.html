/**
 * VAV Diagnostic Analysis Utilities
 * Provides mathematical analysis functions for diagnosing VAV system issues
 */

export interface DiagnosticResult {
  status: 'good' | 'warning' | 'critical';
  value: number;
  message: string;
  details?: Record<string, any>;
  recommendations?: string[];
}

/**
 * Detects oscillation/hunting patterns in timeseries data
 * Uses zero-crossing detection and frequency analysis
 */
export function detectOscillation(
  data: Array<[number, number]>,
  pointName?: string
): DiagnosticResult {
  if (!data || data.length < 10) {
    return {
      status: 'warning',
      value: 0,
      message: 'Insufficient data for oscillation analysis',
    };
  }

  // Calculate mean value
  const values = data.map(d => d[1]);
  const mean = values.reduce((sum, val) => sum + val, 0) / values.length;
  
  // Count zero crossings (crossing the mean)
  let zeroCrossings = 0;
  let lastSign = values[0] > mean ? 1 : -1;
  
  for (let i = 1; i < values.length; i++) {
    const currentSign = values[i] > mean ? 1 : -1;
    if (currentSign !== lastSign) {
      zeroCrossings++;
      lastSign = currentSign;
    }
  }
  
  // Calculate oscillation frequency (crossings per hour)
  const timeSpanMs = data[data.length - 1][0] - data[0][0];
  const timeSpanHours = timeSpanMs / (1000 * 60 * 60);
  const crossingsPerHour = zeroCrossings / timeSpanHours;
  
  // Calculate amplitude (standard deviation)
  const variance = values.reduce((sum, val) => sum + Math.pow(val - mean, 2), 0) / values.length;
  const stdDev = Math.sqrt(variance);
  const amplitude = (stdDev / mean) * 100; // As percentage of mean
  
  // Determine status based on frequency and amplitude
  let status: 'good' | 'warning' | 'critical';
  let message: string;
  let recommendations: string[] = [];
  
  if (crossingsPerHour > 60) {
    status = 'critical';
    message = `Severe hunting detected: ${Math.round(crossingsPerHour)} oscillations/hour`;
    recommendations = [
      'Check PID tuning parameters',
      'Verify damper/valve actuator linkage',
      'Check for oversized equipment',
      'Inspect control loop stability'
    ];
  } else if (crossingsPerHour > 20) {
    status = 'warning';
    message = `Moderate hunting: ${Math.round(crossingsPerHour)} oscillations/hour`;
    recommendations = [
      'Review control loop tuning',
      'Check for mechanical binding',
      'Verify sensor calibration'
    ];
  } else {
    status = 'good';
    message = `Stable control: ${Math.round(crossingsPerHour)} changes/hour`;
  }
  
  return {
    status,
    value: crossingsPerHour,
    message,
    details: {
      frequency: `${crossingsPerHour.toFixed(1)} cycles/hr`,
      amplitude: `${amplitude.toFixed(1)}%`,
      zeroCrossings,
      mean: mean.toFixed(2),
      stdDev: stdDev.toFixed(2)
    },
    recommendations
  };
}

/**
 * Analyzes response between command and feedback signals
 * Detects dead zones, hysteresis, and poor response
 */
export function analyzeResponse(
  command: Array<[number, number]>,
  feedback: Array<[number, number]>
): DiagnosticResult {
  if (!command || !feedback || command.length < 5 || feedback.length < 5) {
    return {
      status: 'warning',
      value: 0,
      message: 'Insufficient data for response analysis',
    };
  }
  
  // Align timestamps and pair command/feedback values
  const pairs: Array<{cmd: number; fb: number}> = [];
  let feedbackIndex = 0;
  
  for (let i = 0; i < command.length; i++) {
    const cmdTime = command[i][0];
    const cmdValue = command[i][1];
    
    // Find closest feedback reading
    while (feedbackIndex < feedback.length - 1 && 
           Math.abs(feedback[feedbackIndex + 1][0] - cmdTime) < 
           Math.abs(feedback[feedbackIndex][0] - cmdTime)) {
      feedbackIndex++;
    }
    
    if (feedbackIndex < feedback.length) {
      pairs.push({
        cmd: cmdValue,
        fb: feedback[feedbackIndex][1]
      });
    }
  }
  
  if (pairs.length < 5) {
    return {
      status: 'warning',
      value: 0,
      message: 'Unable to align command and feedback data',
    };
  }
  
  // Analyze response characteristics
  
  // 1. Dead zone detection (commands that don't produce response)
  let deadZoneStart = 0;
  let deadZoneEnd = 0;
  const sortedPairs = [...pairs].sort((a, b) => a.cmd - b.cmd);
  
  for (let i = 0; i < sortedPairs.length - 1; i++) {
    const cmdDiff = Math.abs(sortedPairs[i + 1].cmd - sortedPairs[i].cmd);
    const fbDiff = Math.abs(sortedPairs[i + 1].fb - sortedPairs[i].fb);
    
    if (cmdDiff > 1 && fbDiff < 0.5) {
      if (sortedPairs[i].cmd < 20) {
        deadZoneEnd = Math.max(deadZoneEnd, sortedPairs[i].cmd);
      }
      if (sortedPairs[i].cmd > 80) {
        deadZoneStart = Math.min(deadZoneStart || 100, sortedPairs[i].cmd);
      }
    }
  }
  
  // 2. Hysteresis calculation (difference between opening and closing curves)
  let maxHysteresis = 0;
  const cmdRange = Math.max(...pairs.map(p => p.cmd)) - Math.min(...pairs.map(p => p.cmd));
  
  for (let cmdLevel = 20; cmdLevel <= 80; cmdLevel += 10) {
    const nearPairs = pairs.filter(p => Math.abs(p.cmd - cmdLevel) < 5);
    if (nearPairs.length > 1) {
      const fbValues = nearPairs.map(p => p.fb);
      const fbRange = Math.max(...fbValues) - Math.min(...fbValues);
      maxHysteresis = Math.max(maxHysteresis, fbRange);
    }
  }
  
  const hysteresisPercent = cmdRange > 0 ? (maxHysteresis / cmdRange) * 100 : 0;
  
  // 3. Linearity (correlation coefficient)
  const cmdMean = pairs.reduce((sum, p) => sum + p.cmd, 0) / pairs.length;
  const fbMean = pairs.reduce((sum, p) => sum + p.fb, 0) / pairs.length;
  
  let numerator = 0;
  let cmdDenominator = 0;
  let fbDenominator = 0;
  
  for (const pair of pairs) {
    const cmdDiff = pair.cmd - cmdMean;
    const fbDiff = pair.fb - fbMean;
    numerator += cmdDiff * fbDiff;
    cmdDenominator += cmdDiff * cmdDiff;
    fbDenominator += fbDiff * fbDiff;
  }
  
  const correlation = cmdDenominator > 0 && fbDenominator > 0 
    ? numerator / (Math.sqrt(cmdDenominator) * Math.sqrt(fbDenominator))
    : 0;
  
  const linearityPercent = Math.abs(correlation) * 100;
  
  // 4. Response time (average delay)
  let totalDelay = 0;
  let delayCount = 0;
  
  for (let i = 1; i < command.length; i++) {
    const cmdChange = Math.abs(command[i][1] - command[i - 1][1]);
    if (cmdChange > 5) {
      // Find when feedback catches up
      const cmdTime = command[i][0];
      const targetFb = command[i][1];
      
      for (let j = 0; j < feedback.length; j++) {
        if (feedback[j][0] > cmdTime && 
            Math.abs(feedback[j][1] - targetFb) < 5) {
          totalDelay += feedback[j][0] - cmdTime;
          delayCount++;
          break;
        }
      }
    }
  }
  
  const avgResponseTimeMs = delayCount > 0 ? totalDelay / delayCount : 0;
  const avgResponseTimeSec = avgResponseTimeMs / 1000;
  
  // Determine overall status
  let status: 'good' | 'warning' | 'critical';
  let message: string;
  let recommendations: string[] = [];
  
  if (deadZoneEnd > 30 || hysteresisPercent > 20 || linearityPercent < 70) {
    status = 'critical';
    message = 'Poor actuator response detected';
    recommendations = [
      'Check actuator linkage and calibration',
      'Inspect for mechanical binding or damage',
      'Verify control signal wiring',
      'Consider actuator replacement'
    ];
  } else if (deadZoneEnd > 15 || hysteresisPercent > 10 || linearityPercent < 85) {
    status = 'warning';
    message = 'Suboptimal actuator response';
    recommendations = [
      'Calibrate actuator travel',
      'Check for worn linkage',
      'Verify proper actuator sizing'
    ];
  } else {
    status = 'good';
    message = 'Good actuator response';
  }
  
  return {
    status,
    value: linearityPercent,
    message,
    details: {
      deadZone: deadZoneEnd > 0 ? `0-${deadZoneEnd.toFixed(0)}%` : 'None',
      hysteresis: `${hysteresisPercent.toFixed(1)}%`,
      linearity: `${linearityPercent.toFixed(1)}%`,
      responseTime: `${avgResponseTimeSec.toFixed(1)} seconds`,
      correlation: correlation.toFixed(3)
    },
    recommendations
  };
}

/**
 * Analyzes thermal performance (heating/cooling efficiency)
 */
export function analyzeThermalPerformance(
  series: Array<{name: string; data: Array<[number, number]>; unit?: string}>
): DiagnosticResult {
  if (!series || series.length < 2) {
    return {
      status: 'warning',
      value: 0,
      message: 'Need at least 2 data series for thermal analysis',
      details: {
        required: 'Valve position and temperature data'
      }
    };
  }
  
  // Try to identify valve/damper and temperature series
  // Users select these, so we work with what we get
  const valveSeries = series[0];  // Assume first is valve/control signal
  const tempSeries = series[1];    // Assume second is temperature
  
  if (!valveSeries.data || !tempSeries.data || 
      valveSeries.data.length < 5 || tempSeries.data.length < 5) {
    return {
      status: 'warning',
      value: 0,
      message: 'Insufficient data for thermal analysis',
    };
  }
  
  // Calculate average valve position and temperature change
  const valveValues = valveSeries.data.map(d => d[1]);
  const tempValues = tempSeries.data.map(d => d[1]);
  
  const avgValvePos = valveValues.reduce((sum, v) => sum + v, 0) / valveValues.length;
  const tempRange = Math.max(...tempValues) - Math.min(...tempValues);
  const tempStdDev = Math.sqrt(
    tempValues.reduce((sum, t) => sum + Math.pow(t - tempValues[0], 2), 0) / tempValues.length
  );
  
  // Analyze heating/cooling capacity
  // When valve is open, how much does temperature change?
  const highValvePeriods = valveValues
    .map((v, i) => ({ valve: v, temp: tempValues[i] || 0, index: i }))
    .filter(p => p.valve > 70);
  
  const lowValvePeriods = valveValues
    .map((v, i) => ({ valve: v, temp: tempValues[i] || 0, index: i }))
    .filter(p => p.valve < 30);
  
  let capacityScore = 100;
  let message = 'Good thermal performance';
  let status: 'good' | 'warning' | 'critical' = 'good';
  let recommendations: string[] = [];
  
  if (highValvePeriods.length > 5 && lowValvePeriods.length > 5) {
    const highValveAvgTemp = highValvePeriods.reduce((sum, p) => sum + p.temp, 0) / highValvePeriods.length;
    const lowValveAvgTemp = lowValvePeriods.reduce((sum, p) => sum + p.temp, 0) / lowValvePeriods.length;
    const effectiveDelta = Math.abs(highValveAvgTemp - lowValveAvgTemp);
    
    // Calculate efficiency based on valve effort vs temperature achievement
    const valveEffort = Math.abs(avgValvePos - 50); // How hard is valve working
    const expectedDelta = valveEffort * 0.3; // Expected temp change per valve %
    capacityScore = Math.min(100, (effectiveDelta / Math.max(expectedDelta, 1)) * 100);
    
    if (capacityScore < 50) {
      status = 'critical';
      message = 'Poor thermal capacity detected';
      recommendations = [
        'Check for fouled coils',
        'Verify hot/chilled water supply temperatures',
        'Inspect for leaking valves',
        'Check for inadequate flow rates',
        'Verify coil sizing for load'
      ];
    } else if (capacityScore < 75) {
      status = 'warning';
      message = 'Reduced thermal capacity';
      recommendations = [
        'Clean coils if dirty',
        'Check water/air flow rates',
        'Verify setpoints are achievable',
        'Consider coil performance testing'
      ];
    } else {
      status = 'good';
      message = 'Good thermal capacity';
    }
  }
  
  // Check for valve working too hard (always near extremes)
  const extremeValvePercent = valveValues.filter(v => v < 10 || v > 90).length / valveValues.length * 100;
  
  if (extremeValvePercent > 50) {
    status = 'warning';
    message = 'Valve frequently at extreme positions';
    recommendations.push('Equipment may be undersized for load');
    recommendations.push('Check for excessive infiltration or heat loss/gain');
    capacityScore = Math.min(capacityScore, 70);
  }
  
  return {
    status,
    value: capacityScore,
    message,
    details: {
      efficiency: `${capacityScore.toFixed(1)}%`,
      avgValvePosition: `${avgValvePos.toFixed(1)}%`,
      temperatureRange: `${tempRange.toFixed(1)}°`,
      valveExtremes: `${extremeValvePercent.toFixed(1)}% of time`,
      temperatureStability: tempStdDev < 1 ? 'Stable' : tempStdDev < 3 ? 'Moderate' : 'Unstable'
    },
    recommendations
  };
}

/**
 * Detects stuck or unresponsive values
 */
export function detectStuckValues(
  data: Array<[number, number]>,
  pointName?: string
): DiagnosticResult {
  if (!data || data.length < 5) {
    return {
      status: 'warning',
      value: 0,
      message: 'Insufficient data for stuck value detection',
    };
  }
  
  const values = data.map(d => d[1]);
  const uniqueValues = new Set(values);
  const uniqueRatio = uniqueValues.size / values.length;
  
  // Check for flatlined data (no change)
  let maxFlatlineDuration = 0;
  let currentFlatlineStart = 0;
  let lastValue = values[0];
  
  for (let i = 1; i < values.length; i++) {
    if (Math.abs(values[i] - lastValue) < 0.01) {
      // Value hasn't changed
      if (currentFlatlineStart === 0) {
        currentFlatlineStart = i - 1;
      }
    } else {
      // Value changed
      if (currentFlatlineStart > 0) {
        const flatlineDuration = i - currentFlatlineStart;
        maxFlatlineDuration = Math.max(maxFlatlineDuration, flatlineDuration);
        currentFlatlineStart = 0;
      }
      lastValue = values[i];
    }
  }
  
  // Check if still flatlined at end
  if (currentFlatlineStart > 0) {
    const flatlineDuration = values.length - currentFlatlineStart;
    maxFlatlineDuration = Math.max(maxFlatlineDuration, flatlineDuration);
  }
  
  const flatlinePercent = (maxFlatlineDuration / values.length) * 100;
  
  // Determine status
  let status: 'good' | 'warning' | 'critical';
  let message: string;
  let recommendations: string[] = [];
  
  if (uniqueRatio < 0.1 || flatlinePercent > 80) {
    status = 'critical';
    message = `Sensor/actuator appears stuck`;
    recommendations = [
      'Check sensor wiring and connections',
      'Verify sensor power supply',
      'Test actuator movement manually',
      'Check for frozen/stuck mechanical components',
      'Verify control system communication'
    ];
  } else if (uniqueRatio < 0.3 || flatlinePercent > 50) {
    status = 'warning';
    message = 'Limited value changes detected';
    recommendations = [
      'Monitor for degrading sensor performance',
      'Check for intermittent connection issues',
      'Verify control logic is updating values'
    ];
  } else {
    status = 'good';
    message = 'Normal value variation';
  }
  
  return {
    status,
    value: uniqueRatio * 100,
    message,
    details: {
      uniqueValues: uniqueValues.size,
      totalReadings: values.length,
      variability: `${(uniqueRatio * 100).toFixed(1)}%`,
      longestFlatline: `${flatlinePercent.toFixed(1)}% of period`,
      lastValue: values[values.length - 1].toFixed(2)
    },
    recommendations
  };
}

/**
 * Run general health check combining all diagnostics
 */
export function runGeneralHealthCheck(
  series: Array<{name: string; data: Array<[number, number]>; unit?: string}>
): DiagnosticResult {
  if (!series || series.length === 0) {
    return {
      status: 'warning',
      value: 0,
      message: 'No data available for analysis',
    };
  }
  
  const results: DiagnosticResult[] = [];
  
  // Run oscillation detection on each series
  for (const s of series) {
    if (s.data && s.data.length > 10) {
      results.push(detectOscillation(s.data, s.name));
      results.push(detectStuckValues(s.data, s.name));
    }
  }
  
  // Run response analysis if we have 2+ series
  if (series.length >= 2) {
    results.push(analyzeResponse(series[0].data, series[1].data));
  }
  
  // Run thermal analysis if we have appropriate data
  if (series.length >= 2) {
    results.push(analyzeThermalPerformance(series));
  }
  
  // Aggregate results
  const criticalCount = results.filter(r => r.status === 'critical').length;
  const warningCount = results.filter(r => r.status === 'warning').length;
  const allRecommendations = results
    .filter(r => r.recommendations)
    .flatMap(r => r.recommendations || []);
  
  // Remove duplicate recommendations
  const uniqueRecommendations = [...new Set(allRecommendations)];
  
  let overallStatus: 'good' | 'warning' | 'critical';
  let overallMessage: string;
  
  if (criticalCount > 0) {
    overallStatus = 'critical';
    overallMessage = `${criticalCount} critical issue${criticalCount > 1 ? 's' : ''} detected`;
  } else if (warningCount > 0) {
    overallStatus = 'warning';
    overallMessage = `${warningCount} warning${warningCount > 1 ? 's' : ''} detected`;
  } else {
    overallStatus = 'good';
    overallMessage = 'System operating normally';
  }
  
  // Calculate overall health score
  const healthScore = Math.max(0, 100 - (criticalCount * 30) - (warningCount * 15));
  
  return {
    status: overallStatus,
    value: healthScore,
    message: overallMessage,
    details: {
      healthScore: `${healthScore}%`,
      criticalIssues: criticalCount,
      warnings: warningCount,
      checksPerformed: results.length,
      diagnostics: results.map(r => ({
        status: r.status,
        message: r.message
      }))
    },
    recommendations: uniqueRecommendations.slice(0, 5) // Top 5 recommendations
  };
