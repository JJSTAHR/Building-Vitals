/**
 * Chart Annotation Service
 * Enables adding annotations to charts using pixel coordinate conversion
 */

import type { ECharts } from 'echarts/core';

export interface ChartAnnotation {
  id: string;
  chartId: string;
  type: 'text' | 'line' | 'rect' | 'circle';
  dataPoint?: {
    seriesIndex: number;
    dataIndex: number;
    value: [number, number]; // [timestamp, value]
  };
  pixelPosition?: {
    x: number;
    y: number;
  };
  content: string;
  style?: {
    color?: string;
    fontSize?: number;
    backgroundColor?: string;
    borderColor?: string;
    borderWidth?: number;
  };
  createdAt: Date;
  createdBy?: string;
}

class ChartAnnotationService {
  private annotations: Map<string, ChartAnnotation[]> = new Map();
  private chartInstances: Map<string, ECharts> = new Map();

  /**
   * Register a chart instance for annotation support
   * @param chartId - Unique identifier for the chart
   * @param chart - ECharts instance
   */
  registerChart(chartId: string, chart: ECharts): void {
    this.chartInstances.set(chartId, chart);

    // Set up click handler for adding annotations
    chart.on('click', (params: any) => {
      if (params.componentType === 'series') {
        this.handleChartClick(chartId, params);
      }
    });
  }

  /**
   * Unregister a chart instance
   * @param chartId - Chart identifier
   */
  unregisterChart(chartId: string): void {
    const chart = this.chartInstances.get(chartId);
    if (chart) {
      chart.off('click');
      this.chartInstances.delete(chartId);
    }
  }

  /**
   * Convert data coordinates to pixel coordinates
   * @param chart - ECharts instance
   * @param value - Data value [x, y]
   * @returns Pixel coordinates {x, y}
   */
  dataToPixel(chart: ECharts, value: [number, number]): { x: number; y: number } | null {
    try {
      const pixel = chart.convertToPixel('grid', value);
      if (Array.isArray(pixel) && pixel.length === 2) {
        return { x: pixel[0], y: pixel[1] };
      }
    } catch (error) {}
    return null;
  }

  /**
   * Convert pixel coordinates to data coordinates
   * @param chart - ECharts instance
   * @param x - X pixel coordinate
   * @param y - Y pixel coordinate
   * @returns Data value [x, y]
   */
  pixelToData(chart: ECharts, x: number, y: number): [number, number] | null {
    try {
      const data = chart.convertFromPixel('grid', [x, y]);
      if (Array.isArray(data) && data.length === 2) {
        return data as [number, number];
      }
    } catch (error) {}
    return null;
  }

  /**
   * Add an annotation to a chart
   * @param annotation - Annotation object
   */
  addAnnotation(annotation: Omit<ChartAnnotation, 'id' | 'createdAt'>): ChartAnnotation {
    const fullAnnotation: ChartAnnotation = {
      ...annotation,
      id: `annotation-${Date.now()}-${Math.random().toString(36).substr(2, 9)}`,
      createdAt: new Date(),
    };

    const chartAnnotations = this.annotations.get(annotation.chartId) || [];
    chartAnnotations.push(fullAnnotation);
    this.annotations.set(annotation.chartId, chartAnnotations);

    // Update chart to show annotation
    this.updateChartAnnotations(annotation.chartId);

    return fullAnnotation;
  }

  /**
   * Remove an annotation
   * @param chartId - Chart identifier
   * @param annotationId - Annotation identifier
   */
  removeAnnotation(chartId: string, annotationId: string): void {
    const chartAnnotations = this.annotations.get(chartId);
    if (chartAnnotations) {
      const filtered = chartAnnotations.filter((a) => a.id !== annotationId);
      this.annotations.set(chartId, filtered);
      this.updateChartAnnotations(chartId);
    }
  }

  /**
   * Get all annotations for a chart
   * @param chartId - Chart identifier
   * @returns Array of annotations
   */
  getAnnotations(chartId: string): ChartAnnotation[] {
    return this.annotations.get(chartId) || [];
  }

  /**
   * Clear all annotations for a chart
   * @param chartId - Chart identifier
   */
  clearAnnotations(chartId: string): void {
    this.annotations.set(chartId, []);
    this.updateChartAnnotations(chartId);
  }

  /**
   * Handle chart click event for adding annotations
   * @param chartId - Chart identifier
   * @param params - Click event parameters
   */
  private handleChartClick(chartId: string, params: any): void {
    // Emit event for UI to handle annotation creation
    const event = new CustomEvent('chart-annotation-click', {
      detail: {
        chartId,
        seriesIndex: params.seriesIndex,
        dataIndex: params.dataIndex,
        value: params.value,
        name: params.name,
        seriesName: params.seriesName,
      },
    });
    window.dispatchEvent(event);
  }

  /**
   * Update chart with current annotations
   * @param chartId - Chart identifier
   */
  private updateChartAnnotations(chartId: string): void {
    const chart = this.chartInstances.get(chartId);
    if (!chart) {
      return;
    }

    const annotations = this.getAnnotations(chartId);
    const markPointData: any[] = [];
    const graphic: any[] = [];

    annotations.forEach((annotation) => {
      if (annotation.type === 'text') {
        if (annotation.dataPoint) {
          // Add as markPoint
          markPointData.push({
            coord: [annotation.dataPoint.value[0], annotation.dataPoint.value[1]],
            value: annotation.content,
            itemStyle: {
              color: annotation.style?.color || '#FF6B6B',
            },
            label: {
              show: true,
              formatter: annotation.content,
              fontSize: annotation.style?.fontSize || 12,
              color: annotation.style?.color || '#333',
              backgroundColor: annotation.style?.backgroundColor || 'rgba(255, 255, 255, 0.9)',
              borderColor: annotation.style?.borderColor || '#ccc',
              borderWidth: annotation.style?.borderWidth || 1,
              padding: [4, 8],
              borderRadius: 4,
            },
          });
        } else if (annotation.pixelPosition) {
          // Add as graphic element
          graphic.push({
            type: 'text',
            id: annotation.id,
            left: annotation.pixelPosition.x,
            top: annotation.pixelPosition.y,
            style: {
              text: annotation.content,
              fontSize: annotation.style?.fontSize || 12,
              fill: annotation.style?.color || '#333',
              backgroundColor: annotation.style?.backgroundColor,
              borderColor: annotation.style?.borderColor,
              borderWidth: annotation.style?.borderWidth,
            },
            draggable: true,
            onmouseover: () => {
              // Show tooltip with annotation details
            },
            ondrag: (e: any) => {
              // Update annotation position
              annotation.pixelPosition = { x: e.target.x, y: e.target.y };
            },
          });
        }
      }
    });

    // Update chart option
    const option = chart.getOption() as any;

    // Add markPoint to first series (can be enhanced to support multiple series)
    if (option.series && option.series.length > 0 && markPointData.length > 0) {
      option.series[0].markPoint = {
        data: markPointData,
        symbol: 'pin',
        symbolSize: 40,
        animation: true,
      };
    }

    // Add graphic elements
    if (graphic.length > 0) {
      option.graphic = graphic;
    }

    chart.setOption(option, { notMerge: false });
  }

  /**
   * Export annotations as JSON
   * @param chartId - Chart identifier
   * @returns JSON string of annotations
   */
  exportAnnotations(chartId: string): string {
    const annotations = this.getAnnotations(chartId);
    return JSON.stringify(annotations, null, 2);
  }

  /**
   * Import annotations from JSON
   * @param chartId - Chart identifier
   * @param json - JSON string of annotations
   */
  importAnnotations(chartId: string, json: string): void {
    try {
      const annotations = JSON.parse(json) as ChartAnnotation[];
      this.annotations.set(chartId, annotations);
      this.updateChartAnnotations(chartId);
    } catch (error) {
      throw new Error('Invalid annotation JSON format');
    }
  }

  /**
   * Find annotations near a data point
   * @param chartId - Chart identifier
   * @param value - Data value [timestamp, value]
   * @param threshold - Distance threshold in data units
   * @returns Array of nearby annotations
   */
  findNearbyAnnotations(
    chartId: string,
    value: [number, number],
    threshold: number = 3600000 // 1 hour in milliseconds for time series
  ): ChartAnnotation[] {
    const annotations = this.getAnnotations(chartId);
    return annotations.filter((annotation) => {
      if (!annotation.dataPoint) {
        return false;
      }

      const timeDiff = Math.abs(annotation.dataPoint.value[0] - value[0]);
      const valueDiff = Math.abs(annotation.dataPoint.value[1] - value[1]);

      return timeDiff <= threshold && valueDiff <= value[1] * 0.1; // Within 10% of value
    });
  }
}

// Export singleton instance
export const chartAnnotationService = new ChartAnnotationService();

// Export convenience functions
export const addChartAnnotation = (annotation: Omit<ChartAnnotation, 'id' | 'createdAt'>) =>
  chartAnnotationService.addAnnotation(annotation);

export const removeChartAnnotation = (chartId: string, annotationId: string) =>
  chartAnnotationService.removeAnnotation(chartId, annotationId);

export const getChartAnnotations = (chartId: string) =>
  chartAnnotationService.getAnnotations(chartId);

export const registerChartForAnnotations = (chartId: string, chart: ECharts) =>
  chartAnnotationService.registerChart(chartId, chart);
