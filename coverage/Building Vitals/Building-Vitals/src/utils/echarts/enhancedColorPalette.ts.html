// Gradient generators for different visualization types
export const GRADIENT_GENERATORS = {
  temperature: (value: number) => {
    // Blue -> Green -> Yellow -> Orange -> Red
    const colors = ['#0288D1', '#00E676', '#FFD600', '#FF9800', '#FF5722'];
    const position = Math.max(0, Math.min(1, value));
    const index = position * (colors.length - 1);
    const lower = Math.floor(index);
    const upper = Math.ceil(index);
    const fraction = index - lower;

    if (lower === upper) {return colors[lower];}

    return interpolateColor(colors[lower], colors[upper], fraction);
  },

  deviation: (value: number, centerValue = 0) => {
    // Blue (below) -> Green (center) -> Red (above)
    const normalizedValue = (value - centerValue) / Math.abs(centerValue || 1);

    if (normalizedValue < -0.5) {return '#0288D1';} // Deep Blue
    if (normalizedValue < -0.25) {return '#03A9F4';} // Light Blue
    if (normalizedValue < 0.25) {return '#00E676';} // Green
    if (normalizedValue < 0.5) {return '#FFB300';} // Amber
    return '#FF5722'; // Red
  },

  efficiency: (value: number) => {
    // Red -> Orange -> Yellow -> Light Green -> Green
    const colors = ['#D32F2F', '#F57C00', '#FBC02D', '#689F38', '#388E3C'];
    const position = Math.max(0, Math.min(1, value));
    const index = position * (colors.length - 1);
    const lower = Math.floor(index);
    const upper = Math.ceil(index);
    const fraction = index - lower;

    if (lower === upper) {return colors[lower];}

    return interpolateColor(colors[lower], colors[upper], fraction);
  },
};

// Helper function to interpolate between two colors
function interpolateColor(color1: string, color2: string, fraction: number): string {
  const rgb1 = hexToRgb(color1);
  const rgb2 = hexToRgb(color2);

  if (!rgb1 || !rgb2) {return color1;}

  const r = Math.round(rgb1.r + (rgb2.r - rgb1.r) * fraction);
  const g = Math.round(rgb1.g + (rgb2.g - rgb1.g) * fraction);
  const b = Math.round(rgb1.b + (rgb2.b - rgb1.b) * fraction);

  return rgbToHex(r, g, b);
}

// Convert hex to RGB
function hexToRgb(hex: string): { r: number; g: number; b: number } | null {
  const result = /^#?([a-f\d]{2})([a-f\d]{2})([a-f\d]{2})$/i.exec(hex);
  return result
    ? {
        r: parseInt(result[1], 16),
        g: parseInt(result[2], 16),
        b: parseInt(result[3], 16),
      }
    : null;
}

// Convert RGB to hex
function rgbToHex(r: number, g: number, b: number): string {
  return '#' + ((1 << 24) + (r << 16) + (g << 8) + b).toString(16).slice(1);
}

// Get colors based on data type and color blind mode
export function getDataTypeColors(
  dataType: keyof typeof DATA_TYPE_PALETTES,
  colorBlindMode: ColorBlindMode = 'normal'
): string[] {
  if (colorBlindMode === 'normal') {
    return DATA_TYPE_PALETTES[dataType].colors;
  }
  return DATA_TYPE_PALETTES[dataType].colorBlindSafe;
}

// Generate a color scale with proper contrast
export function generateAccessibleColorScale(
  count: number,
  backgroundColor: string = '#FFFFFF',
  colorBlindMode: ColorBlindMode = 'normal'
): string[] {
  const palette = COLOR_BLIND_PALETTES[colorBlindMode];
  const colors: string[] = [];

  for (let i = 0; i < count; i++) {
    let color = palette[i % palette.length];

    // Ensure sufficient contrast with background
    while (!hasSufficientContrast(color, backgroundColor, 4.5)) {
      // Darken or lighten the color
      color = adjustColorBrightness(color, backgroundColor === '#FFFFFF' ? -10 : 10);
    }

    colors.push(color);
  }

  return colors;
}

// Adjust color brightness
function adjustColorBrightness(hex: string, percent: number): string {
  const rgb = hexToRgb(hex);
  if (!rgb) {return hex;}

  const r = Math.max(0, Math.min(255, rgb.r + (255 * percent) / 100));
  const g = Math.max(0, Math.min(255, rgb.g + (255 * percent) / 100));
  const b = Math.max(0, Math.min(255, rgb.b + (255 * percent) / 100));

  return rgbToHex(Math.round(r), Math.round(g), Math.round(b));
}

// Check contrast ratio
export function hasSufficientContrast(
  color1: string,
  color2: string,
  minRatio: number = 4.5
): boolean {
  const rgb1 = hexToRgb(color1);
  const rgb2 = hexToRgb(color2);

  if (!rgb1 || !rgb2) {return false;}

  const l1 = getLuminance(rgb1);
  const l2 = getLuminance(rgb2);

  const ratio = (Math.max(l1, l2) + 0.05) / (Math.min(l1, l2) + 0.05);
  return ratio >= minRatio;
}

// Get relative luminance
function getLuminance(rgb: { r: number; g: number; b: number }): number {
  const { r, g, b } = rgb;
  const sRGB = [r, g, b].map((val) => {
    val = val / 255;
    return val <= 0.03928 ? val / 12.92 : Math.pow((val + 0.055) / 1.055, 2.4);
  });
  return 0.2126 * sRGB[0] + 0.7152 * sRGB[1] + 0.0722 * sRGB[2];
}

// Create themed chart colors
export function getThemedChartColors(
  theme: 'light' | 'dark',
  colorBlindMode: ColorBlindMode = 'normal'
): {
  colors: string[];
  background: string;
  text: string;
  grid: string;
  tooltip: {
    background: string;
    border: string;
    text: string;
  };
} {
  const isDark = theme === 'dark';
  const colors = generateAccessibleColorScale(10, isDark ? '#0A0A0B' : '#FFFFFF', colorBlindMode);

  return {
    colors,
    background: isDark ? 'rgba(10, 10, 11, 0.8)' : 'rgba(255, 255, 255, 0.95)',
    text: isDark ? 'rgba(255, 255, 255, 0.9)' : 'rgba(0, 0, 0, 0.87)',
    grid: isDark ? 'rgba(255, 255, 255, 0.1)' : 'rgba(0, 0, 0, 0.1)',
    tooltip: {
      background: isDark ? 'rgba(10, 14, 39, 0.95)' : 'rgba(255, 255, 255, 0.95)',
      border: isDark ? 'rgba(69, 113, 175, 0.3)' : 'rgba(0, 0, 0, 0.2)',
      text: isDark ? '#FFFFFF' : '#000000',
    },
  };
}
