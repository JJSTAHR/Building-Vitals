/**
 * KV Tag Parser Utility
 * Parses the actual KV Tags structure from the point data
 * and extracts clean display names, units, and other metadata
 */

import type { Point } from '../types/api';

// KV Tag structure from the actual data
interface KvTag {
  dis?: string;           // Display name like "RoomTemp", "Damper"
  unit?: string;          // Unit like "Â°F", "%", "cfm", "%RH"
  kind?: string;          // Type: "Number", "Bool", "Str"
  curVal?: string;        // Current value with unit
  enum?: string;          // Enum values for bool/enum types
  navName?: string;       // Navigation name
  axType?: string;        // System type
  precision?: string;     // Decimal precision
  axStatus?: string;      // Status: "ok", "overridden"
  curStatus?: string;     // Current status
  axSlotPath?: string;    // Full slot path
  tz?: string;            // Timezone
  actions?: string;       // Available actions
}

// Extended Point interface with parsed KV tags
export interface EnhancedPointWithKv extends Point {
  // Original fields
  Name?: string;
  Site?: string;
  'Point Type'?: string;
  'Collect Enabled'?: string;
  'Kv Tags'?: string;
  'Marker Tags'?: string;

  // Enhanced fields from KV tags
  display_name?: string;
  unit?: string;
  value?: string | number;
  kind?: string;
  equipment?: string;
  equipmentType?: string;
  pointFunction?: string;
  marker_tags?: string[];
  status?: string;
  precision?: number;
  timezone?: string;
  ai_insights?: {
    recommendations?: string[];
    patterns?: Record<string, any>;
  };
  quality_score?: number;
  _enhanced?: boolean;
  _enhancedAt?: string;
}

/**
 * Parse KV Tags string into structured data
 * Handles both JSON strings and already-parsed objects from Cloudflare Worker
 */
function parseKvTags(kvTagsString: string | any[] | any): KvTag | null {
  if (!kvTagsString || kvTagsString === '[]') return null;

  try {
    // If it's already an array (parsed by Cloudflare Worker), use it directly
    if (Array.isArray(kvTagsString)) {
      if (kvTagsString.length === 0) return null;
      return kvTagsString[0] as KvTag;
    }

    // If it's already an object (single KV tag), return it
    if (typeof kvTagsString === 'object' && kvTagsString !== null) {
      return kvTagsString as KvTag;
    }

    // If it's a string, parse it
    if (typeof kvTagsString === 'string') {
      const kvArray = JSON.parse(kvTagsString);
      if (!Array.isArray(kvArray) || kvArray.length === 0) return null;
      return kvArray[0] as KvTag;
    }

    return null;
  } catch (error) {
    console.error('Failed to parse KV tags:', error, 'Input type:', typeof kvTagsString);
    return null;
  }
}

/**
 * Extract equipment information from point name
 * Handles paths like: "ses/ses_falls_city/Vav707.points.Damper" and "BacnetNetwork.Rtu6_1.points.SaFanStatus"
 * Also handles weather API paths like: "ses/ses_falls_city/weather/wet_bulb"
 */
function extractEquipmentFromName(name: string): {
  type?: string;
  id?: string;
  pointName?: string;
} {
  if (!name) return {};

  // Handle weather API points
  if (name.includes('/weather/')) {
    const weatherMetric = name.split('/').pop() || '';
    return {
      type: 'weather',
      pointName: weatherMetric
    };
  }

  // First, handle paths with slashes (like "ses/ses_falls_city/Vav707.points.Damper")
  // Split by slash and get the last part which contains the equipment and point
  const pathParts = name.split('/');
  const relevantPart = pathParts[pathParts.length - 1]; // e.g., "Vav707.points.Damper"

  // Now split by dots to extract equipment and point
  const parts = relevantPart.split('.');
  if (parts.length < 2) return {};

  // Get equipment and point parts
  // Example: "Vav707.points.Damper" -> Equipment: Vav707, Point: Damper
  // Example: "Rtu6_1.points.SaFanStatus" -> Equipment: Rtu6_1, Point: SaFanStatus
  const pointsIndex = parts.indexOf('points');
  if (pointsIndex > 0 && pointsIndex < parts.length - 1) {
    const equipmentPart = parts[pointsIndex - 1];
    const pointName = parts[pointsIndex + 1];

    // Parse equipment type and ID
    // Handles: Vav707, Rtu6_1, Ahu12, etc.
    const equipmentMatch = equipmentPart.match(/^([A-Za-z]+)(\d+)?(?:[-_](\d+))?/i);
    if (equipmentMatch) {
      return {
        type: equipmentMatch[1].toLowerCase(),
        id: equipmentMatch[2] || equipmentMatch[3] || '',
        pointName
      };
    }

    return {
      type: equipmentPart.toLowerCase(),
      pointName
    };
  }

  return {};
}

/**
 * Generate display name from KV tag dis field and point name
 */
function generateDisplayName(kvTag: KvTag | null, pointName: string, equipment: any): string {
  // Use KV tag display name if available
  if (kvTag?.dis) {
    // Expand common abbreviations in dis field
    let displayName = kvTag.dis;

    // Add spacing between camelCase words
    displayName = displayName.replace(/([a-z])([A-Z])/g, '$1 $2');

    // Expand abbreviations
    displayName = displayName
      .replace(/^Da/i, 'Discharge Air')
      .replace(/^Ra/i, 'Return Air')
      .replace(/^Sa/i, 'Supply Air')
      .replace(/^Oa/i, 'Outside Air')
      .replace(/^Ma/i, 'Mixed Air')
      .replace(/Temp$/i, 'Temperature')
      .replace(/Setpt$/i, 'Setpoint')
      .replace(/Fpm$/i, 'Velocity')
      .replace(/Rh/i, 'Reheat')
      .replace(/Act/i, 'Actual');

    // Add equipment context if available
    if (equipment?.type) {
      const equipmentNames: Record<string, string> = {
        rtu: 'RTU',
        vav: 'VAV',
        ahu: 'AHU',
        mstp: 'MSTP'
      };

      const equipName = equipmentNames[equipment.type] || equipment.type.toUpperCase();
      const equipId = equipment.id || '';

      // Don't duplicate equipment info if already in display name
      if (!displayName.includes(equipName)) {
        displayName = `${equipName}${equipId ? '-' + equipId : ''} ${displayName}`;
      }
    }

    return displayName;
  }

  // Handle weather API points
  if (equipment?.type === 'weather') {
    const weatherMetric = pointName || '';
    if (weatherMetric) {
      const formatted = weatherMetric
        .replace(/_/g, ' ')
        .replace(/\b\w/g, l => l.toUpperCase());
      return `Weather: ${formatted}`;
    }
  }

  // Fallback to point name parsing
  if (pointName) {
    return pointName
      .replace(/([a-z])([A-Z])/g, '$1 $2')
      .replace(/_/g, ' ')
      .replace(/-/g, ' ');
  }

  return 'Unknown Point';
}

/**
 * Generate marker tags based on point characteristics
 */
function generateMarkerTags(kvTag: KvTag | null, equipment: any, pointName: string): string[] {
  const tags = new Set<string>(['point']);

  // Add tags based on KV tag data
  if (kvTag) {
    // Type-based tags
    if (kvTag.kind === 'Number') tags.add('number');
    if (kvTag.kind === 'Bool') tags.add('bool');
    if (kvTag.kind === 'Str') tags.add('string');

    // Unit-based tags
    if (kvTag.unit?.includes('Â°F') || kvTag.unit?.includes('Â°C')) {
      tags.add('temp');
      tags.add('sensor');
    }
    if (kvTag.unit === '%') {
      if (kvTag.dis?.toLowerCase().includes('damper')) tags.add('damper');
      if (kvTag.dis?.toLowerCase().includes('valve')) tags.add('valve');
      if (kvTag.dis?.toLowerCase().includes('signal')) tags.add('signal');
      tags.add('position');
    }
    if (kvTag.unit === 'cfm') {
      tags.add('flow');
      tags.add('air');
    }
    if (kvTag.unit === 'gpm') {
      tags.add('flow');
      tags.add('water');
    }
    if (kvTag.unit === '%RH') {
      tags.add('humidity');
      tags.add('sensor');
    }
    if (kvTag.unit?.includes('ft/min')) {
      tags.add('velocity');
      tags.add('air');
    }

    // Name-based tags
    const disLower = kvTag.dis?.toLowerCase() || '';
    if (disLower.includes('setpt') || disLower.includes('setpoint')) {
      tags.add('sp');
      tags.add('writable');
    }
    if (disLower.includes('status')) tags.add('status');
    if (disLower.includes('mode')) tags.add('mode');
    if (disLower.includes('occ')) tags.add('occupied');
    if (disLower.includes('heat')) tags.add('heat');
    if (disLower.includes('cool')) tags.add('cool');
    if (disLower.includes('fan')) tags.add('fan');

    // Status tags
    if (kvTag.axStatus === 'overridden') tags.add('overridden');
  }

  // Equipment tags
  if (equipment?.type) {
    const equipmentTags: Record<string, string[]> = {
      rtu: ['rtu', 'hvac', 'rooftop'],
      vav: ['vav', 'hvac', 'terminal'],
      ahu: ['ahu', 'hvac'],
      mstp: ['mstp', 'network']
    };
    const eTags = equipmentTags[equipment.type] || [equipment.type];
    eTags.forEach(t => tags.add(t));
  }

  // Point name based tags
  if (pointName) {
    const nameLower = pointName.toLowerCase();
    if (nameLower.includes('da') || nameLower.includes('discharge')) tags.add('discharge');
    if (nameLower.includes('ra') || nameLower.includes('return')) tags.add('return');
    if (nameLower.includes('sa') || nameLower.includes('supply')) tags.add('supply');
    if (nameLower.includes('oa') || nameLower.includes('outside')) tags.add('outside');
  }

  return Array.from(tags);
}

/**
 * Generate AI insights based on point data
 */
function generateAIInsights(kvTag: KvTag | null, equipment: any): {
  recommendations: string[];
  patterns?: Record<string, any>;
} {
  const insights = {
    recommendations: [] as string[],
    patterns: {} as Record<string, any>
  };

  if (!kvTag) return insights;

  // Temperature insights
  if (kvTag.unit?.includes('Â°F')) {
    const disLower = kvTag.dis?.toLowerCase() || '';
    if (disLower.includes('room') || disLower.includes('zone')) {
      insights.recommendations.push('Monitor for ASHRAE 55 comfort compliance (68-76Â°F)');
      insights.patterns.comfort_zone = { min: 68, max: 76, unit: 'Â°F' };
    }
    if (disLower.includes('supply')) {
      insights.recommendations.push('Typical supply air: 55-60Â°F cooling, 90-110Â°F heating');
    }
    if (disLower.includes('discharge')) {
      insights.recommendations.push('Monitor discharge air temperature for proper conditioning');
    }
  }

  // Flow insights
  if (kvTag.unit === 'cfm') {
    insights.recommendations.push('Monitor airflow for ventilation requirements');
    if (kvTag.dis?.toLowerCase().includes('setpt')) {
      insights.recommendations.push('Verify setpoint meets minimum ventilation standards');
    }
  }

  // Damper/Valve insights
  if (kvTag.unit === '%' && (kvTag.dis?.toLowerCase().includes('damper') || kvTag.dis?.toLowerCase().includes('valve'))) {
    const currentVal = parseFloat(kvTag.curVal || '0');
    if (currentVal > 90) {
      insights.recommendations.push('Near full open - equipment may be undersized');
    } else if (currentVal < 10) {
      insights.recommendations.push('Near closed - verify minimum position requirements');
    }
    insights.recommendations.push('Monitor for hunting or oscillation');
  }

  // Humidity insights
  if (kvTag.unit === '%RH') {
    insights.recommendations.push('ASHRAE 55 recommends 30-60% RH for comfort');
    insights.patterns.comfort_zone = { min: 30, max: 60, unit: '%RH' };
  }

  // Equipment-specific insights
  if (equipment?.type === 'vav') {
    if (kvTag.dis?.toLowerCase().includes('damper')) {
      insights.recommendations.push('Ensure minimum ventilation position is maintained');
    }
    if (kvTag.dis?.toLowerCase().includes('heat') && kvTag.dis?.toLowerCase().includes('valve')) {
      insights.recommendations.push('Monitor reheat valve for energy efficiency');
    }
  }

  if (equipment?.type === 'rtu') {
    if (kvTag.dis?.toLowerCase().includes('fan')) {
      insights.recommendations.push('Monitor fan status for proper operation');
    }
  }

  return insights;
}

/**
 * Calculate quality score based on available data
 */
function calculateQualityScore(enhanced: EnhancedPointWithKv): number {
  let score = 0;

  // Has meaningful display name (25 points)
  if (enhanced.display_name && enhanced.display_name !== enhanced.Name) score += 25;

  // Has unit (20 points)
  if (enhanced.unit) score += 20;

  // Has equipment context (20 points)
  if (enhanced.equipment || enhanced.equipmentType) score += 20;

  // Has marker tags (20 points)
  if (enhanced.marker_tags && enhanced.marker_tags.length > 3) score += 20;

  // Has AI insights (15 points)
  if (enhanced.ai_insights?.recommendations && enhanced.ai_insights.recommendations.length > 0) score += 15;

  return score;
}

/**
 * Main enhancement function for points with KV tags
 */
export function enhancePointWithKvTags(point: any): EnhancedPointWithKv {
  // Check if already enhanced
  if (point._enhanced) {
    return point;
  }

  const enhanced: EnhancedPointWithKv = { ...point };

  // Parse KV tags
  const kvTag = parseKvTags(point['Kv Tags'] || point.kv_tags || point.kvTags || '');

  // Extract equipment info from name
  const equipment = extractEquipmentFromName(point.Name || point.name || '');

  // Generate display name
  enhanced.display_name = generateDisplayName(kvTag, equipment.pointName || '', equipment);

  // Extract unit
  if (kvTag?.unit) {
    // Clean up unit format
    enhanced.unit = kvTag.unit
      .replace('degree_Fahrenheit', 'Â°F')
      .replace('degree_Celsius', 'Â°C')
      .replace('percent', '%')
      .replace('percent_relative_humidity', '%RH')
      .replace('cubic_feet_per_minute', 'CFM')
      .replace('foot / minute', 'ft/min')
      .replace('ft/min', 'FPM');
  }

  // Extract current value
  if (kvTag?.curVal) {
    // Parse value from string like "72.0 degree_Fahrenheit"
    const valMatch = kvTag.curVal.match(/^([\d.-]+)/);
    if (valMatch) {
      enhanced.value = parseFloat(valMatch[1]);
    } else if (kvTag.curVal === 'True' || kvTag.curVal === 'False') {
      enhanced.value = kvTag.curVal === 'True' ? 1 : 0;
    }
  }

  // Set equipment info
  if (equipment.type) {
    enhanced.equipment = equipment.type;
    enhanced.equipmentType = equipment.type.toUpperCase();
  }

  // Set point function
  if (kvTag?.dis?.toLowerCase().includes('setpt') || kvTag?.dis?.toLowerCase().includes('setpoint')) {
    enhanced.pointFunction = 'sp';
  } else if (kvTag?.kind === 'Bool' && kvTag?.dis?.toLowerCase().includes('status')) {
    enhanced.pointFunction = 'sensor';
  } else {
    enhanced.pointFunction = 'sensor';
  }

  // Generate marker tags
  enhanced.marker_tags = generateMarkerTags(kvTag, equipment, equipment.pointName || '');

  // Set status
  enhanced.status = kvTag?.axStatus || kvTag?.curStatus || 'ok';

  // Set precision
  if (kvTag?.precision) {
    enhanced.precision = parseFloat(kvTag.precision);
  }

  // Set timezone
  enhanced.timezone = kvTag?.tz;

  // Generate AI insights
  enhanced.ai_insights = generateAIInsights(kvTag, equipment);

  // Calculate quality score
  enhanced.quality_score = calculateQualityScore(enhanced);

  // Mark as enhanced
  enhanced._enhanced = true;
  enhanced._enhancedAt = new Date().toISOString();

  return enhanced;
}

/**
 * Check if a point has KV tags that need parsing
 */
export function hasKvTags(point: any): boolean {
  const kvTags = point['Kv Tags'] || point.kv_tags || point.kvTags;
  return kvTags && kvTags !== '[]' && kvTags !== '';
}
