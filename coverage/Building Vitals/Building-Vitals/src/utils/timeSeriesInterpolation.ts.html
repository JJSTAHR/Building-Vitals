/**
 * Time Series Interpolation Utilities
 * 
 * Provides intelligent interpolation methods for correlating independent time series data
 * without requiring exact timestamp matches. Essential for charts like psychrometric
 * that need to combine temperature and humidity data from different sensors.
 */

export type InterpolationMethod = 'linear' | 'step' | 'nearest' | 'cubic';
export type SamplingStrategy = 'all' | 'periodic' | 'changes' | 'union';

/**
 * Interpolate a value at a specific timestamp from a time series
 * @param series Array of [timestamp, value] pairs
 * @param targetTimestamp The timestamp to interpolate at
 * @param method The interpolation method to use
 * @returns The interpolated value or null if cannot interpolate
 */
export function interpolateValue(
  series: Array<[number, number]>,
  targetTimestamp: number,
  method: InterpolationMethod = 'linear'
): number | null {
  if (!series || series.length === 0) return null;
  
  // Sort series by timestamp (in case it's not sorted)
  const sortedSeries = [...series].sort((a, b) => a[0] - b[0]);
  
  // Find exact match first
  const exactMatch = sortedSeries.find(p => p[0] === targetTimestamp);
  if (exactMatch) return exactMatch[1];
  
  // Find surrounding points
  let before: [number, number] | null = null;
  let after: [number, number] | null = null;
  
  for (let i = 0; i < sortedSeries.length; i++) {
    if (sortedSeries[i][0] < targetTimestamp) {
      before = sortedSeries[i];
    } else if (sortedSeries[i][0] > targetTimestamp) {
      after = sortedSeries[i];
      break;
    }
  }
  
  // Handle edge cases
  if (!before && !after) return null;
  if (!before) return after![1]; // Use first value if before target range
  if (!after) return before[1];   // Use last value if after target range
  
  // Apply interpolation method
  switch (method) {
    case 'nearest': {
      const distToBefore = Math.abs(before[0] - targetTimestamp);
      const distToAfter = Math.abs(after[0] - targetTimestamp);
      return distToBefore < distToAfter ? before[1] : after[1];
    }
    
    case 'step':
      // Step interpolation uses the value from before
      return before[1];
    
    case 'cubic': {
      // For cubic, we need 4 points (2 before, 2 after)
      // For now, fallback to linear if we don't have enough points
      // In production, implement full cubic spline interpolation
      return linearInterpolate(before, after, targetTimestamp);
    }
    
    case 'linear':
    default:
      return linearInterpolate(before, after, targetTimestamp);
  }
}

/**
 * Perform linear interpolation between two points
 */
function linearInterpolate(
  before: [number, number],
  after: [number, number],
  targetTimestamp: number
): number {
  const timeDiff = after[0] - before[0];
  if (timeDiff === 0) return before[1]; // Same timestamp, use before value
  
  const ratio = (targetTimestamp - before[0]) / timeDiff;
  return before[1] + (after[1] - before[1]) * ratio;
}

/**
 * Get all unique timestamps from multiple series
 * @param series Array of series data
 * @returns Sorted array of unique timestamps
 */
export function getAllTimestamps(
  ...series: Array<Array<[number, number]> | undefined | null>
): number[] {
  const timestampSet = new Set<number>();
  
  series.forEach(s => {
    if (s && Array.isArray(s)) {
      s.forEach(point => {
        if (Array.isArray(point) && typeof point[0] === 'number') {
          timestampSet.add(point[0]);
        }
      });
    }
  });
  
  return Array.from(timestampSet).sort((a, b) => a - b);
}

/**
 * Generate sampling timestamps based on strategy
 */
export function generateSamplingTimestamps(
  startTime: number,
  endTime: number,
  strategy: SamplingStrategy,
  existingTimestamps: number[],
  interval?: number
): number[] {
  switch (strategy) {
    case 'all':
      // Use all existing timestamps
      return existingTimestamps;
    
    case 'periodic': {
      // Generate timestamps at regular intervals
      const samplingInterval = interval || 60000; // Default 1 minute
      const timestamps: number[] = [];
      for (let t = startTime; t <= endTime; t += samplingInterval) {
        timestamps.push(t);
      }
      return timestamps;
    }
    
    case 'changes': {
      // Use timestamps where significant changes occur
      // For now, return all timestamps; implement change detection later
      return existingTimestamps;
    }
    
    case 'union':
    default:
      // Use union of all timestamps from all series
      return existingTimestamps;
  }
}

/**
 * Correlate multiple time series using interpolation
 * Returns an array of objects with interpolated values for each series at each timestamp
 */
export function correlateSeriesWithInterpolation(
  seriesMap: Map<string, Array<[number, number]>>,
  timestamps: number[],
  method: InterpolationMethod = 'linear'
): Array<{ timestamp: number; values: Map<string, number | null> }> {
  const results: Array<{ timestamp: number; values: Map<string, number | null> }> = [];
  
  timestamps.forEach(timestamp => {
    const values = new Map<string, number | null>();
    
    seriesMap.forEach((series, name) => {
      const interpolatedValue = interpolateValue(series, timestamp, method);
      values.set(name, interpolatedValue);
    });
    
    results.push({ timestamp, values });
  });
  
  return results;
}

/**
 * Find the time range that contains data from all series
 */
export function findCommonTimeRange(
  ...series: Array<Array<[number, number]> | undefined | null>
): { start: number; end: number } | null {
  let globalStart = -Infinity;
  let globalEnd = Infinity;
  
  series.forEach(s => {
    if (s && Array.isArray(s) && s.length > 0) {
      const sorted = [...s].sort((a, b) => a[0] - b[0]);
      const seriesStart = sorted[0][0];
      const seriesEnd = sorted[sorted.length - 1][0];
      
      globalStart = Math.max(globalStart, seriesStart);
      globalEnd = Math.min(globalEnd, seriesEnd);
    }
  });
  
  if (globalStart === -Infinity || globalEnd === Infinity || globalStart > globalEnd) {
    return null;
  }
  
  return { start: globalStart, end: globalEnd };
}

/**
 * Validate if a series has sufficient data for interpolation
 */
export function validateSeriesForInterpolation(
  series: Array<[number, number]> | undefined | null,
  minPoints: number = 2
): boolean {
  if (!series || !Array.isArray(series)) return false;
  if (series.length < minPoints) return false;
  
  // Check if all points have valid timestamps and values
  return series.every(point => 
    Array.isArray(point) &&
    point.length >= 2 &&
    typeof point[0] === 'number' &&
    typeof point[1] === 'number' &&
    !isNaN(point[0]) &&
    !isNaN(point[1])
  );
}

/**
 * Smart resampling to reduce data points while preserving important features
 */
export function smartResample(
  series: Array<[number, number]>,
  maxPoints: number = 1000
): Array<[number, number]> {
  if (series.length <= maxPoints) return series;
  
  // Use Douglas-Peucker algorithm or similar for smart downsampling
  // For now, use simple uniform sampling
  const step = Math.ceil(series.length / maxPoints);
  const resampled: Array<[number, number]> = [];
  
  for (let i = 0; i < series.length; i += step) {
    resampled.push(series[i]);
  }
  
  // Always include the last point
  if (resampled[resampled.length - 1] !== series[series.length - 1]) {
    resampled.push(series[series.length - 1]);
  }
  
  return resampled;
}
