/**
 * Performance comparison test between batching and direct API
 */
export async function runPerformanceComparison(
  testPoints: TestPoint[],
  timeRange: string = '24h'
): Promise<PerformanceTestResult[]> {
  
  console.group('‚ö° Running Performance Comparison');
  
  const results: PerformanceTestResult[] = [];
  
  // Test scenarios
  const scenarios = [
    { name: 'Direct API (Legacy)', enableBatching: false },
    { name: 'Batching Enabled', enableBatching: true },
  ];
  
  for (const scenario of scenarios) {
    console.log(`\nüî¨ Testing: ${scenario.name}`);
    
    const startTime = performance.now();
    let requestCount = 0;
    let errors = 0;
    let cacheHitRate = 0;
    
    try {
      // Simulate multiple chart requests
      const promises = Array.from({ length: 3 }, async (_, i) => {
        requestCount++;
        
        // Simulate different chart components requesting overlapping data
        const pointSubset = testPoints.slice(i, i + 3);
        
        // This would normally use the actual hook, but for testing we simulate
        return {
          success: true,
          responseTime: Math.random() * 500 + 100, // Simulate 100-600ms response
          cached: scenario.enableBatching && Math.random() > 0.3, // 70% cache hit rate for batching
        };
      });
      
      const responses = await Promise.all(promises);
      const responseTime = performance.now() - startTime;
      
      // Calculate metrics
      const successfulResponses = responses.filter(r => r.success);
      const cachedResponses = responses.filter(r => r.cached);
      cacheHitRate = cachedResponses.length / responses.length;
      
      const result: PerformanceTestResult = {
        testName: scenario.name,
        batchingEnabled: scenario.enableBatching,
        pointCount: testPoints.length,
        responseTime,
        requestCount,
        cacheHitRate,
        errors,
        success: successfulResponses.length === responses.length
      };
      
      results.push(result);
      
      console.log(`‚úì ${scenario.name}:`, {
        responseTime: `${responseTime.toFixed(2)}ms`,
        requests: requestCount,
        cacheHitRate: `${(cacheHitRate * 100).toFixed(1)}%`,
        success: result.success
      });
      
    } catch (error) {
      console.error(`‚ùå ${scenario.name} failed:`, error);
      errors++;
      
      results.push({
        testName: scenario.name,
        batchingEnabled: scenario.enableBatching,
        pointCount: testPoints.length,
        responseTime: 0,
        requestCount,
        cacheHitRate: 0,
        errors,
        success: false
      });
    }
  }
  
  // Summary
  console.log('\nüìä Performance Comparison Summary:');
  const directAPI = results.find(r => !r.batchingEnabled);
  const batchingAPI = results.find(r => r.batchingEnabled);
  
  if (directAPI && batchingAPI && directAPI.success && batchingAPI.success) {
    const improvement = ((directAPI.responseTime - batchingAPI.responseTime) / directAPI.responseTime) * 100;
    console.log(`üöÄ Performance improvement: ${improvement.toFixed(1)}%`);
    console.log(`üìà Cache hit rate: ${(batchingAPI.cacheHitRate * 100).toFixed(1)}%`);
  }
  
  console.groupEnd();
  return results;
}

    id: `test-point-${i}`,
    name: `TestPoint_${String(i).padStart(3, '0')}`,
    unit: ['¬∞F', '¬∞C', '%', 'psi', 'cfm', 'kW'][i % 6],
    markerTag: ['temperature', 'humidity', 'pressure', 'flow', 'power'][i % 5],
    marker_tags: [
      ['temperature', 'zone-temp'],
      ['humidity', 'relative-humidity'],
      ['pressure', 'static-pressure'],
      ['flow', 'air-flow'],
      ['power', 'electrical']
    ][i % 5]
  }));
}
