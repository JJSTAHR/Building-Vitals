/**
 * Value Formatting Utilities
 * Ensures all numeric values display with 4 decimal places
 */

/**
 * Format a numeric value to 4 decimal places
 * Handles both positive and negative numbers
 */
export function formatValue(value: number | null | undefined): string {
  if (value === null || value === undefined || isNaN(value)) {
    return 'N/A';
  }

  // Format to 4 decimal places
  // toFixed returns a string with exactly 4 decimal places
  return value.toFixed(4);
}

/**
 * Format a value with unit
 */
export function formatValueWithUnit(value: number | null | undefined, unit?: string): string {
  const formattedValue = formatValue(value);
  if (formattedValue === 'N/A') {return formattedValue;}

  return unit ? `${formattedValue} ${unit}` : formattedValue;
}

/**
 * Format percentage to 4 decimal places
 */
export function formatPercentage(value: number | null | undefined): string {
  if (value === null || value === undefined || isNaN(value)) {
    return 'N/A';
  }

  return `${value.toFixed(4)}%`;
}

/**
 * Format temperature with sign for deviations
 */
export function formatDeviation(value: number | null | undefined, unit?: string): string {
  if (value === null || value === undefined || isNaN(value)) {
    return 'N/A';
  }

  // Always show sign for deviations
  const sign = value >= 0 ? '+' : '';
  const formatted = `${sign}${value.toFixed(4)}`;

  return unit ? `${formatted} ${unit}` : formatted;
}

/**
 * ECharts tooltip formatter for consistent display
 */
export function createTooltipFormatter(unit?: string) {
  return function (params: any) {
    if (Array.isArray(params)) {
      // Multiple series - Handle different chart types
      const lines = params.map((item) => {
        const value = formatValueWithUnit(item.value[1], unit);
        return `${item.marker} ${item.seriesName}: ${value}`;
      });
      
      // Safely get time/axis label - different for heatmaps vs other charts
      let timeLabel = 'Data Point';
      // Check if params has items and the first item exists
      if (params.length > 0 && params[0]) {
        if (params[0].axisValueLabel) {
          timeLabel = params[0].axisValueLabel;
        } else if (params[0].name) {
          timeLabel = params[0].name;
        } else if (params[0].value && Array.isArray(params[0].value) && params[0].value[0]) {
          // For data in [timestamp, value] format
          const timestamp = params[0].value[0];
          if (typeof timestamp === 'number' && timestamp > 1000000000000) {
            timeLabel = new Date(timestamp).toLocaleString();
          } else {
            timeLabel = String(timestamp);
          }
        }
      }
      
      return `${timeLabel}<br/>${lines.join('<br/>')}`;
    } else {
      // Single series - Handle heatmap vs regular charts
      if (params.componentType === 'series' && params.seriesType === 'heatmap') {
        // Heatmap specific formatting
        const value = Array.isArray(params.value)
          ? formatValueWithUnit(params.value[2], unit) // Heatmap uses [x, y, value] format
          : formatValueWithUnit(params.value, unit);
        return `${params.name || 'Data Point'}: ${value}`;
      } else {
        // Regular chart formatting
        const value = Array.isArray(params.value)
          ? formatValueWithUnit(params.value[1], unit)
          : formatValueWithUnit(params.value, unit);
        return `${params.name}: ${value}`;
      }
    }
  };
}

/**
 * ECharts axis label formatter
 */
export function createAxisLabelFormatter(unit?: string) {
  return function (value: number): string {
    return formatValueWithUnit(value, unit);
  };
}

/**
 * Format large numbers with proper decimal places
 */
export function formatLargeNumber(value: number | null | undefined): string {
  if (value === null || value === undefined || isNaN(value)) {
    return 'N/A';
  }

  const absValue = Math.abs(value);

  // For very large numbers, use scientific notation
  if (absValue >= 1000000) {
    return value.toExponential(4);
  }

  // Otherwise use standard 4 decimal places
  return value.toFixed(4);
}

/**
 * Parse and format a value from API
 */
export function parseAndFormatValue(value: string | number | boolean | null): number {
  if (value === null || value === undefined) {return 0;}

  let numericValue: number;

  if (typeof value === 'number') {
    numericValue = value;
  } else if (typeof value === 'boolean') {
    numericValue = value ? 1 : 0;
  } else if (typeof value === 'string') {
    numericValue = parseFloat(value);
  } else {
    numericValue = 0;
  }

  // Return with 4 decimal precision
  return isNaN(numericValue) ? 0 : parseFloat(numericValue.toFixed(4));
}
