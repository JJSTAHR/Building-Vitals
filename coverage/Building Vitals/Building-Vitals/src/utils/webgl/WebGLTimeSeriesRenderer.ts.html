/**
 * WebGL Time Series Renderer
 * High-performance rendering for millions of time series data points
 */

import type {
  WebGLProgramInfo} from './webglUtils';
import {
  createShader,
  createProgram,
  createBuffer,
  setupAttribute,
  normalizeTimestamps,
  normalizeValues,
  resizeCanvas
} from './webglUtils';

// Vertex shader for time series lines
const VERTEX_SHADER_SOURCE = `
  attribute vec2 a_position;
  attribute float a_series;
  
  uniform vec2 u_resolution;
  uniform vec2 u_pan;
  uniform vec2 u_scale;
  
  varying float v_series;
  
  void main() {
    // Apply pan and scale
    vec2 position = (a_position * u_scale) + u_pan;
    
    // Convert from clip space to pixels
    vec2 zeroToOne = (position + 1.0) / 2.0;
    vec2 zeroToTwo = zeroToOne * 2.0;
    vec2 clipSpace = zeroToTwo - 1.0;
    
    gl_Position = vec4(clipSpace * vec2(1, -1), 0, 1);
    v_series = a_series;
    gl_PointSize = 2.0;
  }
`;

// Fragment shader for time series lines
const FRAGMENT_SHADER_SOURCE = `
  precision mediump float;
  
  varying float v_series;
  uniform vec4 u_colors[10]; // Support up to 10 series
  
  void main() {
    int seriesIndex = int(v_series);
    gl_FragColor = u_colors[seriesIndex];
  }
`;

export interface TimeSeriesData {
  timestamps: Float64Array;
  values: Float32Array[];
  names: string[];
}

export interface RenderOptions {
  lineWidth?: number;
  colors?: string[];
  showPoints?: boolean;
  antialiasing?: boolean;
}

export class WebGLTimeSeriesRenderer {
  private canvas: HTMLCanvasElement;
  private gl: WebGLRenderingContext;
  private programInfo: WebGLProgramInfo | null = null;
  private buffers: {
    position: WebGLBuffer | null;
    series: WebGLBuffer | null;
    indices: WebGLBuffer | null;
  } = {
    position: null,
    series: null,
    indices: null,
  };
  private dataRange = {
    minTime: 0,
    maxTime: 1,
    minValue: 0,
    maxValue: 1,
  };
  private pan = { x: 0, y: 0 };
  private scale = { x: 1, y: 1 };
  private pointCount = 0;
  private seriesCount = 0;
  private colors: number[][] = [];

  constructor(canvas: HTMLCanvasElement) {
    this.canvas = canvas;
    const gl = canvas.getContext('webgl', {
      antialias: true,
      preserveDrawingBuffer: true,
    });

    if (!gl) {
      throw new Error('WebGL not supported');
    }

    this.gl = gl;
    this.initializeWebGL();
  }

  private initializeWebGL(): void {
    const gl = this.gl;

    // Create shaders
    const vertexShader = createShader(gl, gl.VERTEX_SHADER, VERTEX_SHADER_SOURCE);
    const fragmentShader = createShader(gl, gl.FRAGMENT_SHADER, FRAGMENT_SHADER_SOURCE);

    if (!vertexShader || !fragmentShader) {
      throw new Error('Failed to create shaders');
    }

    // Create program
    const program = createProgram(gl, vertexShader, fragmentShader);
    if (!program) {
      throw new Error('Failed to create program');
    }

    // Get locations
    this.programInfo = {
      program,
      attribLocations: {
        position: gl.getAttribLocation(program, 'a_position'),
        series: gl.getAttribLocation(program, 'a_series'),
      },
      uniformLocations: {
        resolution: gl.getUniformLocation(program, 'u_resolution'),
        pan: gl.getUniformLocation(program, 'u_pan'),
        scale: gl.getUniformLocation(program, 'u_scale'),
      },
    };

    // Get color uniform locations
    for (let i = 0; i < 10; i++) {
      this.programInfo.uniformLocations[`color${i}`] = gl.getUniformLocation(
        program,
        `u_colors[${i}]`
      );
    }

    // Set up WebGL state
    gl.enable(gl.BLEND);
    gl.blendFunc(gl.SRC_ALPHA, gl.ONE_MINUS_SRC_ALPHA);
  }

  /**
   * Set data for rendering
   */
  setData(data: TimeSeriesData, options: RenderOptions = {}): void {
    const gl = this.gl;

    // Calculate data range
    this.dataRange.minTime = Math.min(...data.timestamps);
    this.dataRange.maxTime = Math.max(...data.timestamps);
    this.dataRange.minValue = Infinity;
    this.dataRange.maxValue = -Infinity;

    for (const values of data.values) {
      const min = Math.min(...values);
      const max = Math.max(...values);
      this.dataRange.minValue = Math.min(this.dataRange.minValue, min);
      this.dataRange.maxValue = Math.max(this.dataRange.maxValue, max);
    }

    // Normalize data
    const normalizedX = normalizeTimestamps(
      data.timestamps,
      this.dataRange.minTime,
      this.dataRange.maxTime
    );

    // Prepare vertex data
    const pointsPerSeries = data.timestamps.length;
    this.pointCount = pointsPerSeries * data.values.length;
    this.seriesCount = data.values.length;

    const positions = new Float32Array(this.pointCount * 2);
    const series = new Float32Array(this.pointCount);
    const indices = [];

    let vertexIndex = 0;
    for (let seriesIndex = 0; seriesIndex < data.values.length; seriesIndex++) {
      const normalizedY = normalizeValues(
        data.values[seriesIndex],
        this.dataRange.minValue,
        this.dataRange.maxValue
      );

      const baseIndex = seriesIndex * pointsPerSeries;

      for (let i = 0; i < pointsPerSeries; i++) {
        positions[vertexIndex * 2] = normalizedX[i];
        positions[vertexIndex * 2 + 1] = normalizedY[i];
        series[vertexIndex] = seriesIndex;

        // Create line segments
        if (i < pointsPerSeries - 1) {
          indices.push(baseIndex + i, baseIndex + i + 1);
        }

        vertexIndex++;
      }
    }

    // Create buffers
    this.buffers.position = createBuffer(gl, positions);
    this.buffers.series = createBuffer(gl, series);
    this.buffers.indices = createBuffer(gl, new Uint16Array(indices), gl.ELEMENT_ARRAY_BUFFER);

    // Parse colors
    this.colors = this.parseColors(
      options.colors || this.generateDefaultColors(data.values.length)
    );
  }

  /**
   * Parse color strings to RGB arrays
   */
  private parseColors(colors: string[]): number[][] {
    return colors.map((color) => {
      const hex = color.replace('#', '');
      return [
        parseInt(hex.substr(0, 2), 16) / 255,
        parseInt(hex.substr(2, 2), 16) / 255,
        parseInt(hex.substr(4, 2), 16) / 255,
        1.0,
      ];
    });
  }

  /**
   * Generate default colors
   */
  private generateDefaultColors(count: number): string[] {
    const colors = [
      '#2196F3',
      '#F44336',
      '#4CAF50',
      '#FF9800',
      '#9C27B0',
      '#00BCD4',
      '#FFEB3B',
      '#795548',
      '#607D8B',
      '#E91E63',
    ];
    return colors.slice(0, count);
  }

  /**
   * Render the time series
   */
  render(): void {
    const gl = this.gl;
    if (!this.programInfo || !this.buffers.position) {return;}

    // Resize canvas if needed
    resizeCanvas(this.canvas, gl);

    // Clear
    gl.clearColor(0, 0, 0, 0);
    gl.clear(gl.COLOR_BUFFER_BIT);

    // Use program
    gl.useProgram(this.programInfo.program);

    // Set uniforms
    gl.uniform2f(
      this.programInfo.uniformLocations.resolution as WebGLUniformLocation,
      gl.canvas.width,
      gl.canvas.height
    );
    gl.uniform2f(
      this.programInfo.uniformLocations.pan as WebGLUniformLocation,
      this.pan.x,
      this.pan.y
    );
    gl.uniform2f(
      this.programInfo.uniformLocations.scale as WebGLUniformLocation,
      this.scale.x,
      this.scale.y
    );

    // Set colors
    for (let i = 0; i < this.colors.length; i++) {
      const location = this.programInfo.uniformLocations[`color${i}`];
      if (location) {
        gl.uniform4fv(location, this.colors[i]);
      }
    }

    // Set attributes
    setupAttribute(gl, this.buffers.position, this.programInfo.attribLocations.position, 2);
    setupAttribute(gl, this.buffers.series, this.programInfo.attribLocations.series, 1);

    // Draw lines
    if (this.buffers.indices) {
      gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, this.buffers.indices);
      gl.drawElements(gl.LINES, (this.pointCount - this.seriesCount) * 2, gl.UNSIGNED_SHORT, 0);
    }
  }

  /**
   * Set pan offset
   */
  setPan(x: number, y: number): void {
    this.pan.x = x;
    this.pan.y = y;
  }

  /**
   * Set zoom scale
   */
  setScale(x: number, y: number): void {
    this.scale.x = Math.max(0.1, Math.min(100, x));
    this.scale.y = Math.max(0.1, Math.min(100, y));
  }

  /**
   * Convert screen coordinates to data coordinates
   */
  screenToData(screenX: number, screenY: number): { time: number; value: number } {
    const rect = this.canvas.getBoundingClientRect();
    const x = ((screenX - rect.left) / rect.width) * 2 - 1;
    const y = ((screenY - rect.top) / rect.height) * -2 + 1;

    // Apply inverse transform
    const dataX = (x - this.pan.x) / this.scale.x;
    const dataY = (y - this.pan.y) / this.scale.y;

    // Denormalize
    const time =
      ((dataX + 1) / 2) * (this.dataRange.maxTime - this.dataRange.minTime) +
      this.dataRange.minTime;
    const value =
      ((dataY + 1) / 2) * (this.dataRange.maxValue - this.dataRange.minValue) +
      this.dataRange.minValue;

    return { time, value };
  }

  /**
   * Clean up WebGL resources
   */
  dispose(): void {
    const gl = this.gl;

    // Delete buffers
    if (this.buffers.position) {gl.deleteBuffer(this.buffers.position);}
    if (this.buffers.series) {gl.deleteBuffer(this.buffers.series);}
    if (this.buffers.indices) {gl.deleteBuffer(this.buffers.indices);}

    // Delete program
    if (this.programInfo) {
      gl.deleteProgram(this.programInfo.program);
    }
  }
}
