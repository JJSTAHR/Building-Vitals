/**
 * Normalize a unit string to a standard display format
 */
export function normalizeUnit(unit: string | undefined | null): string {
  if (!unit) {return '';}

  // Check direct mapping first
  const normalized = UNIT_MAPPINGS[unit];
  if (normalized !== undefined) {return normalized;}

  // Try lowercase
  const lowerUnit = unit.toLowerCase();
  const lowerNormalized = UNIT_MAPPINGS[lowerUnit];
  if (lowerNormalized !== undefined) {return lowerNormalized;}

  // Try to extract common patterns
  if (lowerUnit.includes('fahrenheit')) {return '°F';}
  if (lowerUnit.includes('celsius')) {return '°C';}
  if (lowerUnit.includes('humidity')) {return '%RH';}
  if (lowerUnit.includes('percent')) {return '%';}
  if (lowerUnit.includes('cfm')) {return 'CFM';}
  if (lowerUnit.includes('gpm')) {return 'GPM';}
  if (lowerUnit.includes('psi')) {return 'PSI';}
  if (lowerUnit.includes('rpm')) {return 'RPM';}
  if (lowerUnit.includes('kwh')) {return 'kWh';}
  if (lowerUnit.includes('kw')) {return 'kW';}
  if (lowerUnit.includes('watts')) {return 'W';}
  if (lowerUnit.includes('volts')) {return 'V';}
  if (lowerUnit.includes('amps') || lowerUnit.includes('amperes')) {return 'A';}

  // Return original if no mapping found
  return unit;
}

/**
 * Get unit category for grouping similar units
 */
export function getUnitCategory(unit: string): string {
  const normalized = normalizeUnit(unit).toLowerCase();

  if (['°f', '°c', 'k'].includes(normalized)) {return 'temperature';}
  if (['%rh', '%'].includes(normalized)) {return 'humidity';}
  if (['pa', 'kpa', 'psi', 'in w.c.', 'in hg', 'mmhg', 'bar', 'mbar'].includes(normalized))
    {return 'pressure';}
  if (['cfm', 'm³/h', 'm³/s', 'l/s', 'l/min', 'gpm'].includes(normalized)) {return 'flow';}
  if (['w', 'kw', 'mw', 'hp', 'btu/hr', 'tons', 'kbtu/hr'].includes(normalized)) {return 'power';}
  if (['wh', 'kwh', 'mwh', 'j', 'kj', 'mj', 'btu', 'therms'].includes(normalized)) {return 'energy';}
  if (
    ['v', 'mv', 'kv', 'a', 'ma', 'ka', 'ω', 'kω', 'mω', 'hz', 'khz', 'mhz', 'pf'].includes(
      normalized
    )
  )
    {return 'electrical';}
  if (['m/s', 'km/h', 'mph', 'ft/s', 'ft/min', 'rpm'].includes(normalized)) {return 'speed';}
  if (['ppm', 'ppb', 'μg/m³', 'mg/m³'].includes(normalized)) {return 'concentration';}

  return 'other';
}

/**
 * Format a value with its unit for display
 */
export function formatWithUnit(
  value: number | null | undefined,
  unit: string | null | undefined,
  precision = 1
): string {
  if (value === null || value === undefined) {return 'N/A';}

  const formattedValue = typeof value === 'number' ? value.toFixed(precision) : String(value);
  const normalizedUnit = normalizeUnit(unit);

  return normalizedUnit ? `${formattedValue} ${normalizedUnit}` : formattedValue;
}

/**
 * Check if two units are compatible for the same chart axis
 */
export function areUnitsCompatible(unit1: string | undefined, unit2: string | undefined): boolean {
  const cat1 = getUnitCategory(unit1 || '');
  const cat2 = getUnitCategory(unit2 || '');

  return cat1 === cat2;
}
