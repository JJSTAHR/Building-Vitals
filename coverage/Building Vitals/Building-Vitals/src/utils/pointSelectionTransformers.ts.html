/**
 * Point Selection Transformers for Building Vitals
 * Handles the transformation of user-selected IoT points into chart-ready data
 * Each transformer is designed for specific building monitoring use cases
 */

import type { TimeSeriesData } from './timeSeriesDataTransformer';

// Types for point selection
export interface SelectedPoint {
  id: string;
  name: string;
  type?: 'temperature' | 'humidity' | 'pressure' | 'power' | 'flow' | 'status' | 'other';
  unit?: string;
  tags?: string[];
  metadata?: {
    equipment?: string;
    location?: string;
    system?: string;
    [key: string]: any;
  };
}

export interface PointData {
  point: SelectedPoint;
  data: TimeSeriesData;
}

/**
 * Bar Chart Transformer - Zone Comparison Dashboard
 * Use Case: Compare current temperatures/power/status across zones
 */
export const transformForZoneComparison = (
  selectedPoints: PointData[],
  options?: {
    sortBy?: 'name' | 'value' | 'deviation';
    showDeviationFrom?: number; // Optional setpoint for deviation
  }
) => {
  const { sortBy = 'name', showDeviationFrom } = options || {};

  const bars = selectedPoints.map(({ point, data }) => {
    const latestValue = data.length > 0 ? data[data.length - 1][1] : 0;
    const latestTimestamp = data.length > 0 ? data[data.length - 1][0] : Date.now();

    const bar = {
      name: point.name,
      value: latestValue,
      unit: point.unit,
      timestamp: latestTimestamp,
      metadata: {
        pointId: point.id,
        location: point.metadata?.location,
        equipment: point.metadata?.equipment,
      },
      // Preserve full time series for drill-down
      rawData: data.map(([ts, val]) => ({
        timestamp: new Date(ts),
        value: val,
      })),
      // Calculate deviation if setpoint provided
      ...(showDeviationFrom !== undefined && {
        deviation: latestValue - showDeviationFrom,
        deviationPercent: ((latestValue - showDeviationFrom) / showDeviationFrom) * 100,
      }),
    };

    return bar;
  });

  // Sort based on criteria
  if (sortBy === 'value') {
    bars.sort((a, b) => b.value - a.value);
  } else if (sortBy === 'deviation' && showDeviationFrom !== undefined) {
    bars.sort((a, b) => Math.abs(b.deviation!) - Math.abs(a.deviation!));
  }

  return bars;
};

/**
 * Device Deviation Heatmap Transformer
 * Use Case: Monitor all VAV boxes or similar equipment against setpoints
 */
export const transformForDeviceDeviationMatrix = (
  selectedPoints: PointData[],
  config: {
    mode: 'setpoint' | 'range' | 'auto';
    setpoint?: number;
    range?: { min: number; max: number };
    timeGranularity?: 'minute' | 'hour' | 'day';
    autoDetectSetpoint?: boolean; // Use equipment metadata
  }
) => {
  const { mode, setpoint, range, timeGranularity = 'hour', autoDetectSetpoint = true } = config;

  // Group by time buckets
  const bucketSize = {
    minute: 60000,
    hour: 3600000,
    day: 86400000,
  }[timeGranularity];

  const heatmapData: Array<{
    deviceId: string;
    deviceName: string;
    timeIndex: number;
    timestamp: number;
    value: number;
    deviation: number;
    status: 'normal' | 'warning' | 'critical';
  }> = [];

  // Find time range
  const allTimestamps = selectedPoints.flatMap((p) => p.data.map(([ts]) => ts));
  const minTime = Math.min(...allTimestamps);
  const maxTime = Math.max(...allTimestamps);

  selectedPoints.forEach(({ point, data }) => {
    // Auto-detect setpoint from equipment metadata if enabled
    let targetSetpoint = setpoint;
    let targetRange = range;

    if (autoDetectSetpoint && point.metadata?.setpoint) {
      targetSetpoint = point.metadata.setpoint;
    }
    if (autoDetectSetpoint && point.metadata?.range) {
      targetRange = point.metadata.range;
    }

    data.forEach(([timestamp, value]) => {
      const timeIndex = Math.floor((timestamp - minTime) / bucketSize);

      let deviation = 0;
      let status: 'normal' | 'warning' | 'critical' = 'normal';

      if (mode === 'setpoint' && targetSetpoint !== undefined) {
        deviation = value - targetSetpoint;
        const absDeviation = Math.abs(deviation);

        if (absDeviation <= 1) {status = 'normal';}
        else if (absDeviation <= 3) {status = 'warning';}
        else {status = 'critical';}
      } else if (mode === 'range' && targetRange) {
        if (value < targetRange.min) {
          deviation = value - targetRange.min;
          status = deviation < -3 ? 'critical' : 'warning';
        } else if (value > targetRange.max) {
          deviation = value - targetRange.max;
          status = deviation > 3 ? 'critical' : 'warning';
        }
      }

      heatmapData.push({
        deviceId: point.id,
        deviceName: point.name,
        timeIndex,
        timestamp,
        value,
        deviation,
        status,
      });
    });
  });

  return {
    data: heatmapData,
    devices: selectedPoints.map((p) => ({
      id: p.point.id,
      name: p.point.name,
      location: p.point.metadata?.location,
      type: p.point.metadata?.equipment,
    })),
    timeRange: { min: minTime, max: maxTime },
    bucketSize,
    config,
  };
};

/**
 * Scatter Plot Transformer - Energy vs Weather Analysis
 * Use Case: Correlate outside temperature with building energy consumption
 */
export const transformForEnergyWeatherCorrelation = (
  temperatureData: PointData,
  energyData: PointData,
  options?: {
    colorByTime?: boolean;
    showRegression?: boolean;
    timeWindow?: 'hourly' | 'daily';
  }
) => {
  const { colorByTime = true, showRegression = true, timeWindow = 'hourly' } = options || {};

  // Pair data points by timestamp
  const correlatedPoints: Array<{
    temperature: number;
    energy: number;
    timestamp: number;
    hour?: number;
    dayOfWeek?: number;
  }> = [];

  const maxTimeDiff = timeWindow === 'hourly' ? 3600000 : 86400000; // 1 hour or 1 day

  temperatureData.data.forEach(([tempTime, tempValue]) => {
    // Find closest energy reading
    let closestEnergy: [number, number] | null = null;
    let minDiff = Infinity;

    for (const [energyTime, energyValue] of energyData.data) {
      const diff = Math.abs(energyTime - tempTime);
      if (diff < minDiff && diff <= maxTimeDiff) {
        minDiff = diff;
        closestEnergy = [energyTime, energyValue];
      }
    }

    if (closestEnergy) {
      const date = new Date(tempTime);
      correlatedPoints.push({
        temperature: tempValue,
        energy: closestEnergy[1],
        timestamp: tempTime,
        hour: date.getHours(),
        dayOfWeek: date.getDay(),
      });
    }
  });

  // Calculate regression if requested
  let regression = null;
  if (showRegression && correlatedPoints.length > 2) {
    const n = correlatedPoints.length;
    const sumX = correlatedPoints.reduce((sum, p) => sum + p.temperature, 0);
    const sumY = correlatedPoints.reduce((sum, p) => sum + p.energy, 0);
    const sumXY = correlatedPoints.reduce((sum, p) => sum + p.temperature * p.energy, 0);
    const sumX2 = correlatedPoints.reduce((sum, p) => sum + p.temperature * p.temperature, 0);

    const slope = (n * sumXY - sumX * sumY) / (n * sumX2 - sumX * sumX);
    const intercept = (sumY - slope * sumX) / n;

    regression = { slope, intercept };
  }

  return {
    data: correlatedPoints.map((p) => ({
      value: [p.temperature, p.energy],
      timestamp: p.timestamp,
      ...(colorByTime && {
        itemStyle: {
          color: getTimeBasedColor(p.hour || 0),
        },
      }),
    })),
    regression,
    correlation: calculateCorrelation(correlatedPoints),
    axes: {
      x: { name: temperatureData.point.name, unit: temperatureData.point.unit },
      y: { name: energyData.point.name, unit: energyData.point.unit },
    },
  };
};

/**
 * Psychrometric Chart Transformer
 * Use Case: Analyze zone comfort conditions
 */
export const transformForComfortAnalysis = (
  temperatureData: PointData,
  humidityData: PointData,
  options?: {
    temperatureUnit?: 'C' | 'F';
    showComfortZone?: boolean;
    comfortZone?: {
      tempMin: number;
      tempMax: number;
      humidityMin: number;
      humidityMax: number;
    };
  }
) => {
  const {
    temperatureUnit = 'F',
    showComfortZone = true,
    comfortZone = {
      tempMin: 68,
      tempMax: 78,
      humidityMin: 30,
      humidityMax: 60,
    },
  } = options || {};

  const pairedData: Array<{
    temperature: number;
    humidity: number;
    timestamp: number;
    inComfortZone: boolean;
    dewPoint: number;
    timeOfDay: 'morning' | 'afternoon' | 'evening' | 'night';
  }> = [];

  // Pair temperature and humidity by timestamp
  const maxTimeDiff = 300000; // 5 minutes

  temperatureData.data.forEach(([tempTime, tempValue]) => {
    const closestHumidity = findClosestValue(humidityData.data, tempTime, maxTimeDiff);

    if (closestHumidity) {
      const hour = new Date(tempTime).getHours();
      const timeOfDay =
        hour < 6 ? 'night' : hour < 12 ? 'morning' : hour < 18 ? 'afternoon' : 'evening';

      // Convert temperature if needed
      const temp =
        temperatureUnit === 'C' && tempValue > 50 ? ((tempValue - 32) * 5) / 9 : tempValue;

      // Calculate dew point (simplified)
      const a = 17.27;
      const b = 237.7;
      const alpha = (a * temp) / (b + temp) + Math.log(closestHumidity[1] / 100);
      const dewPoint = (b * alpha) / (a - alpha);

      const inComfortZone =
        showComfortZone &&
        temp >= comfortZone.tempMin &&
        temp <= comfortZone.tempMax &&
        closestHumidity[1] >= comfortZone.humidityMin &&
        closestHumidity[1] <= comfortZone.humidityMax;

      pairedData.push({
        temperature: temp,
        humidity: closestHumidity[1],
        timestamp: tempTime,
        inComfortZone,
        dewPoint,
        timeOfDay,
      });
    }
  });

  return {
    data: pairedData,
    summary: {
      percentInComfort:
        (pairedData.filter((p) => p.inComfortZone).length / pairedData.length) * 100,
      avgTemp: pairedData.reduce((sum, p) => sum + p.temperature, 0) / pairedData.length,
      avgHumidity: pairedData.reduce((sum, p) => sum + p.humidity, 0) / pairedData.length,
    },
    comfortZone,
  };
};

/**
 * Sankey Diagram Transformer
 * Use Case: Energy flow from main meter to sub-meters
 */
export const transformForEnergyFlow = (
  flows: Array<{
    source: PointData;
    targets: PointData[];
    relationship?: 'splits-to' | 'feeds' | 'measures';
  }>,
  options?: {
    aggregation?: 'sum' | 'average';
    timeWindow?: number;
    balanceFlows?: boolean; // Ensure sum of targets equals source
  }
) => {
  const { aggregation = 'sum', timeWindow, balanceFlows = true } = options || {};

  const nodes = new Map<string, { name: string; level: number }>();
  const links: Array<{
    source: string;
    target: string;
    value: number;
    actualValue?: number;
    balanced?: boolean;
  }> = [];

  flows.forEach((flow, level) => {
    // Add source node
    nodes.set(flow.source.point.name, {
      name: flow.source.point.name,
      level: level,
    });

    // Calculate source value
    const sourceValue = aggregateTimeSeriesData(flow.source.data, aggregation, timeWindow);

    // Process targets
    let totalTargetValue = 0;
    const targetValues = flow.targets.map((target) => {
      nodes.set(target.point.name, {
        name: target.point.name,
        level: level + 1,
      });

      const value = aggregateTimeSeriesData(target.data, aggregation, timeWindow);
      totalTargetValue += value;

      return { name: target.point.name, value };
    });

    // Balance flows if requested
    if (balanceFlows && totalTargetValue > 0) {
      const scaleFactor = sourceValue / totalTargetValue;

      targetValues.forEach((target) => {
        links.push({
          source: flow.source.point.name,
          target: target.name,
          value: target.value * scaleFactor,
          actualValue: target.value,
          balanced: true,
        });
      });
    } else {
      targetValues.forEach((target) => {
        links.push({
          source: flow.source.point.name,
          target: target.name,
          value: target.value,
          balanced: false,
        });
      });
    }
  });

  return {
    nodes: Array.from(nodes.values()),
    links,
    totalFlow: links.reduce((sum, link) => sum + link.value, 0),
  };
};

/**
 * Radar Chart Transformer
 * Use Case: Building performance score across multiple KPIs
 */
export const transformForBuildingScore = (
  kpiPoints: Array<{
    point: PointData;
    category: string;
    targetValue?: number;
    weight?: number;
    normalize?: { min: number; max: number };
  }>
) => {
  const indicators = kpiPoints.map((kpi) => {
    const latestValue =
      kpi.point.data.length > 0 ? kpi.point.data[kpi.point.data.length - 1][1] : 0;

    // Normalize to 0-100 scale
    let normalizedValue = latestValue;
    if (kpi.normalize) {
      normalizedValue =
        ((latestValue - kpi.normalize.min) / (kpi.normalize.max - kpi.normalize.min)) * 100;
      normalizedValue = Math.max(0, Math.min(100, normalizedValue));
    }

    // Calculate score vs target
    let score = normalizedValue;
    if (kpi.targetValue) {
      score = (normalizedValue / kpi.targetValue) * 100;
    }

    return {
      name: kpi.category,
      max: 100,
      value: score,
      actualValue: latestValue,
      unit: kpi.point.point.unit,
      weight: kpi.weight || 1,
    };
  });

  // Calculate weighted overall score
  const totalWeight = indicators.reduce((sum, ind) => sum + ind.weight, 0);
  const overallScore =
    indicators.reduce((sum, ind) => sum + ind.value * ind.weight, 0) / totalWeight;

  return {
    indicators: indicators.map(({ name, max }) => ({ name, max })),
    data: [
      {
        value: indicators.map((ind) => ind.value),
        name: 'Current Performance',
      },
    ],
    overallScore,
    details: indicators,
  };
};

// Helper functions
function findClosestValue(
  data: TimeSeriesData,
  targetTime: number,
  maxDiff: number
): [number, number] | null {
  let closest: [number, number] | null = null;
  let minDiff = Infinity;

  for (const point of data) {
    const diff = Math.abs(point[0] - targetTime);
    if (diff < minDiff && diff <= maxDiff) {
      minDiff = diff;
      closest = point;
    }
    if (point[0] > targetTime + maxDiff) {break;}
  }

  return closest;
}

function aggregateTimeSeriesData(
  data: TimeSeriesData,
  method: 'sum' | 'average',
  timeWindow?: number
): number {
  let relevantData = data;

  if (timeWindow) {
    const cutoff = Date.now() - timeWindow;
    relevantData = data.filter(([ts]) => ts >= cutoff);
  }

  if (relevantData.length === 0) {return 0;}

  const values = relevantData.map(([_, v]) => v);

  return method === 'sum'
    ? values.reduce((sum, v) => sum + v, 0)
    : values.reduce((sum, v) => sum + v, 0) / values.length;
}

function calculateCorrelation(points: Array<{ temperature: number; energy: number }>): number {
  const n = points.length;
  if (n < 2) {return 0;}

  const sumX = points.reduce((sum, p) => sum + p.temperature, 0);
  const sumY = points.reduce((sum, p) => sum + p.energy, 0);
  const sumXY = points.reduce((sum, p) => sum + p.temperature * p.energy, 0);
  const sumX2 = points.reduce((sum, p) => sum + p.temperature * p.temperature, 0);
  const sumY2 = points.reduce((sum, p) => sum + p.energy * p.energy, 0);

  const correlation =
    (n * sumXY - sumX * sumY) / Math.sqrt((n * sumX2 - sumX * sumX) * (n * sumY2 - sumY * sumY));

  return correlation;
}

function getTimeBasedColor(hour: number): string {
  // Morning: blue, Afternoon: orange, Evening: purple, Night: dark blue
  if (hour < 6) {return '#1e3a8a';}
  if (hour < 12) {return '#3b82f6';}
  if (hour < 18) {return '#f97316';}
  return '#7c3aed';
}
