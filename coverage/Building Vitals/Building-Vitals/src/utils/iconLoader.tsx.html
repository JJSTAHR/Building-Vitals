// Dynamic icon loader function
export async function loadIcon(iconName: string): Promise<IconComponent> {
  // Check cache first
  if (iconCache.has(iconName)) {
    return iconCache.get(iconName)!;
  }

  try {
    // Use static imports if available
    const importFn = iconImportMap[iconName];
    if (importFn) {
      const module = await importFn();
      const IconComponent = module.default;

      // Cache the loaded icon
      iconCache.set(iconName, IconComponent);

      return IconComponent;
    }

    // Fallback: try dynamic import (may not work in production)
    const module = await import(/* @vite-ignore */ `@mui/icons-material/${iconName}`);
    const IconComponent = module.default || module[iconName];

    // Cache the loaded icon
    iconCache.set(iconName, IconComponent);

    return IconComponent;
  } catch (error) {
    // Return a fallback component
    return (() => null) as IconComponent;
  }
}

// Helper component for lazy-loaded icons with fallback
export function LazyIcon({
  icon: Icon,
  fallback,
  ...props
}: {
  icon: ComponentType<SvgIconProps>;
  fallback?: React.ReactNode;
} & SvgIconProps) {
  const defaultFallback = <CircularProgress size={20} />;
  return (
    <Suspense fallback={fallback || defaultFallback}>
      <Icon {...props} />
    </Suspense>
  );
}

// Preload function for critical icons
export async function preloadCriticalIcons() {
  // Preload core icons used in the initial render
  const criticalIcons = ['Menu', 'Dashboard', 'Settings', 'Search', 'Refresh'];

  await Promise.all(criticalIcons.map((iconName) => loadIcon(iconName)));
}

// Bundle loader for feature-specific icons
export async function loadIconBundle(bundleName: keyof typeof iconBundles) {
  const bundle = iconBundles[bundleName];
  if (bundle) {
    return bundle();
  }
  return {};
}
