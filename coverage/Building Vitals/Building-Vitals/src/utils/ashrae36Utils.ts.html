/**
 * ASHRAE Guideline 36-2021 Utility Functions
 * 
 * This module provides utility functions for implementing ASHRAE Guideline 36-2021
 * requirements including fault detection, trim & respond logic, request levels,
 * and energy calculations for building automation systems.
 */

/**
 * Sample data point with timestamp and value
 */
export interface DataPoint {
  timestamp: number;
  value: number;
}

/**
 * Control signal data for trim & respond calculations
 */
export interface ControlSignal {
  timestamp: number;
  heating: number;
  cooling: number;
  damper: number;
  valve: number;
}

/**
 * Request level thresholds configuration
 */
export interface RequestLevelThresholds {
  level1: number;  // Minor deviation threshold
  level2: number;  // Moderate deviation threshold
  level3: number;  // Critical deviation threshold
}

/**
 * Energy factors for waste calculations
 */
export interface EnergyFactors {
  heatingFactor: number;  // BTU/hr per % signal
  coolingFactor: number;  // BTU/hr per % signal
  electricityRate: number; // $/kWh
  gasRate: number;        // $/therm
}

/**
 * Fault persistence detection result
 */
export interface FaultPersistenceResult {
  isFaultPresent: boolean;
  faultDuration: number;
  consecutiveFaultSamples: number;
  lastFaultStart: number | null;
}

/**
 * Trim & respond calculation result
 */
export interface TrimRespondResult {
  trimAmount: number;
  respondDirection: 'increase' | 'decrease' | 'none';
  averageControlSignal: number;
  recommendedSetpointAdjustment: number;
}

/**
 * Request level calculation result
 */
export interface RequestLevelResult {
  level: 0 | 1 | 2 | 3;
  deviation: number;
  percentDeviation: number;
  description: string;
}

/**
 * Alarm duration check result
 */
export interface AlarmDurationResult {
  isAlarmActive: boolean;
  alarmDuration: number;
  thresholdMet: boolean;
  consecutiveAlarmSamples: number;
}

/**
 * Energy waste calculation result
 */
export interface EnergyWasteResult {
  simultaneousHeatCool: boolean;
  wastePercentage: number;
  estimatedCostPerHour: number;
  heatingEnergy: number;
  coolingEnergy: number;
  wastedEnergy: number;
}

/**
 * Detects fault persistence according to ASHRAE Guideline 36-2021
 * 
 * Per Guideline 36, a fault is considered persistent if it exists continuously
 * for 5 minutes or longer. This function checks if a measured value exceeds
 * a threshold for the required duration.
 * 
 * @param values - Array of data points with timestamp and value
 * @param threshold - Threshold value for fault detection
 * @param duration - Required fault duration in minutes (default: 5 per ASHRAE 36)
 * @param sampleRate - Sample rate in seconds (default: 60 seconds)
 * @param comparison - Comparison operator ('greater', 'less', 'absolute')
 * @returns Fault persistence detection result
 * 
 * @example
 * ```typescript
 * const tempData = [
 *   { timestamp: Date.now() - 600000, value: 75 },
 *   { timestamp: Date.now() - 300000, value: 85 },
 *   { timestamp: Date.now(), value: 90 }
 * ];
 * const fault = detectFaultPersistence(tempData, 80, 5, 60, 'greater');
 * console.log(fault.isFaultPresent); // true if temp > 80°F for 5+ minutes
 * ```
 */
export function detectFaultPersistence(
  values: DataPoint[],
  threshold: number,
  duration: number = 5,
  sampleRate: number = 60,
  comparison: 'greater' | 'less' | 'absolute' = 'greater'
): FaultPersistenceResult {
  if (values.length === 0) {
    return {
      isFaultPresent: false,
      faultDuration: 0,
      consecutiveFaultSamples: 0,
      lastFaultStart: null
    };
  }

  const requiredSamples = Math.ceil((duration * 60) / sampleRate);
  const sortedValues = [...values].sort((a, b) => a.timestamp - b.timestamp);
  
  let consecutiveFaultSamples = 0;
  let maxConsecutiveFaultSamples = 0;
  let faultStartTime: number | null = null;
  let lastFaultStart: number | null = null;
  
  for (let i = 0; i < sortedValues.length; i++) {
    const point = sortedValues[i];
    let isFault = false;
    
    switch (comparison) {
      case 'greater':
        isFault = point.value > threshold;
        break;
      case 'less':
        isFault = point.value < threshold;
        break;
      case 'absolute':
        isFault = Math.abs(point.value) > threshold;
        break;
    }
    
    if (isFault) {
      if (consecutiveFaultSamples === 0) {
        faultStartTime = point.timestamp;
      }
      consecutiveFaultSamples++;
      maxConsecutiveFaultSamples = Math.max(maxConsecutiveFaultSamples, consecutiveFaultSamples);
    } else {
      if (consecutiveFaultSamples > 0) {
        lastFaultStart = faultStartTime;
      }
      consecutiveFaultSamples = 0;
      faultStartTime = null;
    }
  }
  
  const currentFaultDuration = faultStartTime 
    ? (sortedValues[sortedValues.length - 1].timestamp - faultStartTime) / (60 * 1000)
    : 0;
  
  return {
    isFaultPresent: maxConsecutiveFaultSamples >= requiredSamples,
    faultDuration: currentFaultDuration,
    consecutiveFaultSamples: maxConsecutiveFaultSamples,
    lastFaultStart: lastFaultStart || faultStartTime
  };
}

/**
 * Calculates Trim & Respond logic for setpoint optimization
 * 
 * Per ASHRAE Guideline 36, Trim & Respond logic adjusts setpoints based on
 * control signal positions to optimize energy efficiency while maintaining comfort.
 * 
 * @param controlSignals - Array of control signal data points
 * @param targetPosition - Target control signal position (0-100%)
 * @param responseTime - Response time window in minutes (default: 15)
 * @param trimAmount - Amount to trim setpoint (default: 0.5°F)
 * @returns Trim & respond calculation result
 * 
 * @example
 * ```typescript
 * const signals = [
 *   { timestamp: Date.now() - 900000, heating: 85, cooling: 0, damper: 50, valve: 80 },
 *   { timestamp: Date.now() - 600000, heating: 90, cooling: 0, damper: 55, valve: 85 },
 *   { timestamp: Date.now() - 300000, heating: 95, cooling: 0, damper: 60, valve: 90 }
 * ];
 * const result = calculateTrimRespond(signals, 70, 15);
 * ```
 */
export function calculateTrimRespond(
  controlSignals: ControlSignal[],
  targetPosition: number = 70,
  responseTime: number = 15,
  trimAmount: number = 0.5
): TrimRespondResult {
  if (controlSignals.length === 0) {
    return {
      trimAmount: 0,
      respondDirection: 'none',
      averageControlSignal: 0,
      recommendedSetpointAdjustment: 0
    };
  }
  
  const now = Date.now();
  const timeWindow = responseTime * 60 * 1000;
  const recentSignals = controlSignals.filter(
    signal => (now - signal.timestamp) <= timeWindow
  );
  
  if (recentSignals.length === 0) {
    return {
      trimAmount: 0,
      respondDirection: 'none',
      averageControlSignal: 0,
      recommendedSetpointAdjustment: 0
    };
  }
  
  // Calculate average control signal (use heating + cooling as primary indicator)
  const averageControlSignal = recentSignals.reduce((sum, signal) => {
    return sum + Math.max(signal.heating, signal.cooling);
  }, 0) / recentSignals.length;
  
  let respondDirection: 'increase' | 'decrease' | 'none' = 'none';
  let recommendedSetpointAdjustment = 0;
  
  // Determine trim & respond action based on average control signal
  if (averageControlSignal > targetPosition + 10) {
    // Control signal too high - increase setpoint to reduce load
    respondDirection = 'increase';
    recommendedSetpointAdjustment = trimAmount;
  } else if (averageControlSignal < targetPosition - 10) {
    // Control signal too low - decrease setpoint to increase load
    respondDirection = 'decrease';
    recommendedSetpointAdjustment = -trimAmount;
  }
  
  return {
    trimAmount,
    respondDirection,
    averageControlSignal,
    recommendedSetpointAdjustment
  };
}

/**
 * Calculates request level based on deviation from setpoint
 * 
 * ASHRAE Guideline 36 defines 4 request levels (0-3) based on the severity
 * of deviation from setpoint, used for demand-controlled systems.
 * 
 * @param measurement - Current measured value
 * @param setpoint - Target setpoint value
 * @param thresholds - Threshold configuration for each level
 * @returns Request level calculation result
 * 
 * @example
 * ```typescript
 * const thresholds = { level1: 1.0, level2: 2.0, level3: 3.0 };
 * const result = calculateRequestLevel(75.5, 72.0, thresholds);
 * console.log(result.level); // 3 (critical deviation)
 * ```
 */
export function calculateRequestLevel(
  measurement: number,
  setpoint: number,
  thresholds: RequestLevelThresholds
): RequestLevelResult {
  const deviation = Math.abs(measurement - setpoint);
  const percentDeviation = setpoint !== 0 ? (deviation / Math.abs(setpoint)) * 100 : 0;
  
  let level: 0 | 1 | 2 | 3 = 0;
  let description = 'Normal operation - no request';
  
  if (deviation >= thresholds.level3) {
    level = 3;
    description = 'Critical deviation - maximum request';
  } else if (deviation >= thresholds.level2) {
    level = 2;
    description = 'Moderate deviation - high request';
  } else if (deviation >= thresholds.level1) {
    level = 1;
    description = 'Minor deviation - low request';
  }
  
  return {
    level,
    deviation,
    percentDeviation,
    description
  };
}

/**
 * Checks if alarm condition has persisted for required duration
 * 
 * Similar to fault persistence but specifically for alarm conditions
 * that require time-based validation before triggering notifications.
 * 
 * @param condition - Array of boolean values indicating alarm condition
 * @param requiredMinutes - Required duration in minutes for alarm activation
 * @param samples - Array of sample timestamps corresponding to conditions
 * @param sampleRate - Sample rate in seconds (default: 60)
 * @returns Alarm duration check result
 * 
 * @example
 * ```typescript
 * const conditions = [true, true, true, false, true, true];
 * const timestamps = [1000, 2000, 3000, 4000, 5000, 6000];
 * const result = checkAlarmDuration(conditions, 3, timestamps, 60);
 * ```
 */
export function checkAlarmDuration(
  condition: boolean[],
  requiredMinutes: number,
  samples: number[],
  sampleRate: number = 60
): AlarmDurationResult {
  if (condition.length === 0 || condition.length !== samples.length) {
    return {
      isAlarmActive: false,
      alarmDuration: 0,
      thresholdMet: false,
      consecutiveAlarmSamples: 0
    };
  }
  
  const requiredSamples = Math.ceil((requiredMinutes * 60) / sampleRate);
  let consecutiveAlarmSamples = 0;
  let maxConsecutiveAlarmSamples = 0;
  let alarmStartTime: number | null = null;
  
  for (let i = 0; i < condition.length; i++) {
    if (condition[i]) {
      if (consecutiveAlarmSamples === 0) {
        alarmStartTime = samples[i];
      }
      consecutiveAlarmSamples++;
      maxConsecutiveAlarmSamples = Math.max(maxConsecutiveAlarmSamples, consecutiveAlarmSamples);
    } else {
      consecutiveAlarmSamples = 0;
      alarmStartTime = null;
    }
  }
  
  const currentAlarmDuration = alarmStartTime 
    ? (samples[samples.length - 1] - alarmStartTime) / (60 * 1000)
    : 0;
  
  return {
    isAlarmActive: consecutiveAlarmSamples > 0,
    alarmDuration: currentAlarmDuration,
    thresholdMet: maxConsecutiveAlarmSamples >= requiredSamples,
    consecutiveAlarmSamples: maxConsecutiveAlarmSamples
  };
}

/**
 * Calculates energy waste from simultaneous heating and cooling
 * 
 * Detects and quantifies energy waste when heating and cooling systems
 * operate simultaneously, a common fault in HVAC systems.
 * 
 * @param heatingSignal - Heating control signal (0-100%)
 * @param coolingSignal - Cooling control signal (0-100%)
 * @param energyFactors - Energy conversion factors and rates
 * @param simultaneousThreshold - Minimum signal level to consider simultaneous (default: 10%)
 * @returns Energy waste calculation result
 * 
 * @example
 * ```typescript
 * const energyFactors = {
 *   heatingFactor: 1000,    // BTU/hr per %
 *   coolingFactor: 1200,    // BTU/hr per %
 *   electricityRate: 0.12,  // $/kWh
 *   gasRate: 1.50          // $/therm
 * };
 * const waste = calculateEnergyWaste(75, 45, energyFactors);
 * console.log(waste.estimatedCostPerHour); // Cost of wasted energy
 * ```
 */
export function calculateEnergyWaste(
  heatingSignal: number,
  coolingSignal: number,
  energyFactors: EnergyFactors,
  simultaneousThreshold: number = 10
): EnergyWasteResult {
  const simultaneousHeatCool = heatingSignal >= simultaneousThreshold && 
                               coolingSignal >= simultaneousThreshold;
  
  if (!simultaneousHeatCool) {
    return {
      simultaneousHeatCool: false,
      wastePercentage: 0,
      estimatedCostPerHour: 0,
      heatingEnergy: 0,
      coolingEnergy: 0,
      wastedEnergy: 0
    };
  }
  
  // Calculate energy consumption
  const heatingEnergy = (heatingSignal / 100) * energyFactors.heatingFactor; // BTU/hr
  const coolingEnergy = (coolingSignal / 100) * energyFactors.coolingFactor; // BTU/hr
  
  // Wasted energy is the smaller of the two signals
  const wastedSignal = Math.min(heatingSignal, coolingSignal);
  const wastePercentage = wastedSignal;
  
  // Calculate wasted energy in BTU/hr
  const wastedHeatingEnergy = (wastedSignal / 100) * energyFactors.heatingFactor;
  const wastedCoolingEnergy = (wastedSignal / 100) * energyFactors.coolingFactor;
  const wastedEnergy = wastedHeatingEnergy + wastedCoolingEnergy;
  
  // Convert to cost per hour
  // Assuming heating is gas (100,000 BTU = 1 therm) and cooling is electric (3412 BTU = 1 kWh)
  const wastedGasCost = (wastedHeatingEnergy / 100000) * energyFactors.gasRate;
  const wastedElectricCost = (wastedCoolingEnergy / 3412) * energyFactors.electricityRate;
  const estimatedCostPerHour = wastedGasCost + wastedElectricCost;
  
  return {
    simultaneousHeatCool,
    wastePercentage,
    estimatedCostPerHour,
    heatingEnergy,
    coolingEnergy,
    wastedEnergy
  };
}

/**
 * Utility function to convert Fahrenheit to Celsius
 */
export function fahrenheitToCelsius(fahrenheit: number): number {
  return (fahrenheit - 32) * (5 / 9);
}

/**
 * Utility function to convert Celsius to Fahrenheit
 */
export function celsiusToFahrenheit(celsius: number): number {
  return (celsius * (9 / 5)) + 32;
}

 * Utility function to convert PSI to kPa

  return psi * 6.89476;
}

/**
 * Utility function to convert kPa to PSI

export function kpaToPsi(kpa: number): number {

/**
 * Validates that a value is within acceptable range for HVAC systems
 */
export function validateHvacRange(value: number, min: number, max: number): boolean {
  return value >= min && value <= max && !isNaN(value) && isFinite(value);
}

/**
 * Calculates moving average for smoothing noisy sensor data
 */
export function calculateMovingAverage(values: number[], windowSize: number): number[] {
  if (windowSize <= 0 || values.length === 0) return [];
  
  const result: number[] = [];
  for (let i = 0; i < values.length; i++) {
    const start = Math.max(0, i - Math.floor(windowSize / 2));
    const end = Math.min(values.length, start + windowSize);
    const window = values.slice(start, end);
    const average = window.reduce((sum, val) => sum + val, 0) / window.length;
    result.push(average);
  }
  
  return result;
}
