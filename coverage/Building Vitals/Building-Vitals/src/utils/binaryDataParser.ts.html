/**
 * Binary Data Parser for High-Resolution Time Series
 * Handles conversion between JSON and binary formats for efficient data transfer
 */

export interface BinaryHeader {
  version: number; // Format version (1 byte)
  pointCount: number; // Total points (4 bytes)
  startTime: number; // First timestamp (4 bytes)
  endTime: number; // Last timestamp (4 bytes)
  checksum: number; // Data integrity (1 byte)
}

export interface TimeSeriesPoint {
  timestamp: number;
  value: number;
  point_name?: string;
}

export interface BinaryTimeSeriesData {
  header: BinaryHeader;
  points: TimeSeriesPoint[];
}

/**
 * Convert JSON timeseries data to binary ArrayBuffer
 * Format: [header(14 bytes)] + [points(8 bytes each)]
 */
export function jsonToBinary(data: TimeSeriesPoint[]): ArrayBuffer {
  if (!data || data.length === 0) {
    return new ArrayBuffer(0);
  }

  // Sort by timestamp for consistency
  const sortedData = [...data].sort((a, b) => a.timestamp - b.timestamp);

  // Calculate buffer size: header (14 bytes) + points (8 bytes each)
  const headerSize = 14;
  const pointSize = 8;
  const totalSize = headerSize + sortedData.length * pointSize;

  const buffer = new ArrayBuffer(totalSize);
  const view = new DataView(buffer);

  // Write header
  let offset = 0;

  // Version (1 byte)
  view.setUint8(offset, 1);
  offset += 1;

  // Point count (4 bytes)
  view.setUint32(offset, sortedData.length, true); // little-endian
  offset += 4;

  // Start time (4 bytes) - Unix timestamp in seconds
  const startTime = Math.floor(sortedData[0].timestamp / 1000);
  view.setUint32(offset, startTime, true);
  offset += 4;

  // End time (4 bytes) - Unix timestamp in seconds
  const endTime = Math.floor(sortedData[sortedData.length - 1].timestamp / 1000);
  view.setUint32(offset, endTime, true);
  offset += 4;

  // Calculate checksum (simple XOR of all values)
  let checksum = 0;
  sortedData.forEach((point) => {
    checksum ^= Math.floor(point.value);
  });
  view.setUint8(offset, checksum & 0xff);
  offset += 1;

  // Write points
  sortedData.forEach((point) => {
    // Timestamp (4 bytes) - Unix timestamp in seconds
    view.setUint32(offset, Math.floor(point.timestamp / 1000), true);
    offset += 4;

    // Value (4 bytes) - float32
    view.setFloat32(offset, point.value, true);
    offset += 4;
  });

  return buffer;
}

/**
 * Parse binary ArrayBuffer back to JSON timeseries data
 */
export function binaryToJson(buffer: ArrayBuffer): BinaryTimeSeriesData {
  if (!buffer || buffer.byteLength === 0) {
    return {
      header: {
        version: 0,
        pointCount: 0,
        startTime: 0,
        endTime: 0,
        checksum: 0,
      },
      points: [],
    };
  }

  const view = new DataView(buffer);
  let offset = 0;

  // Read header
  const version = view.getUint8(offset);
  offset += 1;

  const pointCount = view.getUint32(offset, true);
  offset += 4;

  const startTime = view.getUint32(offset, true);
  offset += 4;

  const endTime = view.getUint32(offset, true);
  offset += 4;

  const checksum = view.getUint8(offset);
  offset += 1;

  // Read points
  const points: TimeSeriesPoint[] = [];
  for (let i = 0; i < pointCount; i++) {
    const timestamp = view.getUint32(offset, true) * 1000; // Convert back to milliseconds
    offset += 4;

    const value = view.getFloat32(offset, true);
    offset += 4;

    points.push({ timestamp, value });
  }

  // Verify checksum
  let calculatedChecksum = 0;
  points.forEach((point) => {
    calculatedChecksum ^= Math.floor(point.value);
  });

  if ((calculatedChecksum & 0xff) !== checksum) {
    console.warn('Binary data checksum mismatch - data may be corrupted');
  }

  return {
    header: {
      version,
      pointCount,
      startTime: startTime * 1000, // Convert to milliseconds
      endTime: endTime * 1000,
      checksum,
    },
    points,
  };
}

/**
 * Calculate size reduction from JSON to binary
 */
export function calculateSizeReduction(jsonData: TimeSeriesPoint[]): {
  jsonSize: number;
  binarySize: number;
  reduction: number;
  reductionPercent: number;
} {
  const jsonStr = JSON.stringify(jsonData);
  const jsonSize = new TextEncoder().encode(jsonStr).length;

  const binarySize = 14 + jsonData.length * 8; // header + points

  const reduction = jsonSize - binarySize;
  const reductionPercent = (reduction / jsonSize) * 100;

  return {
    jsonSize,
    binarySize,
    reduction,
    reductionPercent,
  };
}

/**
 * Create a streaming parser for large datasets
 */
export class BinaryStreamParser {
  private buffer: ArrayBuffer;
  private offset: number = 0;
  private header: BinaryHeader | null = null;

  constructor(buffer: ArrayBuffer) {
    this.buffer = buffer;
    this.parseHeader();
  }

  private parseHeader(): void {
    if (this.buffer.byteLength < 14) {
      throw new Error('Buffer too small to contain header');
    }

    const view = new DataView(this.buffer);

    this.header = {
      version: view.getUint8(0),
      pointCount: view.getUint32(1, true),
      startTime: view.getUint32(5, true) * 1000,
      endTime: view.getUint32(9, true) * 1000,
      checksum: view.getUint8(13),
    };

    this.offset = 14; // Start after header
  }

  getHeader(): BinaryHeader | null {
    return this.header;
  }

  /**
   * Read next batch of points
   */
  readPoints(count: number): TimeSeriesPoint[] {
    if (!this.header) {
      throw new Error('Header not parsed');
    }

    const points: TimeSeriesPoint[] = [];
    const view = new DataView(this.buffer);
    const maxPoints = Math.min(count, this.header.pointCount - (this.offset - 14) / 8);

    for (let i = 0; i < maxPoints; i++) {
      if (this.offset + 8 > this.buffer.byteLength) {
        break;
      }

      const timestamp = view.getUint32(this.offset, true) * 1000;
      this.offset += 4;

      const value = view.getFloat32(this.offset, true);
      this.offset += 4;

      points.push({ timestamp, value });
    }

    return points;
  }

  hasMorePoints(): boolean {
    return this.offset < this.buffer.byteLength;
  }

  reset(): void {
    this.offset = 14;
  }
}

/**
 * Utility to merge multiple binary buffers (for multi-point responses)
 */
export function mergeBinaryBuffers(buffers: ArrayBuffer[]): ArrayBuffer {
  // Calculate total points
  let totalPoints = 0;
  const headers: BinaryHeader[] = [];

  buffers.forEach((buffer) => {
    const data = binaryToJson(buffer);
    headers.push(data.header);
    totalPoints += data.header.pointCount;
  });

  // Find overall time range
  const startTime = Math.min(...headers.map((h) => h.startTime));
  const endTime = Math.max(...headers.map((h) => h.endTime));

  // Merge all points
  const allPoints: TimeSeriesPoint[] = [];
  buffers.forEach((buffer) => {
    const data = binaryToJson(buffer);
    allPoints.push(...data.points);
  });

  // Sort by timestamp
  allPoints.sort((a, b) => a.timestamp - b.timestamp);

  // Convert back to binary
  return jsonToBinary(allPoints);
}
