/**
 * Common data transformation utilities for Building Vitals
 * This module consolidates all data transformation functions used across the application
 */

import { mean, median, standardDeviation, min, max, sum } from 'simple-statistics';

// Re-export existing chart data transformers
export * from './chartDataTransformers';

/**
 * Generic data point interface
 */
export interface DataPoint {
  timestamp: number | string | Date;
  value: number;
  [key: string]: any;
}

/**
 * Transform raw API data to chart-ready format
 */
export function transformToChartData(
  data: DataPoint[],
  options: {
    xField?: string;
    yField?: string;
    sortByX?: boolean;
    fillGaps?: boolean;
    gapValue?: number;
  } = {}
): Array<[number, number | null]> {
  const {
    xField = 'timestamp',
    yField = 'value',
    sortByX = true,
    fillGaps = false,
    gapValue = 0,
  } = options;

  const chartData = data.map((point) => {
    const x =
      typeof point[xField] === 'string' || point[xField] instanceof Date
        ? new Date(point[xField]).getTime()
        : point[xField];
    const y = point[yField] ?? gapValue;
    return [x, y] as [number, number];
  });

  if (sortByX) {
    chartData.sort((a, b) => a[0] - b[0]);
  }

  if (fillGaps && chartData.length > 1) {
    const filledData: Array<[number, number]> = [];
    for (let i = 0; i < chartData.length - 1; i++) {
      filledData.push(chartData[i]);

      const currentTime = chartData[i][0];
      const nextTime = chartData[i + 1][0];
      const timeDiff = nextTime - currentTime;

      // Fill gaps if time difference is too large (customize threshold as needed)
      const threshold = 60 * 60 * 1000; // 1 hour
      if (timeDiff > threshold) {
        const steps = Math.floor(timeDiff / threshold);
        for (let j = 1; j < steps; j++) {
          filledData.push([currentTime + j * threshold, gapValue]);
        }
      }
    }
    filledData.push(chartData[chartData.length - 1]);
    return filledData;
  }

  return chartData;
}

/**
 * Aggregate data points by time buckets
 */
export function aggregateByTimeBucket(
  data: DataPoint[],
  bucketSize: number, // in milliseconds
  aggregationFn: 'mean' | 'sum' | 'min' | 'max' | 'count' = 'mean'
): DataPoint[] {
  if (data.length === 0) {return [];}

  const buckets = new Map<number, number[]>();

  data.forEach((point) => {
    const timestamp =
      typeof point.timestamp === 'string' || point.timestamp instanceof Date
        ? new Date(point.timestamp).getTime()
        : point.timestamp;

    const bucketKey = Math.floor(timestamp / bucketSize) * bucketSize;

    if (!buckets.has(bucketKey)) {
      buckets.set(bucketKey, []);
    }
    buckets.get(bucketKey)!.push(point.value);
  });

  const aggregatedData: DataPoint[] = [];

  buckets.forEach((values, bucketKey) => {
    let aggregatedValue: number;

    switch (aggregationFn) {
      case 'mean':
        aggregatedValue = mean(values);
        break;
      case 'sum':
        aggregatedValue = sum(values);
        break;
      case 'min':
        aggregatedValue = min(values);
        break;
      case 'max':
        aggregatedValue = max(values);
        break;
      case 'count':
        aggregatedValue = values.length;
        break;
    }

    aggregatedData.push({
      timestamp: bucketKey,
      value: aggregatedValue,
      count: values.length,
    });
  });

  return aggregatedData.sort(
    (a, b) =>
      (typeof a.timestamp === 'number' ? a.timestamp : 0) -
      (typeof b.timestamp === 'number' ? b.timestamp : 0)
  );
}

/**
 * Calculate statistics for a dataset
 */
export function calculateStatistics(values: number[]): {
  count: number;
  mean: number;
  median: number;
  min: number;
  max: number;
  sum: number;
  stdDev: number;
  variance: number;
  percentiles: {
    p25: number;
    p75: number;
    p90: number;
    p95: number;
    p99: number;
  };
} {
  if (values.length === 0) {
    return {
      count: 0,
      mean: 0,
      median: 0,
      min: 0,
      max: 0,
      sum: 0,
      stdDev: 0,
      variance: 0,
      percentiles: {
        p25: 0,
        p75: 0,
        p90: 0,
        p95: 0,
        p99: 0,
      },
    };
  }

  const sortedValues = [...values].sort((a, b) => a - b);
  const n = values.length;

  return {
    count: n,
    mean: mean(values),
    median: median(sortedValues),
    min: min(values),
    max: max(values),
    sum: sum(values),
    stdDev: n > 1 ? standardDeviation(values) : 0,
    variance: n > 1 ? Math.pow(standardDeviation(values), 2) : 0,
    percentiles: {
      p25: sortedValues[Math.floor(n * 0.25)],
      p75: sortedValues[Math.floor(n * 0.75)],
      p90: sortedValues[Math.floor(n * 0.9)],
      p95: sortedValues[Math.floor(n * 0.95)],
      p99: sortedValues[Math.floor(n * 0.99)],
    },
  };
}

/**
 * Normalize values to 0-1 range
 */
export function normalizeValues(values: number[]): number[] {
  if (values.length === 0) {return [];}

  const minVal = min(values);
  const maxVal = max(values);
  const range = maxVal - minVal;

  if (range === 0) {
    return values.map(() => 0.5);
  }

  return values.map((v) => (v - minVal) / range);
}

/**
 * Apply moving average smoothing
 */
export function movingAverage(data: DataPoint[], windowSize: number): DataPoint[] {
  if (data.length === 0 || windowSize <= 0) {return data;}

  const result: DataPoint[] = [];

  for (let i = 0; i < data.length; i++) {
    const startIdx = Math.max(0, i - Math.floor(windowSize / 2));
    const endIdx = Math.min(data.length, i + Math.floor(windowSize / 2) + 1);
    const window = data.slice(startIdx, endIdx);

    const avgValue = mean(window.map((d) => d.value));

    result.push({
      ...data[i],
      value: avgValue,
      originalValue: data[i].value,
    });
  }

  return result;
}

/**
 * Detect outliers using IQR method
 */
export function detectOutliers(
  values: number[],
  multiplier: number = 1.5
): {
  outliers: number[];
  lowerBound: number;
  upperBound: number;
} {
  if (values.length < 4) {
    return { outliers: [], lowerBound: 0, upperBound: 0 };
  }

  const sortedValues = [...values].sort((a, b) => a - b);
  const q1 = sortedValues[Math.floor(sortedValues.length * 0.25)];
  const q3 = sortedValues[Math.floor(sortedValues.length * 0.75)];
  const iqr = q3 - q1;

  const lowerBound = q1 - multiplier * iqr;
  const upperBound = q3 + multiplier * iqr;

  const outliers = values.filter((v) => v < lowerBound || v > upperBound);

  return { outliers, lowerBound, upperBound };
}

/**
 * Resample time series data to regular intervals
 */
export function resampleTimeSeries(
  data: DataPoint[],
  interval: number,
  interpolation: 'linear' | 'nearest' | 'none' = 'linear'
): DataPoint[] {
  if (data.length < 2) {return data;}

  const sortedData = [...data].sort((a, b) => {
    const timeA = typeof a.timestamp === 'number' ? a.timestamp : new Date(a.timestamp).getTime();
    const timeB = typeof b.timestamp === 'number' ? b.timestamp : new Date(b.timestamp).getTime();
    return timeA - timeB;
  });

  const startTime: number =
    typeof sortedData[0].timestamp === 'number'
      ? sortedData[0].timestamp
      : new Date(sortedData[0].timestamp).getTime();
  const endTime: number =
    typeof sortedData[sortedData.length - 1].timestamp === 'number'
      ? sortedData[sortedData.length - 1].timestamp
      : new Date(sortedData[sortedData.length - 1].timestamp).getTime();

  const resampled: DataPoint[] = [];
  let dataIndex = 0;

  for (let time = startTime; time <= endTime; time += interval) {
    // Find surrounding data points
    while (dataIndex < sortedData.length - 1) {
      const nextTime: number =
        typeof sortedData[dataIndex + 1].timestamp === 'number'
          ? sortedData[dataIndex + 1].timestamp
          : new Date(sortedData[dataIndex + 1].timestamp).getTime();

      if (nextTime > time) {break;}
      dataIndex++;
    }

    let value: number;

    if (interpolation === 'none') {
      // Skip if no exact match
      const currentTime: number =
        typeof sortedData[dataIndex].timestamp === 'number'
          ? sortedData[dataIndex].timestamp
          : new Date(sortedData[dataIndex].timestamp).getTime();

      if (Math.abs(currentTime - time) > interval / 2) {continue;}
      value = sortedData[dataIndex].value;
    } else if (interpolation === 'nearest') {
      // Use nearest point
      if (dataIndex < sortedData.length - 1) {
        const currentTime: number =
          typeof sortedData[dataIndex].timestamp === 'number'
            ? sortedData[dataIndex].timestamp
            : new Date(sortedData[dataIndex].timestamp).getTime();
        const nextTime: number =
          typeof sortedData[dataIndex + 1].timestamp === 'number'
            ? sortedData[dataIndex + 1].timestamp
            : new Date(sortedData[dataIndex + 1].timestamp).getTime();

        if (Math.abs(time - currentTime) <= Math.abs(time - nextTime)) {
          value = sortedData[dataIndex].value;
        } else {
          value = sortedData[dataIndex + 1].value;
        }
      } else {
        value = sortedData[dataIndex].value;
      }
    } else {
      // Linear interpolation
      if (dataIndex < sortedData.length - 1) {
        const currentTime: number =
          typeof sortedData[dataIndex].timestamp === 'number'
            ? sortedData[dataIndex].timestamp
            : new Date(sortedData[dataIndex].timestamp).getTime();
        const nextTime: number =
          typeof sortedData[dataIndex + 1].timestamp === 'number'
            ? sortedData[dataIndex + 1].timestamp
            : new Date(sortedData[dataIndex + 1].timestamp).getTime();

        const ratio = (time - currentTime) / (nextTime - currentTime);
        value =
          sortedData[dataIndex].value +
          ratio * (sortedData[dataIndex + 1].value - sortedData[dataIndex].value);
      } else {
        value = sortedData[dataIndex].value;
      }
    }

    resampled.push({
      timestamp: time,
      value,
    });
  }

  return resampled;
}

/**
 * Group data by a key and aggregate values
 */
export function groupByAndAggregate<T extends Record<string, any>>(
  data: T[],
  groupKey: keyof T,
  valueKey: keyof T,
  aggregationFn: 'sum' | 'mean' | 'count' | 'min' | 'max' = 'sum'
): Array<{ group: any; value: number; items: T[] }> {
  const groups = new Map<any, T[]>();

  data.forEach((item) => {
    const key = item[groupKey];
    if (!groups.has(key)) {
      groups.set(key, []);
    }
    groups.get(key)!.push(item);
  });

  const result: Array<{ group: any; value: number; items: T[] }> = [];

  groups.forEach((items, group) => {
    const values = items.map((item) => Number(item[valueKey]) || 0);
    let aggregatedValue: number;

    switch (aggregationFn) {
      case 'sum':
        aggregatedValue = sum(values);
        break;
      case 'mean':
        aggregatedValue = mean(values);
        break;
      case 'count':
        aggregatedValue = items.length;
        break;
      case 'min':
        aggregatedValue = min(values);
        break;
      case 'max':
        aggregatedValue = max(values);
        break;
    }

    result.push({
      group,
      value: aggregatedValue,
      items,
    });
  });

  return result;
}

/**
 * Calculate rate of change between consecutive points
 */
export function calculateRateOfChange(data: DataPoint[]): DataPoint[] {
  if (data.length < 2) {return [];}

  const result: DataPoint[] = [];

  for (let i = 1; i < data.length; i++) {
    const prevTime: number =
      typeof data[i - 1].timestamp === 'number'
        ? data[i - 1].timestamp
        : new Date(data[i - 1].timestamp).getTime();
    const currTime: number =
      typeof data[i].timestamp === 'number'
        ? data[i].timestamp
        : new Date(data[i].timestamp).getTime();

    const timeDiff = (currTime - prevTime) / 1000; // Convert to seconds
    const valueDiff = data[i].value - data[i - 1].value;
    const rate = timeDiff > 0 ? valueDiff / timeDiff : 0;

    result.push({
      timestamp: data[i].timestamp,
      value: rate,
      originalValue: data[i].value,
    });
  }

  return result;
}
