/**
 * Pure utility functions for deviation calculations
 * No React dependencies, completely testable
 */

export interface DeviationConfig {
  mode: 'target' | 'range';
  targetValue?: number;
  minValue?: number;
  maxValue?: number;
  tolerance?: number;
  unit?: string;
}

export interface DeviationResult {
  deviation: number;
  normalized: number;
  status: 'on-target' | 'normal' | 'warning' | 'critical';
  color: string;
  percentage: number;
}

/**
 * Calculate deviation from target value
 */
export function calculateTargetDeviation(
  value: number,
  targetValue: number,
  tolerance: number = 0.1
): DeviationResult {
  const deviation = (value - targetValue) / Math.abs(targetValue || 1);
  const absDeviation = Math.abs(deviation);
  const percentage = deviation * 100;
  
  let status: DeviationResult['status'];
  let color: string;
  let normalized: number;
  
  if (absDeviation < tolerance * 0.5) {
    status = 'on-target';
    color = '#4caf50';
    normalized = absDeviation / tolerance * 0.2;
  } else if (absDeviation < tolerance) {
    status = 'normal';
    color = '#8bc34a';
    normalized = 0.2 + (absDeviation - tolerance * 0.5) / (tolerance * 0.5) * 0.3;
  } else if (absDeviation < tolerance * 3) {
    status = 'warning';
    color = '#ff9800';
    normalized = 0.5 + (absDeviation - tolerance) / (tolerance * 2) * 0.3;
  } else {
    status = 'critical';
    color = '#f44336';
    normalized = Math.min(1, 0.8 + (absDeviation - tolerance * 3) / (tolerance * 3) * 0.2);
  }
  
  return {
    deviation,
    normalized: deviation < 0 ? -normalized : normalized,
    status,
    color,
    percentage
  };
}

/**
 * Calculate deviation from range
 */
export function calculateRangeDeviation(
  value: number,
  minValue: number,
  maxValue: number,
  tolerance: number = 0.1
): DeviationResult {
  const range = maxValue - minValue;
  const center = (minValue + maxValue) / 2;
  
  let deviation: number;
  let status: DeviationResult['status'];
  let color: string;
  let normalized: number;
  
  if (value < minValue) {
    // Below range
    deviation = (minValue - value) / range;
    const severity = deviation / tolerance;
    
    if (severity < 1) {
      status = 'warning';
      color = '#ff9800';
      normalized = -0.5 - severity * 0.3;
    } else {
      status = 'critical';
      color = '#f44336';
      normalized = -0.8 - Math.min(0.2, (severity - 1) * 0.1);
    }
  } else if (value > maxValue) {
    // Above range
    deviation = (value - maxValue) / range;
    const severity = deviation / tolerance;
    
    if (severity < 1) {
      status = 'warning';
      color = '#ff9800';
      normalized = 0.5 + severity * 0.3;
    } else {
      status = 'critical';
      color = '#f44336';
      normalized = 0.8 + Math.min(0.2, (severity - 1) * 0.1);
    }
  } else {
    // Within range
    const distanceFromCenter = Math.abs(value - center);
    const maxDistance = range / 2;
    deviation = (value - center) / range;
    normalized = distanceFromCenter / maxDistance * 0.3;
    
    if (distanceFromCenter < maxDistance * 0.2) {
      status = 'on-target';
      color = '#4caf50';
    } else if (distanceFromCenter < maxDistance * 0.6) {
      status = 'normal';
      color = '#8bc34a';
    } else {
      status = 'warning';
      color = '#ffeb3b';
    }
    
    normalized = value > center ? normalized : -normalized;
  }
  
  const percentage = deviation * 100;
  
  return {
    deviation,
    normalized,
    status,
    color,
    percentage
  };
}

/**
 * Main deviation calculation function
 */
export function calculateDeviation(
  value: number,
  config: DeviationConfig
): DeviationResult {
  if (config.mode === 'target' && config.targetValue !== undefined) {
    return calculateTargetDeviation(
      value,
      config.targetValue,
      config.tolerance
    );
  } else if (config.mode === 'range' && config.minValue !== undefined && config.maxValue !== undefined) {
    return calculateRangeDeviation(
      value,
      config.minValue,
      config.maxValue,
      config.tolerance
    );
  }
  
  // Fallback for invalid config
  return {
    deviation: 0,
    normalized: 0,
    status: 'normal',
    color: '#9e9e9e',
    percentage: 0
  };
}

/**
 * Generate gradient colors for heatmap visualization with ultra-smooth transitions
 * Uses continuous cubic Hermite spline interpolation for perfectly smooth gradients
 */
export function generateHeatmapGradient(steps: number = 100): string[] {
  const colors: string[] = [];
  
  // Define key color points with values and RGB colors
  // Using more intermediate points for smoother transitions
  const keyPoints = [
    { value: -1.0, r: 0, g: 20, b: 255 },      // Deep blue (critical negative)
    { value: -0.8, r: 0, g: 50, b: 255 },      // Blue
    { value: -0.6, r: 20, g: 100, b: 230 },    // Medium blue
    { value: -0.4, r: 40, g: 140, b: 200 },    // Light blue
    { value: -0.25, r: 60, g: 170, b: 170 },   // Cyan-blue
    { value: -0.15, r: 76, g: 190, b: 140 },   // Cyan
    { value: -0.08, r: 86, g: 200, b: 110 },   // Cyan-green
    { value: -0.04, r: 90, g: 210, b: 90 },    // Light green-cyan
    { value: -0.02, r: 92, g: 220, b: 70 },    // Near perfect green
    { value: 0.0, r: 96, g: 230, b: 50 },      // Perfect green (on target)
    { value: 0.02, r: 110, g: 225, b: 45 },    // Slightly yellow-green
    { value: 0.04, r: 130, g: 220, b: 40 },    // Light yellow-green
    { value: 0.08, r: 160, g: 210, b: 35 },    // Yellow-green
    { value: 0.15, r: 190, g: 200, b: 30 },    // More yellow
    { value: 0.25, r: 220, g: 190, b: 25 },    // Yellow
    { value: 0.4, r: 240, g: 170, b: 20 },     // Yellow-orange
    { value: 0.6, r: 250, g: 140, b: 10 },     // Light orange
    { value: 0.8, r: 255, g: 100, b: 0 },      // Orange
    { value: 1.0, r: 255, g: 50, b: 0 }        // Red (critical positive)
  ];
  
  // Cubic Hermite spline interpolation for ultra-smooth transitions
  const cubicHermite = (t: number, p0: number, p1: number, m0: number, m1: number): number => {
    const t2 = t * t;
    const t3 = t2 * t;
    return (2 * t3 - 3 * t2 + 1) * p0 + 
           (t3 - 2 * t2 + t) * m0 + 
           (-2 * t3 + 3 * t2) * p1 + 
           (t3 - t2) * m1;
  };
  
  // Calculate tangents for smooth interpolation
  const calculateTangent = (prev: number, curr: number, next: number): number => {
    return (next - prev) * 0.5;
  };
  
  for (let i = 0; i < steps; i++) {
    const position = i / (steps - 1); // 0 to 1
    const value = position * 2 - 1; // -1 to 1
    
    // Find the two key points that bracket this value
    let startIdx = 0;
    for (let j = 0; j < keyPoints.length - 1; j++) {
      if (value >= keyPoints[j].value && value <= keyPoints[j + 1].value) {
        startIdx = j;
        break;
      }
    }
    
    // Ensure we're within bounds
    startIdx = Math.max(0, Math.min(startIdx, keyPoints.length - 2));
    
    const p0 = keyPoints[startIdx];
    const p1 = keyPoints[startIdx + 1];
    
    // Calculate normalized position between the two points
    const range = p1.value - p0.value;
    const t = range > 0 ? (value - p0.value) / range : 0;
    
    // Calculate tangents for each color channel
    const prevIdx = Math.max(0, startIdx - 1);
    const nextIdx = Math.min(keyPoints.length - 1, startIdx + 2);
    
    // Red channel
    const rm0 = calculateTangent(
      keyPoints[prevIdx].r,
      p0.r,
      p1.r
    );
    const rm1 = calculateTangent(
      p0.r,
      p1.r,
      keyPoints[nextIdx].r
    );
    
    // Green channel
    const gm0 = calculateTangent(
      keyPoints[prevIdx].g,
      p0.g,
      p1.g
    );
    const gm1 = calculateTangent(
      p0.g,
      p1.g,
      keyPoints[nextIdx].g
    );
    
    // Blue channel
    const bm0 = calculateTangent(
      keyPoints[prevIdx].b,
      p0.b,
      p1.b
    );
    const bm1 = calculateTangent(
      p0.b,
      p1.b,
      keyPoints[nextIdx].b
    );
    
    // Apply smooth interpolation with easing
    const easedT = t * t * (3 - 2 * t); // Smoothstep function for extra smoothness
    
    const r = Math.round(Math.max(0, Math.min(255, 
      cubicHermite(easedT, p0.r, p1.r, rm0, rm1)
    )));
    const g = Math.round(Math.max(0, Math.min(255, 
      cubicHermite(easedT, p0.g, p1.g, gm0, gm1)
    )));
    const b = Math.round(Math.max(0, Math.min(255, 
      cubicHermite(easedT, p0.b, p1.b, bm0, bm1)
    )));
    
    const toHex = (n: number) => n.toString(16).padStart(2, '0');
    colors.push(`#${toHex(r)}${toHex(g)}${toHex(b)}`);
  }
  
  return colors;
}
