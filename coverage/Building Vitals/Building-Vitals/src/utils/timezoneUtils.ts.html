/**
 * Timezone utilities for proper time handling in Building Vitals
 *
 * This utility helps convert between:
 * 1. User's local time (what they see/select in UI)
 * 2. Site/building timezone (where the data was recorded)
 * 3. API format (ISO 8601 with UTC)
 */

import { format, parseISO } from 'date-fns';
import { toZonedTime, fromZonedTime } from 'date-fns-tz';

/**
 * Format a local date/time for the ACE API
 *
 * This converts the date to UTC and formats it with the 'Z' suffix
 * as required by the ACE IoT API.
 *
 * Example:
 * - User selects: 9:00 AM EST (UTC-5)
 * - This function returns: "2024-01-26T14:00:00Z" (9 AM EST = 2 PM UTC)
 *
 * @param date - The date in user's local time
 * @returns ISO 8601 string with 'Z' suffix for API
 */
export function formatLocalTimeForAPI(date: Date): string {
  // Convert to UTC and format as ISO string
  return date.toISOString();
}

/**
 * Format a date for API considering building timezone
 *
 * This function ensures that when a user selects a time, it's interpreted
 * as the building's local time, not the user's browser time.
 *
 * Example:
 * - User in New York selects: 9:00 AM
 * - Building is in Chicago (1 hour behind)
 * - Function returns: "2024-01-26T09:00:00Z" (9 AM Chicago time)
 *
 * @param date - The date to format
 * @param buildingTimezone - The IANA timezone of the building
 * @param userTimezone - The user's browser timezone (optional)
 * @returns ISO 8601 string for API
 */
export function formatBuildingTimeForAPI(
  date: Date,
  buildingTimezone: string,
  userTimezone?: string
): string {
  const userTz = userTimezone || getBrowserTimezone();

  if (buildingTimezone === userTz) {
    // Same timezone, no conversion needed
    return formatLocalTimeForAPI(date);
  }

  // Convert the date from user's timezone to building's timezone
  // First, get the date as it appears in the user's timezone
  const userZonedTime = toZonedTime(date, userTz);

  // Then interpret that same wall clock time in the building's timezone
  const buildingDate = fromZonedTime(userZonedTime, buildingTimezone);

  // Format for API using the building's local time
  return formatLocalTimeForAPI(buildingDate);
}

/**
 * Convert a date from user's timezone to a specific site timezone
 *
 * This is used when we need to query data for a building in a different
 * timezone than the user's current location.
 *
 * @param date - Date in user's local time
 * @param siteTimezone - IANA timezone of the site (e.g., 'America/Chicago')
 * @returns Date adjusted to site timezone
 */
export function convertToSiteTimezone(date: Date, siteTimezone: string): Date {
  // Get the user's timezone
  const userTimezone = Intl.DateTimeFormat().resolvedOptions().timeZone;

  // Convert from user timezone to UTC, then to site timezone
  const utcDate = fromZonedTime(date, userTimezone);
  return toZonedTime(utcDate, siteTimezone);
}

/**
 * Format a date for API with site timezone consideration
 *
 * Use this when you have site timezone information and want to query
 * data based on the site's local time.
 *
 * @param date - Date in user's selected time
 * @param siteTimezone - Optional site timezone
 * @returns ISO 8601 string for API
 */
export function formatTimeForAPI(date: Date, siteTimezone?: string): string {
  if (siteTimezone) {
    // Convert to site timezone first
    const siteDate = convertToSiteTimezone(date, siteTimezone);
    return formatLocalTimeForAPI(siteDate);
  }

  // No site timezone info - use user's intended time directly
  return formatLocalTimeForAPI(date);
}

/**
 * Parse API timestamp to local display time
 *
 * @param isoString - ISO 8601 timestamp from API
 * @param displayTimezone - Optional timezone for display (defaults to user's local)
 * @returns Date object for display
 */
export function parseAPITimestamp(isoString: string, displayTimezone?: string): Date {
  const date = parseISO(isoString);

  if (displayTimezone) {
    return toZonedTime(date, displayTimezone);
  }

  // Return as-is for local display
  return date;
}

/**
 * Get a human-readable timezone offset
 *
 * @param timezone - IANA timezone string
 * @returns Formatted offset like "EST (UTC-5)"
 */
export function getTimezoneDisplay(timezone: string): string {
  const now = new Date();
  const formatter = new Intl.DateTimeFormat('en-US', {
    timeZone: timezone,
    timeZoneName: 'short',
  });

  const parts = formatter.formatToParts(now);
  const timeZoneName = parts.find((part) => part.type === 'timeZoneName')?.value || timezone;

  // Calculate offset
  const utcDate = new Date(now.toLocaleString('en-US', { timeZone: 'UTC' }));
  const tzDate = new Date(now.toLocaleString('en-US', { timeZone: timezone }));
  const offsetHours = Math.round((utcDate.getTime() - tzDate.getTime()) / 3600000);
  const offsetString = offsetHours >= 0 ? `+${offsetHours}` : `${offsetHours}`;

  return `${timeZoneName} (UTC${offsetString})`;
}

/**
 * Calculate time range in building's timezone
 *
 * @param timeRange - Time range string (e.g., '24h', '7d')
 * @param buildingTimezone - The building's timezone
 * @param userTimezone - The user's timezone (optional)
 * @returns Start and end times formatted for API
 */
export function calculateBuildingTimeRange(
  timeRange: string,
  buildingTimezone: string,
  userTimezone?: string
): { startTime: string; endTime: string } {
  const userTz = userTimezone || getBrowserTimezone();

  // Get current time in building's timezone
  const now = new Date();
  const buildingNow = buildingTimezone === userTz ? now : toZonedTime(now, buildingTimezone);

  let start: Date;
  let end: Date = buildingNow;

  // Calculate start time based on range
  switch (timeRange) {
    case '1h':
      start = new Date(buildingNow.getTime() - 60 * 60 * 1000);
      break;
    case '3h':
      start = new Date(buildingNow.getTime() - 3 * 60 * 60 * 1000);
      break;
    case '6h':
      start = new Date(buildingNow.getTime() - 6 * 60 * 60 * 1000);
      break;
    case '12h':
      start = new Date(buildingNow.getTime() - 12 * 60 * 60 * 1000);
      break;
    case '24h':
    case '1d':
      start = new Date(buildingNow.getTime() - 24 * 60 * 60 * 1000);
      break;
    case '3d':
      start = new Date(buildingNow.getTime() - 3 * 24 * 60 * 60 * 1000);
      break;
    case '7d':
    case '1w':
      start = new Date(buildingNow.getTime() - 7 * 24 * 60 * 60 * 1000);
      break;
    case '30d':
    case '1m':
      start = new Date(buildingNow.getTime() - 30 * 24 * 60 * 60 * 1000);
      break;
    case 'today':
      start = new Date(buildingNow);
      start.setHours(0, 0, 0, 0);
      break;
    case 'yesterday':
      const yesterday = new Date(buildingNow);
      yesterday.setDate(yesterday.getDate() - 1);
      start = new Date(yesterday);
      start.setHours(0, 0, 0, 0);
      end = new Date(yesterday);
      end.setHours(23, 59, 59, 999);
      break;
    default:
      // Check for custom range
      if (timeRange.includes(' - ')) {
        const [startStr, endStr] = timeRange.split(' - ');
        const startDate = new Date(startStr.trim());
        const endDate = new Date(endStr.trim());
        return {
          startTime: formatBuildingTimeForAPI(startDate, buildingTimezone, userTz),
          endTime: formatBuildingTimeForAPI(endDate, buildingTimezone, userTz),
        };
      }
      start = new Date(buildingNow.getTime() - 24 * 60 * 60 * 1000); // Default to 24h
  }

  return {
    startTime: formatLocalTimeForAPI(start),
    endTime: formatLocalTimeForAPI(end),
  };
}

/**
 * Get the browser's current timezone
 */
export function getBrowserTimezone(): string {
  return Intl.DateTimeFormat().resolvedOptions().timeZone;
}

/**
 * Format a date range for display with timezone info
 */
export function formatDateRangeDisplay(startDate: Date, endDate: Date, timezone?: string): string {
  const tz = timezone || getBrowserTimezone();
  const start = format(startDate, 'MMM d, yyyy h:mm a');
  const end = format(endDate, 'MMM d, yyyy h:mm a');
  const tzDisplay = getTimezoneDisplay(tz);

  return `${start} - ${end} ${tzDisplay}`;
}
