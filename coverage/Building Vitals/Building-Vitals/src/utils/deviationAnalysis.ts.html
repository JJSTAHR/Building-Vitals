/**
 * Deviation Analysis Utilities
 * Reusable functions for deviation calculations and visualization across chart types
 */

export interface DeviationConfig {
  setpoint: number;
  tolerance: number;
  unit?: string;
  mode?: 'absolute' | 'percentage';
  colorScheme?: DeviationColorScheme;
}

export interface DeviationColorScheme {
  within: string;
  slightlyBelow: string;
  below: string;
  farBelow: string;
  slightlyAbove: string;
  above: string;
  farAbove: string;
}

export interface DeviationPoint {
  value: number;
  deviation: number;
  percentDeviation: number;
  severity: DeviationSeverity;
  color: string;
  withinTolerance: boolean;
}

export type DeviationSeverity =
  | 'within'
  | 'slightly-below'
  | 'below'
  | 'far-below'
  | 'slightly-above'
  | 'above'
  | 'far-above';

/**
 * Calculate deviation metrics for a single value
 */
export function calculateDeviation(value: number, config: DeviationConfig): DeviationPoint {
  const deviation = value - config.setpoint;
  const percentDeviation = (deviation / config.setpoint) * 100;
  const withinTolerance = Math.abs(deviation) <= config.tolerance;

  // Determine severity based on multiples of tolerance
  let severity: DeviationSeverity;
  const deviationRatio = Math.abs(deviation) / config.tolerance;

  if (withinTolerance) {
    severity = 'within';
  } else if (deviation < 0) {
    if (deviationRatio <= 2) {severity = 'slightly-below';}
    else if (deviationRatio <= 4) {severity = 'below';}
    else {severity = 'far-below';}
  } else {
    if (deviationRatio <= 2) {severity = 'slightly-above';}
    else if (deviationRatio <= 4) {severity = 'above';}
    else {severity = 'far-above';}
  }

  const colorScheme = config.colorScheme || DEVIATION_COLOR_SCHEMES.standard;
  const color = getDeviationColor(severity, colorScheme);

  return {
    value,
    deviation,
    percentDeviation,
    severity,
    color,
    withinTolerance,
  };
}

/**
 * Get color for deviation severity
 */
export function getDeviationColor(
  severity: DeviationSeverity,
  colorScheme: DeviationColorScheme
): string {
  const severityMap: Record<DeviationSeverity, keyof DeviationColorScheme> = {
    within: 'within',
    'slightly-below': 'slightlyBelow',
    below: 'below',
    'far-below': 'farBelow',
    'slightly-above': 'slightlyAbove',
    above: 'above',
    'far-above': 'farAbove',
  };

  return colorScheme[severityMap[severity]];
}

/**
 * Calculate deviation color on a gradient
 */
export function getDeviationGradientColor(
  value: number,
  setpoint: number,
  tolerance: number,
  maxDeviation: number
): string {
  const deviation = value - setpoint;

  if (Math.abs(deviation) <= tolerance) {
    return '#4CAF50'; // Green for within tolerance
  }

  const normalizedDeviation = Math.abs(deviation) / maxDeviation;

  if (deviation < 0) {
    // Blue gradient for below setpoint
    const r = 0;
    const g = Math.round(255 * (1 - normalizedDeviation));
    const b = 255;
    return `rgb(${r}, ${g}, ${b})`;
  } else {
    // Yellow to red gradient for above setpoint
    const r = 255;
    const g = Math.round(255 * (1 - normalizedDeviation));
    const b = 0;
    return `rgb(${r}, ${g}, ${b})`;
  }
}

/**
 * Calculate statistics for an array of values
 */
export function calculateDeviationStats(values: number[], config: DeviationConfig) {
  const deviations = values.map((v) => calculateDeviation(v, config));

  const stats = {
    count: values.length,
    withinTolerance: deviations.filter((d) => d.withinTolerance).length,
    belowSetpoint: deviations.filter((d) => d.deviation < -config.tolerance).length,
    aboveSetpoint: deviations.filter((d) => d.deviation > config.tolerance).length,
    maxPositiveDeviation: Math.max(...deviations.map((d) => d.deviation)),
    maxNegativeDeviation: Math.min(...deviations.map((d) => d.deviation)),
    averageDeviation: deviations.reduce((sum, d) => sum + Math.abs(d.deviation), 0) / values.length,
    standardDeviation: 0,
    percentWithinTolerance: 0,
  };

  // Calculate standard deviation
  const mean = values.reduce((sum, v) => sum + v, 0) / values.length;
  const squaredDiffs = values.map((v) => Math.pow(v - mean, 2));
  stats.standardDeviation = Math.sqrt(squaredDiffs.reduce((sum, d) => sum + d, 0) / values.length);

  stats.percentWithinTolerance = (stats.withinTolerance / stats.count) * 100;

  return stats;
}

/**
 * Format deviation value for display
 */
export function formatDeviation(
  deviation: number,
  unit: string = '',
  mode: 'absolute' | 'percentage' = 'absolute'
): string {
  if (mode === 'percentage') {
    return `${deviation > 0 ? '+' : ''}${deviation.toFixed(1)}%`;
  }

  const formatted = deviation.toFixed(3);
  const sign = deviation > 0 ? '+' : '';
  return `${sign}${formatted}${unit ? ' ' + unit : ''}`;
}

/**
 * Determine if a point is an outlier based on statistical analysis
 */
export function isOutlier(
  value: number,
  mean: number,
  stdDev: number,
  threshold: number = 3
): boolean {
  return Math.abs(value - mean) > threshold * stdDev;
}

/**
 * Calculate comfort zone deviations (for psychrometric charts)
 */
export function calculateComfortDeviation(
  temperature: number,
  humidity: number,
  comfortZone: {
    tempMin: number;
    tempMax: number;
    humidityMin: number;
    humidityMax: number;
  }
) {
  const tempDeviation =
    temperature < comfortZone.tempMin
      ? temperature - comfortZone.tempMin
      : temperature > comfortZone.tempMax
        ? temperature - comfortZone.tempMax
        : 0;

  const humidityDeviation =
    humidity < comfortZone.humidityMin
      ? humidity - comfortZone.humidityMin
      : humidity > comfortZone.humidityMax
        ? humidity - comfortZone.humidityMax
        : 0;

  const totalDeviation = Math.sqrt(
    Math.pow(tempDeviation / (comfortZone.tempMax - comfortZone.tempMin), 2) +
      Math.pow(humidityDeviation / (comfortZone.humidityMax - comfortZone.humidityMin), 2)
  );

  return {
    temperature: tempDeviation,
    humidity: humidityDeviation,
    total: totalDeviation,
    withinComfort: tempDeviation === 0 && humidityDeviation === 0,
  };
}
