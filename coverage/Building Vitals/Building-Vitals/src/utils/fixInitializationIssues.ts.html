/**
 * Fix for initialization issues that cause loading loops
 */

export function fixInitializationIssues() {
  // Check if we're stuck in a loading loop
  const loadingStates = {
    auth: false,
    profile: false,
    token: false,
  };

  // Monitor Redux state for stuck loading
  if (typeof window !== 'undefined' && (window as any).__REDUX_STORE__) {
    const store = (window as any).__REDUX_STORE__;

    const checkLoadingStates = () => {
      const state = store.getState();
      const auth = state.auth;

      if (auth) {
        // If any loading state is stuck for more than 10 seconds, force clear it
        if (auth.loading || auth.loadingProfile || auth.tokenLoading) {
          setTimeout(() => {
            const currentState = store.getState().auth;
            if (currentState.loading || currentState.loadingProfile || currentState.tokenLoading) {
              // Dispatch actions to clear loading states
              if (currentState.loading) {
                store.dispatch({
                  type: 'auth/authFailure',
                  payload: 'Loading timeout - please refresh',
                });
              }
            }
          }, 10000);
        }
      }
    };

    // Check immediately and periodically
    checkLoadingStates();
    setInterval(checkLoadingStates, 5000);
  }

  // Fix for BatchingContext errors
  if (typeof window !== 'undefined') {
    // Disable prefetch if it's causing issues
    (window as any).__DISABLE_BATCHING_PREFETCH__ = true;

    // Clear any stuck promises
    const originalPromiseReject = Promise.reject;
    Promise.reject = function (reason) {
      if (reason === null || reason === undefined) {
        console.warn('Promise rejected with null/undefined, providing default error');
        return originalPromiseReject.call(this, new Error('Promise rejected with null value'));
      }
      return originalPromiseReject.call(this, reason);
    };
  }
}

// Auto-run on import
fixInitializationIssues();

// Add to window for manual debugging
if (typeof window !== 'undefined') {
  (window as any).fixInitializationIssues = fixInitializationIssues;

  // Add helper to manually clear loading states
  (window as any).clearLoadingStates = () => {
    if ((window as any).__REDUX_STORE__) {
      const store = (window as any).__REDUX_STORE__;
      store.dispatch({ type: 'auth/tokenLoadComplete' });
      store.dispatch({ type: 'auth/loadProfileComplete' });
    }
  };
}
