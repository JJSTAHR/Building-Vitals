/**
 * Time Series Alignment Utilities
 * Handles synchronization of data points from multiple sensors with different timestamps
 */

export interface TimeSeriesPoint {
  timestamp: number;
  value: number;
}

export interface AlignmentOptions {
  method: 'nearest' | 'linear' | 'previous' | 'next' | 'forward-fill' | 'backward-fill';
  maxGap?: number; // Maximum time gap in milliseconds to consider for alignment
  tolerance?: number; // Time tolerance in milliseconds for "same" timestamp
  requireAllSeries?: boolean; // If false, allows partial data when some series are missing
  minSeriesRequired?: number; // Minimum number of series that must have data (default: all)
  fillGaps?: boolean; // If true, uses last known value for gaps
}

const DEFAULT_OPTIONS: AlignmentOptions = {
  method: 'nearest',
  maxGap: 5 * 60 * 1000, // 5 minutes
  tolerance: 30 * 1000, // 30 seconds
  requireAllSeries: false, // Allow partial data by default
  minSeriesRequired: 1, // At least one series must have data
  fillGaps: true, // Fill gaps with last known values
};

/**
 * Aligns multiple time series with potentially different timestamps
 * @param series Array of time series to align
 * @param options Alignment configuration
 * @returns Aligned data points where all series have values
 */
export function alignTimeSeries(
  series: Array<{ name: string; data: Array<[number, number]> }>,
  options: Partial<AlignmentOptions> = {}
): Array<{ timestamp: number; values: number[] }> {
  const config = { ...DEFAULT_OPTIONS, ...options };
  
  if (!series || series.length === 0) {
    return [];
  }
  
  // Create a union of all unique timestamps within tolerance
  const allTimestamps = new Set<number>();
  const timestampGroups = new Map<number, number[]>();
  
  // Group timestamps that are within tolerance of each other
  series.forEach(s => {
    s.data.forEach(([ts]) => {
      let foundGroup = false;
      
      for (const [groupTs] of timestampGroups) {
        if (Math.abs(ts - groupTs) <= config.tolerance!) {
          timestampGroups.get(groupTs)!.push(ts);
          foundGroup = true;
          break;
        }
      }
      
      if (!foundGroup) {
        timestampGroups.set(ts, [ts]);
        allTimestamps.add(ts);
      }
    });
  });
  
  // Calculate representative timestamp for each group (median)
  const representativeTimestamps = new Map<number, number>();
  timestampGroups.forEach((group, originalTs) => {
    const sorted = group.sort((a, b) => a - b);
    const median = sorted[Math.floor(sorted.length / 2)];
    representativeTimestamps.set(originalTs, median);
  });
  
  // Track last known values for forward-fill
  const lastKnownValues: (number | null)[] = new Array(series.length).fill(null);
  
  // Align data points
  const alignedData: Array<{ timestamp: number; values: number[] }> = [];
  
  Array.from(allTimestamps).sort((a, b) => a - b).forEach(timestamp => {
    const representativeTs = representativeTimestamps.get(timestamp) || timestamp;
    const values: number[] = [];
    let foundCount = 0;
    
    for (let i = 0; i < series.length; i++) {
      let value = findValueAtTimestamp(series[i].data, representativeTs, config);
      
      // If no value found and fillGaps is enabled, use last known value
      if (value === null && config.fillGaps && lastKnownValues[i] !== null) {
        value = lastKnownValues[i];
      }
      
      // Update last known value if we found a real value
      if (value !== null && findValueAtTimestamp(series[i].data, representativeTs, { ...config, fillGaps: false }) !== null) {
        lastKnownValues[i] = value;
      }
      
      if (value !== null) {
        foundCount++;
      }
      
      values.push(value!);
    }
    
    // Determine if we should include this data point
    const minRequired = config.minSeriesRequired ?? series.length;
    const shouldInclude = config.requireAllSeries ? 
      foundCount === series.length : 
      foundCount >= Math.min(minRequired, series.length);
    
    if (shouldInclude) {
      // Replace nulls with NaN for better handling in charts
      const finalValues = values.map(v => v === null ? NaN : v);
      alignedData.push({ timestamp: representativeTs, values: finalValues });
    }
  });
  
  return alignedData;
}

/**
 * Finds value at or near a timestamp using specified method
 */
function findValueAtTimestamp(
  data: Array<[number, number]>,
  targetTimestamp: number,
  config: AlignmentOptions
): number | null {
  if (!data || data.length === 0) {
    return null;
  }
  
  // Sort data by timestamp
  const sorted = [...data].sort((a, b) => a[0] - b[0]);
  
  // Find exact or nearest match
  let closest: [number, number] | null = null;
  let minDiff = Infinity;
  
  for (const point of sorted) {
    const diff = Math.abs(point[0] - targetTimestamp);
    
    if (diff <= config.tolerance!) {
      // Within tolerance, consider it a match
      if (diff < minDiff) {
        minDiff = diff;
        closest = point;
      }
    }
  }
  
  if (closest) {
    return closest[1];
  }
  
  // If no exact match, use interpolation based on method
  switch (config.method) {
    case 'nearest':
      return findNearestValue(sorted, targetTimestamp, config.maxGap!);
    
    case 'linear':
      return interpolateLinear(sorted, targetTimestamp, config.maxGap!);
    
    case 'previous':
      return findPreviousValue(sorted, targetTimestamp, config.maxGap!);
    
    case 'next':
      return findNextValue(sorted, targetTimestamp, config.maxGap!);
    
    default:
      return null;
  }
}

/**
 * Find the nearest value within maxGap
 */
function findNearestValue(
  data: Array<[number, number]>,
  timestamp: number,
  maxGap: number
): number | null {
  let nearest: [number, number] | null = null;
  let minDiff = Infinity;
  
  for (const point of data) {
    const diff = Math.abs(point[0] - timestamp);
    if (diff < minDiff && diff <= maxGap) {
      minDiff = diff;
      nearest = point;
    }
  }
  
  return nearest ? nearest[1] : null;
}

/**
 * Linear interpolation between two points
 */
function interpolateLinear(
  data: Array<[number, number]>,
  timestamp: number,
  maxGap: number
): number | null {
  // Find surrounding points
  let before: [number, number] | null = null;
  let after: [number, number] | null = null;
  
  for (let i = 0; i < data.length; i++) {
    if (data[i][0] <= timestamp) {
      before = data[i];
    }
    if (data[i][0] >= timestamp && !after) {
      after = data[i];
      break;
    }
  }
  
  if (!before || !after) {
    // Try nearest if we're at the edge
    return findNearestValue(data, timestamp, maxGap);
  }
  
  // Check if gap is too large
  if (after[0] - before[0] > maxGap) {
    return null;
  }
  
  // Linear interpolation
  const ratio = (timestamp - before[0]) / (after[0] - before[0]);
  return before[1] + ratio * (after[1] - before[1]);
}

/**
 * Find the previous value within maxGap
 */
function findPreviousValue(
  data: Array<[number, number]>,
  timestamp: number,
  maxGap: number
): number | null {
  let previous: [number, number] | null = null;
  
  for (const point of data) {
    if (point[0] <= timestamp) {
      previous = point;
    } else {
      break;
    }
  }
  
  if (previous && timestamp - previous[0] <= maxGap) {
    return previous[1];
  }
  
  return null;
}

/**
 * Find the next value within maxGap
 */
function findNextValue(
  data: Array<[number, number]>,
  timestamp: number,
  maxGap: number
): number | null {
  for (const point of data) {
    if (point[0] >= timestamp) {
      if (point[0] - timestamp <= maxGap) {
        return point[1];
      }
      break;
    }
  }
  
  return null;
}

/**
 * Detects the typical sampling interval of a time series
 */
export function detectSamplingInterval(data: Array<[number, number]>): number | null {
  if (!data || data.length < 2) {
    return null;
  }
  
  const sorted = [...data].sort((a, b) => a[0] - b[0]);
  const intervals: number[] = [];
  
  for (let i = 1; i < sorted.length; i++) {
    intervals.push(sorted[i][0] - sorted[i - 1][0]);
  }
  
  // Return median interval
  intervals.sort((a, b) => a - b);
  return intervals[Math.floor(intervals.length / 2)];
}

/**
 * Analyzes time series alignment quality
 */
export function analyzeAlignmentQuality(
  series: Array<{ name: string; data: Array<[number, number]> }>
): {
  totalPoints: number;
  alignedPoints: number;
  alignmentRate: number;
  samplingIntervals: Array<{ name: string; interval: number | null }>;
  recommendation: string;
} {
  // Get total unique timestamps
  const allTimestamps = new Set<number>();
  series.forEach(s => {
    s.data.forEach(([ts]) => allTimestamps.add(ts));
  });
  
  // Get aligned points with default settings
  const aligned = alignTimeSeries(series);
  
  // Detect sampling intervals
  const samplingIntervals = series.map(s => ({
    name: s.name,
    interval: detectSamplingInterval(s.data),
  }));
  
  const alignmentRate = aligned.length / allTimestamps.size;
  
  // Generate recommendation
  let recommendation = '';
  if (alignmentRate > 0.9) {
    recommendation = 'Excellent alignment - data points are well synchronized';
  } else if (alignmentRate > 0.7) {
    recommendation = 'Good alignment - minor timestamp differences detected';
  } else if (alignmentRate > 0.5) {
    recommendation = 'Fair alignment - consider adjusting sensor sampling rates';
  } else {
    recommendation = 'Poor alignment - significant timestamp mismatches. Check sensor configurations and time synchronization';
  }
  
  return {
    totalPoints: allTimestamps.size,
    alignedPoints: aligned.length,
    alignmentRate,
    samplingIntervals,
    recommendation,
  };
}
