/**
 * Bundle optimization utilities
 *
 * These utilities help monitor and optimize bundle size
 */

/**
 * Track component load times
 */
class ComponentLoadTracker {
  private loadTimes = new Map<string, number[]>();

  recordLoad(componentName: string, loadTime: number) {
    if (!this.loadTimes.has(componentName)) {
      this.loadTimes.set(componentName, []);
    }

    const times = this.loadTimes.get(componentName)!;
    times.push(loadTime);

    // Keep only last 100 measurements
    if (times.length > 100) {
      times.shift();
    }
  }

  getAverageLoadTime(componentName: string): number {
    const times = this.loadTimes.get(componentName);
    if (!times || times.length === 0) {return 0;}

    const sum = times.reduce((a, b) => a + b, 0);
    return sum / times.length;
  }

  getSlowComponents(threshold: number = 100): Array<{ name: string; avgTime: number }> {
    const slow: Array<{ name: string; avgTime: number }> = [];

    this.loadTimes.forEach((times, name) => {
      const avg = this.getAverageLoadTime(name);
      if (avg > threshold) {
        slow.push({ name, avgTime: avg });
      }
    });

    return slow.sort((a, b) => b.avgTime - a.avgTime);
  }

  generateReport() {
    const report = {
      totalComponents: this.loadTimes.size,
      slowComponents: this.getSlowComponents(),
      averageLoadTimes: {} as Record<string, number>,
    };

    this.loadTimes.forEach((_, name) => {
      report.averageLoadTimes[name] = Math.round(this.getAverageLoadTime(name));
    });

    return report;
  }
}

export const componentLoadTracker = new ComponentLoadTracker();

/**
 * Measure component load time
 */
export function measureComponentLoad<T>(
  componentName: string,
  loadFn: () => Promise<T>
): Promise<T> {
  const startTime = performance.now();

  return loadFn().then(
    (result) => {
      const loadTime = performance.now() - startTime;
      componentLoadTracker.recordLoad(componentName, loadTime);

      if (loadTime > 200) {
      }

      return result;
    },
    (error) => {
      const loadTime = performance.now() - startTime;
      componentLoadTracker.recordLoad(componentName, loadTime);
      throw error;
    }
  );
}

/**
 * Lazy load with performance tracking
 */
export function lazyWithTracking<T extends React.ComponentType<any>>(
  importFn: () => Promise<{ default: T }>,
  componentName: string
): React.LazyExoticComponent<T> {
  return React.lazy(() => measureComponentLoad(componentName, importFn));
}

/**
 * Bundle size estimation utilities
 */
export const bundleSizeEstimator = {
  /**
   * Estimate the size impact of importing a module
   */
  estimateImportSize(modulePath: string): number {
    // Rough estimates based on common libraries
    const sizeMap: Record<string, number> = {
      echarts: 750, // KB
      'echarts-gl': 350,
      '@mui/material': 200,
      '@mui/icons-material': 1500, // All icons
      firebase: 150,
      react: 45,
      'react-dom': 120,
      lodash: 70,
      'date-fns': 80,
      jspdf: 280,
      html2canvas: 170,
    };

    for (const [lib, size] of Object.entries(sizeMap)) {
      if (modulePath.includes(lib)) {
        return size;
      }
    }

    return 10; // Default estimate for unknown modules
  },

  /**
   * Check if a component should be lazy loaded
   */
  shouldLazyLoad(
    componentPath: string,
    options?: {
      sizeThreshold?: number;
      routeBased?: boolean;
    }
  ): boolean {
    const { sizeThreshold = 50, routeBased = false } = options || {};

    // Always lazy load route-based components
    if (routeBased) {return true;}

    // Lazy load large components
    const estimatedSize = this.estimateImportSize(componentPath);
    if (estimatedSize > sizeThreshold) {return true;}

    // Lazy load certain types of components
    const lazyPatterns = [
      /charts?\//i,
      /modal/i,
      /dialog/i,
      /export/i,
      /pdf/i,
      /print/i,
      /settings/i,
      /admin/i,
    ];

    return lazyPatterns.some((pattern) => pattern.test(componentPath));
  },
};

/**
 * Resource hints for optimizing loading
 */
export const resourceHints = {
  /**
   * Preload critical resources
   */
  preloadCriticalResources() {
    const criticalResources = [
      '/assets/react-core-[hash].js',
      '/assets/mui-core-components-[hash].js',
      '/assets/firebase-auth-[hash].js',
    ];

    criticalResources.forEach((resource) => {
      const link = document.createElement('link');
      link.rel = 'preload';
      link.as = 'script';
      link.href = resource;
      document.head.appendChild(link);
    });
  },

  /**
   * Prefetch resources for likely next navigation
   */
  prefetchForRoute(routeName: string) {
    const routeResources: Record<string, string[]> = {
      dashboard: ['/assets/echarts-core-[hash].js', '/assets/echarts-basic-[hash].js'],
      settings: ['/assets/mui-form-components-[hash].js'],
      export: ['/assets/pdf-export-[hash].js'],
    };

    const resources = routeResources[routeName] || [];
    resources.forEach((resource) => {
      const link = document.createElement('link');
      link.rel = 'prefetch';
      link.as = 'script';
      link.href = resource;
      document.head.appendChild(link);
    });
  },
};

/**
 * Monitor and report bundle metrics
 */
export function reportBundleMetrics() {
  if (!performance.getEntriesByType) {return;}

  const resources = performance.getEntriesByType('resource');
  const scripts = resources.filter((r) => r.name.endsWith('.js'));

  const metrics = {
    totalScripts: scripts.length,
    totalSize: 0,
    totalLoadTime: 0,
    largestScript: { name: '', size: 0 },
    slowestScript: { name: '', time: 0 },
  };

  scripts.forEach((script) => {
    const size = script.transferSize || 0;
    const time = script.duration || 0;

    metrics.totalSize += size;
    metrics.totalLoadTime += time;

    if (size > metrics.largestScript.size) {
      metrics.largestScript = { name: script.name, size };
    }

    if (time > metrics.slowestScript.time) {
      metrics.slowestScript = { name: script.name, time };
    }
  });

  return metrics;
}

// Auto-report metrics in development
if (import.meta.env.DEV) {
  window.addEventListener('load', () => {
    setTimeout(reportBundleMetrics, 2000);
  });
}
