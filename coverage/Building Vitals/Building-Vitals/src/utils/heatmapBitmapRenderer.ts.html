/**
 * Bitmap renderer for extreme dataset heatmaps
 * Renders millions of data points as pixels with proper gradients
 */

import { generateHeatmapGradient } from './deviationCalculations';

interface HeatmapDataPoint {
  xIndex: number;
  yIndex: number;
  value: number; // Normalized -1 to 1
}

interface BitmapOptions {
  width: number;
  height: number;
  data: HeatmapDataPoint[];
  colorSteps?: number;
  blurRadius?: number;
}

/**
 * Creates a bitmap image from heatmap data for extreme datasets
 * This renders each data point as a pixel, allowing millions of points
 * to be visualized with proper gradients
 */
export function createHeatmapBitmap(options: BitmapOptions): string {
  const { width, height, data, colorSteps = 256, blurRadius = 1 } = options;
  
  // Create off-screen canvas
  const canvas = document.createElement('canvas');
  canvas.width = width;
  canvas.height = height;
  const ctx = canvas.getContext('2d', { willReadFrequently: true });
  
  if (!ctx) {
    console.error('Failed to get canvas context');
    return '';
  }
  
  // Generate color map
  const colors = generateHeatmapGradient(colorSteps);
  
  // Create image data
  const imageData = ctx.createImageData(width, height);
  const pixels = imageData.data;
  
  // Initialize with transparent background
  for (let i = 0; i < pixels.length; i += 4) {
    pixels[i] = 200;     // R
    pixels[i + 1] = 200; // G
    pixels[i + 2] = 200; // B
    pixels[i + 3] = 30;  // A (mostly transparent)
  }
  
  // Map normalized values to color indices
  const valueToColorIndex = (value: number): number => {
    // Value is -1 to 1, map to 0 to colorSteps-1
    const normalized = (value + 1) / 2; // 0 to 1
    const index = Math.floor(normalized * (colorSteps - 1));
    return Math.max(0, Math.min(colorSteps - 1, index));
  };
  
  // Parse hex color to RGB
  const hexToRgb = (hex: string): { r: number; g: number; b: number } => {
    const result = /^#?([a-f\d]{2})([a-f\d]{2})([a-f\d]{2})$/i.exec(hex);
    return result ? {
      r: parseInt(result[1], 16),
      g: parseInt(result[2], 16),
      b: parseInt(result[3], 16)
    } : { r: 128, g: 128, b: 128 };
  };
  
  // Plot each data point as a pixel
  data.forEach(point => {
    const x = Math.floor(point.xIndex);
    const y = Math.floor(point.yIndex);
    
    if (x >= 0 && x < width && y >= 0 && y < height) {
      const pixelIndex = (y * width + x) * 4;
      const colorIndex = valueToColorIndex(point.value);
      const color = hexToRgb(colors[colorIndex]);
      
      pixels[pixelIndex] = color.r;
      pixels[pixelIndex + 1] = color.g;
      pixels[pixelIndex + 2] = color.b;
      pixels[pixelIndex + 3] = 255; // Full opacity for data points
    }
  });
  
  // Apply the image data
  ctx.putImageData(imageData, 0, 0);
  
  // Apply Gaussian blur for smooth gradients
  if (blurRadius > 0) {
    ctx.filter = `blur(${blurRadius}px)`;
    ctx.drawImage(canvas, 0, 0);
  }
  
  // Convert to data URL
  return canvas.toDataURL('image/png');
}

/**
 * Creates a high-resolution heatmap by rendering in tiles
 * This allows rendering of extremely large datasets
 */
export function createTiledHeatmapBitmap(
  data: HeatmapDataPoint[],
  totalWidth: number,
  totalHeight: number,
  tileSize: number = 512
): string {
  // For very large datasets, we render in tiles and compose them
  const canvas = document.createElement('canvas');
  canvas.width = Math.min(totalWidth, 2048); // Limit to reasonable size
  canvas.height = Math.min(totalHeight, 2048);
  const ctx = canvas.getContext('2d');
  
  if (!ctx) {
    return '';
  }
  
  // Scale factor if we're downsampling
  const scaleX = canvas.width / totalWidth;
  const scaleY = canvas.height / totalHeight;
  
  // Scale and filter data
  const scaledData = data.map(point => ({
    xIndex: point.xIndex * scaleX,
    yIndex: point.yIndex * scaleY,
    value: point.value
  }));
  
  // Render the scaled version
  const bitmap = createHeatmapBitmap({
    width: canvas.width,
    height: canvas.height,
    data: scaledData,
    blurRadius: Math.max(1, Math.min(5, 1000000 / data.length)) // Adaptive blur
  });
  
  return bitmap;
}

/**
 * Optimized renderer for year-long datasets
 */
export function renderYearHeatmap(
  data: Array<{
    timestamp: number;
    seriesIndex: number;
    value: number;
    normalizedValue: number;
  }>,
  seriesCount: number,
  startTime: number,
  endTime: number
): string {
  const msPerDay = 24 * 60 * 60 * 1000;
  const totalDays = Math.ceil((endTime - startTime) / msPerDay);
  
  // Determine canvas dimensions
  // For a year: 365 days width, series count height
  const width = Math.min(totalDays, 1000); // Cap at 1000px for performance
  const height = Math.min(seriesCount * 20, 500); // 20px per series, cap at 500px
  
  // Convert data to bitmap format
  const bitmapData: HeatmapDataPoint[] = [];
  
  data.forEach(point => {
    const dayIndex = Math.floor((point.timestamp - startTime) / msPerDay);
    const x = Math.floor((dayIndex / totalDays) * width);
    const y = Math.floor((point.seriesIndex / seriesCount) * height);
    
    bitmapData.push({
      xIndex: x,
      yIndex: y,
      value: point.normalizedValue
    });
  });
  
  // Render with adaptive settings based on data density
  const density = bitmapData.length / (width * height);
  const blurRadius = density > 10 ? 3 : density > 5 ? 2 : 1;
  
  return createHeatmapBitmap({
    width,
    height,
    data: bitmapData,
    blurRadius
  });
}
