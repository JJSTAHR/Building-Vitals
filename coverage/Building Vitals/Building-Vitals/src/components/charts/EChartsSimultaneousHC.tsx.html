export const EChartsSimultaneousHC: React.FC<SimultaneousHCProps> = ({
  series = [],
  selectedPoints = [],
  loading = false,
  error = null,
  height = 400,
  title = 'Simultaneous Heating/Cooling Detection',
  config = {},
  onChartReady,
  showToolbox = true,
  enableDataView = false,
  showDataZoom = true,
  showLegend = true,
  onEdit,
  onDelete
}) => {
  const theme = useTheme();
  const baseOptions = useBaseChartOptions();
  
  // Get available points
  const { data: availablePoints = [] } = usePoints();
  
  // Setup state
  const [setupMode, setSetupMode] = useState(!series.length);
  const [setupComplete, setSetupComplete] = useState(false);
  const [pointSelections, setPointSelections] = useState<Map<string, any[]>>(new Map());
  const [showCalculationInfo, setShowCalculationInfo] = useState(false);
  const [detectionConfig, setDetectionConfig] = useState({
    wasteThreshold: config.wasteThreshold || 5, // kW
    overlapDuration: config.overlapDuration || 15, // minutes
    energyUnit: config.energyUnit || 'kW',
    temperatureUnit: config.temperatureUnit || 'F',
    heatingThreshold: config.heatingThreshold || 10, // % output
    coolingThreshold: config.coolingThreshold || 10 // % output
  });
  
  // Get all selected points for data fetching
  const allSelectedPoints = useMemo(() => {
    const points: any[] = [];
    pointSelections.forEach((categoryPoints) => {
      points.push(...categoryPoints);
    });
    return points;
  }, [pointSelections]);
  
  // Fetch actual timeseries data for selected points
  const {
    data: chartData,
    isLoading: dataLoading,
    error: dataError
  } = useChartData({
    selectedPoints: allSelectedPoints,
    enabled: setupComplete && allSelectedPoints.length > 0,
    timeRange: '24h' // Default time range, could be made configurable
  });

  // Process HC data and detect simultaneous operations
  const processedData = useMemo(() => {
    // Use fetched data if available from guided setup, otherwise fallback to props
    const dataSource = setupComplete && chartData?.series ? chartData.series : series;
    
    if (!dataSource.length) return null;

    // Get selected points by category
    const ahuHeatingPoints = pointSelections.get('ahuHeating') || [];
    const ahuCoolingPoints = pointSelections.get('ahuCooling') || [];
    const zoneReheatPoints = pointSelections.get('zoneReheat') || [];
    const supplyTempPoints = pointSelections.get('supplyTemp') || [];
    
    // Map series to selected points
    let heatingSeries: any[] = [];
    let coolingSeries: any[] = [];
    let tempSeries: any[] = [];
    
    if (setupComplete && (ahuHeatingPoints.length > 0 || zoneReheatPoints.length > 0)) {
      // Use exact point mappings from guided selection with actual fetched data
      heatingSeries = dataSource.filter(s => 
        [...ahuHeatingPoints, ...zoneReheatPoints].some(p => 
          p.id === s.id || p.name === s.name || p.id === s.name
        )
      );
      
      coolingSeries = dataSource.filter(s => 
        ahuCoolingPoints.some(p => p.id === s.id || p.name === s.name || p.id === s.name)
      );
      
      tempSeries = dataSource.filter(s => 
        supplyTempPoints.some(p => p.id === s.id || p.name === s.name || p.id === s.name)
      );
    } else {
      // Fallback to name-based filtering for backward compatibility
      heatingSeries = dataSource.filter(s => 
        s.name?.toLowerCase().includes('heat') || 
        s.name?.toLowerCase().includes('htg') ||
        s.name?.toLowerCase().includes('reheat')
      );

      coolingSeries = dataSource.filter(s => 
        s.name?.toLowerCase().includes('cool') || 
        s.name?.toLowerCase().includes('clg') ||
        s.name?.toLowerCase().includes('chw')
      );

      tempSeries = dataSource.filter(s => 
        s.name?.toLowerCase().includes('temp') ||
        s.name?.toLowerCase().includes('sat') ||
        s.name?.toLowerCase().includes('supply')
      );
    }

    const totalZones = Math.max(heatingSeries.length, coolingSeries.length);
    
    // Determine visualization mode based on zone count
    let vizMode: 'individual' | 'aggregated' | 'heatmap';
    if (totalZones <= 10) {
      vizMode = 'individual';
    } else if (totalZones <= 50) {
      vizMode = 'aggregated';
    } else {
      vizMode = 'heatmap';
    }

    // Get all unique timestamps
    const allTimestamps = new Set<number>();
    [...heatingSeries, ...coolingSeries, ...tempSeries].forEach(s => {
      s.data?.forEach((d: any) => allTimestamps.add(d[0]));
    });
    const timestamps = Array.from(allTimestamps).sort();

    // Detect simultaneous heating/cooling events
    const detectSimultaneousHC = (timestamp: number): WasteEvent | null => {
      const heatingValues = heatingSeries
        .map(s => ({
          name: s.name,
          value: s.data?.find((d: any) => Math.abs(d[0] - timestamp) < 60000)?.[1] || 0
        }))
        .filter(v => v.value > detectionConfig.heatingThreshold);

      const coolingValues = coolingSeries
        .map(s => ({
          name: s.name,
          value: s.data?.find((d: any) => Math.abs(d[0] - timestamp) < 60000)?.[1] || 0
        }))
        .filter(v => v.value > detectionConfig.coolingThreshold);

      // Count active heating and cooling zones (ASHRAE 36 approach)
      const activeHeatingZones = heatingValues.length;
      const activeCoolingZones = coolingValues.length;
      
      if (activeHeatingZones === 0 || activeCoolingZones === 0) {
        return null; // No simultaneous operation
      }

      // Calculate energy waste
      const totalHeating = heatingValues.reduce((sum, h) => sum + h.value, 0);
      const totalCooling = coolingValues.reduce((sum, c) => sum + c.value, 0);
      
      // Calculate actual energy waste based on simultaneous operation
      // Waste = Min(Heating%, Cooling%) × Capacity × Efficiency Loss
      const overlapPercent = Math.min(totalHeating, totalCooling);
      
      // Estimate energy based on number of zones and typical capacities
      const factors = ENERGY_FACTORS[detectionConfig.energyUnit];
      const ahuCapacity = ahuHeating.length * factors.typicalAHUCapacity;
      const zoneCapacity = affectedZones.length * factors.typicalZoneCapacity;
      
      // The waste is the overlapping energy that fights against each other
      // Formula: (Overlap% / 100) × Capacity × (1/HeatingEff + 1/CoolingEff)
      const efficiencyPenalty = detectionConfig.energyUnit === 'kW' 
        ? (1/factors.heatingCOP + 1/factors.coolingCOP)
        : (1/factors.heatingEfficiency + 1/factors.coolingEfficiency);
      
      const wasteKW = (overlapPercent / 100) * 
                     Math.min(ahuCapacity, zoneCapacity) * 
                     efficiencyPenalty * 0.5; // 0.5 factor for partial load

      if (wasteKW < detectionConfig.wasteThreshold) {
        return null; // Below threshold
      }

      // Determine waste type based on zone counts and overlap
      let wasteType: WasteEvent['type'] = 'simultaneous';
      let message = '';

      if (heatingValues.some(h => h.name.toLowerCase().includes('reheat'))) {
        wasteType = 'reheat';
        message = `${activeHeatingZones} zone(s) reheat active while ${activeCoolingZones} zone(s) cooling (${wasteKW.toFixed(1)} ${detectionConfig.energyUnit} waste)`;
      } else if (totalHeating > totalCooling * 1.5) {
        wasteType = 'mixed';
        message = `${activeHeatingZones} heating vs ${activeCoolingZones} cooling zones (${wastePercent.toFixed(1)}% waste)`;
      } else {
        message = `${activeHeatingZones} heating + ${activeCoolingZones} cooling zones active (${wasteKW.toFixed(1)} ${detectionConfig.energyUnit} waste)`;
      }

      return {
        timestamp,
        duration: detectionConfig.overlapDuration,
        wasteKW,
        wastePercent,
        affectedZones: [...heatingValues, ...coolingValues].map(v => v.name),
        type: wasteType,
        message
      };
    };

    const wasteEvents = timestamps
      .map(ts => detectSimultaneousHC(ts))
      .filter((event): event is WasteEvent => event !== null);

    // Calculate aggregated statistics
    const calculateAggregatedStats = (timestamp: number) => {
      const allHeating = heatingSeries
        .map(s => s.data?.find((d: any) => Math.abs(d[0] - timestamp) < 60000)?.[1] || 0);
      const allCooling = coolingSeries
        .map(s => s.data?.find((d: any) => Math.abs(d[0] - timestamp) < 60000)?.[1] || 0);

      const totalHeating = allHeating.reduce((sum, val) => sum + val, 0);
      const totalCooling = allCooling.reduce((sum, val) => sum + val, 0);
      const avgTemp = tempSeries.reduce((sum, s) => {
        const val = s.data?.find((d: any) => Math.abs(d[0] - timestamp) < 60000)?.[1] || 0;
        return sum + val;
      }, 0) / Math.max(tempSeries.length, 1);

      const activeHeating = allHeating.filter(v => v > detectionConfig.heatingThreshold).length;
      const activeCooling = allCooling.filter(v => v > detectionConfig.coolingThreshold).length;

      return {
        timestamp,
        totalHeating,
        totalCooling,
        avgTemp,
        activeHeating,
        activeCooling,
        conflict: activeHeating > 0 && activeCooling > 0
      };
    };

    const aggregatedStats = timestamps.map(calculateAggregatedStats);

    return {
      vizMode,
      totalZones,
      heatingSeries,
      coolingSeries,
      tempSeries,
      wasteEvents,
      aggregatedStats,
      timestamps
    };
  }, [series, chartData?.series, detectionConfig, setupComplete, pointSelections]);

  // Generate chart options
  const chartOptions = useMemo((): EChartsOption => {
    if (!processedData) {
      return mergeChartOptions(baseOptions, {
        title: title ? {
          ...CHART_DESIGN_TOKENS.title.base,
          text: 'No Data Available',
          textStyle: {
            ...CHART_DESIGN_TOKENS.title.base.textStyle,
            color: theme.palette.text.primary,
          },
        } : undefined,
        series: []
      });
    }

    // Calculate adaptive animation mode
    const totalPoints = [...processedData.heatingSeries, ...processedData.coolingSeries]
      .reduce((sum, s) => sum + (s.data?.length || 0), 0);
    const animationMode = CHART_DESIGN_TOKENS.adaptive.animationMode(totalPoints);

    const { vizMode, heatingSeries, coolingSeries, wasteEvents, aggregatedStats } = processedData;
    const chartSeries: any[] = [];

    if (vizMode === 'individual') {
      // Show individual heating and cooling signals
      heatingSeries.forEach((s, idx) => {
        chartSeries.push({
          name: s.name,
          type: 'line',
          data: s.data,
          yAxisIndex: 0,
          smooth: true,
          symbol: 'none',
          lineStyle: { width: 2, color: '#ff5722' },
          areaStyle: { opacity: 0.2, color: '#ff5722' },
          stack: 'heating'
        });
      });

      coolingSeries.forEach((s, idx) => {
        chartSeries.push({
          name: s.name,
          type: 'line',
          data: s.data,
          yAxisIndex: 0,
          smooth: true,
          symbol: 'none',
          lineStyle: { width: 2, color: '#2196f3' },
          areaStyle: { opacity: 0.2, color: '#2196f3' },
          stack: 'cooling'
        });
      });
    } else if (vizMode === 'aggregated') {
      // Show aggregated heating/cooling totals
      chartSeries.push(
        {
          name: 'Total Heating',
          type: 'line',
          data: aggregatedStats.map(s => [s.timestamp, s.totalHeating]),
          yAxisIndex: 0,
          smooth: true,
          lineStyle: { width: 3, color: '#ff5722' },
          areaStyle: { opacity: 0.3, color: '#ff5722' }
        },
        {
          name: 'Total Cooling',
          type: 'line',
          data: aggregatedStats.map(s => [s.timestamp, s.totalCooling]),
          yAxisIndex: 0,
          smooth: true,
          lineStyle: { width: 3, color: '#2196f3' },
          areaStyle: { opacity: 0.3, color: '#2196f3' }
        },
        {
          name: 'Active H/C Zones',
          type: 'bar',
          data: aggregatedStats.map(s => [s.timestamp, s.activeHeating + s.activeCooling]),
          yAxisIndex: 1,
          itemStyle: {
            color: (params: any) => {
              const stats = aggregatedStats[params.dataIndex];
              return stats?.conflict ? '#ff9800' : '#4caf50';
            }
          },
          barWidth: '60%',
          z: 1
        }
      );
    } else {
      // Heatmap for building-wide view
      const heatmapData = heatingSeries.flatMap((hSeries, hIdx) => 
        coolingSeries.flatMap((cSeries, cIdx) => {
          // Find overlapping timestamps and calculate conflict intensity
          const conflicts: any[] = [];
          hSeries.data?.forEach((hPoint: any) => {
            const cPoint = cSeries.data?.find((d: any) => Math.abs(d[0] - hPoint[0]) < 60000);
            if (cPoint && hPoint[1] > 10 && cPoint[1] > 10) {
              const intensity = Math.min(hPoint[1], cPoint[1]);
              conflicts.push([hPoint[0], `${hIdx}-${cIdx}`, intensity]);
            }
          });
          return conflicts;
        })
      );

      chartSeries.push({
        name: 'HC Conflicts',
        type: 'heatmap',
        data: heatmapData,
        yAxisIndex: 0,
        label: { show: false },
        emphasis: {
          itemStyle: { shadowBlur: 10 }
        }
      });
    }

    // Add waste events as scatter plot
    if (wasteEvents.length > 0) {
      chartSeries.push({
        name: 'Waste Events',
        type: 'scatter',
        data: wasteEvents.map(event => [
          event.timestamp,
          100, // Top of chart
          event.wasteKW,
          event.type
        ]),
        yAxisIndex: 0,
        symbolSize: (value: any) => Math.max(8, Math.min(30, value[2] * 2)),
        itemStyle: {
          color: (params: any) => {
            switch (params.data[3]) {
              case 'reheat': return '#ff9800';
              case 'mixed': return '#ff5722';
              default: return '#f44336';
            }
          },
          opacity: 0.8
        },
        emphasis: {
          itemStyle: { opacity: 1 }
        },
        z: 100
      });

      // Add waste regions
      const wasteRegions = wasteEvents.map(event => ({
        xAxis: event.timestamp,
        itemStyle: {
          color: 'rgba(244, 67, 54, 0.1)',
          borderColor: '#f44336',
          borderWidth: 1,
          borderType: 'dashed'
        }
      }));

      if (chartSeries[0]) {
        chartSeries[0].markArea = {
          silent: true,
          data: wasteRegions.map(region => [region, region])
        };
      }
    }

    return mergeChartOptions(baseOptions, {
      ...CHART_DESIGN_TOKENS.animation[animationMode],
      title: title ? {
        ...CHART_DESIGN_TOKENS.title.base,
        text: title,
        subtext: `${processedData.totalZones} zones monitored - ${wasteEvents.length} waste events detected`,
        textStyle: {
          ...CHART_DESIGN_TOKENS.title.base.textStyle,
          color: theme.palette.text.primary,
        },
      } : undefined,
      tooltip: {
        ...CHART_DESIGN_TOKENS.tooltip.base,
        trigger: 'axis',
        axisPointer: { type: 'cross' },
        backgroundColor: theme.palette.mode === 'dark'
          ? 'rgba(50, 50, 50, 0.95)'
          : 'rgba(255, 255, 255, 0.95)',
        borderColor: theme.palette.mode === 'dark'
          ? 'rgba(255, 255, 255, 0.12)'
          : 'rgba(0, 0, 0, 0.12)',
        textStyle: {
          ...CHART_DESIGN_TOKENS.tooltip.base.textStyle,
          color: theme.palette.text.primary,
        },
        formatter: (params: any) => {
          const time = new Date(params[0].value[0]).toLocaleString();
          let html = `<strong>${time}</strong><br/>`;
          
          params.forEach((p: any) => {
            if (p.seriesName === 'Waste Events') {
              html += `${p.marker} <strong>WASTE DETECTED</strong><br/>`;
              html += `Energy Waste: ${p.value[2]?.toFixed(1)} ${detectionConfig.energyUnit}<br/>`;
              html += `Type: ${p.value[3]}<br/>`;
            } else {
              const unit = p.seriesName.includes('Temp') ? `°${detectionConfig.temperatureUnit}` : '%';
              html += `${p.marker} ${p.seriesName}: ${p.value[1]?.toFixed(1)}${unit}<br/>`;
            }
          });

          // Add waste event details if present
          const wasteEvent = wasteEvents.find(e => Math.abs(e.timestamp - params[0].value[0]) < 60000);
          if (wasteEvent) {
            html += `<br/><strong style="color: #f44336">⚠️ ${wasteEvent.message}</strong>`;
            html += `<br/>Affected Zones: ${wasteEvent.affectedZones.length}`;
            html += `<br/>Energy Waste: ${wasteEvent.wasteKW.toFixed(1)} ${detectionConfig.energyUnit}`;
            html += `<br/><em style="font-size: 0.9em">Calc: ${wasteEvent.wastePercent.toFixed(0)}% overlap × capacity</em>`;
          }
          
          return html;
        }
      },
      grid: {
        ...CHART_DESIGN_TOKENS.grid.base,
        left: '3%',
        right: '4%',
        bottom: showDataZoom ? '15%' : '3%',
        top: calculateGridTop({
          hasTitle: !!title,
          hasToolbox: true,
        }),
        containLabel: true
      },
      xAxis: {
        type: 'time',
        boundaryGap: false
      },
      yAxis: [
        {
          type: 'value',
          name: 'H/C Output (%)',
          position: 'left',
          min: 0,
          max: 100,
          axisLabel: { formatter: '{value}%' }
        },
        {
          type: 'value',
          name: 'Zone Count',
          position: 'right',
          min: 0,
          axisLabel: { formatter: '{value}' },
          show: vizMode === 'aggregated'
        }
      ],
      visualMap: vizMode === 'heatmap' ? {
        min: 0,
        max: 100,
        calculable: true,
        orient: 'vertical',
        right: '10',
        top: 'center',
        inRange: {
          color: ['#4caf50', '#ffeb3b', '#ff9800', '#f44336']
        },
        text: ['High Conflict', 'Low Conflict']
      } : undefined,
      series: chartSeries,
      dataZoom: showDataZoom ? [
        {
          type: 'inside',
          start: 0,
          end: 100
        },
        {
          type: 'slider',
          start: 0,
          end: 100
        }
      ] : undefined,
      toolbox: showToolbox ? {
        feature: {
          dataView: { show: enableDataView },
          restore: { show: true },
          saveAsImage: { show: true },
          dataZoom: { show: showDataZoom },
          magicType: { 
            show: true,
            type: ['line', 'bar', 'stack']
          }
        }
      } : undefined,
      legend: showLegend ? {
        data: chartSeries.map(s => s.name),
        bottom: 0,
        selected: vizMode === 'individual' ? 
          Object.fromEntries(chartSeries.map((s, i) => [s.name, i < 6])) : 
          undefined
      } : undefined
    }, {
      showToolbox,
      enableDataView,
      showDataZoom,
      showLegend,
      onEdit,
      onDelete
    });
  }, [processedData, detectionConfig, baseOptions, title, showToolbox, enableDataView, showDataZoom, showLegend, onEdit, onDelete]);

  // Point selection handlers
  const handlePointSelectionChange = (categoryId: string, points: any[]) => {
    setPointSelections(prev => new Map(prev).set(categoryId, points));
  };

  const handleSetupComplete = () => {
    setSetupMode(false);
    setSetupComplete(true);
    // Data fetching will be triggered automatically by useChartData hook
    console.log('Simultaneous H/C setup complete. Selected points:', {
      totalPoints: allSelectedPoints.length,
      ahuHeating: pointSelections.get('ahuHeating')?.length || 0,
      ahuCooling: pointSelections.get('ahuCooling')?.length || 0,
      zoneReheat: pointSelections.get('zoneReheat')?.length || 0,
      supplyTemp: pointSelections.get('supplyTemp')?.length || 0
    });
  };

  const handleSetupBack = () => {
    setSetupMode(true);
  };

  // Render guided setup
  if (setupMode) {
    return (
      <Paper sx={{ p: 3, height, overflow: 'auto' }}>
        <GuidedPointSelection
          title="Simultaneous Heating/Cooling Detection Setup"
          description="Select points to monitor for simultaneous heating and cooling detection"
          categories={SIMULTANEOUS_HC_CATEGORIES}
          availablePoints={availablePoints}
          selectedPoints={pointSelections}
          onSelectionChange={handlePointSelectionChange}
          onComplete={handleSetupComplete}
        />
        
        {/* Configuration section after points are selected */}
        {Array.from(pointSelections.values()).some(pts => pts.length > 0) && (
          <Box sx={{ mt: 3 }}>
            <Typography variant="h6" gutterBottom>
              Detection Configuration
            </Typography>
            <Grid container spacing={2}>
              <Grid item xs={12}>
                <Typography gutterBottom>
                  Waste Threshold ({detectionConfig.energyUnit})
                </Typography>
                <Slider
                  value={detectionConfig.wasteThreshold}
                  onChange={(e, value) => setDetectionConfig(prev => ({
                    ...prev,
                    wasteThreshold: value as number
                  }))}
                  valueLabelDisplay="auto"
                  min={0}
                  max={50}
                  step={1}
                  marks
                />
              </Grid>
              <Grid item xs={6}>
                <TextField
                  label="Heating Threshold (%)"
                  type="number"
                  value={detectionConfig.heatingThreshold}
                  onChange={(e) => setDetectionConfig(prev => ({
                    ...prev,
                    heatingThreshold: Number(e.target.value)
                  }))}
                  fullWidth
                />
              </Grid>
              <Grid item xs={6}>
                <TextField
                  label="Cooling Threshold (%)"
                  type="number"
                  value={detectionConfig.coolingThreshold}
                  onChange={(e) => setDetectionConfig(prev => ({
                    ...prev,
                    coolingThreshold: Number(e.target.value)
                  }))}
                  fullWidth
                />
              </Grid>
              <Grid item xs={6}>
                <FormControl fullWidth>
                  <InputLabel>Energy Unit</InputLabel>
                  <Select
                    value={detectionConfig.energyUnit}
                    onChange={(e) => setDetectionConfig(prev => ({
                      ...prev,
                      energyUnit: e.target.value as 'kW' | 'BTU/hr'
                    }))}
                  >
                    <MenuItem value="kW">kW</MenuItem>
                    <MenuItem value="BTU/hr">BTU/hr</MenuItem>
                  </Select>
                </FormControl>
              </Grid>
              <Grid item xs={6}>
                <TextField
                  label="Overlap Duration (min)"
                  type="number"
                  value={detectionConfig.overlapDuration}
                  onChange={(e) => setDetectionConfig(prev => ({
                    ...prev,
                    overlapDuration: Number(e.target.value)
                  }))}
                  fullWidth
                />
              </Grid>
            </Grid>
          </Box>
        )}
      </Paper>
    );
  }

  // Loading state
  if (loading || (setupComplete && dataLoading)) {
    return (
      <Box sx={{ display: 'flex', justifyContent: 'center', alignItems: 'center', height }}>
        <CircularProgress />
        {setupComplete && dataLoading && (
          <Typography variant="body2" sx={{ ml: 2 }}>
            Fetching data for {allSelectedPoints.length} selected points...
          </Typography>
        )}
      </Box>
    );
  }

  // Error state
  if (error || (setupComplete && dataError)) {
    return (
      <Alert severity="error" sx={{ height }}>
        {error?.message || dataError?.message}
      </Alert>
    );
  }

  // Main chart render
  return (
    <Box sx={{ height, position: 'relative' }}>
      {processedData && (
        <Box sx={{ position: 'absolute', top: 8, right: 8, zIndex: 1000 }}>
          <Stack direction="row" spacing={1} alignItems="center">
            <Chip 
              label={`${processedData.totalZones} Zones`}
              size="small"
              color="primary"
            />
            <Chip 
              label={processedData.vizMode.toUpperCase()}
              size="small"
              variant="outlined"
            />
            {processedData.wasteEvents.length > 0 && (
              <Chip 
                label={`${processedData.wasteEvents.length} Waste Events`}
                size="small"
                color="error"
                icon={<ErrorIcon />}
              />
            )}
            {processedData.wasteEvents.length > 0 && (
              <Chip 
                label={`${processedData.wasteEvents.reduce((sum, e) => sum + e.wasteKW, 0).toFixed(1)} ${detectionConfig.energyUnit} waste`}
                size="small"
                color="warning"
                icon={<Battery4Bar />}
              />
            )}
            <Tooltip title="How is energy waste calculated?">
              <IconButton size="small" onClick={() => setShowCalculationInfo(true)}>
                <Info />
              </IconButton>
            </Tooltip>
            <IconButton size="small" onClick={handleSetupBack}>
              <Settings />
            </IconButton>
          </Stack>
        </Box>
      )}
      
      <ReactECharts
        option={chartOptions}
        style={{ height: '100%', width: '100%' }}
        onChartReady={onChartReady}
        theme={theme.palette.mode}
      />
      
      {/* Calculation Methodology Dialog */}
      <Dialog 
        open={showCalculationInfo} 
        onClose={() => setShowCalculationInfo(false)}
        maxWidth="md"
        fullWidth
      >
        <DialogTitle>
          <Box sx={{ display: 'flex', alignItems: 'center', gap: 1 }}>
            <Calculate />
            Energy Waste Calculation Methodology
          </Box>
        </DialogTitle>
        <DialogContent>
          <Stack spacing={3} sx={{ mt: 2 }}>
            <Alert severity="info">
              Energy waste is calculated based on the simultaneous operation of heating and cooling systems, which represents energy fighting against itself.
            </Alert>
            
            <Box>
              <Typography variant="h6" gutterBottom>
                Calculation Formula
              </Typography>
              <Paper sx={{ p: 2, bgcolor: 'grey.50' }}>
                <Typography variant="body2" component="pre" sx={{ fontFamily: 'monospace' }}>
                  Energy Waste = Overlap% × System Capacity × Efficiency Penalty × Load Factor
                </Typography>
              </Paper>
            </Box>

            <Box>
              <Typography variant="h6" gutterBottom>
                Components Explained
              </Typography>
              <List>
                <ListItem>
                  <ListItemText
                    primary="Overlap Percentage"
                    secondary="The minimum of heating and cooling percentages. If heating is at 60% and cooling at 40%, overlap is 40%."
                  />
                </ListItem>
                <ListItem>
                  <ListItemText
                    primary="System Capacity"
                    secondary={`Estimated based on equipment count: ${ENERGY_FACTORS[detectionConfig.energyUnit].typicalAHUCapacity} ${detectionConfig.energyUnit} per AHU, ${ENERGY_FACTORS[detectionConfig.energyUnit].typicalZoneCapacity} ${detectionConfig.energyUnit} per zone`}
                  />
                </ListItem>
                <ListItem>
                  <ListItemText
                    primary="Efficiency Penalty"
                    secondary={`Combined inefficiency of simultaneous operation: 1/HeatingCOP (${ENERGY_FACTORS[detectionConfig.energyUnit].heatingCOP || ENERGY_FACTORS[detectionConfig.energyUnit].heatingEfficiency}) + 1/CoolingCOP (${ENERGY_FACTORS[detectionConfig.energyUnit].coolingCOP || ENERGY_FACTORS[detectionConfig.energyUnit].coolingEfficiency})`}
                  />
                </ListItem>
                <ListItem>
                  <ListItemText
                    primary="Load Factor (0.5)"
                    secondary="Accounts for partial load conditions and diversity - systems rarely operate at full capacity"
                  />
                </ListItem>
              </List>
            </Box>

            <Box>
              <Typography variant="h6" gutterBottom>
                Detection Thresholds
              </Typography>
              <Grid container spacing={2}>
                <Grid item xs={6}>
                  <Typography variant="body2" color="text.secondary">
                    Minimum Heating Signal: {detectionConfig.heatingThreshold}%
                  </Typography>
                </Grid>
                <Grid item xs={6}>
                  <Typography variant="body2" color="text.secondary">
                    Minimum Cooling Signal: {detectionConfig.coolingThreshold}%
                  </Typography>
                </Grid>
                <Grid item xs={6}>
                  <Typography variant="body2" color="text.secondary">
                    Waste Threshold: {detectionConfig.wasteThreshold} {detectionConfig.energyUnit}
                  </Typography>
                </Grid>
                <Grid item xs={6}>
                  <Typography variant="body2" color="text.secondary">
                    Duration Required: {detectionConfig.overlapDuration} minutes
                  </Typography>
                </Grid>
              </Grid>
            </Box>

            <Alert severity="warning">
              <Typography variant="body2">
                <strong>Note:</strong> These calculations are estimates based on typical equipment capacities. 
                For precise energy analysis, actual equipment capacities and measured flow rates should be used.
                The calculation assumes standard efficiency values which may vary based on your specific equipment.
              </Typography>
            </Alert>
          </Stack>
        </DialogContent>
        <DialogActions>
          <Button onClick={() => setShowCalculationInfo(false)}>Close</Button>
        </DialogActions>
      </Dialog>
    </Box>
  );
};
