/**
 * Generate heatmap gradient colors with smooth transitions
 *
 * Uses design tokens for deviation colors to ensure consistency.
 * For 'deviation' scheme, uses DEVIATION_COLORS.gradient (13-stop bluegreenred).
 *
 * @param scheme - Color scheme: 'deviation' uses design tokens, 'temperature' uses legacy gradient
 * @param steps - Number of interpolation steps (default 256 for smooth gradients)
 * @returns Array of hex color strings
 */
const generateHeatmapGradient = (scheme: string = 'deviation', steps: number = 256): string[] => {
  // Define key color stops
  let colorStops: Array<{ position: number; color: string }> = [];

  switch (scheme) {
    case 'temperature':
      // Legacy temperature scheme - kept for backward compatibility
      colorStops = [
        { position: 0, color: '#313695' },    // Deep blue
        { position: 0.25, color: '#74add1' }, // Light blue
        { position: 0.5, color: '#ffffbf' },  // Yellow (neutral)
        { position: 0.75, color: '#fdae61' }, // Orange
        { position: 1, color: '#a50026' },    // Deep red
      ];
      break;
    case 'deviation':
    default:
      // Use standard deviation gradient from design tokens
      // 13-stop gradient: blue (under)  green (optimal)  red (over)
      // Maps to -6 through 0 (optimal) to +6 deviations
      const deviationGradient = DEVIATION_COLORS.gradient;
      const deviationCount = deviationGradient.length;

      // Create evenly-spaced color stops from design tokens
      colorStops = deviationGradient.map((color, index) => ({
        position: index / (deviationCount - 1),
        color: color,
      }));
  }

  // Generate smooth gradient with specified number of steps
  const colors: string[] = [];

  for (let i = 0; i < steps; i++) {
    const position = i / (steps - 1);

    // Find the two color stops this position falls between
    let lowerStop = colorStops[0];
    let upperStop = colorStops[colorStops.length - 1];

    for (let j = 0; j < colorStops.length - 1; j++) {
      if (position >= colorStops[j].position && position <= colorStops[j + 1].position) {
        lowerStop = colorStops[j];
        upperStop = colorStops[j + 1];
        break;
      }
    }

    // Interpolate between the two colors
    const range = upperStop.position - lowerStop.position;
    const localPosition = range > 0 ? (position - lowerStop.position) / range : 0;

    const color = interpolateColor(lowerStop.color, upperStop.color, localPosition);
    colors.push(color);
  }

  return colors;
};

// Helper function to interpolate between two colors
const interpolateColor = (color1: string, color2: string, factor: number): string => {
  const c1 = hexToRgb(color1);
  const c2 = hexToRgb(color2);
  
  const r = Math.round(c1.r + (c2.r - c1.r) * factor);
  const g = Math.round(c1.g + (c2.g - c1.g) * factor);
  const b = Math.round(c1.b + (c2.b - c1.b) * factor);
  
  return rgbToHex(r, g, b);
};

// Convert hex to RGB
const hexToRgb = (hex: string): { r: number; g: number; b: number } => {
  const result = /^#?([a-f\d]{2})([a-f\d]{2})([a-f\d]{2})$/i.exec(hex);
  return result ? {
    r: parseInt(result[1], 16),
    g: parseInt(result[2], 16),
    b: parseInt(result[3], 16)
  } : { r: 0, g: 0, b: 0 };
};

// Convert RGB to hex
const rgbToHex = (r: number, g: number, b: number): string => {
  return "#" + ((1 << 24) + (r << 16) + (g << 8) + b).toString(16).slice(1);
};

// Calculate deviation from target/range
const calculateDeviation = (
  value: number,
  config?: {
    targetValue?: number;
    minValue?: number;
    maxValue?: number;
    tolerance?: number;
  }
): number => {
  if (!config) return 0;
  
  if (config.targetValue !== undefined) {
    // Target mode: deviation from target
    const deviation = value - config.targetValue;
    
    // Smart tolerance calculation based on target magnitude
    let effectiveTolerance: number;
    
    if (config.tolerance !== undefined) {
      // Use explicit tolerance if provided
      effectiveTolerance = config.tolerance;
    } else {
      // Auto-calculate tolerance based on target value magnitude
      const targetMagnitude = Math.abs(config.targetValue);
      
      if (targetMagnitude < 0.1) {
        // For very small values (like 0.02), use 10% of target as tolerance
        effectiveTolerance = targetMagnitude * 0.1;
      } else if (targetMagnitude < 1) {
        // For small values, use 5% of target
        effectiveTolerance = targetMagnitude * 0.05;
      } else if (targetMagnitude < 10) {
        // For medium values, use 2% of target
        effectiveTolerance = targetMagnitude * 0.02;
      } else {
        // For large values, use 1% of target
        effectiveTolerance = targetMagnitude * 0.01;
      }
      
      // Ensure minimum tolerance to avoid division issues
      effectiveTolerance = Math.max(effectiveTolerance, 0.0001);
    }
    
    // Normalize deviation by tolerance
    // This gives us a value where 1 = 100% of tolerance
    const normalizedDeviation = deviation / effectiveTolerance;
    
    // Cap at reasonable limits for visualization (-2 to +2 = -200% to +200%)
    return Math.max(-2, Math.min(2, normalizedDeviation));
    
  } else if (config.minValue !== undefined && config.maxValue !== undefined) {
    // Range mode: deviation from range
    const range = config.maxValue - config.minValue;
    
    if (value < config.minValue) {
      // Below range: negative deviation
      const deviation = (config.minValue - value) / range;
      return -Math.min(2, deviation); // Cap at -2
    } else if (value > config.maxValue) {
      // Above range: positive deviation
      const deviation = (value - config.maxValue) / range;
      return Math.min(2, deviation); // Cap at +2
    }
    return 0; // Within range
  }
  
  return 0;
};

// Calculate statistics for a series
const calculateSeriesStats = (values: number[]) => {
  const sorted = [...values].sort((a, b) => a - b);
  const n = sorted.length;
  
  const mean = values.reduce((sum, v) => sum + v, 0) / n;
  const median = n % 2 === 0 
    ? (sorted[n / 2 - 1] + sorted[n / 2]) / 2 
    : sorted[Math.floor(n / 2)];
  
  const q1 = sorted[Math.floor(n * 0.25)];
  const q3 = sorted[Math.floor(n * 0.75)];
  const iqr = q3 - q1;
  
  const variance = values.reduce((sum, v) => sum + Math.pow(v - mean, 2), 0) / n;
  const stdDev = Math.sqrt(variance);
  
  const min = sorted[0];
  const max = sorted[n - 1];
  const range = max - min;
  
  return { mean, median, stdDev, min, max, range, q1, q3, iqr };
};

// Calculate deviation using specified method
const calculateDeviationWithMethod = (
  value: number,
  stats: ReturnType<typeof calculateSeriesStats>,
  method: string,
  config?: {
    targetValue?: number;
    minValue?: number;
    maxValue?: number;
    tolerance?: number;
  }
): number => {
  switch (method) {
    case 'target':
      // Use configured target if available
      if (config?.targetValue !== undefined) {
        return calculateDeviation(value, config);
      }
      // Fall back to mean
      return (value - stats.mean) / (stats.stdDev || 1);
      
    case 'mean':
      // Deviation from mean in standard deviations (z-score)
      return stats.stdDev > 0 ? (value - stats.mean) / stats.stdDev : 0;
      
    case 'median':
      // Deviation from median in MAD (Median Absolute Deviation)
      const mad = stats.iqr / 1.35; // Approximate MAD from IQR
      return mad > 0 ? (value - stats.median) / mad : 0;
      
    case 'minmax':
      // Position within min-max range (-1 to +1)
      if (stats.range > 0) {
        const normalized = (value - stats.min) / stats.range; // 0 to 1
        return (normalized - 0.5) * 4; // Convert to -2 to +2
      }
      return 0;
      
    case 'percentage':
      // Percentage deviation from mean (good for large values)
      if (stats.mean !== 0) {
        const pctDev = ((value - stats.mean) / Math.abs(stats.mean)) * 100;
        return pctDev / 50; // Normalize to -2 to +2 range (100% = 2)
      }
      return 0;
      
    case 'iqr':
      // Deviation based on IQR (robust to outliers)
      if (stats.iqr > 0) {
        if (value < stats.q1) {
          return (value - stats.q1) / stats.iqr;
        } else if (value > stats.q3) {
          return (value - stats.q3) / stats.iqr;
        }
        return 0; // Within IQR = normal
      }
      return 0;
      
    case 'auto':
    default:
      // Auto-select best method based on data characteristics
      const cv = stats.stdDev / Math.abs(stats.mean); // Coefficient of variation
      
      // If configured target exists, use it
      if (config?.targetValue !== undefined) {
        return calculateDeviation(value, config);
      }
      
      // For data with high variation coefficient (>0.5), use percentage
      if (cv > 0.5 && stats.mean !== 0) {
        const pctDev = ((value - stats.mean) / Math.abs(stats.mean)) * 100;
        return pctDev / 50;
      }
      
      // For data with outliers (high IQR relative to range), use median/IQR
      if (stats.iqr > 0 && stats.iqr < stats.range * 0.5) {
        const mad = stats.iqr / 1.35;
        return mad > 0 ? (value - stats.median) / mad : 0;
      }
      
      // Default to z-score
      return stats.stdDev > 0 ? (value - stats.mean) / stats.stdDev : 0;
  }
};

      maxTime = Date.now();

      startDate,
      endDate,

        deviceBuckets.get(bucketTimestamp)!.push(point[1]);

          aggregatedValue = bucketValues.reduce((sum, v) => sum + v, 0) / bucketValues.length;

          const sorted = [...bucketValues].sort((a, b) => a - b);

                    <strong> How aggregation works:</strong> We combine ~{aggregationInfo.pointsPerCell} raw data points into each cell 
                    representing {getResolutionText(strategy.cellResolution)} of time. 
                    {strategy.aggregationMethod === 'max' ? 
                      ' We use the MAXIMUM value to ensure no anomalies or critical deviations are hidden during aggregation.' : 
                    strategy.aggregationMethod === 'mean' ? 
                      ' We use the AVERAGE to show typical behavior over the time period.' : 
                      ' We use the 95th PERCENTILE to show significant deviations while filtering noise.'}
                    {' '}<strong>Green = Normal</strong>, Blue = Below Target, Red/Orange = Above Target.
                    {' For full 5-minute resolution, select a time range under 7 days.'}
                  </Typography>
                </Alert>
              )}
            </Stack>
          </Paper>

          {/* Method and Stats Bar */}
          <Box sx={{ display: 'flex', gap: 1, flexWrap: 'wrap' }}>
            <Chip
              label={`Analysis: ${getMethodDisplayName(deviationMethod)}`}
              size="small"
              variant="outlined"
              sx={{ fontWeight: 500 }}
            />
            <Chip
              label={`${series.length} devices`}
              size="small"
              variant="outlined"
            />
            <Chip
              label={`${chartData.xAxis.length} time points`}
              size="small"
              variant="outlined"
            />
            <Chip
              label={`${aggregatedData.length.toLocaleString()} cells`}
              size="small"
              variant="outlined"
            />
          </Box>
        </Stack>
      </Box>
        
      {/* Chart Container */}
      <Box sx={{ 
        flexGrow: 1, 
        minHeight: 0,
        position: 'relative',
        p: 1
      }}>
        <ReactECharts
          ref={chartRef}
          option={chartOptions}
          style={{ height: '100%', width: '100%' }}
          theme={theme.palette.mode}
          opts={{ renderer: strategy.useWebGL ? 'canvas' : 'svg' }}
          onEvents={{
            click: handleCellClick,
          }}
        />
      </Box>
    </Paper>
  );
};
