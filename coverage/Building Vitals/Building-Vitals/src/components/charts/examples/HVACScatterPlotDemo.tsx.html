/**
 * HVAC Scatter Plot Demo Component
 * Demonstrates the enhanced scatter plot with comprehensive HVAC correlation analysis
 */

import React, { useState } from 'react';
import {
  Box,
  Card,
  CardContent,
  FormControl,
  InputLabel,
  Select,
  MenuItem,
  Switch,
  FormControlLabel,
  Chip,
  Alert,
  Typography,
  Divider,
  Grid,
  Button,
  Paper,
} from '@mui/material';
import { useTheme } from '@mui/material/styles';
import ScatterPlotContainer from '../containers/ScatterPlotContainer';
import { 
  HVAC_CORRELATION_PRESETS,
  type CorrelationResult,
  type HVACInsights,
  type CorrelationConfig 
} from '../../../utils/hvac/correlationAnalysis';

// Sample HVAC data for demonstration
const SAMPLE_HVAC_POINTS = [
  { name: 'outdoor_air_temp', unit: '°F', markerTag: 'OAT' },
  { name: 'total_energy', unit: 'kWh', markerTag: 'Energy' },
  { name: 'zone_temp_1', unit: '°F', markerTag: 'Zone1' },
  { name: 'damper_position_1', unit: '%', markerTag: 'Damper1' },
  { name: 'supply_air_temp', unit: '°F', markerTag: 'SAT' },
  { name: 'return_air_temp', unit: '°F', markerTag: 'RAT' },
  { name: 'relative_humidity', unit: '%RH', markerTag: 'RH' },
  { name: 'hvac_energy', unit: 'kWh', markerTag: 'HVAC' },
  { name: 'airflow_rate', unit: 'CFM', markerTag: 'Airflow' },
  { name: 'static_pressure', unit: 'inWC', markerTag: 'Pressure' },
];

interface DemoState {
  selectedPoints: typeof SAMPLE_HVAC_POINTS;
  xAxisPoint: string;
  yAxisPoint: string;
  correlationMode: 'time-value' | 'point-correlation';
  enableCorrelationAnalysis: boolean;
  hvacPreset: string;
  showConfidenceIntervals: boolean;
  showOutliers: boolean;
  enableClustering: boolean;
  showCorrelationStats: boolean;
  colorByTime: 'hour' | 'day' | 'week' | 'month' | 'none';
  enableBrushSelection: boolean;
  timeRange: string;
  autoSelectHVACPreset: boolean;
}

const HVACScatterPlotDemo: React.FC = () => {
  const theme = useTheme();
  
  const [state, setState] = useState<DemoState>({
    selectedPoints: SAMPLE_HVAC_POINTS,
    xAxisPoint: 'outdoor_air_temp',
    yAxisPoint: 'total_energy',
    correlationMode: 'point-correlation',
    enableCorrelationAnalysis: true,
    hvacPreset: '',
    showConfidenceIntervals: true,
    showOutliers: true,
    enableClustering: true,
    showCorrelationStats: true,
    colorByTime: 'hour',
    enableBrushSelection: true,
    timeRange: '24h',
    autoSelectHVACPreset: true,
  });

  const [analysisResults, setAnalysisResults] = useState<CorrelationResult | null>(null);
  const [hvacInsights, setHVACInsights] = useState<HVACInsights | null>(null);
  const [selectedPointsCount, setSelectedPointsCount] = useState(0);

  // Update state helper
  const updateState = (updates: Partial<DemoState>) => {
    setState(prev => ({ ...prev, ...updates }));
  };

  // Correlation configuration
  const correlationConfig: Partial<CorrelationConfig> = {
    method: 'pearson',
    confidenceLevel: 0.95,
    outlierThreshold: 2.5,
    enableClustering: state.enableClustering,
    clusterCount: 3,
    timeBasedAnalysis: state.colorByTime !== 'none',
  };

  // Handle correlation analysis results
  const handleCorrelationAnalysis = (analysis: CorrelationResult) => {
    setAnalysisResults(analysis);
  };

  // Handle HVAC insights
  const handleHVACInsights = (insights: HVACInsights) => {
    setHVACInsights(insights);
  };

  // Handle point selection
  const handlePointsSelected = (points: Array<{ x: number; y: number; timestamp?: number }>) => {
    setSelectedPointsCount(points.length);
  };

  // Load preset configuration
  const loadPreset = (presetId: string) => {
    const preset = HVAC_CORRELATION_PRESETS.find(p => p.id === presetId);
    if (preset) {
      updateState({
        hvacPreset: presetId,
        xAxisPoint: preset.xAxisPoint,
        yAxisPoint: preset.yAxisPoint,
        correlationMode: 'point-correlation',
      });
    }
  };

  return (
    <Box sx={{ p: 3, backgroundColor: theme.palette.background.default }}>
      <Typography variant="h4" gutterBottom color="primary">
        🏢 HVAC Correlation Analysis Demo
      </Typography>
      
      <Typography variant="body1" sx={{ mb: 3, color: 'text.secondary' }}>
        This demo showcases the enhanced scatter plot component with comprehensive HVAC correlation analysis capabilities.
        Configure the settings below to explore different analysis modes and visualizations.
      </Typography>

      <Grid container spacing={3}>
        {/* Configuration Panel */}
        <Grid item xs={12} lg={4}>
          <Card>
            <CardContent>
              <Typography variant="h6" gutterBottom>Configuration</Typography>
              
              {/* HVAC Presets */}
              <FormControl fullWidth sx={{ mb: 2 }}>
                <InputLabel>HVAC Preset</InputLabel>
                <Select
                  value={state.hvacPreset}
                  label="HVAC Preset"
                  onChange={(e) => loadPreset(e.target.value)}
                >
                  <MenuItem value="">
                    <em>Custom Configuration</em>
                  </MenuItem>
                  {HVAC_CORRELATION_PRESETS.map(preset => (
                    <MenuItem key={preset.id} value={preset.id}>
                      <Box>
                        <Typography variant="body2">{preset.name}</Typography>
                        <Typography variant="caption" color="text.secondary">
                          {preset.description}
                        </Typography>
                      </Box>
                    </MenuItem>
                  ))}
                </Select>
              </FormControl>

              {/* Point Selection */}
              <FormControl fullWidth sx={{ mb: 2 }}>
                <InputLabel>X-Axis Point</InputLabel>
                <Select
                  value={state.xAxisPoint}
                  label="X-Axis Point"
                  onChange={(e) => updateState({ xAxisPoint: e.target.value })}
                >
                  {SAMPLE_HVAC_POINTS.map(point => (
                    <MenuItem key={point.name} value={point.name}>
                      {point.name} ({point.unit})
                    </MenuItem>
                  ))}
                </Select>
              </FormControl>

              <FormControl fullWidth sx={{ mb: 2 }}>
                <InputLabel>Y-Axis Point</InputLabel>
                <Select
                  value={state.yAxisPoint}
                  label="Y-Axis Point"
                  onChange={(e) => updateState({ yAxisPoint: e.target.value })}
                >
                  {SAMPLE_HVAC_POINTS.map(point => (
                    <MenuItem key={point.name} value={point.name}>
                      {point.name} ({point.unit})
                    </MenuItem>
                  ))}
                </Select>
              </FormControl>

              {/* Time Range */}
              <FormControl fullWidth sx={{ mb: 2 }}>
                <InputLabel>Time Range</InputLabel>
                <Select
                  value={state.timeRange}
                  label="Time Range"
                  onChange={(e) => updateState({ timeRange: e.target.value })}
                >
                  <MenuItem value="1h">Last Hour</MenuItem>
                  <MenuItem value="6h">Last 6 Hours</MenuItem>
                  <MenuItem value="24h">Last 24 Hours</MenuItem>
                  <MenuItem value="7d">Last 7 Days</MenuItem>
                  <MenuItem value="30d">Last 30 Days</MenuItem>
                </Select>
              </FormControl>

              {/* Color By Time */}
              <FormControl fullWidth sx={{ mb: 2 }}>
                <InputLabel>Color By Time</InputLabel>
                <Select
                  value={state.colorByTime}
                  label="Color By Time"
                  onChange={(e) => updateState({ colorByTime: e.target.value as any })}
                >
                  <MenuItem value="none">No Time Coloring</MenuItem>
                  <MenuItem value="hour">Hour of Day</MenuItem>
                  <MenuItem value="day">Day of Week</MenuItem>
                  <MenuItem value="week">Week of Year</MenuItem>
                  <MenuItem value="month">Month</MenuItem>
                </Select>
              </FormControl>

              <Divider sx={{ my: 2 }} />

              {/* Analysis Options */}
              <Typography variant="subtitle2" gutterBottom>Analysis Options</Typography>
              
              <FormControlLabel
                control={
                  <Switch
                    checked={state.enableCorrelationAnalysis}
                    onChange={(e) => updateState({ enableCorrelationAnalysis: e.target.checked })}
                  />
                }
                label="Enable Correlation Analysis"
              />

              <FormControlLabel
                control={
                  <Switch
                    checked={state.showConfidenceIntervals}
                    onChange={(e) => updateState({ showConfidenceIntervals: e.target.checked })}
                  />
                }
                label="Show Confidence Intervals"
              />

              <FormControlLabel
                control={
                  <Switch
                    checked={state.showOutliers}
                    onChange={(e) => updateState({ showOutliers: e.target.checked })}
                  />
                }
                label="Highlight Outliers"
              />

              <FormControlLabel
                control={
                  <Switch
                    checked={state.enableClustering}
                    onChange={(e) => updateState({ enableClustering: e.target.checked })}
                  />
                }
                label="Enable Clustering"
              />

              <FormControlLabel
                control={
                  <Switch
                    checked={state.showCorrelationStats}
                    onChange={(e) => updateState({ showCorrelationStats: e.target.checked })}
                  />
                }
                label="Show Statistics Panel"
              />

              <FormControlLabel
                control={
                  <Switch
                    checked={state.enableBrushSelection}
                    onChange={(e) => updateState({ enableBrushSelection: e.target.checked })}
                  />
                }
                label="Enable Brush Selection"
              />

              <FormControlLabel
                control={
                  <Switch
                    checked={state.autoSelectHVACPreset}
                    onChange={(e) => updateState({ autoSelectHVACPreset: e.target.checked })}
                  />
                }
                label="Auto-detect HVAC Preset"
              />
            </CardContent>
          </Card>

          {/* Analysis Results */}
          {analysisResults && (
            <Card sx={{ mt: 2 }}>
              <CardContent>
                <Typography variant="h6" gutterBottom>Analysis Results</Typography>
                
                <Box sx={{ display: 'flex', flexWrap: 'wrap', gap: 1, mb: 2 }}>
                  <Chip 
                    label={`r = ${analysisResults.coefficient.toFixed(3)}`}
                    color={Math.abs(analysisResults.coefficient) > 0.7 ? 'success' : 'default'}
                    size="small"
                  />
                  <Chip 
                    label={`R² = ${analysisResults.rSquared.toFixed(3)}`}
                    color="primary"
                    size="small"
                  />
                  <Chip 
                    label={`p = ${analysisResults.pValue.toFixed(4)}`}
                    color={analysisResults.isStatisticallySignificant ? 'success' : 'warning'}
                    size="small"
                  />
                  {analysisResults.outlierCount > 0 && (
                    <Chip 
                      label={`${analysisResults.outlierCount} outliers`}
                      color="error"
                      size="small"
                    />
                  )}
                </Box>

                <Typography variant="body2" sx={{ fontFamily: 'monospace', mb: 1 }}>
                  {analysisResults.equation.formula}
                </Typography>

                <Typography variant="caption" color="text.secondary">
                  Correlation Strength: {analysisResults.significance}
                </Typography>
              </CardContent>
            </Card>
          )}

          {/* HVAC Insights */}
          {hvacInsights && (
            <Card sx={{ mt: 2 }}>
              <CardContent>
                <Typography variant="h6" gutterBottom>HVAC Insights</Typography>
                
                <Alert 
                  severity={
                    hvacInsights.systemHealth === 'excellent' || hvacInsights.systemHealth === 'good' 
                      ? 'success' 
                      : hvacInsights.systemHealth === 'fair' 
                        ? 'warning' 
                        : 'error'
                  }
                  sx={{ mb: 2 }}
                >
                  System Health: <strong>{hvacInsights.systemHealth.toUpperCase()}</strong>
                </Alert>

                <Typography variant="body2" sx={{ mb: 2 }}>
                  {hvacInsights.interpretation}
                </Typography>

                {hvacInsights.faultIndicators.length > 0 && (
                  <Box sx={{ mb: 2 }}>
                    <Typography variant="subtitle2" color="error.main" gutterBottom>
                      ⚠️ Fault Indicators
                    </Typography>
                    {hvacInsights.faultIndicators.map((fault, index) => (
                      <Alert key={index} severity="error" sx={{ mb: 1 }}>
                        <Typography variant="caption">
                          {fault.message}
                        </Typography>
                      </Alert>
                    ))}
                  </Box>
                )}

                {hvacInsights.energyOptimization.length > 0 && (
                  <Box>
                    <Typography variant="subtitle2" color="success.main" gutterBottom>
                      💡 Optimization Opportunities
                    </Typography>
                    {hvacInsights.energyOptimization.map((opp, index) => (
                      <Alert key={index} severity="info" sx={{ mb: 1 }}>
                        <Typography variant="caption">
                          {opp.opportunity}
                          {opp.estimatedSavings && (
                            <Box component="span" sx={{ fontWeight: 'bold', ml: 0.5 }}>
                              ({opp.estimatedSavings})
                            </Box>
                          )}
                        </Typography>
                      </Alert>
                    ))}
                  </Box>
                )}
              </CardContent>
            </Card>
          )}

          {/* Selection Info */}
          {selectedPointsCount > 0 && (
            <Paper sx={{ p: 2, mt: 2, backgroundColor: theme.palette.primary.main }}>
              <Typography variant="body1" sx={{ color: 'primary.contrastText' }}>
                Selected Points: {selectedPointsCount}
              </Typography>
              <Button 
                variant="outlined" 
                size="small" 
                sx={{ mt: 1, color: 'primary.contrastText', borderColor: 'primary.contrastText' }}
                onClick={() => setSelectedPointsCount(0)}
              >
                Clear Selection
              </Button>
            </Paper>
          )}
        </Grid>

        {/* Chart Area */}
        <Grid item xs={12} lg={8}>
          <Card sx={{ height: '800px' }}>
            <CardContent sx={{ height: '100%', p: 1 }}>
              <ScatterPlotContainer
                selectedPoints={state.selectedPoints}
                timeRange={state.timeRange}
                xAxisPoint={state.xAxisPoint}
                yAxisPoint={state.yAxisPoint}
                correlationMode={state.correlationMode}
                enableCorrelationAnalysis={state.enableCorrelationAnalysis}
                correlationConfig={correlationConfig}
                hvacPreset={state.hvacPreset}
                showConfidenceIntervals={state.showConfidenceIntervals}
                showOutliers={state.showOutliers}
                enableClustering={state.enableClustering}
                showCorrelationStats={state.showCorrelationStats}
                colorByTime={state.colorByTime}
                enableBrushSelection={state.enableBrushSelection}
                autoSelectHVACPreset={state.autoSelectHVACPreset}
                onPointsSelected={handlePointsSelected}
                onCorrelationAnalysis={handleCorrelationAnalysis}
                onHVACInsights={handleHVACInsights}
                height="100%"
                title={`${state.xAxisPoint} vs ${state.yAxisPoint} Analysis`}
                showRegression
                enableToolbox
                showExportToolbar
              />
            </CardContent>
          </Card>
        </Grid>
      </Grid>

      {/* Instructions */}
      <Card sx={{ mt: 3 }}>
        <CardContent>
          <Typography variant="h6" gutterBottom>How to Use</Typography>
          <Typography variant="body2" component="div">
            <strong>1. Select HVAC Preset:</strong> Choose from predefined HVAC correlation patterns or create custom configurations.
            <br />
            <strong>2. Configure Analysis:</strong> Enable different analysis features like clustering, outlier detection, and time-based coloring.
            <br />
            <strong>3. Interpret Results:</strong> Review correlation statistics, HVAC insights, and system health assessments.
            <br />
            <strong>4. Interactive Features:</strong> Use brush selection to analyze specific data regions, hover for detailed tooltips.
            <br />
            <strong>5. Export Data:</strong> Use the export toolbar to save charts and analysis data for reports.
          </Typography>
        </CardContent>
      </Card>
    </Box>
  );
};

export default HVACScatterPlotDemo;
