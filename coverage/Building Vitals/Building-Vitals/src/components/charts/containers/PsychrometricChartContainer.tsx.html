import React, { useMemo, useState, useCallback } from 'react';
import { Box, Card, CardContent, Typography, Switch, FormControlLabel, Select, MenuItem, FormControl, InputLabel, Alert } from '@mui/material';
import EChartsPsychrometric from '../EChartsPsychrometric';
import { useChartData } from '../../../hooks/useChartData';
import { formatNumberForDisplay } from '../../../utils/formatters';
import { 
  calculatePsychrometricProperties, 
  generateASHRAE55ComfortZone,
  calculateASHRAE55Comfort,
  calculateAdaptiveComfort,
  type PsychrometricProperties,
  type ASHRAE55ComfortResult,
  type AdaptiveComfortResult
} from '../../../utils/hvac/psychrometricCalculations';

interface PsychrometricDataPoint {
  temperature: number;
  humidity: number;
  name?: string;
  timestamp?: number;
  dewPoint?: number;
  wetBulb?: number;
  markerTags?: string[];
  pointMetadata?: any;
  properties?: PsychrometricProperties;
  comfort?: ASHRAE55ComfortResult;
  adaptiveComfort?: AdaptiveComfortResult;
}

interface PsychrometricChartContainerProps {
  // Chart configuration
  title?: string;
  height?: string | number;
  temperatureUnit?: 'C' | 'F';
  
  // API data points
  temperaturePoints?: Array<{
    id?: string;
    name: string;
    unit?: string;
    markerTag?: string;
  }>;
  humidityPoints?: Array<{
    id?: string;
    name: string;
    unit?: string;
    markerTag?: string;
  }>;
  outdoorTemperaturePoints?: Array<{
    id?: string;
    name: string;
    unit?: string;
    markerTag?: string;
  }>;
  
  // Time configuration
  timeRange?: string;
  
  // Display options
  showComfortZones?: boolean;
  showGridLines?: boolean;
  showEnthalpy?: boolean;
  showWetBulb?: boolean;
  showSpecificVolume?: boolean;
  colorByTime?: boolean;
  enableToolbox?: boolean;
  
  // ASHRAE 55 settings
  clothingInsulation?: number; // clo
  metabolicRate?: number; // met
  airVelocity?: number; // m/s
  comfortCategory?: 'I' | 'II' | 'III';
  enableAdaptiveComfort?: boolean;
  
  // Action handlers
  onEdit?: () => void;
  onDelete?: () => void;
}

const PsychrometricChartContainer: React.FC<PsychrometricChartContainerProps> = ({
  title = 'Psychrometric Chart',
  height = 600,
  temperatureUnit = 'C',
  temperaturePoints = [],
  humidityPoints = [],
  outdoorTemperaturePoints = [],
  timeRange, // Use timeRange from parent, no default
  showComfortZones = true,
  showGridLines = true,
  showEnthalpy = true,
  showWetBulb = true,
  showSpecificVolume = true,
  colorByTime = false,
  enableToolbox = true,
  clothingInsulation = 0.5,
  metabolicRate = 1.2,
  airVelocity = 0.1,
  comfortCategory = 'II',
  enableAdaptiveComfort = false,
  onEdit,
  onDelete
}) => {
  // Local state for interactive features
  const [unitSystem, setUnitSystem] = useState<'SI' | 'IP'>(temperatureUnit === 'C' ? 'SI' : 'IP');
  const [activeComfortZones, setActiveComfortZones] = useState({
    summer: true,
    winter: true,
    categoryI: false,
    categoryII: true,
    categoryIII: false
  });
  const [selectedSeason, setSelectedSeason] = useState<'summer' | 'winter'>('summer');
  const [pinnedPoint, setPinnedPoint] = useState<PsychrometricDataPoint | null>(null);

  // Fetch temperature data
  const {
    data: tempData,
    isLoading: tempLoading,
    error: tempError
  } = useChartData({
    selectedPoints: temperaturePoints,
    timeRange,
    enabled: temperaturePoints.length > 0
  });

  // Fetch humidity data
  const {
    data: humidityData,
    isLoading: humidityLoading,
    error: humidityError
  } = useChartData({
    selectedPoints: humidityPoints,
    timeRange,
    enabled: humidityPoints.length > 0
  });

  // Fetch outdoor temperature for adaptive comfort
  const {
    data: outdoorTempData,
    isLoading: outdoorTempLoading,
    error: outdoorTempError
  } = useChartData({
    selectedPoints: outdoorTemperaturePoints,
    timeRange,
    enabled: enableAdaptiveComfort && outdoorTemperaturePoints.length > 0
  });

  // Convert temperature units
  const convertTemperature = useCallback((temp: number, fromUnit: 'C' | 'F', toUnit: 'C' | 'F') => {
    if (fromUnit === toUnit) {return temp;}
    if (fromUnit === 'C' && toUnit === 'F') {return (temp * 9/5) + 32;}
    if (fromUnit === 'F' && toUnit === 'C') {return (temp - 32) * 5/9;}
    return temp;
  }, []);

  // Correlate temperature and humidity data into psychrometric points
  const psychrometricData = useMemo((): PsychrometricDataPoint[] => {
    if (!tempData?.series?.length || !humidityData?.series?.length) {return [];}

    const result: PsychrometricDataPoint[] = [];
    const tempSeries = tempData.series[0]; // Use first temperature series
    const humiditySeries = humidityData.series[0]; // Use first humidity series

    if (!tempSeries?.data || !humiditySeries?.data) {return [];}

    // Calculate average outdoor temperature for adaptive comfort
    let averageOutdoorTemp: number | undefined;
    if (enableAdaptiveComfort && outdoorTempData?.series?.length) {
      const outdoorData = outdoorTempData.series[0]?.data || [];
      if (outdoorData.length > 0) {
        const sum = outdoorData.reduce((acc, [_, value]) => acc + value, 0);
        averageOutdoorTemp = sum / outdoorData.length;
      }
    }

    // Create psychrometric data points by correlating temperature and humidity
    // Use latest values approach for real-time monitoring
    const latestTemp = tempSeries.data[tempSeries.data.length - 1];
    const latestHumidity = humiditySeries.data[humiditySeries.data.length - 1];

    if (latestTemp && latestHumidity) {
      const temperature = convertTemperature(latestTemp[1], 'C', unitSystem === 'SI' ? 'C' : 'F');
      const humidity = latestHumidity[1];

      // Calculate all psychrometric properties
      const tempCelsius = unitSystem === 'SI' ? temperature : convertTemperature(temperature, 'F', 'C');
      const properties = calculatePsychrometricProperties(tempCelsius, humidity);
      
      // Calculate ASHRAE 55 comfort
      const comfort = calculateASHRAE55Comfort(
        tempCelsius,
        tempCelsius, // Assume operative temp equals air temp
        airVelocity,
        humidity,
        metabolicRate,
        clothingInsulation
      );

      // Calculate adaptive comfort if enabled
      let adaptiveComfort: AdaptiveComfortResult | undefined;
      if (enableAdaptiveComfort && averageOutdoorTemp !== undefined) {
        const outdoorTempC = convertTemperature(averageOutdoorTemp, 'C', 'C');
        adaptiveComfort = calculateAdaptiveComfort(outdoorTempC, tempCelsius, comfortCategory);
      }

      const point: PsychrometricDataPoint = {
        temperature,
        humidity,
        timestamp: latestTemp[0],
        name: `${tempSeries.name} / ${humiditySeries.name}`,
        dewPoint: convertTemperature(properties.dewPointTemp, 'C', unitSystem === 'SI' ? 'C' : 'F'),
        wetBulb: convertTemperature(properties.wetBulbTemp, 'C', unitSystem === 'SI' ? 'C' : 'F'),
        markerTags: [
          ...(tempSeries.markerTags || []),
          ...(humiditySeries.markerTags || [])
        ],
        properties,
        comfort,
        adaptiveComfort
      };

      result.push(point);
    }

    // Add historical data points for time-based visualization
    if (colorByTime && tempSeries.data.length > 10) {
      const stepSize = Math.max(1, Math.floor(tempSeries.data.length / 20)); // Max 20 points
      
      for (let i = 0; i < tempSeries.data.length; i += stepSize) {
        const tempPoint = tempSeries.data[i];
        const humidityPoint = humiditySeries.data.find(([timestamp]) => 
          Math.abs(timestamp - tempPoint[0]) <= 5 * 60 * 1000 // Within 5 minutes
        );

        if (tempPoint && humidityPoint) {
          const temperature = convertTemperature(tempPoint[1], 'C', unitSystem === 'SI' ? 'C' : 'F');
          const humidity = humidityPoint[1];
          const tempCelsius = unitSystem === 'SI' ? temperature : convertTemperature(temperature, 'F', 'C');
          
          const properties = calculatePsychrometricProperties(tempCelsius, humidity);
          
          result.push({
            temperature,
            humidity,
            timestamp: tempPoint[0],
            name: `Historical ${new Date(tempPoint[0]).toLocaleTimeString('en-US', { hour: 'numeric', minute: '2-digit', hour12: true })}`,
            dewPoint: convertTemperature(properties.dewPointTemp, 'C', unitSystem === 'SI' ? 'C' : 'F'),
            wetBulb: convertTemperature(properties.wetBulbTemp, 'C', unitSystem === 'SI' ? 'C' : 'F'),
            properties
          });
        }
      }
    }

    return result;
  }, [
    tempData, 
    humidityData, 
    outdoorTempData, 
    unitSystem, 
    colorByTime, 
    enableAdaptiveComfort,
    clothingInsulation,
    metabolicRate,
    airVelocity,
    comfortCategory,
    convertTemperature
  ]);

  // Generate ASHRAE 55 comfort zones
  const comfortZones = useMemo(() => {
    const zones: Array<{ name: string; data: Array<[number, number]>; style: any }> = [];
    
    if (!showComfortZones) {return zones;}

    // Summer comfort zone
    if (activeComfortZones.summer) {
      const summerZone = generateASHRAE55ComfortZone('summer', clothingInsulation, metabolicRate, airVelocity);
      if (summerZone.length > 0) {
        const zoneData = summerZone.map(point => [
          convertTemperature(point.temperature, 'C', unitSystem === 'SI' ? 'C' : 'F'),
          point.humidity
        ] as [number, number]);
        
        zones.push({
          name: 'Summer Comfort Zone',
          data: zoneData,
          style: { fill: 'rgba(255, 193, 7, 0.2)', stroke: '#ffc107', strokeWidth: 2 }
        });
      }
    }

    // Winter comfort zone
    if (activeComfortZones.winter) {
      const winterZone = generateASHRAE55ComfortZone('winter', clothingInsulation * 1.5, metabolicRate, airVelocity);
      if (winterZone.length > 0) {
        const zoneData = winterZone.map(point => [
          convertTemperature(point.temperature, 'C', unitSystem === 'SI' ? 'C' : 'F'),
          point.humidity
        ] as [number, number]);
        
        zones.push({
          name: 'Winter Comfort Zone',
          data: zoneData,
          style: { fill: 'rgba(33, 150, 243, 0.2)', stroke: '#2196f3', strokeWidth: 2 }
        });
      }
    }

    return zones;
  }, [showComfortZones, activeComfortZones, clothingInsulation, metabolicRate, airVelocity, unitSystem, convertTemperature]);

  // Handle loading and error states
  const isLoading = tempLoading || humidityLoading || (enableAdaptiveComfort && outdoorTempLoading);
  const error = tempError || humidityError || (enableAdaptiveComfort && outdoorTempError);

  // Handle point click for pinning properties display
  const handlePointClick = useCallback((point: PsychrometricDataPoint) => {
    setPinnedPoint(point === pinnedPoint ? null : point);
  }, [pinnedPoint]);

  return (
    <Box>
      {/* Control Panel */}
      <Card sx={{ mb: 2 }}>
        <CardContent>
          <Box sx={{ display: 'flex', flexWrap: 'wrap', gap: 2, alignItems: 'center' }}>
            {/* Unit System */}
            <FormControl size="small" sx={{ minWidth: 120 }}>
              <InputLabel>Units</InputLabel>
              <Select
                value={unitSystem}
                onChange={(e) => setUnitSystem(e.target.value as 'SI' | 'IP')}
                label="Units"
              >
                <MenuItem value="SI">SI (°C)</MenuItem>
                <MenuItem value="IP">IP (°F)</MenuItem>
              </Select>
            </FormControl>

            {/* Season Selection */}
            <FormControl size="small" sx={{ minWidth: 120 }}>
              <InputLabel>Season</InputLabel>
              <Select
                value={selectedSeason}
                onChange={(e) => setSelectedSeason(e.target.value as 'summer' | 'winter')}
                label="Season"
              >
                <MenuItem value="summer">Summer</MenuItem>
                <MenuItem value="winter">Winter</MenuItem>
              </Select>
            </FormControl>

            {/* Comfort Zone Toggles */}
            <FormControlLabel
              control={
                <Switch
                  checked={activeComfortZones.summer}
                  onChange={(e) => setActiveComfortZones(prev => ({ 
                    ...prev, 
                    summer: e.target.checked 
                  }))}
                />
              }
              label="Summer Zone"
            />
            
            <FormControlLabel
              control={
                <Switch
                  checked={activeComfortZones.winter}
                  onChange={(e) => setActiveComfortZones(prev => ({ 
                    ...prev, 
                    winter: e.target.checked 
                  }))}
                />
              }
              label="Winter Zone"
            />

            {/* Property Lines Toggles */}
            <FormControlLabel
              control={
                <Switch
                  checked={showEnthalpy}
                  onChange={() => {/* handled by parent */}}
                />
              }
              label="Enthalpy Lines"
            />
          </Box>
        </CardContent>
      </Card>

      {/* Error Display */}
      {error && (
        <Alert severity="error" sx={{ mb: 2 }}>
          {error.message || 'Failed to load psychrometric data'}
        </Alert>
      )}

      {/* Pinned Point Properties */}
      {pinnedPoint && (
        <Card sx={{ mb: 2 }}>
          <CardContent>
            <Typography variant="h6" gutterBottom>
              Pinned Properties: {pinnedPoint.name}
            </Typography>
            <Box sx={{ display: 'grid', gridTemplateColumns: 'repeat(auto-fit, minmax(200px, 1fr))', gap: 2 }}>
              <Box>
                <Typography variant="body2" color="textSecondary">Dry Bulb Temperature</Typography>
                <Typography variant="body1">
                  {formatNumberForDisplay(pinnedPoint.temperature)}°{unitSystem === 'SI' ? 'C' : 'F'}
                </Typography>
              </Box>
              <Box>
                <Typography variant="body2" color="textSecondary">Relative Humidity</Typography>
                <Typography variant="body1">{formatNumberForDisplay(pinnedPoint.humidity)}%</Typography>
              </Box>
              <Box>
                <Typography variant="body2" color="textSecondary">Dew Point</Typography>
                <Typography variant="body1">
                  {formatNumberForDisplay(pinnedPoint.dewPoint || 0)}°{unitSystem === 'SI' ? 'C' : 'F'}
                </Typography>
              </Box>
              <Box>
                <Typography variant="body2" color="textSecondary">Wet Bulb</Typography>
                <Typography variant="body1">
                  {formatNumberForDisplay(pinnedPoint.wetBulb || 0)}°{unitSystem === 'SI' ? 'C' : 'F'}
                </Typography>
              </Box>
              {pinnedPoint.properties && (
                <>
                  <Box>
                    <Typography variant="body2" color="textSecondary">Enthalpy</Typography>
                    <Typography variant="body1">
                      {formatNumberForDisplay(pinnedPoint.properties.enthalpy)} kJ/kg
                    </Typography>
                  </Box>
                  <Box>
                    <Typography variant="body2" color="textSecondary">Specific Volume</Typography>
                    <Typography variant="body1">
                      {formatNumberForDisplay(pinnedPoint.properties.specificVolume)} m³/kg
                    </Typography>
                  </Box>
                </>
              )}
              {pinnedPoint.comfort && (
                <Box>
                  <Typography variant="body2" color="textSecondary">Comfort Assessment</Typography>
                  <Typography variant="body1">
                    Category {pinnedPoint.comfort.comfortCategory} - 
                    PMV: {formatNumberForDisplay(pinnedPoint.comfort.pmv)} - 
                    PPD: {formatNumberForDisplay(pinnedPoint.comfort.ppd)}%
                  </Typography>
                </Box>
              )}
            </Box>
          </CardContent>
        </Card>
      )}

      {/* Main Chart */}
      <EChartsPsychrometric
        data={psychrometricData}
        loading={isLoading}
        error={error}
        height={height}
        title={title}
        temperatureUnit={unitSystem === 'SI' ? 'C' : 'F'}
        showComfortZone={showComfortZones}
        showGridLines={showGridLines}
        colorByTime={colorByTime}
        enableToolbox={enableToolbox}
        comfortZones={comfortZones}
        showEnthalpy={showEnthalpy}
        showWetBulb={showWetBulb}
        showSpecificVolume={showSpecificVolume}
        onPointClick={handlePointClick}
        onEdit={onEdit}
        onDelete={onDelete}
      />
    </Box>
  );
};

export default PsychrometricChartContainer;
