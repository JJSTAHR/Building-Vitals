import React, { useMemo, useState } from 'react';
import { Box, Alert, Paper, Tabs, Tab, Typography } from '@mui/material';
import EChartsPerfectEconomizer from '../EChartsPerfectEconomizer';
import { useChartData } from '../../../hooks/useChartData';
import TimeSeriesAnalysis from '../economizer/TimeSeriesAnalysis';
import PerformanceMetrics from '../economizer/PerformanceMetrics';
import HVACCorrelations from '../economizer/HVACCorrelations';
import type {
  EconomizerConfig,
  EconomizerState} from '../../../utils/hvac/economizerCalculations';
import { 
  calculateEconomizerControl,
  generateEconomizerPerformanceReport,
  EconomizerMode,
  EconomizerType
} from '../../../utils/hvac/economizerCalculations';

interface PerfectEconomizerContainerProps {
  // Required points for economizer analysis
  oatPointName: string;    // Outdoor Air Temperature
  matPointName: string;    // Mixed Air Temperature
  ratPointName: string;    // Return Air Temperature
  
  // Optional points for enhanced analysis
  oaHumidityPointName?: string;     // Outdoor Air Humidity
  raHumidityPointName?: string;     // Return Air Humidity
  damperPositionPointName?: string; // OA Damper Position
  coolingStagePointName?: string;   // Cooling Stage Status
  heatingStagePointName?: string;   // Heating Stage Status
  
  // Configuration
  economizer?: Partial<EconomizerConfig>;
  timeRange?: string;
  height?: string | number;
  title?: string;
  enabled?: boolean;
  
  // Event handlers
  onEdit?: () => void;
  onDelete?: () => void;
}

interface TabPanelProps {
  children?: React.ReactNode;
  index: number;
  value: number;
}

function TabPanel(props: TabPanelProps) {
  const { children, value, index, ...other } = props;

  return (
    <div
      role="tabpanel"
      hidden={value !== index}
      id={`economizer-tabpanel-${index}`}
      aria-labelledby={`economizer-tab-${index}`}
      {...other}
    >
      {value === index && (
        <Box sx={{ p: 3 }}>
          {children}
        </Box>
      )}
    </div>
  );
}

const PerfectEconomizerContainer: React.FC<PerfectEconomizerContainerProps> = ({
  oatPointName,
  matPointName,
  ratPointName,
  oaHumidityPointName,
  raHumidityPointName,
  damperPositionPointName,
  coolingStagePointName,
  heatingStagePointName,
  economizer,
  timeRange, // Use timeRange from parent, no default
  height = 600,
  title = 'Perfect Economizer Analysis',
  enabled = true,
  onEdit,
  onDelete,
}) => {
  const [activeTab, setActiveTab] = useState(0);

  // Default economizer configuration
  const economizerConfig: EconomizerConfig = useMemo(() => ({
    type: EconomizerType.DIFFERENTIAL_DRY_BULB,
    highLimitShutoff: 70,
    minimumOADamperPosition: 15,
    mixedAirSetpoint: 55,
    returnAirTemp: 72,
    controlDeadband: 2,
    integrationSettings: {
      enabled: true,
      maxCoolingStage: 4,
      damperResponseTime: 300,
    },
    ...economizer,
  }), [economizer]);

  // Collect all point names for data fetching
  const pointNames = useMemo(() => {
    const points = [
      { name: oatPointName, unit: '°F', markerTag: 'oat' },
      { name: matPointName, unit: '°F', markerTag: 'mat' },
      { name: ratPointName, unit: '°F', markerTag: 'rat' },
    ];

    if (oaHumidityPointName) {
      points.push({ name: oaHumidityPointName, unit: '%RH', markerTag: 'oah' });
    }
    if (raHumidityPointName) {
      points.push({ name: raHumidityPointName, unit: '%RH', markerTag: 'rah' });
    }
    if (damperPositionPointName) {
      points.push({ name: damperPositionPointName, unit: '%', markerTag: 'damper' });
    }
    if (coolingStagePointName) {
      points.push({ name: coolingStagePointName, unit: '', markerTag: 'cooling' });
    }
    if (heatingStagePointName) {
      points.push({ name: heatingStagePointName, unit: '', markerTag: 'heating' });
    }

    return points;
  }, [
    oatPointName,
    matPointName, 
    ratPointName,
    oaHumidityPointName,
    raHumidityPointName,
    damperPositionPointName,
    coolingStagePointName,
    heatingStagePointName,
  ]);

  // Fetch time series data
  const { data, isLoading, error } = useChartData({
    selectedPoints: pointNames,
    timeRange,
    enabled,
  });

  // Process time series data for economizer analysis
  const processedData = useMemo(() => {
    if (!data?.series || data.series.length === 0) {
      return {
        economizerData: [],
        timeSeriesData: {},
        performanceReport: null,
        correlationData: null,
      };
    }

    // Extract individual time series
    const oatSeries = data.series.find(s => s.markerTag === 'oat' || s.name === oatPointName);
    const matSeries = data.series.find(s => s.markerTag === 'mat' || s.name === matPointName);
    const ratSeries = data.series.find(s => s.markerTag === 'rat' || s.name === ratPointName);
    const oaHumidSeries = data.series.find(s => s.markerTag === 'oah' || s.name === oaHumidityPointName);
    const raHumidSeries = data.series.find(s => s.markerTag === 'rah' || s.name === raHumidityPointName);
    const damperSeries = data.series.find(s => s.markerTag === 'damper' || s.name === damperPositionPointName);
    const coolingSeries = data.series.find(s => s.markerTag === 'cooling' || s.name === coolingStagePointName);
    const heatingSeries = data.series.find(s => s.markerTag === 'heating' || s.name === heatingStagePointName);

    if (!oatSeries?.data || !matSeries?.data || !ratSeries?.data) {
      console.warn('[PerfectEconomizerContainer] Missing required temperature data');
      return {
        economizerData: [],
        timeSeriesData: {},
        performanceReport: null,
        correlationData: null,
      };
    }

    // Create a combined time series with interpolated values
    const allTimestamps = new Set<number>();
    [oatSeries.data, matSeries.data, ratSeries.data].forEach(series => {
      series.forEach(([timestamp]) => allTimestamps.add(timestamp));
    });

    const sortedTimestamps = Array.from(allTimestamps).sort((a, b) => a - b);

    // Create economizer states for each timestamp
    const economizerStates: EconomizerState[] = [];
    const economizerDataPoints: Array<{
      outdoorTemp: number;
      mixedAirTemp: number;
      returnAirTemp: number;
      timestamp: number;
      damperPosition?: number;
      preheatActive?: boolean;
      coolingActive?: boolean;
      reheatActive?: boolean;
      mode?: string;
      optimalDamperPosition?: number;
      energySavingsPotential?: number;
      faultDetected?: boolean;
      faultSeverity?: 'low' | 'medium' | 'high' | 'critical';
    }> = [];

    sortedTimestamps.forEach(timestamp => {
      const oat = interpolateValue(oatSeries.data, timestamp);
      const mat = interpolateValue(matSeries.data, timestamp);
      const rat = interpolateValue(ratSeries.data, timestamp);
      
      if (oat === null || mat === null || rat === null) {return;}

      const oaHumidity = oaHumidSeries ? interpolateValue(oaHumidSeries.data, timestamp) || 50 : 50;
      const raHumidity = raHumidSeries ? interpolateValue(raHumidSeries.data, timestamp) || 50 : 50;
      const damperPos = damperSeries ? interpolateValue(damperSeries.data, timestamp) || 15 : 15;
      const coolingStage = coolingSeries ? interpolateValue(coolingSeries.data, timestamp) || 0 : 0;
      const heatingStage = heatingSeries ? interpolateValue(heatingSeries.data, timestamp) || 0 : 0;

      // Create economizer state
      const state: EconomizerState = {
        outdoorAirTemp: oat,
        outdoorAirHumidity: oaHumidity,
        returnAirTemp: rat,
        returnAirHumidity: raHumidity,
        mixedAirTemp: mat,
        coolingDemand: Math.max(0, coolingStage * 25), // Convert stage to percentage
        damperPosition: damperPos,
        mode: EconomizerMode.MINIMUM_OA, // Will be calculated
        timestamp: new Date(timestamp),
      };

      // Calculate optimal economizer control
      const optimalControl = calculateEconomizerControl(state, economizerConfig);
      
      // Detect faults by comparing actual vs optimal
      const damperDeviation = Math.abs(damperPos - optimalControl.damperPosition);
      const tempDeviation = Math.abs(mat - economizerConfig.mixedAirSetpoint);
      
      let faultSeverity: 'low' | 'medium' | 'high' | 'critical' | undefined;
      if (damperDeviation > 20 || tempDeviation > 5) {faultSeverity = 'high';}
      else if (damperDeviation > 10 || tempDeviation > 3) {faultSeverity = 'medium';}
      else if (damperDeviation > 5 || tempDeviation > 2) {faultSeverity = 'low';}

      // Check for simultaneous heating/cooling
      const simultaneousOperation = heatingStage > 0 && coolingStage > 0;

      economizerStates.push(state);

      economizerDataPoints.push({
        outdoorTemp: oat,
        mixedAirTemp: mat,
        returnAirTemp: rat,
        timestamp,
        damperPosition: damperPos,
        preheatActive: heatingStage > 0,
        coolingActive: coolingStage > 0,
        reheatActive: simultaneousOperation,
        mode: optimalControl.mode,
        optimalDamperPosition: optimalControl.damperPosition,
        energySavingsPotential: optimalControl.energySavings || 0,
        faultDetected: faultSeverity !== undefined || simultaneousOperation,
        faultSeverity,
      });
    });

    // Generate performance report
    const performanceReport = generateEconomizerPerformanceReport(economizerStates, economizerConfig);

    // Calculate correlation data for enhanced analysis
    const correlationData = calculateHVACCorrelations(data.series, economizerDataPoints);

    return {
      economizerData: economizerDataPoints,
      timeSeriesData: {
        oat: oatSeries.data,
        mat: matSeries.data,
        rat: ratSeries.data,
        damper: damperSeries?.data || [],
        cooling: coolingSeries?.data || [],
        heating: heatingSeries?.data || [],
      },
      performanceReport,
      correlationData,
    };
  }, [data, economizerConfig, oatPointName, matPointName, ratPointName]);

  const handleTabChange = (event: React.SyntheticEvent, newValue: number) => {
    setActiveTab(newValue);
  };

  if (error) {
    return (
      <Alert severity="error" sx={{ mt: 2 }}>
        Failed to load economizer data: {error.message}
      </Alert>
    );
  }

  return (
    <Box sx={{ width: '100%' }}>
      <Paper elevation={2}>
        <Tabs
          value={activeTab}
          onChange={handleTabChange}
          aria-label="economizer analysis tabs"
          sx={{ borderBottom: 1, borderColor: 'divider' }}
        >
          <Tab label="Perfect Economizer" />
          <Tab label="Time Series Analysis" />
          <Tab label="Performance Metrics" />
          <Tab label="HVAC Correlations" />
        </Tabs>

        <TabPanel value={activeTab} index={0}>
          <EChartsPerfectEconomizer
            data={processedData.economizerData}
            config={{
              mixedAirSetpoint: economizerConfig.mixedAirSetpoint,
              minimumOAPercentage: economizerConfig.minimumOADamperPosition,
              preheatSetpoint: economizerConfig.lowLimitShutoff || 32,
              coolingSetpoint: economizerConfig.highLimitShutoff,
              returnAirTemp: economizerConfig.returnAirTemp,
              economizerHighLimit: economizerConfig.highLimitShutoff,
              unit: 'F',
            }}
            loading={isLoading}
            error={error}
            height={height}
            title={title}
            showDiagnostics
            onEdit={onEdit}
            onDelete={onDelete}
          />
        </TabPanel>

        <TabPanel value={activeTab} index={1}>
          <TimeSeriesAnalysis
            timeSeriesData={processedData.timeSeriesData}
            economizerData={processedData.economizerData}
            config={economizerConfig}
            isLoading={isLoading}
            height={typeof height === 'number' ? height - 100 : '500px'}
          />
        </TabPanel>

        <TabPanel value={activeTab} index={2}>
          <PerformanceMetrics
            performanceReport={processedData.performanceReport}
            timeRange={timeRange}
            isLoading={isLoading}
          />
        </TabPanel>

        <TabPanel value={activeTab} index={3}>
          <HVACCorrelations
            correlationData={processedData.correlationData}
            isLoading={isLoading}
            height={typeof height === 'number' ? height - 100 : '500px'}
          />
        </TabPanel>
      </Paper>
    </Box>
  );
};

// Helper function to interpolate values at specific timestamps
function interpolateValue(data: Array<[number, number]>, timestamp: number): number | null {
  if (!data || data.length === 0) {return null;}

  // Find exact match first
  const exactMatch = data.find(([t]) => t === timestamp);
  if (exactMatch) {return exactMatch[1];}

  // Find surrounding points for interpolation
  let before: [number, number] | null = null;
  let after: [number, number] | null = null;

  for (const point of data) {
    if (point[0] <= timestamp) {
      before = point;
    } else if (point[0] > timestamp && !after) {
      after = point;
      break;
    }
  }

  // Extrapolation cases
  if (!before) {return after ? after[1] : null;}
  if (!after) {return before[1];}

  // Linear interpolation
  const timeDiff = after[0] - before[0];
  if (timeDiff === 0) {return before[1];}

  const valueRange = after[1] - before[1];
  const timeOffset = timestamp - before[0];
  
  return before[1] + (valueRange * timeOffset / timeDiff);
}
