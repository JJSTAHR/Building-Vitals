/**
 * ChartToolbar Component
 *
 * A unified, customizable toolbar for charts following the unified chart design system.
 * Provides standard actions (export, fullscreen, refresh) and supports custom toolbar items.
 *
 * Features:
 * - Standard toolbar items with icons and tooltips
 * - Customizable toolbar items via props
 * - Configurable 48px height (standard)
 * - Top/bottom positioning support
 * - Theme integration with glassmorphism
 * - Full accessibility support (ARIA labels, keyboard navigation)
 * - ECharts toolbox feature integration
 * - Responsive design with proper z-index management
 */

import React, { useState, useCallback } from 'react';
import {
  Box,
  IconButton,
  Tooltip,
  Divider,
  ButtonGroup,
  useTheme,
  alpha,
} from '@mui/material';
import {
  GetApp as ExportIcon,
  Fullscreen as FullscreenIcon,
  FullscreenExit as FullscreenExitIcon,
  Refresh as RefreshIcon,
  Settings as SettingsIcon,
  ZoomIn as ZoomInIcon,
  ZoomOut as ZoomOutIcon,
  RestartAlt as ResetIcon,
  Image as ImageIcon,
  TableChart as CsvIcon,
} from '@mui/icons-material';
import type { SxProps, Theme as MuiTheme } from '@mui/material/styles';
import { Z_INDEX_CONFIG } from '../../../utils/toolbarPositioning';
import { glassEffect } from '../../../theme/unifiedTheme';

// Toolbar position types
export type ToolbarPosition = 'top' | 'bottom';

// Standard toolbar item types
export type StandardToolbarItem =
  | 'export'
  | 'fullscreen'
  | 'refresh'
  | 'settings'
  | 'zoomIn'
  | 'zoomOut'
  | 'reset';

// Custom toolbar item definition
export interface CustomToolbarItem {
  id: string;
  icon: React.ReactNode;
  label: string;
  onClick: () => void;
  disabled?: boolean;
  hidden?: boolean;
  ariaLabel?: string;
  color?: 'primary' | 'secondary' | 'default' | 'error' | 'info' | 'success' | 'warning';
}

export interface ChartToolbarProps {
  /**
   * Position of the toolbar (top or bottom of chart)
   * @default 'top'
   */
  position?: ToolbarPosition;

  /**
   * Standard toolbar items to display
   * @default ['export', 'fullscreen', 'refresh']
   */
  standardItems?: StandardToolbarItem[];

  /**
   * Custom toolbar items to append after standard items
   */
  customItems?: CustomToolbarItem[];

  /**
   * Export handler
   */
  onExport?: (format: 'png' | 'csv') => void;

  /**
   * Fullscreen toggle handler
   */
  onFullscreenToggle?: () => void;

  /**
   * Refresh/reload data handler
   */
  onRefresh?: () => void;

  /**
   * Settings handler
   */
  onSettings?: () => void;

  /**
   * Zoom in handler
   */
  onZoomIn?: () => void;

  /**
   * Zoom out handler
   */
  onZoomOut?: () => void;

  /**
   * Reset chart handler (reset zoom, pan, etc.)
   */
  onReset?: () => void;

  /**
   * Current fullscreen state
   * @default false
   */
  isFullscreen?: boolean;

  /**
   * Whether any operation is in progress (shows loading state)
   * @default false
   */
  loading?: boolean;

  /**
   * Custom styles
   */
  sx?: SxProps<MuiTheme>;

  /**
   * Accessible label for the toolbar
   * @default 'Chart toolbar'
   */
  ariaLabel?: string;
}

/**
 * ChartToolbar component for unified chart system
 */
export const ChartToolbar: React.FC<ChartToolbarProps> = ({
  position = 'top',
  standardItems = ['export', 'fullscreen', 'refresh'],
  customItems = [],
  onExport,
  onFullscreenToggle,
  onRefresh,
  onSettings,
  onZoomIn,
  onZoomOut,
  onReset,
  isFullscreen = false,
  loading = false,
  sx,
  ariaLabel = 'Chart toolbar',
}) => {
  const theme = useTheme();
  const [exportMenuAnchor, setExportMenuAnchor] = useState<HTMLElement | null>(null);

  // Handle export with sub-menu for format selection
  const handleExportClick = useCallback((event: React.MouseEvent<HTMLButtonElement>) => {
    if (onExport) {
      setExportMenuAnchor(event.currentTarget);
    }
  }, [onExport]);

  const handleExportFormat = useCallback((format: 'png' | 'csv') => {
    onExport?.(format);
    setExportMenuAnchor(null);
  }, [onExport]);

  // Standard toolbar item configurations
  const standardItemConfigs: Record<StandardToolbarItem, {
    icon: React.ReactNode;
    label: string;
    onClick?: () => void;
    disabled?: boolean;
    ariaLabel: string;
  }> = {
    export: {
      icon: <ExportIcon fontSize="small" />,
      label: 'Export chart',
      onClick: handleExportClick as any,
      disabled: !onExport || loading,
      ariaLabel: 'Export chart data',
    },
    fullscreen: {
      icon: isFullscreen ? <FullscreenExitIcon fontSize="small" /> : <FullscreenIcon fontSize="small" />,
      label: isFullscreen ? 'Exit fullscreen' : 'Enter fullscreen',
      onClick: onFullscreenToggle,
      disabled: !onFullscreenToggle || loading,
      ariaLabel: isFullscreen ? 'Exit fullscreen mode' : 'Enter fullscreen mode',
    },
    refresh: {
      icon: <RefreshIcon fontSize="small" />,
      label: 'Refresh data',
      onClick: onRefresh,
      disabled: !onRefresh || loading,
      ariaLabel: 'Refresh chart data',
    },
    settings: {
      icon: <SettingsIcon fontSize="small" />,
      label: 'Chart settings',
      onClick: onSettings,
      disabled: !onSettings || loading,
      ariaLabel: 'Open chart settings',
    },
    zoomIn: {
      icon: <ZoomInIcon fontSize="small" />,
      label: 'Zoom in',
      onClick: onZoomIn,
      disabled: !onZoomIn || loading,
      ariaLabel: 'Zoom in on chart',
    },
    zoomOut: {
      icon: <ZoomOutIcon fontSize="small" />,
      label: 'Zoom out',
      onClick: onZoomOut,
      disabled: !onZoomOut || loading,
      ariaLabel: 'Zoom out on chart',
    },
    reset: {
      icon: <ResetIcon fontSize="small" />,
      label: 'Reset view',
      onClick: onReset,
      disabled: !onReset || loading,
      ariaLabel: 'Reset chart view',
    },
  };

  // Calculate position styles
  const positionStyles: SxProps<MuiTheme> = position === 'top'
    ? {
        top: 0,
        left: 0,
        right: 0,
        borderTopLeftRadius: theme.shape.borderRadius,
        borderTopRightRadius: theme.shape.borderRadius,
        borderBottom: `1px solid ${theme.palette.divider}`,
      }
    : {
        bottom: 0,
        left: 0,
        right: 0,
        borderBottomLeftRadius: theme.shape.borderRadius,
        borderBottomRightRadius: theme.shape.borderRadius,
        borderTop: `1px solid ${theme.palette.divider}`,
      };

  // Export sub-menu
  const renderExportMenu = () => {
    if (!exportMenuAnchor) return null;

    return (
      <Box
        sx={{
          position: 'absolute',
          top: position === 'top' ? '48px' : 'auto',
          bottom: position === 'bottom' ? '48px' : 'auto',
          left: exportMenuAnchor ? exportMenuAnchor.offsetLeft : 0,
          zIndex: Z_INDEX_CONFIG.UNIFIED_CHART_TOOLBAR + 1,
          ...glassEffect.medium,
          borderRadius: 1,
          padding: 0.5,
          display: 'flex',
          flexDirection: 'column',
          gap: 0.5,
        }}
      >
        <Tooltip title="Export as PNG image" placement="right">
          <IconButton
            size="small"
            onClick={() => handleExportFormat('png')}
            aria-label="Export as PNG"
            sx={{
              '&:hover': {
                backgroundColor: alpha(theme.palette.primary.main, 0.1),
              },
            }}
          >
            <ImageIcon fontSize="small" />
          </IconButton>
        </Tooltip>
        <Tooltip title="Export as CSV data" placement="right">
          <IconButton
            size="small"
            onClick={() => handleExportFormat('csv')}
            aria-label="Export as CSV"
            sx={{
              '&:hover': {
                backgroundColor: alpha(theme.palette.primary.main, 0.1),
              },
            }}
          >
            <CsvIcon fontSize="small" />
          </IconButton>
        </Tooltip>
      </Box>
    );
  };

  return (
    <>
      <Box
        role="toolbar"
        aria-label={ariaLabel}
        sx={{
          position: 'absolute',
          height: 48,
          display: 'flex',
          alignItems: 'center',
          justifyContent: 'flex-end',
          px: 2,
          gap: 0.5,
          zIndex: Z_INDEX_CONFIG.UNIFIED_CHART_TOOLBAR,
          ...glassEffect.light,
          backgroundColor: theme.palette.mode === 'dark'
            ? 'rgba(0, 0, 0, 0.6)'
            : 'rgba(255, 255, 255, 0.8)',
          backdropFilter: 'blur(10px)',
          WebkitBackdropFilter: 'blur(10px)',
          transition: 'all 0.3s cubic-bezier(0.4, 0, 0.2, 1)',
          '&:hover': {
            backgroundColor: theme.palette.mode === 'dark'
              ? 'rgba(0, 0, 0, 0.75)'
              : 'rgba(255, 255, 255, 0.95)',
          },
          ...positionStyles,
          ...sx,
        }}
      >
        <ButtonGroup
          variant="text"
          size="small"
          sx={{
            '& .MuiButtonGroup-grouped': {
              minWidth: 'auto',
              padding: theme.spacing(0.75),
            },
          }}
        >
          {/* Render standard items */}
          {standardItems.map((itemKey, index) => {
            const config = standardItemConfigs[itemKey];
            if (!config) return null;

            return (
              <React.Fragment key={itemKey}>
                {index > 0 && index % 3 === 0 && (
                  <Divider
                    orientation="vertical"
                    flexItem
                    sx={{
                      mx: 0.5,
                      borderColor: alpha(theme.palette.divider, 0.3),
                    }}
                  />
                )}
                <Tooltip title={config.label} placement={position === 'top' ? 'bottom' : 'top'}>
                  <span>
                    <IconButton
                      size="small"
                      onClick={config.onClick}
                      disabled={config.disabled}
                      aria-label={config.ariaLabel}
                      sx={{
                        color: theme.palette.text.primary,
                        '&:hover': {
                          backgroundColor: alpha(theme.palette.primary.main, 0.1),
                          color: theme.palette.primary.main,
                        },
                        '&:disabled': {
                          color: theme.palette.action.disabled,
                        },
                        transition: 'all 0.2s ease-in-out',
                      }}
                    >
                      {config.icon}
                    </IconButton>
                  </span>
                </Tooltip>
              </React.Fragment>
            );
          })}

          {/* Render custom items */}
          {customItems.length > 0 && standardItems.length > 0 && (
            <Divider
              orientation="vertical"
              flexItem
              sx={{
                mx: 0.5,
                borderColor: alpha(theme.palette.divider, 0.3),
              }}
            />
          )}

          {customItems.map((item) => {
            if (item.hidden) return null;

            return (
              <Tooltip key={item.id} title={item.label} placement={position === 'top' ? 'bottom' : 'top'}>
                <span>
                  <IconButton
                    size="small"
                    onClick={item.onClick}
                    disabled={item.disabled || loading}
                    aria-label={item.ariaLabel || item.label}
                    color={item.color || 'default'}
                    sx={{
                      '&:hover': {
                        backgroundColor: alpha(
                          item.color ? theme.palette[item.color].main : theme.palette.primary.main,
                          0.1
                        ),
                      },
                      transition: 'all 0.2s ease-in-out',
                    }}
                  >
                    {item.icon}
                  </IconButton>
                </span>
              </Tooltip>
            );
          })}
        </ButtonGroup>
      </Box>

      {/* Export format sub-menu */}
      {renderExportMenu()}

      {/* Clickaway listener for export menu */}
      {exportMenuAnchor && (
        <Box
          onClick={() => setExportMenuAnchor(null)}
          sx={{
            position: 'fixed',
            top: 0,
            left: 0,
            right: 0,
            bottom: 0,
            zIndex: Z_INDEX_CONFIG.UNIFIED_CHART_TOOLBAR,
          }}
        />
      )}
    </>
  );
};

export default ChartToolbar;
