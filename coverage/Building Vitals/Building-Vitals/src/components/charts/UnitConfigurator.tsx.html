import React, { useState } from 'react';
import {
  Dialog,
  DialogTitle,
  DialogContent,
  DialogActions,
  Button,
  TextField,
  List,
  ListItem,
  ListItemText,
  ListItemSecondaryAction,
  IconButton,
  Typography,
  Box,
  Alert,
  Autocomplete,
} from '@mui/material';
import { Edit, Check, Close } from '@mui/icons-material';
import { normalizeUnit } from '../../utils/unitNormalization';

interface UnitConfiguratorProps {
  open: boolean;
  onClose: () => void;
  points: Array<{
    id: string;
    name: string;
    unit?: string;
    markerTags?: string[];
  }>;
  onUnitsUpdated: (unitOverrides: Record<string, string>) => void;
  initialOverrides?: Record<string, string>;
}

// Common unit suggestions by category
const UNIT_SUGGESTIONS = {
  temperature: ['°F', '°C', 'K'],
  humidity: ['%RH', '%'],
  pressure: ['PSI', 'Pa', 'kPa', 'in w.c.', 'in Hg', 'bar', 'mbar'],
  flow: ['CFM', 'GPM', 'm³/h', 'L/s', 'L/min'],
  power: ['W', 'kW', 'MW', 'HP', 'BTU/hr', 'tons'],
  energy: ['kWh', 'MWh', 'BTU', 'therms'],
  electrical: ['V', 'A', 'kW', 'kVA', 'kVAR', 'PF', 'Hz'],
  speed: ['RPM', 'm/s', 'ft/min'],
  concentration: ['ppm', 'ppb', 'mg/m³'],
  percentage: ['%', 'ratio'],
  count: ['', 'units', 'cycles'],
};

// Flatten all suggestions for autocomplete
const ALL_UNITS = Array.from(new Set(Object.values(UNIT_SUGGESTIONS).flat()));

export const UnitConfigurator: React.FC<UnitConfiguratorProps> = ({
  open,
  onClose,
  points,
  onUnitsUpdated,
  initialOverrides = {},
}) => {
  const [unitOverrides, setUnitOverrides] = useState<Record<string, string>>(initialOverrides);
  const [editingId, setEditingId] = useState<string | null>(null);
  const [editValue, setEditValue] = useState('');

  const handleEdit = (pointId: string, currentUnit: string) => {
    setEditingId(pointId);
    setEditValue(unitOverrides[pointId] || currentUnit || '');
  };

  const handleSave = (pointId: string) => {
    setUnitOverrides((prev) => ({
      ...prev,
      [pointId]: editValue,
    }));
    setEditingId(null);
    setEditValue('');
  };

  const handleCancel = () => {
    setEditingId(null);
    setEditValue('');
  };

  const handleApply = () => {
    onUnitsUpdated(unitOverrides);
    onClose();
  };

  const getUnitCategory = (point: { markerTags?: string[] }): string => {
    if (!point.markerTags) {return '';}

    const tags = point.markerTags.map((t) => t.toLowerCase());

    if (tags.some((t) => t.includes('temp') || t.includes('temperature'))) {return 'temperature';}
    if (tags.some((t) => t.includes('humid') || t.includes('rh'))) {return 'humidity';}
    if (tags.some((t) => t.includes('press'))) {return 'pressure';}
    if (tags.some((t) => t.includes('flow') || t.includes('cfm') || t.includes('gpm')))
      {return 'flow';}
    if (tags.some((t) => t.includes('power') || t.includes('kw'))) {return 'power';}
    if (tags.some((t) => t.includes('energy') || t.includes('kwh'))) {return 'energy';}
    if (tags.some((t) => t.includes('volt') || t.includes('amp') || t.includes('current')))
      {return 'electrical';}
    if (tags.some((t) => t.includes('speed') || t.includes('rpm'))) {return 'speed';}
    if (tags.some((t) => t.includes('co2') || t.includes('voc') || t.includes('ppm')))
      {return 'concentration';}

    return '';
  };

  const getSuggestedUnits = (point: { markerTags?: string[] }): string[] => {
    const category = getUnitCategory(point);
    return category ? UNIT_SUGGESTIONS[category as keyof typeof UNIT_SUGGESTIONS] || [] : ALL_UNITS;
  };

  return (
    <Dialog open={open} onClose={onClose} maxWidth="md" fullWidth>
      <DialogTitle>Configure Units</DialogTitle>
      <DialogContent>
        <Alert severity="info" sx={{ mb: 2 }}>
          Configure display units for each point. The original units from the data source are shown,
          but you can override them for better display.
        </Alert>

        <List>
          {points.map((point) => {
            const originalUnit = point.unit || '(no unit)';
            const normalizedUnit = normalizeUnit(originalUnit);
            const overrideUnit = unitOverrides[point.id];
            const displayUnit = overrideUnit || normalizedUnit;
            const isEditing = editingId === point.id;
            const suggestedUnits = getSuggestedUnits(point);

            return (
              <ListItem key={point.id} divider>
                <ListItemText
                  primary={
                    <Box sx={{ display: 'flex', alignItems: 'center', gap: 1 }}>
                      <Typography variant="body1">{point.name}</Typography>
                      {point.markerTags && point.markerTags.length > 0 && (
                        <Typography variant="caption" color="text.secondary">
                          [{point.markerTags.slice(0, 2).join(', ')}]
                        </Typography>
                      )}
                    </Box>
                  }
                  secondary={
                    <Box sx={{ mt: 1 }}>
                      {isEditing ? (
                        <Autocomplete
                          freeSolo
                          options={suggestedUnits}
                          value={editValue}
                          onChange={(_, newValue) => setEditValue(newValue || '')}
                          onInputChange={(_, newValue) => setEditValue(newValue)}
                          renderInput={(params) => (
                            <TextField
                              {...params}
                              size="small"
                              autoFocus
                              placeholder="Enter unit"
                              helperText={`Original: ${originalUnit}`}
                            />
                          )}
                          sx={{ width: 200 }}
                        />
                      ) : (
                        <Box>
                          <Typography variant="body2">
                            Display: <strong>{displayUnit || '(no unit)'}</strong>
                          </Typography>
                          {overrideUnit && (
                            <Typography variant="caption" color="text.secondary">
                              Original: {normalizedUnit}
                            </Typography>
                          )}
                        </Box>
                      )}
                    </Box>
                  }
                />
                <ListItemSecondaryAction>
                  {isEditing ? (
                    <>
                      <IconButton onClick={() => handleSave(point.id)} color="primary" size="small">
                        <Check />
                      </IconButton>
                      <IconButton onClick={handleCancel} size="small">
                        <Close />
                      </IconButton>
                    </>
                  ) : (
                    <IconButton onClick={() => handleEdit(point.id, displayUnit)} size="small">
                      <Edit />
                    </IconButton>
                  )}
                </ListItemSecondaryAction>
              </ListItem>
            );
          })}
        </List>
      </DialogContent>
      <DialogActions>
        <Button onClick={onClose}>Cancel</Button>
        <Button onClick={handleApply} variant="contained">
          Apply Units
        </Button>
      </DialogActions>
    </Dialog>
  );
};
