/**
 * Calculate aggregated value from timeseries data
 */
const calculateAggregatedValue = (
  data: Array<[number, number]>,
  aggregation: 'sum' | 'average' | 'latest' | 'max' | 'min' = 'sum'
): number => {
  if (!data || data.length === 0) { return 0; }

  switch (aggregation) {
    case 'sum':
      return data.reduce((sum, [_, value]) => sum + value, 0);
    case 'average':
      return data.reduce((sum, [_, value]) => sum + value, 0) / data.length;
    case 'latest':
      return data[data.length - 1][1];
    case 'max':
      return Math.max(...data.map(([_, value]) => value));
    case 'min':
      return Math.min(...data.map(([_, value]) => value));
    default:
      return data.reduce((sum, [_, value]) => sum + value, 0);
  }
};

/**
 * Parse hierarchical structure from point names
 * Example: "AHU-1/Supply/Temp" -> ["AHU-1", "Supply", "Temp"]
 */
const parseHierarchyFromName = (name: string, separator: string = '/'): string[] => {
  if (!name) { return ['Unknown']; }
  
  // Split by separator and clean up parts
  const parts = name.split(separator)
    .map(part => part.trim())
    .filter(part => part.length > 0);
  
  return parts.length > 0 ? parts : [name];
};

/**
 * Build hierarchical tree structure from flat series data
 */
const buildHierarchicalTree = (
  seriesData: ChartSeriesData[],
  options: {
    separator: string;
    maxLevels: number;
    aggregation: 'sum' | 'average' | 'latest' | 'max' | 'min';
  }
): TreemapNode[] => {
  const { separator, maxLevels, aggregation } = options;
  const tree = new Map<string, TreemapNode>();
  
  // Process each series to build tree
  seriesData.forEach((series) => {
    const hierarchy = parseHierarchyFromName(series.name, separator);
    const truncatedHierarchy = hierarchy.slice(0, maxLevels);
    
    // Calculate aggregated value for this series
    const value = calculateAggregatedValue(series.data, aggregation);
    
    // Build path through hierarchy
    let currentPath = '';
    let currentLevel = tree;
    
    truncatedHierarchy.forEach((part, index) => {
      const isLast = index === truncatedHierarchy.length - 1;
      const pathKey = currentPath ? `${currentPath}${separator}${part}` : part;
      
      if (!currentLevel.has(pathKey)) {
        const node: TreemapNode = {
          name: part,
          level: index,
          path: truncatedHierarchy.slice(0, index + 1),
          parentName: index > 0 ? truncatedHierarchy[index - 1] : undefined,
          originalPointName: series.name,
          unit: series.unit,
          category: truncatedHierarchy[0], // Top-level category
          children: isLast ? undefined : [],
          value: isLast ? Math.abs(value) : 0, // Only leaf nodes get values initially
          timeseriesData: isLast ? series.data : undefined,
          aggregation,
        };
        currentLevel.set(pathKey, node);
      }
      
      currentPath = pathKey;
      
      // If not a leaf node, prepare for next level
      if (!isLast) {
        const node = currentLevel.get(pathKey)!;
        if (!node.children) {
          node.children = [];
        }
        // Convert children array to Map for easier building
        if (!node.childrenMap) {
          (node as any).childrenMap = new Map();
        }
        currentLevel = (node as any).childrenMap;
      }
    });
  });
  
  // Convert Maps back to arrays and calculate parent values
  const convertMapToArray = (nodeMap: Map<string, TreemapNode>): TreemapNode[] => {
    const nodes = Array.from(nodeMap.values());
    
    return nodes.map(node => {
      if ((node as any).childrenMap) {
        node.children = convertMapToArray((node as any).childrenMap);
        delete (node as any).childrenMap;
        
        // Calculate aggregated value for parent nodes
        if (node.children && node.children.length > 0) {
          const childValues = node.children.map(child => child.value || 0);
          switch (aggregation) {
            case 'sum':
              node.value = childValues.reduce((sum, val) => sum + val, 0);
              break;
            case 'average':
              node.value = childValues.reduce((sum, val) => sum + val, 0) / childValues.length;
              break;
            case 'max':
              node.value = Math.max(...childValues);
              break;
            case 'min':
              node.value = Math.min(...childValues);
              break;
            case 'latest':
            default:
              node.value = childValues.reduce((sum, val) => sum + val, 0); // Fallback to sum
              break;
          }
        }
      }
      
      return node;
    });
  };
  
  return convertMapToArray(tree);
};

/**
 * Calculate efficiency score for color mapping
 * This is a placeholder - can be enhanced with domain-specific logic
 */
const calculateEfficiency = (node: TreemapNode): number => {
  if (!node.timeseriesData || node.timeseriesData.length === 0) {
    return 0.5; // Neutral efficiency for non-leaf nodes
  }
  
  // Simple efficiency calculation based on value consistency
  const values = node.timeseriesData.map(([_, value]) => value);
  const mean = values.reduce((sum, val) => sum + val, 0) / values.length;
  const variance = values.reduce((sum, val) => sum + Math.pow(val - mean, 2), 0) / values.length;
  const coefficient = variance / (mean || 1);
  
  // Lower coefficient of variation = higher efficiency
  return Math.max(0, Math.min(1, 1 - coefficient / 10));
};

/**
 * Transform raw series data to TreemapNode structure
 */
const transformToTreemapData = (
  data: Array<{ name: string; data: Array<[number, number]>; unit?: string; markerTags?: string[] }>,
  props: EChartsTreemapProps
): TreemapNode[] => {
  if (!data || data.length === 0) return [];
  
  const {
    aggregationMethod = 'sum',
    hierarchySeparator = '/',
    maxHierarchyLevels = 4,
    autoDetectHierarchy = true,
    colorByEfficiency = false,
    efficiencyThresholds = { good: 0.8, warning: 0.6, critical: 0.4 },
  } = props;
  
  if (!autoDetectHierarchy) {
    // Simple flat structure
    return data.map((series) => ({
      name: series.name,
      value: Math.abs(calculateAggregatedValue(series.data, aggregationMethod)),
      unit: series.unit,
      timeseriesData: series.data,
      aggregation: aggregationMethod,
      originalPointName: series.name,
      level: 0,
      path: [series.name],
      efficiency: colorByEfficiency ? calculateEfficiency({ 
        name: series.name, 
        timeseriesData: series.data 
      } as TreemapNode) : undefined,
    }));
  }
  
  // Build hierarchical structure
  const hierarchicalData = buildHierarchicalTree(data, {
    separator: hierarchySeparator,
    maxLevels: maxHierarchyLevels,
    aggregation: aggregationMethod,
  });
  
  // Add efficiency calculations if enabled
  if (colorByEfficiency) {
    const addEfficiencyToTree = (nodes: TreemapNode[]): TreemapNode[] => {
      return nodes.map(node => ({
        ...node,
        efficiency: calculateEfficiency(node),
        children: node.children ? addEfficiencyToTree(node.children) : undefined,
      }));
    };
    
    return addEfficiencyToTree(hierarchicalData);
  }
  
  return hierarchicalData;
};

/**
 * Process treemap data to handle timeseries values and apply styling
 */
const processTreemapData = (nodes: TreemapNode[], efficiencyThresholds?: { good: number; warning: number; critical: number }): TreemapNode[] => {
  return nodes.map((node) => {
    if (node.children) {
      // Recursively process children
      return {
        ...node,
        children: processTreemapData(node.children, efficiencyThresholds),
      };
    } else if (node.timeseriesData && node.timeseriesData.length > 0) {
      // Leaf node with timeseries data
      const processedNode = {
        ...node,
        value: calculateAggregatedValue(node.timeseriesData, node.aggregation),
      };
      
      // Apply efficiency-based coloring
      if (node.efficiency !== undefined && efficiencyThresholds) {
        const { good, warning, critical } = efficiencyThresholds;
        let color: string;
        
        if (node.efficiency >= good) {
          color = '#4CAF50'; // Green
        } else if (node.efficiency >= warning) {
          color = '#FF9800'; // Orange
        } else if (node.efficiency >= critical) {
          color = '#F44336'; // Red
        } else {
          color = '#9E9E9E'; // Gray
        }
        
        processedNode.itemStyle = {
          ...processedNode.itemStyle,
          color,
        };
      }
      
      return processedNode;
    }
    return node;
  });
};

          onNodeSelect(params.data, path);
        }
      });
    }
    
    if (onChartReady) {
      onChartReady(chart);
    }
  };

  const validData = useMemo(() => {
    if (!Array.isArray(data)) { return []; }

    return data.filter(series => 
      series && 
      typeof series.name === 'string' &&

      series.data.length > 0

  }, [data]);

    <Box 
      ref={containerRef}
      sx={{ 
        position: 'relative', 
        height, 
        width: '100%',
        overflow: 'hidden',

        error={error || (validData.length === 0 ? new Error('No valid data provided for treemap visualization') : null)}
        height="100%"

        onChartReady={(chart) => {
          chartInstanceRef.current = chart;
          handleChartReady(chart);
        }}
      />
    </Box>
  );
};

const arePropsEqual = (prevProps: EChartsTreemapProps, nextProps: EChartsTreemapProps) => {
  // Check primitive props
  if (
    prevProps.loading !== nextProps.loading ||
    prevProps.error !== nextProps.error ||

    prevProps.sizeByValue !== nextProps.sizeByValue ||
    prevProps.enableDrillDown !== nextProps.enableDrillDown
  ) {
    return false;
  }
  
  // Check efficiency thresholds object
  if (prevProps.efficiencyThresholds !== nextProps.efficiencyThresholds) {
    if (!prevProps.efficiencyThresholds || !nextProps.efficiencyThresholds) {
      return false;
    }
    if (
      prevProps.efficiencyThresholds.good !== nextProps.efficiencyThresholds.good ||
      prevProps.efficiencyThresholds.warning !== nextProps.efficiencyThresholds.warning ||
      prevProps.efficiencyThresholds.critical !== nextProps.efficiencyThresholds.critical
    ) {
      return false;
    }
  }

  // Check data array - use reference equality for large datasets
  if (prevProps.data !== nextProps.data) {
    // Treemap data is often hierarchical and complex, so prefer reference equality
    // Only do deep comparison for very small datasets
    if (prevProps.data.length > 10 || nextProps.data.length > 10) {
      return false; // Use reference equality for large datasets
    }
    // Deep comparison for small datasets (checking only top level)
    if (prevProps.data.length !== nextProps.data.length) {
      return false;
    }
    for (let i = 0; i < prevProps.data.length; i++) {
      const prevNode = prevProps.data[i];
      const nextNode = nextProps.data[i];
      if (
        prevNode.name !== nextNode.name ||
        prevNode.value !== nextNode.value ||
        prevNode.children !== nextNode.children ||
        prevNode.timeseriesData !== nextNode.timeseriesData ||
        prevNode.aggregation !== nextNode.aggregation ||
        prevNode.unit !== nextNode.unit ||
        prevNode.itemStyle !== nextNode.itemStyle
      ) {
        return false;
      }
    }
  }

  // Check customColors array
  if (prevProps.customColors !== nextProps.customColors) {
    if (!prevProps.customColors || !nextProps.customColors) {
      return false;
    }
    if (prevProps.customColors.length !== nextProps.customColors.length) {
      return false;
    }
    for (let i = 0; i < prevProps.customColors.length; i++) {
      if (prevProps.customColors[i] !== nextProps.customColors[i]) {
        return false;
      }
    }
  }

  // Check callback references
  if (
    prevProps.onChartReady !== nextProps.onChartReady ||
    prevProps.onNodeSelect !== nextProps.onNodeSelect
  ) {
    return false;
  }

  return true;
};
