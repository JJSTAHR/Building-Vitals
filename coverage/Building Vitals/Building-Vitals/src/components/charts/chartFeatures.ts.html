/**
 * Standardized chart features configuration
 * Provides consistent toolbox, zoom, and interactive features across all chart types
 */

import type { EChartsOption } from 'echarts';

// Chart type categories for feature determination
export type ChartCategory = 'timeseries' | 'statistical' | 'distribution' | 'correlation' | 'gauge';

export interface ChartFeatureConfig {
  toolbox?: boolean;
  dataZoom?: boolean;
  dataView?: boolean;
  saveAsImage?: boolean;
  restore?: boolean;
  magicType?: boolean;
  brush?: boolean;
  timeline?: boolean;
  markPoint?: boolean;
  markLine?: boolean;
  markArea?: boolean;
  // Custom action handlers
  onEdit?: () => void;
  onDelete?: () => void;
}

// Default features by chart category
const categoryFeatures: Record<ChartCategory, ChartFeatureConfig> = {
  timeseries: {
    toolbox: true,
    dataZoom: true,
    dataView: true,
    saveAsImage: true,
    restore: true,
    magicType: true,
    markPoint: true,
    markLine: true,
    markArea: true,
    brush: true,
  },
  statistical: {
    toolbox: true,
    dataZoom: true,
    dataView: true,
    saveAsImage: true,
    restore: true,
    markLine: true,
  },
  distribution: {
    toolbox: true,
    dataView: true,
    saveAsImage: true,
    restore: true,
    brush: true,
  },
  correlation: {
    toolbox: true,
    dataView: true,
    saveAsImage: true,
    restore: true,
    brush: true,
  },
  gauge: {
    toolbox: true,
    saveAsImage: true,
    dataView: true,
  },
};

// Map chart types to categories
export const chartTypeCategory: Record<string, ChartCategory> = {
  // Timeseries charts
  TimeSeries: 'timeseries',
  AreaChart: 'timeseries',
  StackedArea: 'timeseries',
  Candlestick: 'timeseries',
  EnhancedLine: 'timeseries',

  // Statistical charts
  BoxPlot: 'statistical',
  BarChart: 'statistical',
  ControlBand: 'statistical', // Control Band charts (formerly SPC)
  SPC: 'statistical', // Backward compatibility alias

  // Distribution charts
  Heatmap: 'distribution',
  CalendarHeatmap: 'distribution',
  DeviceDeviationHeatmap: 'distribution',
  Psychrometric: 'distribution',

  // Correlation charts
  ScatterPlot: 'correlation',
  ParallelCoordinates: 'correlation',

  // HVAC Diagnostic Charts
  PerfectEconomizer: 'timeseries',
  ChilledWaterReset: 'timeseries',
  DPOptimization: 'timeseries',
  SimultaneousHC: 'timeseries',

  // Single value charts
  Gauge: 'gauge',
};

/**
 * Get standard toolbox configuration for a chart
 * @deprecated Use getStandardToolboxConfig from chartToolkitConfig.ts instead
 */
export const getToolboxConfig = (
  chartType: string,
  features?: Partial<ChartFeatureConfig>,
  theme?: 'light' | 'dark'
) => {
  const category = chartTypeCategory[chartType] || 'timeseries';
  const defaultFeatures = categoryFeatures[category];
  const mergedFeatures = { ...defaultFeatures, ...features };

  const toolboxConfig: any = {
    show: mergedFeatures.toolbox !== false,
    orient: 'horizontal',
    itemSize: 16,
    itemGap: 12,
    right: '20px',
    top: '60px', // Increased from 10px to avoid overlap with chart controls
    feature: {},
    iconStyle: {
      borderColor: theme === 'dark' ? '#666' : '#999',
      borderWidth: 1,
    },
    emphasis: {
      iconStyle: {
        borderColor: theme === 'dark' ? '#fff' : '#333',
        shadowColor: theme === 'dark' ? 'rgba(255,255,255,0.3)' : 'rgba(0,0,0,0.3)',
        shadowBlur: 4,
        shadowOffsetY: 2,
      },
    },
    tooltip: {
      show: true,
      position: 'bottom',
      backgroundColor: theme === 'dark' ? 'rgba(30,30,30,0.9)' : 'rgba(255,255,255,0.9)',
      borderColor: theme === 'dark' ? '#555' : '#ccc',
      textStyle: {
        color: theme === 'dark' ? '#fff' : '#333',
        fontSize: 12,
      },
    },
  };

  // Save as Image - Available for all charts
  if (mergedFeatures.saveAsImage) {
    toolboxConfig.feature.saveAsImage = {
      type: 'png',
      name: 'chart',
      backgroundColor: 'auto',
      title: 'Save as Image',
      pixelRatio: 2,
    };
  }

  // Data View - Show raw data in table format
  if (mergedFeatures.dataView) {
    toolboxConfig.feature.dataView = {
      title: 'Data View',
      readOnly: false,
      lang: ['Data View', 'Close', 'Refresh'],
      backgroundColor: theme === 'dark' ? '#1a1a1a' : '#fff',
      textareaColor: theme === 'dark' ? '#1a1a1a' : '#fff',
      textareaBorderColor: theme === 'dark' ? '#333' : '#ddd',
      textColor: theme === 'dark' ? '#fff' : '#000',
      buttonColor: theme === 'dark' ? '#3a84ff' : '#1976d2',
      buttonTextColor: '#fff',
    };
  }

  // Restore - Reset zoom and other interactions
  if (mergedFeatures.restore) {
    toolboxConfig.feature.restore = {
      title: 'Reset',
    };
  }

  // Data Zoom - Enable selection zoom
  if (mergedFeatures.dataZoom && category === 'timeseries') {
    toolboxConfig.feature.dataZoom = {
      title: {
        zoom: 'Area Zoom',
        back: 'Zoom Reset',
      },
      yAxisIndex: false, // Only zoom X axis for time series
    };
  }

  // Magic Type - Switch between chart types
  if (mergedFeatures.magicType && (chartType === 'TimeSeries' || chartType === 'BarChart')) {
    toolboxConfig.feature.magicType = {
      type: ['line', 'bar', 'stack'],
      title: {
        line: 'Line Chart',
        bar: 'Bar Chart',
        stack: 'Stack',
      },
    };
  }

  // Brush selection
  if (mergedFeatures.brush) {
    toolboxConfig.feature.brush = {
      type: ['rect', 'polygon', 'lineX', 'lineY', 'keep', 'clear'],
      title: {
        rect: 'Rectangle Selection',
        polygon: 'Polygon Selection',
        lineX: 'Horizontal Selection',
        lineY: 'Vertical Selection',
        keep: 'Keep Selection',
        clear: 'Clear Selection',
      },
    };
  }

  // Custom Edit button - positioned first for better visibility
  if (mergedFeatures.onEdit) {
    toolboxConfig.feature.myEdit = {
      show: true,
      title: 'Edit Chart Settings',
      icon: 'path://M3 17.25V21h3.75L17.81 9.94l-3.75-3.75L3 17.25zM20.71 7.04c.39-.39.39-1.02 0-1.41l-2.34-2.34c-.39-.39-1.02-.39-1.41 0l-1.83 1.83 3.75 3.75 1.83-1.83z',
      onclick: mergedFeatures.onEdit,
      emphasis: {
        iconStyle: {
          borderColor: '#4CAF50',
          shadowColor: 'rgba(76, 175, 80, 0.3)',
        },
      },
    };
  }

  // Custom Delete button - positioned second for visibility
  if (mergedFeatures.onDelete) {
    toolboxConfig.feature.myDelete = {
      show: true,
      title: 'Delete Chart',
      icon: 'path://M6 19c0 1.1.9 2 2 2h8c1.1 0 2-.9 2-2V7H6v12zM19 4h-3.5l-1-1h-5l-1 1H5v2h14V4z',
      onclick: mergedFeatures.onDelete,
      emphasis: {
        iconStyle: {
          borderColor: '#F44336',
          shadowColor: 'rgba(244, 67, 54, 0.3)',
        },
      },
    };
  }

  return toolboxConfig;
};

/**
 * Get standard data zoom configuration
 */
export const getDataZoomConfig = (
  chartType: string,
  showSlider: boolean = true,
  startPercent: number = 0,
  endPercent: number = 100
) => {
  const category = chartTypeCategory[chartType];

  // Only time series and statistical charts get data zoom
  if (category !== 'timeseries' && category !== 'statistical') {
    return [];
  }

  const config = [
    {
      type: 'inside',
      start: startPercent,
      end: endPercent,
      minValueSpan: 10,
      maxValueSpan: 100,
      zoomOnMouseWheel: true,
      moveOnMouseMove: true,
      moveOnMouseWheel: false,
      preventDefaultMouseMove: true,
    },
  ];

  if (showSlider) {
    config.push({
      type: 'slider',
      show: true,
      start: startPercent,
      end: endPercent,
      height: 20,
      bottom: 10,
      borderColor: '#ccc',
      fillerColor: 'rgba(135, 175, 255, 0.2)',
      handleIcon:
        'path://M10.7,11.9v-1.3H9.3v1.3c-4.9,0.3-8.8,4.4-8.8,9.4c0,5,3.9,9.1,8.8,9.4v1.3h1.3v-1.3c4.9-0.3,8.8-4.4,8.8-9.4C19.5,16.3,15.6,12.2,10.7,11.9z',
      handleSize: '100%',
      handleStyle: {
        color: '#a7b7cc',
      },
      textStyle: {
        color: '#666',
      },
      filterMode: 'filter',
    } as any);
  }

  return config;
};

/**
 * Get mark line configuration for thresholds
 */
export const getMarkLineConfig = (
  thresholds?: Array<{ value: number; name: string; color?: string }>,
  unit?: string
) => {
  if (!thresholds || thresholds.length === 0) {
    return undefined;
  }

  return {
    symbol: 'none',
    animation: false,
    label: {
      formatter: (params: any) => {
        const threshold = thresholds.find((t) => t.value === params.value);
        return threshold ? `${threshold.name}: ${params.value}${unit || ''}` : '';
      },
      position: 'end',
    },
    lineStyle: {
      type: 'dashed',
      width: 2,
    },
    data: thresholds.map((threshold) => ({
      yAxis: threshold.value,
      name: threshold.name,
      lineStyle: {
        color: threshold.color || '#ff6b6b',
      },
      label: {
        color: threshold.color || '#ff6b6b',
      },
    })),
  };
};

/**
 * Get mark area configuration for ranges
 */
export const getMarkAreaConfig = (
  ranges?: Array<{ start: number; end: number; name: string; color?: string }>,
  _unit?: string
) => {
  if (!ranges || ranges.length === 0) {
    return undefined;
  }

  return {
    silent: true,
    itemStyle: {
      opacity: 0.3,
    },
    data: ranges.map((range) => [
      {
        yAxis: range.start,
        name: range.name,
        itemStyle: {
          color: range.color || 'rgba(255, 173, 177, 0.4)',
        },
      },
      {
        yAxis: range.end,
      },
    ]),
  };
};

/**
 * Enhanced tooltip configuration with better formatting
 */
export const getEnhancedTooltipConfig = (
  chartType: string,
  unit?: string,
  theme?: 'light' | 'dark'
) => {
  const category = chartTypeCategory[chartType];

  return {
    trigger: category === 'gauge' ? 'item' : 'axis',
    axisPointer: {
      type: category === 'correlation' ? 'cross' : 'line',
      label: {
        backgroundColor: theme === 'dark' ? '#6a7985' : '#333',
      },
      crossStyle: {
        color: '#999',
      },
    },
    backgroundColor: theme === 'dark' ? 'rgba(30, 30, 30, 0.95)' : 'rgba(255, 255, 255, 0.95)',
    borderColor: theme === 'dark' ? '#333' : '#ddd',
    borderWidth: 1,
    padding: [10, 15],
    textStyle: {
      color: theme === 'dark' ? '#fff' : '#333',
      fontSize: 12,
    },
    extraCssText: 'box-shadow: 0 2px 12px rgba(0,0,0,0.1); border-radius: 4px;',
    confine: true, // Keep tooltip within chart bounds
    formatter: (params: any) => {
      // Custom formatter can be overridden by individual charts
      if (Array.isArray(params) && params.length > 0) {
        // Safely get time/axis label - different for heatmaps vs other charts
        let time = 'Data Point';
        if (params[0].axisValueLabel) {
          time = params[0].axisValueLabel;
        } else if (params[0].name) {
          time = params[0].name;
        } else if (params[0].value && Array.isArray(params[0].value) && params[0].value[0]) {
          // For data in [timestamp, value] format
          const timestamp = params[0].value[0];
          if (typeof timestamp === 'number' && timestamp > 1000000000000) {
            time = new Date(timestamp).toLocaleString();
          } else {
            time = String(timestamp);
          }
        }
        
        let content = `<div style="margin-bottom: 8px; font-weight: 600;">${time}</div>`;

        params.forEach((item: any) => {
          if (item.value !== null && item.value !== undefined) {
            const value = Array.isArray(item.value) ? item.value[1] : item.value;
            const formattedValue = typeof value === 'number' ? value.toFixed(2) : value;
            content += `
              <div style="display: flex; align-items: center; margin: 4px 0;">
                <span style="display: inline-block; width: 10px; height: 10px; 
                  background-color: ${item.color}; border-radius: 50%; margin-right: 8px;"></span>
                <span style="flex: 1;">${item.seriesName}:</span>
                <span style="font-weight: 600; margin-left: 8px;">
                  ${formattedValue}${unit ? ` ${unit}` : ''}
                </span>
              </div>
            `;
          }
        });

        return content;
      }
      return '';
    },
  };
};

/**
 * Apply standard features to a chart option
 * @deprecated Use UniversalChartWrapper component instead for new charts
 */
export const applyStandardFeatures = (
  option: EChartsOption,
  chartType: string,
  config?: {
    features?: Partial<ChartFeatureConfig>;
    theme?: 'light' | 'dark';
    unit?: string;
    thresholds?: Array<{ value: number; name: string; color?: string }>;
    ranges?: Array<{ start: number; end: number; name: string; color?: string }>;
    showDataZoomSlider?: boolean;
    dataZoomStart?: number;
    dataZoomEnd?: number;
    onEdit?: () => void;
    onDelete?: () => void;
    hasHeader?: boolean; // New parameter to account for chart headers
  }
): EChartsOption => {
  const {
    features,
    theme = 'light',
    unit,
    thresholds,
    ranges,
    showDataZoomSlider = true,
    dataZoomStart = 0,
    dataZoomEnd = 100,
    onEdit,
    onDelete,
    hasHeader = true, // Default to true for backward compatibility
  } = config || {};

  // Apply toolbox
  const featuresWithHandlers = { ...features, onEdit, onDelete };
  const toolbox = getToolboxConfig(chartType, featuresWithHandlers, theme);
  if (toolbox.show) {
    option.toolbox = { ...option.toolbox, ...toolbox };
  }

  // Apply data zoom
  const dataZoom = getDataZoomConfig(chartType, showDataZoomSlider, dataZoomStart, dataZoomEnd);
  if (dataZoom.length > 0) {
    option.dataZoom = dataZoom;
  }

  // Enhance tooltip
  const enhancedTooltip = getEnhancedTooltipConfig(chartType, unit, theme);
  if (enhancedTooltip) {
    option.tooltip = {
      ...enhancedTooltip,
      ...(option.tooltip || {}),
    } as any;
  }

  // Ensure proper grid spacing to account for toolbox positioning and headers
  if (option.grid && typeof option.grid === 'object' && !Array.isArray(option.grid)) {
    option.grid = {
      top: hasHeader ? 80 : 50, // Increased top padding for toolbox and header
      right: 60, // Space for toolbox
      bottom: showDataZoomSlider ? 60 : 40, // Space for data zoom slider
      left: 60,
      containLabel: true,
      ...option.grid,
    };
  } else if (!option.grid) {
    option.grid = {
      top: hasHeader ? 80 : 50,
      right: 60,
      bottom: showDataZoomSlider ? 60 : 40,
      left: 60,
      containLabel: true,
    };
  }

  // Apply mark lines and areas for series
  if (option.series && Array.isArray(option.series)) {
    option.series = option.series.map((series: any) => {
      if (series.type === 'line' || series.type === 'bar') {
        return {
          ...series,
          markLine: thresholds ? getMarkLineConfig(thresholds, unit) : series.markLine,
          markArea: ranges ? getMarkAreaConfig(ranges, unit) : series.markArea,
        };
      }
      return series;
    });
  }

  return option;
};
