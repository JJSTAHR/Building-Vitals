/**
 * EChartsHeatmap - 2D heatmap visualization for time-series data patterns and correlations
 *
 * Transforms time-series data into a 2D heatmap where X-axis represents time buckets,
 * Y-axis represents different points/metrics, and color intensity shows value magnitude.
 * Perfect for visualizing patterns across multiple sensors over time, identifying
 * anomalies, and comparing performance across building systems.
 *
 * Features:
 * - Automatic time-series to grid transformation with configurable buckets
 * - Flexible time bucketing (5min, 15min, hourly, daily)
 * - Multiple normalization methods (minmax, zscore, baseline comparison)
 * - Color scheme presets for temperature, pressure, energy, etc.
 * - Deviation visualization with custom color scales
 * - Marker tags integration for point identification
 * - Interactive tooltips with original and normalized values
 * - Custom color scale configuration
 * - Progressive rendering for large heatmaps (>10k cells)
 * - Dark/light theme support
 *
 * @example
 * Basic heatmap from time-series data:
 * ```tsx
 * import EChartsHeatmap from '@/components/charts/EChartsHeatmap';
 *
 * function TemperatureHeatmap() {
 *   const data = [
 *     {
 *       name: 'Zone 1 Temp',
 *       data: [[timestamp1, 72], [timestamp2, 73], ...],
 *       unit: '°F'
 *     },
 *     {
 *       name: 'Zone 2 Temp',
 *       data: [[timestamp1, 71], [timestamp2, 72], ...],
 *       unit: '°F'
 *     }
 *   ];
 *
 *   return (
 *     <EChartsHeatmap
 *       data={data}
 *       title="Multi-Zone Temperature Patterns"
 *       timeBucket="hourly"
 *       height={600}
 *       showValues
 *     />
 *   );
 * }
 * ```
 *
 * @example
 * Advanced usage with normalization and baseline comparison:
 * ```tsx
 * function DeviationHeatmap() {
 *   const baselineValues = {
 *     'Zone 1': 72,
 *     'Zone 2': 71,
 *     'Zone 3': 73
 *   };
 *
 *   return (
 *     <EChartsHeatmap
 *       selectedPoints={zonePoints}
 *       timeRange="7d"
 *       timeBucket="daily"
 *       enableNormalization
 *       normalizationMethod="baseline"
 *       baselineValues={baselineValues}
 *       showAbsoluteValues={false}
 *       title="Temperature Deviation from Setpoints"
 *       presetType="deviation"
 *       showMarkerTags
 *       height={800}
 *     />
 *   );
 * }
 * ```
 *
 * @param props - Heatmap props extending UnifiedChartRendererProps
 * @returns Rendered heatmap with time-bucketed data visualization
 *
 * @see {@link transformTimeseriesToHeatmap} for data transformation details
 * @see {@link HEATMAP_COLOR_SCALES} for available color schemes
 *
 * @remarks
 * - Time bucketing aggregates values within each bucket using average by default
 * - Normalization is applied after bucketing for consistent scaling
 * - Baseline mode shows deviation from expected values
 * - Color scales automatically adjust to data range
 * - Large heatmaps (>10k cells) enable progressive rendering automatically
 */

/**
 * Transform timeseries data into 2D heatmap format
 * X-axis: time buckets, Y-axis: different points, Z-value: data values
 */
function transformTimeseriesToHeatmap(
  seriesData: Array<{
    name: string;
    data: Array<[number, number]>;
    unit?: string;
    markerTags?: string[];
    pointMetadata?: PointMetadata;
  }>,
  timeBucket: 'hourly' | 'daily' | '15min' | '5min' = 'hourly',
  options: {
    enableNormalization?: boolean;
    normalizationMethod?: 'minmax' | 'zscore' | 'baseline';
    baselineValues?: Record<string, number>;
    showAbsoluteValues?: boolean;
  } = {}
): {
  data: HeatmapDataPoint[];
  xCategories: string[];
  yCategories: string[];
  stats: { min: number; max: number; mean: number };
} {
  if (!seriesData || seriesData.length === 0) {
    return { data: [], xCategories: [], yCategories: [], stats: { min: 0, max: 0, mean: 0 } };
  }

  // Find time range across all series
  let minTime = Infinity;
  let maxTime = -Infinity;
  seriesData.forEach(series => {
    series.data.forEach(([timestamp]) => {
      minTime = Math.min(minTime, timestamp);
      maxTime = Math.max(maxTime, timestamp);
    });
  });

  if (!isFinite(minTime) || !isFinite(maxTime)) {
    return { data: [], xCategories: [], yCategories: [], stats: { min: 0, max: 0, mean: 0 } };
  }

  // Create time buckets based on interval using shared utilities
  const intervalMs = TimeAggregation.getIntervalMs(
    timeBucket === 'hourly' ? 'hourly' :
    timeBucket === 'daily' ? 'daily' :
    timeBucket === '15min' ? '15min' : '5min'
  );

  const timeBuckets: Array<{ timestamp: number; label: string }> = [];
  for (let t = Math.floor(minTime / intervalMs) * intervalMs; t <= maxTime; t += intervalMs) {
    const date = new Date(t);
    let label: string;
    
    if (timeBucket === 'hourly') {
      label = format(date, 'MM/dd h:mm a');
    } else if (timeBucket === 'daily') {
      label = format(date, 'MM/dd');
    } else {
      label = format(date, 'h:mm a');
    }
    
    timeBuckets.push({ timestamp: t, label });
  }

  // Create Y-axis categories (point names)
  const yCategories = seriesData.map(series => series.name);

  // Aggregate data into time buckets
  const bucketedData = new Map<string, Map<string, number[]>>();
  
  seriesData.forEach((series, yIndex) => {
    series.data.forEach(([timestamp, value]) => {
      const bucketTime = Math.floor(timestamp / intervalMs) * intervalMs;
      const bucketKey = bucketTime.toString();
      
      if (!bucketedData.has(bucketKey)) {
        bucketedData.set(bucketKey, new Map());
      }
      
      const bucket = bucketedData.get(bucketKey)!;
      if (!bucket.has(series.name)) {
        bucket.set(series.name, []);
      }
      
      bucket.get(series.name)!.push(value);
    });
  });

  // Calculate aggregated values and collect all values for normalization
  const allValues: number[] = [];
  const heatmapData: HeatmapDataPoint[] = [];

  timeBuckets.forEach((bucket, xIndex) => {
    const bucketData = bucketedData.get(bucket.timestamp.toString());
    
    yCategories.forEach((pointName, yIndex) => {
      const values = bucketData?.get(pointName);
      
      if (values && values.length > 0) {
        // Average values in bucket using shared utilities
        const avgValue = ChartStatistics.mean(values);
        allValues.push(avgValue);
        
        const series = seriesData.find(s => s.name === pointName);
        
        heatmapData.push({
          x: xIndex,
          y: yIndex,
          value: avgValue,
          pointName,
          formattedName: series?.name || pointName,
          markerTags: series?.markerTags,
          pointMetadata: series?.pointMetadata,
          timestamp: bucket.timestamp,
          actualValue: avgValue
        });
      }
    });
  });

  // Apply normalization if enabled
  if (options.enableNormalization && heatmapData.length > 0) {
    if (options.normalizationMethod === 'minmax') {
      const { min: globalMin, max: globalMax } = ChartStatistics.range(allValues);
      const range = globalMax - globalMin;
      
      if (range > 0) {
        heatmapData.forEach(point => {
          point.value = (point.actualValue! - globalMin) / range;
        });
      }
    } else if (options.normalizationMethod === 'zscore') {
      const mean = ChartStatistics.mean(allValues);
      const stdDev = ChartStatistics.standardDeviation(allValues);
      
      if (stdDev > 0) {
        heatmapData.forEach(point => {
          point.value = (point.actualValue! - mean) / stdDev;
        });
      }
    } else if (options.normalizationMethod === 'baseline' && options.baselineValues) {
      heatmapData.forEach(point => {
        const baseline = options.baselineValues![point.pointName!];
        if (baseline !== undefined) {
          point.deviation = DeviationCalculations.fromTarget(point.actualValue!, baseline);
          point.value = options.showAbsoluteValues ? point.actualValue! : point.deviation;
        }
      });
    }
  }

  // Calculate stats using shared utilities
  const finalValues = heatmapData.map(d => d.value);
  const range = ChartStatistics.range(finalValues);
  const stats = {
    min: range.min,
    max: range.max,
    mean: ChartStatistics.mean(finalValues)
  };

  return {
    data: heatmapData,
    xCategories: timeBuckets.map(b => b.label),
    yCategories,
    stats
  };
}

    return () => {
      if (chartInstanceRef.current?._resizeCleanup) {
        (chartInstanceRef.current)._resizeCleanup();
      }
    };
  }, []);


  // Get centralized base chart options
  const baseChartOptions = useBaseChartOptions();

  const option: EChartsOption = useMemo(() => {
    let heatmapData: HeatmapDataPoint[] = [];
    let xCats: string[] = [];
    let yCats: string[] = [];
    let stats = { min: 0, max: 0, mean: 0 };
    
    // Apply chart-specific transformation: Grid transformation for heatmaps
    if (validatedData.length > 0 && validatedData[0]?.data) {
      // Transform timeseries data to 2D heatmap format
      const transformResult = transformTimeseriesToHeatmap(
        validatedData,
        timeBucket,
        {
          enableNormalization,
          normalizationMethod,
          baselineValues,
          showAbsoluteValues
        }
      );
      
      heatmapData = transformResult.data;
      xCats = transformResult.xCategories;
      yCats = transformResult.yCategories;
      stats = transformResult.stats;
    } else if (validatedData.length > 0 && typeof validatedData[0] === 'object' && 'value' in validatedData[0]) {
      // Use provided data directly
      heatmapData = validatedData;
      
      // Extract unique categories if not provided
      xCats = xCategories || Array.from(new Set(validatedData.map((d) => String(d.x)))).sort();
      yCats = yCategories || Array.from(new Set(validatedData.map((d) => 
        d.formattedName || String(d.y)
      ))).sort();
      
      // Calculate stats using shared utilities
      const values = validatedData.map(d => d.value);
      if (values.length > 0) {
        const range = ChartStatistics.range(values);
        stats = {
          min: range.min,
          max: range.max,
          mean: ChartStatistics.mean(values)
        };
      }
    }

    // Handle empty data case using modular builders
    if (heatmapData.length === 0) {
      const titleOptions = buildChartTitle(title, theme);
      const xAxisOptions = buildXAxis({ type: 'category', data: [] });
      const yAxisOptions = buildYAxis({ type: 'category', data: [] });
      
      return combineOptions(baseChartOptions, {
        title: titleOptions,
        xAxis: xAxisOptions,
        yAxis: yAxisOptions,
        series: [],
        graphic: [
          {
            type: 'text',
            left: 'center',
            top: 'center',
            style: {
              text: 'No data available',
              fontSize: 16,
              fill: theme.palette.text.secondary,
            },
          },
        ],
      });
    }
    
    // Transform data for heatmap
    const values = heatmapData.map((d) => d.value);
    let displayValues = values;
    let minValue = stats.min;
    let maxValue = stats.max;

    // If deviation config is provided, transform values to deviations
    if (deviationConfig) {
      displayValues = transformToDeviationData(values, deviationConfig);
      minValue = displayValues.length > 0 ? Math.min(...displayValues) : 0;
      maxValue = displayValues.length > 0 ? Math.max(...displayValues) : 1;
    }

    // Transform data to array format [xIndex, yIndex, value] for ECharts
    const echartsHeatmapData = heatmapData.map((item, index) => {
      const xIndex = typeof item.x === 'number' ? item.x : xCats.indexOf(String(item.x));
      const yIndex = typeof item.y === 'number' ? item.y : yCats.indexOf(String(item.y));
      const displayValue = deviationConfig ? displayValues[index] : item.value;
      return [xIndex, yIndex, displayValue, item.actualValue || item.value]; // Keep original value for tooltip
    });

    // Create custom tooltip formatter
    const tooltipFormatter = (params: any) => {
      const xLabel = xCats[params.value[0]];
      const yLabel = yCats[params.value[1]];
      const originalValue = params.value[3] || params.value[2]; // Original value
      const displayValue = params.value[2]; // Display value (could be deviation)

      const pointData = heatmapData.find((d) => {
        const xIndex = typeof d.x === 'number' ? d.x : xCats.indexOf(String(d.x));
        const yIndex = typeof d.y === 'number' ? d.y : yCats.indexOf(String(d.y));
        return xIndex === params.value[0] && yIndex === params.value[1];
      });

      let result = `<div style="padding: 8px;">`;

      // Use marker tags if available
      if (pointData) {
        if (showMarkerTags && (pointData.markerTags || pointData.pointMetadata?.markerTags)) {
          const metadata: PointMetadata = {
            id: pointData.pointName || `${xLabel}-${yLabel}`,
            name: pointData.pointName || `${xLabel} - ${yLabel}`,
            markerTags: pointData.markerTags || pointData.pointMetadata?.markerTags,
            unit: unit,
            ...pointData.pointMetadata,
          };

          const formattedName = formatPointWithTags(metadata, {
            maxTags: 3,
            includeUnit: false,
          });
          result += `<strong>${formattedName}</strong><br/>`;
        } else if (pointData.pointName) {
          result += `<strong>${pointData.pointName}</strong><br/>`;
        }
        
        // Add timestamp for timeseries data
        if (pointData.timestamp) {
          result += `Time: ${new Date(pointData.timestamp).toLocaleString()}<br/>`;
        }
      }

      result += `${xLabel} - ${yLabel}<br/>`;
      
      // Show different value types based on configuration
      if (enableNormalization) {
        result += `Normalized: <strong>${formatNumberForDisplay(displayValue, 3)}</strong><br/>`;
        result += `Actual: <strong>${formatNumberForDisplay(originalValue, 2)}${unit ? ` ${unit}` : ''}</strong>`;
        
        if (pointData?.deviation !== undefined) {
          const deviationSign = pointData.deviation >= 0 ? '+' : '';
          result += `<br/>Deviation: <strong>${deviationSign}${formatNumberForDisplay(pointData.deviation, 2)}${unit ? ` ${unit}` : ''}</strong>`;
        }
      } else {
        result += `Value: <strong>${formatNumberForDisplay(originalValue, 2)}${unit ? ` ${unit}` : ''}</strong>`;
      }

      // Add marker tags as separate line if available
      if (showMarkerTags && pointData?.markerTags && pointData.markerTags.length > 0) {
        result += `<br/><span style="color: rgba(255,255,255,0.7); font-size: 11px;">`;
        result += `Tags: ${pointData.markerTags.slice(0, 5).join(', ')}`;
        result += `</span>`;
      }

      if (deviationConfig) {
        const deviation = calculateDeviation(originalValue, deviationConfig);
        const deviationSign = deviation.deviation >= 0 ? '+' : '';
        const severityColor =
          deviation.severity === 'critical'
            ? '#f44336'
            : deviation.severity === 'warning'
              ? '#ff9800'
              : '#4caf50';

        result += `<br/>Deviation: <span style="color: ${severityColor}; font-weight: bold;">`;
        result += `${deviationSign}${formatNumberForDisplay(deviation.deviation, 3)}${unit ? ` ${unit}` : ''}</span>`;

        if (deviationConfig.mode === 'single' && deviationConfig.setpoint !== undefined) {
          result += `<br/>Setpoint: ${deviationConfig.setpoint}${unit ? ` ${unit}` : ''}`;
        } else if (deviationConfig.mode === 'range') {
          result += `<br/>Range: ${deviationConfig.minSetpoint}-${deviationConfig.maxSetpoint}${unit ? ` ${unit}` : ''}`;
        }

        result += `<br/>Status: <span style="color: ${severityColor}; font-weight: bold;">`;
        result += `${deviation.isInRange ? 'In Range' : deviation.severity.toUpperCase()}</span>`;
      }

      result += '</div>';
      return result;
    };

    // Build chart options using modular builders
    const isDarkMode = theme.palette.mode === 'dark';

    const titleOptions = title ? {
      ...CHART_DESIGN_TOKENS.title.base,
      text: title,
      textStyle: {
        ...CHART_DESIGN_TOKENS.title.base.textStyle,
        color: theme.palette.text.primary,
      },
    } : undefined;

    const tooltipOptions = {
      ...CHART_DESIGN_TOKENS.tooltip.base,
      trigger: 'item',
      position: 'top',
      formatter: tooltipFormatter,
      backgroundColor: isDarkMode
        ? 'rgba(50, 50, 50, 0.95)'
        : 'rgba(255, 255, 255, 0.95)',
      borderColor: isDarkMode
        ? 'rgba(255, 255, 255, 0.12)'
        : 'rgba(0, 0, 0, 0.12)',
      textStyle: {
        ...CHART_DESIGN_TOKENS.tooltip.base.textStyle,
        color: theme.palette.text.primary,
      },
    };

    const gridOptions = {
      ...CHART_DESIGN_TOKENS.grid.base,
      height: '70%',
      top: calculateGridTop({
        hasTitle: !!title,
        hasToolbox: true,
      }),
      right: '10%',
      left: '15%',
    };
    
    const xAxisOptions = buildXAxis({
      type: 'category',
      data: xCats,
      axisLabel: {
        rotate: xCats.length > 10 ? 45 : 0,
        color: theme.palette.text.secondary,
      },
      splitLine: {
        show: true,
      },
    });
    
    const yAxisOptions = buildYAxis({
      type: 'category',
      data: yCats,
      axisLabel: {
        color: theme.palette.text.secondary,
      },
      splitLine: {
        show: true,
      },
    });
    
    const visualMapOptions = deviationConfig
      ? createDeviationVisualMap(displayValues, deviationConfig, {
          orient: 'horizontal',
          left: 'center',
          bottom: '5%',
        })
      : buildVisualMap({
          min: minValue,
          max: maxValue,
          calculable: true,
          orient: 'horizontal',
          left: 'center',
          bottom: '5%',
          textStyle: {
            color: theme.palette.text.secondary,
          },
          inRange: enableColorScaleConfig
            ? {
                color: generateColorScale(minValue, maxValue, colorStops).map(
                  (stop) => stop.color
                ),
              }
            : {
                color: HEATMAP_COLOR_SCHEMES[colorScheme],
              },
        });
    
    const heatmapSeries = SeriesBuilders.heatmap({
      name: deviationConfig ? 'Deviation Heatmap' : 
            enableNormalization ? 'Normalized Heatmap' : 
            'Heatmap',
      data: echartsHeatmapData,
      label: {
        show: showValues,
        formatter: (params: any) => {
          const value = showAbsoluteValues || !enableNormalization ? 
                        params.value[3] || params.value[2] : // Show original value
                        params.value[2]; // Show normalized value
          return formatNumberForDisplay(value, 1);
        },
        color: theme.palette.text.primary,
        fontSize: 10,
      },
      emphasis: {
        itemStyle: {
          shadowBlur: 10,
          shadowColor:
            theme.palette.mode === 'dark' ? 'rgba(255, 255, 255, 0.3)' : 'rgba(0, 0, 0, 0.5)',
          opacity: 1,
        },
      },
    });
    
    // Calculate adaptive animation
    const totalDataPoints = heatmapData.length;
    const animationMode = CHART_DESIGN_TOKENS.adaptive.animationMode(totalDataPoints);

    // Build chart options using modular approach
    const heatmapOptions = combineOptions(
      baseChartOptions,
      {
        ...CHART_DESIGN_TOKENS.animation[animationMode],
        title: titleOptions,
        tooltip: tooltipOptions,
        grid: gridOptions,
        xAxis: xAxisOptions,
        yAxis: yAxisOptions,
        visualMap: visualMapOptions,
        series: [heatmapSeries],
      },
      {
        showDataZoom: false, // Heatmaps typically don't need data zoom
        showLegend: false, // Heatmaps use visual map instead of legend
        showToolbox: enableToolbox,
        enableDataView: true, // Enabled for data export
        onEdit,
        onDelete,
      }
    );

    // Apply optimizations for large heatmaps
    const finalOption =
      totalDataPoints > 10000
        ? getOptimizedChartOptions(heatmapOptions, {
            dataSize: totalDataPoints,
            seriesCount: 1,
            enableProgressive: true,
            enableLargeMode: true,
            enableDataSampling: false, // Heatmaps don't need sampling
          })
        : heatmapOptions;

    return finalOption;
  }, [
    baseChartOptions,
    validatedData,
    title,
    colorScheme,
    xCategories,
    yCategories,
    timeRange,
    selectedPoints,
    timeBucket,
    enableNormalization,
    normalizationMethod,
    baselineValues,
    showAbsoluteValues,
    deviationConfig,
    showValues,
    theme.palette.mode,
    theme.palette.text.primary,
    theme.palette.text.secondary,
    enableToolbox,
    unit,
    showMarkerTags,
    enableColorScaleConfig,
    colorStops,
    onEdit,
    onDelete,
  ]);

  return (
    <Box sx={{ 
      position: 'relative', 
      height: typeof height === 'string' ? height : `${height}px`, 
      width: width || '100%',
      display: 'flex',
      flexDirection: 'column',
      minHeight: typeof height === 'number' ? height : 300, // Ensure minimum height
      // Force explicit height for proper chart rendering
      '& .echarts-wrapper': {
        height: '100% !important',
        width: '100% !important',
      },
    }}>
      
      <Box
        sx={{
          flex: '1 1 auto',
          position: 'relative',
          minHeight: 0,
          minWidth: 0,
          width: '100%',
          height: '100%',
        }}
      >
        <EChartsWrapper
          error={finalError}
          loading={finalLoading}
          option={option}
          height="100%"
          width="100%"
          theme={theme.palette.mode === 'dark' ? 'dark' : 'light'}
          style={{ 
            position: 'absolute',
            top: 0,
            left: 0,
            right: 0,
            bottom: 0,
            width: '100%',
            height: '100%',
          }}
          getChartInstance={(instance) => {
            chartInstanceRef.current = instance;
            
            // Call onChartReady if provided
            if (onChartReady) {
              onChartReady(instance);
            }
            
            // Set up resize listener for better responsiveness
            const handleResize = () => {
              if (instance) {
                instance.resize();
              }
            };
            
            // Listen for both standard resize and grid resize events
            window.addEventListener('resize', handleResize);
            window.addEventListener('grid-resize', handleResize);
            
            // Cleanup on unmount
            const cleanup = () => {
              window.removeEventListener('resize', handleResize);
              window.removeEventListener('grid-resize', handleResize);
            };
            
            // Store cleanup function on instance
            (instance)._resizeCleanup = cleanup;
          }}
        />
      </Box>
      
    </Box>
  );
};
