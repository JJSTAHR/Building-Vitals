  onVisualRangeChange?: (min: number | undefined, max: number | undefined) => void;
  showInline?: boolean;
}

  visualRangeMin,

    if (onVisualRangeChange) {
      onVisualRangeChange(numValue, visualRangeMax);
    }
  };

  const handleMaxChange = (value: string) => {
    const numValue = value === '' ? undefined : parseFloat(value);
    if (onVisualRangeChange) {
      onVisualRangeChange(visualRangeMin, numValue);

    <Box sx={{ maxWidth: 400, p: 1 }}>
      <Typography variant="subtitle2" gutterBottom sx={{ fontWeight: 'bold' }}>

    return (
      <Box sx={{ mt: 2 }}>
        <Box sx={{ display: 'flex', alignItems: 'center', gap: 1, mb: 1 }}>
          <Typography variant="subtitle2">
            Color Scale Focus

          <Tooltip title={helpTooltip} placement="right">
            <IconButton size="small">
