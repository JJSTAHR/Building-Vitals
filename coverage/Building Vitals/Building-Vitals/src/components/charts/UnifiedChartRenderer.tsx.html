interface UnifiedChartRendererProps {
  // Core props
  monitor: MonitorData;
  // REMOVED: timeRange prop - each monitor manages its own timeRange

  // Feature flags
  enableTools?: string[]; // ['edit', 'delete', 'export', 'fullscreen', 'refresh', 'zoom', 'dataView']

export const UnifiedChartRenderer: React.FC<UnifiedChartRendererProps> = ({

  enableTools = ['edit', 'delete', 'export', 'refresh', 'fullscreen'],
  enableLazyLoad = true,
  enableAutoResize = true,
  onEdit,
  onDelete,
  onRefresh,
  onExport,
  onFullscreen,
  onUpdate,
  height = 400,
  minHeight = 300,
  showLoading = true,
  loadingMessage = 'Loading chart...',
  errorMessage = 'Error loading chart',
  emptyMessage = 'No data available',
  debounceResize = 300,
  lazyLoadThreshold = 0.1,
}) => {
  const theme = useTheme();
  const containerRef = useRef<HTMLDivElement>(null);
  const chartInstanceRef = useRef<any>(null);
  const [isFullscreen, setIsFullscreen] = useState(false);
  const [exportMenuAnchor, setExportMenuAnchor] = useState<null | HTMLElement>(null);
  const [showDataView, setShowDataView] = useState(false);
  const [isHovered, setIsHovered] = useState(false);
  const [isRefreshing, setIsRefreshing] = useState(false);

  // Lazy loading setup
  const { ref: inViewRef, inView } = useInView({
    threshold: lazyLoadThreshold,
    triggerOnce: true,
    skip: !enableLazyLoad,
  });

  // Combine refs for intersection observer and container
  const setRefs = useCallback(
    (node: HTMLDivElement | null) => {
      containerRef.current = node;
      if (enableLazyLoad) {
        inViewRef(node);
      }
    },
    [inViewRef, enableLazyLoad]
  );

  // Fetch data
  // Debug logging for both Perfect Economizer AND Line charts for comparison
  const isDebugChart = monitor.chartType === 'PerfectEconomizer' || 
                       monitor.chartType === 'TimeSeries' || 
                       monitor.chartType === 'line';
  
  if (isDebugChart) {
    console.log('Debug data fetch setup for chart type:', {
      chartType: monitor.chartType,
      monitorId: monitor.id,
      monitorTitle: monitor.title,
      selectedPoints: monitor.selectedPoints?.map(p => ({
        id: p.id,
        name: p.name,
        unit: p.unit,
        hasId: !!p.id,
        hasName: !!p.name
      })),
      selectedPointsCount: monitor.selectedPoints?.length || 0,
      timeRange: monitor.timeRange || '24h', // Use monitor's own timeRange
      enabled: !!monitor.selectedPoints && monitor.selectedPoints.length > 0 && (!enableLazyLoad || inView),
      lazyLoadEnabled: enableLazyLoad,
      inView,
    });
  }
  
  // 🔍 Use ONLY monitor's own timeRange, default to 24h if not set
  const effectiveTimeRange = monitor.timeRange || '24h';
  
  console.log('🎯 [UnifiedChartRenderer] TimeRange for chart:', {
    monitorId: monitor.id,
    chartType: monitor.chartType,
    monitorTimeRange: monitor.timeRange,
    effectiveTimeRange,
    selectedPointsCount: monitor.selectedPoints?.length || 0,
    enabled: !!monitor.selectedPoints && monitor.selectedPoints.length > 0 && (!enableLazyLoad || inView),
  });

  const { data, series, isLoading, error } = useChartData({
    selectedPoints: monitor.selectedPoints || [],
    timeRange: effectiveTimeRange, // Use monitor's timeRange first, then global, then default to 24h
    monitorId: monitor.id, // Pass monitor ID for global time range synchronization
    enabled:
      !!monitor.selectedPoints && monitor.selectedPoints.length > 0 && (!enableLazyLoad || inView),
  });
  
  // Debug logging for data response comparison
  if (isDebugChart) {
    console.log('Debug data response for chart type:', {
      chartType: monitor.chartType,
      monitorId: monitor.id,
      monitorTitle: monitor.title,
      isLoading,
      hasError: !!error,
      errorMessage: error?.message,
      hasData: !!data,
      hasSeries: !!series,
      seriesLength: series?.length || 0,
      dataLength: data?.length || 0,
      seriesDetails: series?.map(s => ({
        name: s.name,
        dataPoints: s.data?.length || 0,
        hasData: !!(s.data && s.data.length > 0),
        firstDataPoint: s.data?.[0],
        lastDataPoint: s.data?.[s.data.length - 1],
        unit: s.unit
      })) || [],
      dataStructure: data ? {
        isArray: Array.isArray(data),
        hasPointSamples: !!(data && typeof data === 'object' && data.point_samples),
        keys: typeof data === 'object' ? Object.keys(data) : []
      } : null
    });
  }

  // Auto-resize handling
  useEffect(() => {
    if (!enableAutoResize || !containerRef.current || !chartInstanceRef.current) {return;}

    const handleResize = debounce(() => {
      if (chartInstanceRef.current && typeof chartInstanceRef.current.resize === 'function') {
        chartInstanceRef.current.resize();
      }
    }, debounceResize);

    const resizeObserver = new ResizeObserver(handleResize);
    resizeObserver.observe(containerRef.current);

    window.addEventListener('resize', handleResize);

    return () => {
      resizeObserver.disconnect();
      window.removeEventListener('resize', handleResize);
      handleResize.cancel();
    };
  }, [enableAutoResize, debounceResize]);

  // Handle chart ready
  const handleChartReady = useCallback((instance: any) => {
    chartInstanceRef.current = instance;
  }, []);

  // Cleanup on unmount
  useEffect(() => {
    return () => {
      // Dispose chart instance if it exists
      if (chartInstanceRef.current && typeof chartInstanceRef.current.dispose === 'function') {
        try {
          chartInstanceRef.current.dispose();
          chartInstanceRef.current = null;
        } catch (error) {
          console.error('Failed to dispose chart instance:', error);
        }
      }
    };
  }, []);

  // Handle fullscreen
  const handleFullscreenToggle = useCallback(() => {
    setIsFullscreen(!isFullscreen);
    onFullscreen?.();
    
    // Trigger chart resize after fullscreen toggle
    setTimeout(() => {
      if (chartInstanceRef.current && typeof chartInstanceRef.current.resize === 'function') {
        chartInstanceRef.current.resize();
      }
    }, 100);
  }, [isFullscreen, onFullscreen]);

  // Handle refresh with loading state
  const handleRefresh = useCallback(async () => {
    if (isRefreshing) return;
    
    setIsRefreshing(true);
    try {
      await onRefresh?.();
    } catch (error) {
      console.error('Failed to refresh data:', error);
    } finally {
      setIsRefreshing(false);
    }
  }, [onRefresh, isRefreshing]);

  // Handle export
  const handleExport = useCallback(
    (format: 'png' | 'csv' | 'json') => {
      setExportMenuAnchor(null);
      if (onExport) {
        onExport(format);
      } else {
        // Default export implementation
        if (format === 'png' && chartInstanceRef.current) {
          try {
            // Use the ECharts instance to get the data URL
            const echartsInstance = chartInstanceRef.current.getEchartsInstance 
              ? chartInstanceRef.current.getEchartsInstance() 
              : chartInstanceRef.current;
            
            if (echartsInstance && echartsInstance.getDataURL) {
              const url = echartsInstance.getDataURL({
                type: 'png',
                pixelRatio: 2,
                backgroundColor: '#ffffff',
              });
              const link = document.createElement('a');
              link.download = `${monitor.title}-${new Date().toISOString().slice(0, 10)}.png`;
              link.href = url;
              document.body.appendChild(link);
              link.click();
              document.body.removeChild(link);
            } else {
              console.error('Chart instance not available for PNG export');
            }
          } catch (error) {
            console.error('Failed to export PNG:', error);
          }
        }
        
        // CSV Export - simplified approach
        if (format === 'csv') {
          try {
            // Use series data directly for CSV export
            if (series && series.length > 0) {
              const csvData = series.flatMap(s => 
                s.data?.map((point: any) => ({
                  timestamp: new Date(point[0] || point.timestamp).toISOString(),
                  value: point[1] || point.value,
                  pointName: s.name,
                  unit: s.unit || '',
                })) || []
              );
              
              if (csvData.length > 0) {
                exportTimeseriesAsCSV(csvData, {
                  filename: `${monitor.title}-${new Date().toISOString().slice(0, 10)}.csv`,
                  includeHeaders: true,
                  dateFormat: 'iso',
                  precision: 3,
                  includeUnits: true,
                });
              } else {
                console.warn('No data available for CSV export');
              }
            }
          } catch (error) {
            console.error('Failed to export CSV:', error);
          }
        }
        
        // JSON Export with all chart data
        if (format === 'json') {
          try {
            const jsonData = {
              chartType: monitor.chartType,
              title: monitor.title,
              exportedAt: new Date().toISOString(),
              config: monitor.config,
              series: series || [],
              selectedPoints: monitor.selectedPoints || [],
              timeRange: effectiveTimeRange, // Use the chart's own timeRange
            };
            exportTimeseriesAsJSON(
              jsonData as any,
              `${monitor.title}-${new Date().toISOString().slice(0, 10)}.json`,
              true
            );
          } catch (error) {
            console.error('Failed to export JSON:', error);
          }
        }
      }
    },
    [onExport, monitor.title, series, monitor.chartType, monitor.config, monitor.selectedPoints, effectiveTimeRange]
  );

  // Get chart component
  const ChartComponent = CHART_COMPONENTS[monitor.chartType] || EChartsTimeSeriesChart;

  // Loading state
  if (enableLazyLoad && !inView) {
    return (
      <Box
        ref={setRefs}
        sx={{
          height,
          minHeight,
          display: 'flex',
          alignItems: 'center',
          justifyContent: 'center',
          bgcolor: alpha(theme.palette.background.paper, 0.8),
          borderRadius: 1,
        }}
      >
        <Typography variant="body2" color="text.secondary">
          Scroll to load chart
        </Typography>
      </Box>
    );
  }

  if (isLoading && showLoading) {
    return (
      <Box
        ref={setRefs}
        sx={{
          height,
          minHeight,
          display: 'flex',
          alignItems: 'center',
          justifyContent: 'center',
          flexDirection: 'column',
          gap: 2,
        }}
      >
        <CircularProgress size={32} />
        <Typography variant="body2" color="text.secondary">
          {loadingMessage}
        </Typography>
      </Box>
    );
  }

  // We'll handle error and empty states inside the main render
  // This ensures the toolbar is always visible
  const hasError = !!error;
  
  // All charts now use series data directly
  const isEmpty = !series || series.length === 0 || series.every(s => !s.data || s.data.length === 0);

  // Chart props - handle different data formats for different chart types
  const getChartSpecificProps = () => {
    const baseProps = {
      title: monitor.title,
      timeRange: effectiveTimeRange, // Use the chart's own timeRange
      height: '100%',
      width: '100%',
      onChartReady: handleChartReady,
      enableToolbox: true,
      onEdit: enableTools.includes('edit') ? onEdit : undefined,
      onDelete: enableTools.includes('delete') ? onDelete : undefined,
      selectedPoints: monitor.selectedPoints,
      loading: isLoading,
      error: error,
      ...monitor.config, // Include all monitor config as default
    };

    const chartType = monitor.chartType;

    // GROUP 1: Line/Time Series Charts - Need series array with [timestamp, value] format
    if (['TimeSeries', 'line', 'Line', 'LineChart', 'timeline', 'TimelineChart'].includes(chartType)) {
      // Transform series to use formatted names (with marker tags) instead of raw point names
      const formattedSeries = series?.map(s => ({
        ...s,
        name: s.formattedName || s.name // Use formattedName (includes marker tags) for chart display
      })) || [];
      
      return {
        ...baseProps,
        data: formattedSeries, // Line charts expect the series array directly
        series: formattedSeries,
      };
    }

    // GROUP 2: Area Charts - Similar to line charts
    if (['area', 'AreaChart', 'StackedArea'].includes(chartType)) {
      // Transform series to use formatted names (with marker tags) instead of raw point names
      const formattedSeries = series?.map(s => ({
        ...s,
        name: s.formattedName || s.name // Use formattedName (includes marker tags) for chart display
      })) || [];
      
      return {
        ...baseProps,
        data: formattedSeries, // Area charts use same format as line charts
        series: formattedSeries,
        stacked: chartType === 'StackedArea', // Stack only for StackedArea type
      };
    }

    // GROUP 3: Bar Charts - Keep existing working behavior
    if (['bar', 'Bar', 'BarChart'].includes(chartType)) {
      // Transform series to use formatted names (with marker tags) instead of raw point names
      const formattedSeries = series?.map(s => ({
        ...s,
        name: s.formattedName || s.name // Use formattedName (includes marker tags) for chart display
      })) || [];
      
      return {
        ...baseProps,
        data: formattedSeries, // Bar charts already work with series
        series: formattedSeries,
      };
    }

    // GROUP 4: Heatmaps - Keep existing working behavior
    if (['heatmap', 'Heatmap'].includes(chartType)) {
      return {
        ...baseProps,
        data: series || [], // Heatmaps already work with series
        series: series,
      };
    }

    // GROUP 5: Scatter Plots - Need paired x,y data
    if (['scatter', 'Scatter', 'ScatterPlot'].includes(chartType)) {
      return {
        ...baseProps,
        data: series || [], // Scatter can work with series data
        series: series,
      };
    }

    // GROUP 6: Gauge Charts - Need single value or gauge-specific format
    if (['gauge', 'Gauge', 'GaugeChart'].includes(chartType)) {
      // Extract latest value from series for gauge
      const latestValue = series?.[0]?.data?.slice(-1)[0]?.[1] || 0;
      return {
        ...baseProps,
        data: series || [], // Pass series data
        series: series,
        value: latestValue, // Pass latest value for gauge display
      };
    }

    // GROUP 7: Statistical Charts
    if (['BoxPlot', 'boxplot'].includes(chartType)) {
      return {
        ...baseProps,
        data: series || [], // Box plots work with series data
        series: series,
      };
    }

    if (['Candlestick', 'candlestick'].includes(chartType)) {
      return {
        ...baseProps,
        data: series || [], // Candlestick works with series data
        series: series,
      };
    }

    // GROUP 8: Calendar Heatmaps
    if (['CalendarHeatmap', 'CalendarYearHeatmap'].includes(chartType)) {
      return {
        ...baseProps,
        data: series || [], // Calendar works with series data
        series: series,
      };
    }

    // GROUP 9: HVAC Specialized Charts
    if (chartType === 'Psychrometric') {
      return {
        ...baseProps,
        data: series || [], // Psychrometric uses series data
        series: series,
      };
    }

    if (chartType === 'VAVDiagnostic') {
      return {
        ...baseProps,
        data: series || [], // VAV Diagnostic uses series data
        series: series,
        diagnosticMode: monitor.config?.diagnosticMode || 'general',
      };
    }

    if (chartType === 'VAVComprehensive') {
      return {
        ...baseProps,
        series: series || [], // VAV Enhanced uses all 7 series
        data: series || [],
      };
    }

    if (chartType === 'ChilledWaterReset') {
      return {
        ...baseProps,
        series: series || [],
        data: series || [],
        config: {
          mode: monitor.config?.mode || 'basic',
          resetStrategy: monitor.config?.resetStrategy || 'trim-respond',
          minSetpoint: monitor.config?.minSetpoint || 42,
          maxSetpoint: monitor.config?.maxSetpoint || 48,
          targetValvePosition: monitor.config?.targetValvePosition || 85,
          responseTime: monitor.config?.responseTime || 15,
          unit: monitor.config?.unit || 'F',
        },
      };
    }

    if (chartType === 'DPOptimization') {
      return {
        ...baseProps,
        series: series || [],
        data: series || [],
        config: {
          mode: monitor.config?.mode || 'basic',
          resetStrategy: monitor.config?.resetStrategy || 'trim-respond',
          minSetpoint: monitor.config?.minSetpoint || 0.5,
          maxSetpoint: monitor.config?.maxSetpoint || 3.0,
          targetDamperPosition: monitor.config?.targetDamperPosition || 85,
          responseTime: monitor.config?.responseTime || 10,
          unit: monitor.config?.unit || 'inWC',
        },
      };
    }

    if (chartType === 'SimultaneousHC') {
      return {
        ...baseProps,
        series: series || [],
        data: series || [],
        config: {
          mode: monitor.config?.mode || 'basic',
          heatingThreshold: monitor.config?.heatingThreshold || 5,
          coolingThreshold: monitor.config?.coolingThreshold || 5,
          temperatureDifferenceThreshold: monitor.config?.temperatureDifferenceThreshold || 2,
          unit: monitor.config?.unit || 'F',
          detectionWindow: monitor.config?.detectionWindow || 15,
        },
      };
    }

    if (chartType === 'PerfectEconomizer') {
      // PerfectEconomizer needs specific config structure
      const economizerConfig = {
        mixedAirSetpoint: monitor.config?.mixedAirSetpoint || 55,
        minimumOAPercentage: monitor.config?.minimumOAPercentage || 20,
        preheatSetpoint: monitor.config?.preheatSetpoint || 45,
        coolingSetpoint: monitor.config?.coolingSetpoint || 75,
        returnAirTemp: monitor.config?.returnAirTemp || 72,
        economizerHighLimit: monitor.config?.economizerHighLimit || 75,
        unit: monitor.config?.unit || 'F',
      };
      

      
      return {
        ...baseProps,
        data: series || [], // Same as line chart
        series: series,     // Same as line chart - NO FALLBACK
        config: economizerConfig, // Pass config as a nested object
      };
    }

    // GROUP 10: Device Deviation Heatmap - Special handling
    if (chartType === 'DeviceDeviationHeatmap') {
      return {
        ...baseProps,
        data: undefined,    // Don't pass series as data - wrong format
        series: series,     // Pass as series prop for proper processing
        pointConfigs: monitor.config?.pointConfigs || [],
        selectedPoints: monitor.selectedPoints,
        timeRange: effectiveTimeRange, // Use the chart's own timeRange
        enableLiveData: false,
        onConfigsChange: (configs: any[]) => {
          // Save configs to monitor for persistence
          if (onUpdate) {
            onUpdate(monitor.id, {
              config: {
                ...monitor.config,
                pointConfigs: configs
              }
            });
          }
        },
      };
    }

    // GROUP 11: Multi-dimensional Charts
    if (chartType === 'ParallelCoordinates') {
      return {
        ...baseProps,
        data: series || [], // Parallel coordinates works with series data
        series: series,
      };
    }

    if (['Radar', 'radar'].includes(chartType)) {
      return {
        ...baseProps,
        data: series || [], // Radar works with series data
        series: series,
      };
    }

    // GROUP 12: Flow/Hierarchy Charts
    if (['Sankey', 'sankey'].includes(chartType)) {
      return {
        ...baseProps,
        data: series || [], // Sankey works with series data
        series: series,
      };
    }

    if (['Treemap', 'treemap'].includes(chartType)) {
      return {
        ...baseProps,
        data: series || [], // Treemap works with series data
        series: series,
      };
    }

    // GROUP 13: Control Band Charts (formerly SPC) - Need specific configuration
    if (['ControlBand', 'controlband', 'ControlBandChart', 'SPC', 'spc', 'SPCChart'].includes(chartType)) {
      // Control Band charts expect series data and limit configuration
      return {
        ...baseProps,
        data: { series: series || [] }, // Wrap series in data object
        series: series,
        // Pass Control Band configuration - handle string values from form inputs
        upperLimit: typeof monitor.config?.upperLimit === 'string' 
          ? parseFloat(monitor.config.upperLimit) 
          : (monitor.config?.upperLimit || monitor.config?.upperControlLimit || 100),
        lowerLimit: typeof monitor.config?.lowerLimit === 'string'
          ? parseFloat(monitor.config.lowerLimit)
          : (monitor.config?.lowerLimit || monitor.config?.lowerControlLimit || 0),
        targetValue: typeof monitor.config?.targetValue === 'string'
          ? parseFloat(monitor.config.targetValue)
          : (monitor.config?.targetValue || monitor.config?.centerLine),
        warningThreshold: monitor.config?.warningThreshold || 10,
        showBackgroundZones: monitor.config?.showBackgroundZones !== false,
        showAlerts: monitor.config?.showAlerts !== false,
      };
    }

    // DEFAULT: Pass all data formats and let the chart component decide
    // This ensures backward compatibility and handles any new chart types
    return {
      ...baseProps,
      data: series || [],
      series: series,
    };
  };

  const chartProps = getChartSpecificProps();

  return (
    <Box
      ref={setRefs}
      onMouseEnter={() => setIsHovered(true)}
      onMouseLeave={() => setIsHovered(false)}
      sx={{
        height,
        minHeight,
        position: 'relative',
        bgcolor: theme.palette.background.paper,
        borderRadius: 1,
        overflow: 'hidden',
        transition: 'all 0.3s ease-in-out',
        ...(isFullscreen && {
          position: 'fixed',
          top: 0,
          left: 0,
          right: 0,
          bottom: 0,
          zIndex: theme.zIndex.modal,
          height: '100vh',
          borderRadius: 0,
        }),
      }}
    >
      {/* Toolbar - positioned to not overlap with ECharts toolbox */}
      <Box
        sx={{
          position: 'absolute',
          top: 8,
          left: 8, // Move to left side to avoid ECharts toolbox on right
          display: 'flex',
          gap: 0.5,
          zIndex: 100, // Lower z-index than ECharts toolbox
          opacity: isHovered ? 1 : 0.7, // Changed to always show at 70% opacity, 100% on hover
          transition: 'opacity 0.3s ease-in-out',
          bgcolor: alpha(theme.palette.background.paper, 0.9),
          borderRadius: 1,
          p: 0.5,
          backdropFilter: 'blur(8px)',
          pointerEvents: 'auto', // Ensure it's clickable
        }}
      >
        {enableTools.includes('refresh') && (
          <Tooltip title="Refresh data">
            <IconButton size="small" onClick={handleRefresh} disabled={isRefreshing}>
              <RefreshIcon fontSize="small" sx={{ 
                animation: isRefreshing ? 'spin 1s linear infinite' : 'none',
                '@keyframes spin': {
                  '0%': { transform: 'rotate(0deg)' },
                  '100%': { transform: 'rotate(360deg)' }
                }
              }} />
            </IconButton>
          </Tooltip>
        )}

        {/* Export button with menu for CSV/JSON */}
        {enableTools.includes('export') && (
          <Tooltip title="Export data (CSV, JSON, PNG)">
            <IconButton size="small" onClick={(e) => setExportMenuAnchor(e.currentTarget)}>
              <ExportIcon fontSize="small" />
            </IconButton>
          </Tooltip>
        )}


        {enableTools.includes('edit') && (
          <Tooltip title="Edit">
            <IconButton size="small" onClick={onEdit}>
              <EditIcon fontSize="small" />
            </IconButton>
          </Tooltip>
        )}

        {enableTools.includes('delete') && (
          <Tooltip title="Delete">
            <IconButton size="small" onClick={onDelete} color="error">
              <DeleteIcon fontSize="small" />
            </IconButton>
          </Tooltip>
        )}

        {enableTools.includes('fullscreen') && (
          <Tooltip title={isFullscreen ? "Exit fullscreen" : "Enter fullscreen"}>
            <IconButton size="small" onClick={handleFullscreenToggle}>
              {isFullscreen ? (
                <FullscreenExitIcon fontSize="small" />
              ) : (
                <FullscreenIcon fontSize="small" />
              )}
            </IconButton>
          </Tooltip>
        )}

      </Box>

      {/* Chart */}
      <Box sx={{ 
        height: '100%', 
        width: '100%',
        position: 'relative',
        zIndex: 1, // Base z-index
        // Ensure ECharts toolbox is not blocked
        '& > div': {
          position: 'relative !important',
          zIndex: 'auto !important',
        }
      }}>
        {hasError ? (
          <Box
            sx={{
              height: '100%',
              display: 'flex',
              alignItems: 'center',
              justifyContent: 'center',
              p: 2,
            }}
          >
            <Alert severity="error" sx={{ maxWidth: 400 }}>
              <Typography variant="body2">
                {errorMessage}: {error.message}
              </Typography>
              <Typography variant="caption" display="block" sx={{ mt: 1, opacity: 0.8 }}>
                You can still edit or delete this chart.
              </Typography>
            </Alert>
          </Box>
        ) : isEmpty ? (
          <Box
            sx={{
              height: '100%',
              display: 'flex',
              alignItems: 'center',
              justifyContent: 'center',
            }}
          >
            <Typography variant="h6" color="text.secondary">
              No Data Available
            </Typography>
            <Typography variant="body2" color="text.secondary" sx={{ mt: 2 }}>
              The selected points do not have data for the chosen time range.
            </Typography>
            {monitor.selectedPoints?.length > 0 && (
              <Box sx={{ mt: 2, p: 2, bgcolor: 'background.paper', borderRadius: 1 }}>
                <Typography variant="caption" color="text.secondary">
                  Selected points:
                </Typography>
                <Typography variant="body2" sx={{ mt: 0.5 }}>
                  {monitor.selectedPoints.map(p => p.name || p).join(', ')}
                </Typography>
              </Box>
            )}
            <Typography variant="caption" display="block" color="text.secondary" sx={{ mt: 2 }}>
              Try selecting a different time range or checking if data collection is enabled for these points.
            </Typography>
          </Box>
        ) : (
          <Suspense fallback={<CircularProgress />}>
            <SafeChartWrapper
              chartId={monitor.id}
              fallback={
                <Box
                  sx={{
                    p: 2,
                    height: '100%',
                    minHeight: 300,
                    display: 'flex',
                    flexDirection: 'column',
                    alignItems: 'center',
                    justifyContent: 'center',
                  }}
                >
                  <Alert
                    severity="error"
                    sx={{
                      maxWidth: 400,
                      backgroundColor: 'rgba(255, 82, 82, 0.1)',
                      border: '1px solid rgba(255, 82, 82, 0.2)',
                      borderRadius: 2,
                    }}
                  >
                    <Typography variant="body2" sx={{ color: 'error.main' }}>
                      Failed to render {monitor.chartType || 'chart'}
                    </Typography>
                    <Typography variant="caption" display="block" sx={{ mt: 1, opacity: 0.8 }}>
                      The chart encountered an error but can still be edited or deleted.
                    </Typography>
                  </Alert>
                </Box>
              }
            >
              <ChartComponent {...chartProps} />
            </SafeChartWrapper>
          </Suspense>
        )}
      </Box>

      {/* Export Menu with clear descriptions */}
      <Menu
        anchorEl={exportMenuAnchor}
        open={Boolean(exportMenuAnchor)}
        onClose={() => setExportMenuAnchor(null)}
        PaperProps={{
          sx: {
            minWidth: 250,
            '& .MuiMenuItem-root': {
              px: 2,
              py: 1.5,
            }
          }
        }}
      >
        <MenuItem onClick={() => handleExport('csv')}>
          <Box>
            <Typography variant="body2" fontWeight={600}>
              📊 Export as CSV
            </Typography>
            <Typography variant="caption" color="text.secondary" display="block">
              Spreadsheet format (Excel compatible)
            </Typography>
          </Box>
        </MenuItem>
        <MenuItem onClick={() => handleExport('json')}>
          <Box>
            <Typography variant="body2" fontWeight={600}>
              📄 Export as JSON
            </Typography>
            <Typography variant="caption" color="text.secondary" display="block">
              Raw data with full precision
            </Typography>
          </Box>
        </MenuItem>
        <MenuItem onClick={() => handleExport('png')}>
          <Box>
            <Typography variant="body2" fontWeight={600}>
              🖼️ Export as PNG
            </Typography>
            <Typography variant="caption" color="text.secondary" display="block">
              High-quality image of the chart
            </Typography>
          </Box>
        </MenuItem>
      </Menu>

      {/* More Menu - Removed non-functional items */}
    </Box>
  );
};

export default UnifiedChartRenderer;
