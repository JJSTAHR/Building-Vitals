import React, { useMemo, useState } from 'react';
import {
  Box,
  Paper,
  Typography,
  ToggleButton,
  ToggleButtonGroup,
  Chip,
  Stack,
  IconButton,
  Tooltip,
  Alert,
} from '@mui/material';
import {
  Warning as WarningIcon,
  CheckCircle as CheckCircleIcon,
  Error as ErrorIcon,
  TrendingUp as TrendingUpIcon,
  TrendingDown as TrendingDownIcon,
  Info as InfoIcon,
} from '@mui/icons-material';
import { useTheme } from '@mui/material/styles';
import EChartsWrapper from './EChartsWrapper';
import type { EChartsOption } from 'echarts';
import { formatNumberForDisplay } from '../../utils/formatters';
interface FaultDetectionData {
  timestamp: number;
  zone: string;
  value: number;
  faultType?: 'temperature' | 'humidity' | 'pressure' | 'flow' | 'energy';
  severity?: 'low' | 'medium' | 'high' | 'critical';
  deviation?: number; // Deviation from setpoint or expected value
  baseline?: number;
}

interface FaultDetectionHeatmapProps {
  data: FaultDetectionData[];
  loading?: boolean;
  error?: Error | null;
  height?: string | number;
  title?: string;
  faultThresholds?: {
    low: number;
    medium: number;
    high: number;
    critical: number;
  };
  showFaultOverlay?: boolean;
  showTrends?: boolean;
  unit?: string;
  onFaultClick?: (fault: FaultDetectionData) => void;
}

const FaultDetectionHeatmap: React.FC<FaultDetectionHeatmapProps> = ({
  data = [],
  loading = false,
  error = null,
  height = 600,
  title = 'Fault Detection Analysis',
  faultThresholds = {
    low: 2,
    medium: 5,
    high: 10,
    critical: 15,
  },
  showFaultOverlay = true,
  showTrends = true,
  unit = '',
  onFaultClick,
}) => {
  const theme = useTheme();
  const [viewMode, setViewMode] = useState<'heatmap' | 'faults' | 'combined'>('combined');
  const [selectedFaultType, setSelectedFaultType] = useState<string>('all');

  // Process data for heatmap
  const processedData = useMemo(() => {
    // Get unique zones and timestamps
    const zones = [...new Set(data.map((d) => d.zone))].sort();
    const timestamps = [...new Set(data.map((d) => d.timestamp))].sort((a, b) => a - b);

    // Create time labels
    const timeLabels = timestamps.map((ts) => {
      const date = new Date(ts);
      return date.toLocaleTimeString('en-US', { hour: 'numeric', minute: '2-digit', hour12: true });
    });

    // Create heatmap data
    const heatmapData: any[] = [];
    const faultMarkers: any[] = [];

    zones.forEach((zone, zoneIndex) => {
      timestamps.forEach((timestamp, timeIndex) => {
        const point = data.find((d) => d.zone === zone && d.timestamp === timestamp);

        if (point) {
          heatmapData.push([timeIndex, zoneIndex, point.value]);

          // Detect faults based on deviation or thresholds
          const deviation = Math.abs(point.deviation || 0);
          let severity: string | null = null;

          if (deviation >= faultThresholds.critical) {
            severity = 'critical';
          } else if (deviation >= faultThresholds.high) {
            severity = 'high';
          } else if (deviation >= faultThresholds.medium) {
            severity = 'medium';
          } else if (deviation >= faultThresholds.low) {
            severity = 'low';
          }

          if (severity && showFaultOverlay) {
            faultMarkers.push({
              coord: [timeIndex, zoneIndex],
              value: point.value,
              severity,
              faultType: point.faultType,
              zone: zone,
              timestamp: timestamp,
              deviation: deviation,
            });
          }
        }
      });
    });

    return {
      zones,
      timeLabels,
      heatmapData,
      faultMarkers,
      minValue: data.length > 0 ? Math.min(...data.map((d) => d.value)) : 0,
      maxValue: data.length > 0 ? Math.max(...data.map((d) => d.value)) : 1,
    };
  }, [data, faultThresholds, showFaultOverlay]);

  // Calculate fault statistics
  const faultStats = useMemo(() => {
    const stats = {
      total: 0,
      bySeverity: { low: 0, medium: 0, high: 0, critical: 0 },
      byType: {} as Record<string, number>,
      byZone: {} as Record<string, number>,
    };

    processedData.faultMarkers.forEach((fault) => {
      stats.total++;
      stats.bySeverity[fault.severity as keyof typeof stats.bySeverity]++;

      if (fault.faultType) {
        stats.byType[fault.faultType] = (stats.byType[fault.faultType] || 0) + 1;
      }

      stats.byZone[fault.zone] = (stats.byZone[fault.zone] || 0) + 1;
    });

    return stats;
  }, [processedData.faultMarkers]);

  const option = useMemo((): EChartsOption => {
    const series: any[] = [];

    // Base heatmap series
    if (viewMode === 'heatmap' || viewMode === 'combined') {
      series.push({
        name: 'Values',
        type: 'heatmap',
        data: processedData.heatmapData,
        label: {
          show: false,
        },
        emphasis: {
          itemStyle: {
            shadowBlur: 10,
            shadowColor: 'rgba(0, 0, 0, 0.5)',
          },
        },
      });
    }

    // Fault overlay series
    if (
      (viewMode === 'faults' || viewMode === 'combined') &&
      processedData.faultMarkers.length > 0
    ) {
      // Group faults by severity for different visual representations
      const severityGroups = processedData.faultMarkers.reduce(
        (acc, fault) => {
          if (!acc[fault.severity]) {acc[fault.severity] = [];}
          acc[fault.severity].push(fault);
          return acc;
        },
        {} as Record<string, any[]>
      );

      const severityConfig = {
        critical: { symbol: 'rect', size: 30, color: theme.palette.error.main },
        high: { symbol: 'circle', size: 25, color: theme.palette.warning.main },
        medium: { symbol: 'triangle', size: 20, color: theme.palette.warning.light },
        low: { symbol: 'diamond', size: 15, color: theme.palette.info.main },
      };

      Object.entries(severityGroups).forEach(([severity, faults]) => {
        const config = severityConfig[severity as keyof typeof severityConfig];

        series.push({
          name: `${severity.charAt(0).toUpperCase() + severity.slice(1)} Faults`,
          type: 'scatter',
          coordinateSystem: 'cartesian2d',
          data: faults.map((f) => ({
            value: f.coord,
            ...f,
          })),
          symbol: config.symbol,
          symbolSize: config.size,
          itemStyle: {
            color: config.color,
            borderColor: '#fff',
            borderWidth: 2,
          },
          z: 10,
          emphasis: {
            scale: 1.5,
            itemStyle: {
              shadowBlur: 20,
              shadowColor: 'rgba(0,0,0,0.3)',
            },
          },
        });
      });
    }

    return {
      title: {
        text: title,
        left: 'center',
        textStyle: {
          color: theme.palette.text.primary,
        },
      },
      tooltip: {
        trigger: 'item',
        formatter: (params: any) => {
          if (params.seriesType === 'heatmap') {
            const zone = processedData.zones[params.value[1]];
            const time = processedData.timeLabels[params.value[0]];
            const value = formatNumberForDisplay(params.value[2], 2);
            return `
              <div style="padding: 8px;">
                <strong>${zone}</strong><br/>
                Time: ${time}<br/>
                Value: ${value} ${unit}
              </div>
            `;
          } else {
            // Fault marker tooltip
            const fault = params.data;
            return `
              <div style="padding: 8px;">
                <strong>${fault.severity.toUpperCase()} FAULT</strong><br/>
                Zone: ${fault.zone}<br/>
                Type: ${fault.faultType || 'Unknown'}<br/>
                Deviation: ${formatNumberForDisplay(fault.deviation, 2)} ${unit}<br/>
                Value: ${formatNumberForDisplay(fault.value, 2)} ${unit}
              </div>
            `;
          }
        },
        backgroundColor: theme.palette.background.paper,
        borderColor: theme.palette.divider,
        textStyle: {
          color: theme.palette.text.primary,
        },
      },
      grid: {
        left: '15%',
        right: '10%',
        bottom: '15%',
        top: '10%',
        containLabel: true,
      },
      xAxis: {
        type: 'category',
        data: processedData.timeLabels,
        name: 'Time',
        nameLocation: 'middle',
        nameGap: 35,
        splitArea: {
          show: true,
        },
        axisLabel: {
          rotate: 45,
          color: theme.palette.text.secondary,
        },
      },
      yAxis: {
        type: 'category',
        data: processedData.zones,
        name: 'Zone',
        nameLocation: 'middle',
        nameGap: 70,
        splitArea: {
          show: true,
        },
        axisLabel: {
          color: theme.palette.text.secondary,
        },
      },
      visualMap:
        viewMode !== 'faults'
          ? {
              min: processedData.minValue,
              max: processedData.maxValue,
              calculable: true,
              orient: 'horizontal',
              left: 'center',
              bottom: '5%',
              inRange: {
                color: [
                  '#313695',
                  '#4575b4',
                  '#74add1',
                  '#abd9e9',
                  '#e0f3f8',
                  '#ffffbf',
                  '#fee090',
                  '#fdae61',
                  '#f46d43',
                  '#d73027',
                  '#a50026',
                ],
              },
              text: [`${processedData.maxValue} ${unit}`, `${processedData.minValue} ${unit}`],
              textStyle: {
                color: theme.palette.text.primary,
              },
            }
          : undefined,
      series: series,
      toolbox: {
        feature: {
          dataView: { show: true },
          restore: { show: true },
          saveAsImage: { show: true },
        },
      },
    };
  }, [processedData, viewMode, title, unit, theme]);

  if (loading) {
    return (
      <Box sx={{ height, display: 'flex', alignItems: 'center', justifyContent: 'center' }}>
        <Typography>Loading fault detection data...</Typography>
      </Box>
    );
  }

  if (error) {
    return (
      <Box sx={{ height, display: 'flex', alignItems: 'center', justifyContent: 'center' }}>
        <Typography color="error">Error loading chart: {error.message}</Typography>
      </Box>
    );
  }

  return (
    <Box sx={{ height: '100%' }}>
      <Paper elevation={0} sx={{ p: 2, mb: 2 }}>
        <Stack direction="row" spacing={2} alignItems="center" justifyContent="space-between">
          <ToggleButtonGroup
            value={viewMode}
            exclusive
            onChange={(_, newMode) => newMode && setViewMode(newMode)}
            size="small"
          >
            <ToggleButton value="heatmap">Heatmap Only</ToggleButton>
            <ToggleButton value="faults">Faults Only</ToggleButton>
            <ToggleButton value="combined">Combined View</ToggleButton>
          </ToggleButtonGroup>

          <Stack direction="row" spacing={1}>
            {faultStats.total > 0 && (
              <>
                <Chip
                  icon={<ErrorIcon />}
                  label={`Critical: ${faultStats.bySeverity.critical}`}
                  color="error"
                  size="small"
                  variant={faultStats.bySeverity.critical > 0 ? 'filled' : 'outlined'}
                />
                <Chip
                  icon={<WarningIcon />}
                  label={`High: ${faultStats.bySeverity.high}`}
                  color="warning"
                  size="small"
                  variant={faultStats.bySeverity.high > 0 ? 'filled' : 'outlined'}
                />
                <Chip
                  icon={<InfoIcon />}
                  label={`Medium: ${faultStats.bySeverity.medium}`}
                  color="info"
                  size="small"
                  variant={faultStats.bySeverity.medium > 0 ? 'filled' : 'outlined'}
                />
                <Chip label={`Total Faults: ${faultStats.total}`} size="small" variant="outlined" />
              </>
            )}
          </Stack>
        </Stack>
      </Paper>

      <EChartsWrapper
        option={option}
        height={typeof height === 'number' ? height - 100 : `calc(${height} - 100px)`}
        loading={loading}
        onClick={(params: any) => {
          if (params.seriesType === 'scatter' && onFaultClick) {
            onFaultClick(params.data);
          }
        }}
      />

      {faultStats.total === 0 && (
        <Alert severity="success" sx={{ mt: 2 }}>
          No faults detected in the current time range
        </Alert>
      )}
    </Box>
  );
};

export default FaultDetectionHeatmap;
