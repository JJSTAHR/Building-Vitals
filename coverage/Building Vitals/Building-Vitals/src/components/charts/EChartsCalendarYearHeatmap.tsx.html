import React, { useMemo } from 'react';
import EChartsWrapper from './EChartsWrapper';
import type { EChartsOption } from 'echarts';
import { useTheme } from '@mui/material/styles';
import { formatNumberForDisplay } from '../../utils/formatters';
import { registerChartForActions } from '../../utils/echarts/chartActionDispatcher';
import { addChartToGroup } from '../../utils/echarts/chartConnectionManager';
import { enhanceChart } from '../../utils/chartEnhancements';
import { useBaseChartOptions, mergeChartOptions } from '../../hooks/useBaseChartOptions';
import { ChartTimeFormatter } from '../../utils/chartTimeFormatter';
import { format } from 'date-fns';
import { CHART_DESIGN_TOKENS, calculateGridTop, calculateGridBottom } from '@/utils/chartDesignTokens';

interface EChartsCalendarYearHeatmapProps {
  // STANDARD DATA PROP - matches line/area charts EXACTLY
  data: Array<{
    name: string;
    data: Array<[number, number]>; // [timestamp, value] pairs
    unit?: string;
    markerTags?: string[];
  }>;
  year: number;
  loading?: boolean;
  error?: Error | null;
  height?: string | number;
  title?: string;
  unit?: string;
  metric?: string; // What the value represents (e.g., "Energy Consumption", "Temperature")
  colorScale?: {
    min?: string;
    max?: string;
    pieces?: Array<{
      min: number;
      max: number;
      color: string;
      label?: string;
    }>;
  };
  groupId?: string;
  chartId?: string;
  onChartReady?: (chart: any) => void;
  // Action handlers
  onEdit?: () => void;
  onDelete?: () => void;
}

const EChartsCalendarYearHeatmap: React.FC<EChartsCalendarYearHeatmapProps> = ({
  data,
  year,
  loading,
  error,
  height = '400px',
  title,
  unit = '',
  metric = 'Value',
  colorScale,
  groupId,
  chartId,
  onChartReady,
  onEdit,
  onDelete,
}) => {
  const theme = useTheme();
  const baseOptions = useBaseChartOptions();

  // Transform series data to calendar format (extracted from option useMemo)
  const calendarData = useMemo(() => {
    if (!data || data.length === 0) return [];

    // Combine all series data and group by date
    const dateGroups = new Map<string, number[]>();

    data.forEach(series => {
      series.data.forEach(([timestamp, value]) => {
        const dateStr = format(new Date(timestamp), 'yyyy-MM-dd');
        if (!dateGroups.has(dateStr)) {
          dateGroups.set(dateStr, []);
        }
        dateGroups.get(dateStr)!.push(value);
      });
    });

    // Calculate daily aggregates (average by default)
    return Array.from(dateGroups.entries()).map(([dateStr, values]) => ({
      date: dateStr,
      value: values.reduce((sum, val) => sum + val, 0) / values.length,
      metadata: {
        count: values.length,
        min: Math.min(...values),
        max: Math.max(...values)
      }
    }));
  }, [data]);

  // Calculate min and max values for color scale
  const { minValue, maxValue } = useMemo(() => {
    const values = calendarData.map((d) => d.value).filter((v) => !isNaN(v));
    return {
      minValue: values.length > 0 ? Math.min(...values) : 0,
      maxValue: values.length > 0 ? Math.max(...values) : 1
    };
  }, [calendarData]);

  const option = useMemo((): EChartsOption => {

    // Process data to ensure correct date format
    const processedData = calendarData.map((item) => {
      let dateStr: string;
      if (typeof item.date === 'string') {
        dateStr = item.date;
      } else if (item.date instanceof Date) {
        dateStr = item.date.toISOString().split('T')[0];
      } else if (typeof item.date === 'number') {
        dateStr = new Date(item.date).toISOString().split('T')[0];
      } else {
        dateStr = '';
      }

      return [dateStr, item.value, item.metadata];
    });

    // Default color scale based on theme
    const defaultMinColor = theme.palette.mode === 'dark' ? '#1e3a5f' : '#e3f2fd';
    const defaultMaxColor = theme.palette.primary.main;
    const isDarkMode = theme.palette.mode === 'dark';

    const calendarSpecificOptions = {
      title: title
        ? {
            ...CHART_DESIGN_TOKENS.title.base,
            text: title,
            textStyle: {
              ...CHART_DESIGN_TOKENS.title.base.textStyle,
              color: theme.palette.text.primary,
            },
          }
        : undefined,
      tooltip: {
        ...CHART_DESIGN_TOKENS.tooltip.base,
        trigger: 'item',
        backgroundColor: isDarkMode
          ? 'rgba(50, 50, 50, 0.95)'
          : 'rgba(255, 255, 255, 0.95)',
        borderColor: isDarkMode
          ? 'rgba(255, 255, 255, 0.12)'
          : 'rgba(0, 0, 0, 0.12)',
        textStyle: {
          ...CHART_DESIGN_TOKENS.tooltip.base.textStyle,
          color: theme.palette.text.primary,
        },
        formatter: (params: any) => {
          const date = new Date(params.data[0]);
          const value = params.data[1];
          const metadata = params.data[2];

          const dateStr = ChartTimeFormatter.calendar(date);

          let tooltip = `<strong>${dateStr}</strong><br/>`;
          tooltip += `${metric}: ${formatNumberForDisplay(value, 2)} ${unit}`;

          if (metadata) {
            if (metadata.details) {
              tooltip += `<br/>${metadata.details}`;
            }
            if (metadata.anomaly) {
              tooltip += `<br/><span style="color: ${theme.palette.error.main}">⚠ Anomaly detected</span>`;
            }
          }

          return tooltip;
        },
      },
      visualMap: {
        type: 'piecewise',
        orient: 'horizontal',
        left: 'center',
        bottom: 20,
        pieces: colorScale?.pieces || [
          {
            min: minValue,
            max: minValue + (maxValue - minValue) * 0.2,
            label: 'Low',
            color: defaultMinColor,
          },
          {
            min: minValue + (maxValue - minValue) * 0.2,
            max: minValue + (maxValue - minValue) * 0.4,
            label: 'Below Average',
            color: theme.palette.info.light,
          },
          {
            min: minValue + (maxValue - minValue) * 0.4,
            max: minValue + (maxValue - minValue) * 0.6,
            label: 'Average',
            color: theme.palette.warning.light,
          },
          {
            min: minValue + (maxValue - minValue) * 0.6,
            max: minValue + (maxValue - minValue) * 0.8,
            label: 'Above Average',
            color: theme.palette.warning.main,
          },
          {
            min: minValue + (maxValue - minValue) * 0.8,
            max: maxValue,
            label: 'High',
            color: defaultMaxColor,
          },
        ],
        inRange: {
          color: colorScale
            ? [colorScale.min || defaultMinColor, colorScale.max || defaultMaxColor]
            : undefined,
        },
        textStyle: {
          color: theme.palette.text.secondary,
        },
      },
      calendar: {
        top: title ? 80 : 60,
        left: 60,
        right: 60,
        cellSize: ['auto', 20],
        range: year.toString(),
        itemStyle: {
          borderWidth: 0.5,
          borderColor: theme.palette.divider,
        },
        yearLabel: {
          show: true,
          position: 'top',
          margin: 40,
          fontSize: 20,
          fontWeight: 'bold',
          color: theme.palette.text.primary,
        },
        monthLabel: {
          show: true,
          nameMap: 'en',
          color: theme.palette.text.secondary,
        },
        dayLabel: {
          show: true,
          nameMap: ['S', 'M', 'T', 'W', 'T', 'F', 'S'],
          color: theme.palette.text.secondary,
        },
        splitLine: {
          show: true,
          lineStyle: {
            color: theme.palette.divider,
            width: 1,
            type: 'solid',
          },
        },
      },
      series: [
        {
          type: 'heatmap',
          coordinateSystem: 'calendar',
          data: processedData,
          label: {
            show: false,
          },
          emphasis: {
            itemStyle: {
              shadowBlur: 10,
              shadowColor: 'rgba(0, 0, 0, 0.5)',
            },
          },
          animation: true,
          animationDuration: 1000,
          animationEasing: 'cubicOut',
        },
      ],
      aria: {
        enabled: true,
        label: {
          enabled: true,
          description: `Calendar heatmap showing daily ${metric} for the year ${year}`,
        },
      },
    };

    // Calculate adaptive animation
    const totalPoints = calendarData.reduce((sum, _) => sum + 1, 0);
    const animationMode = CHART_DESIGN_TOKENS.adaptive.animationMode(totalPoints);

    return mergeChartOptions(baseOptions, {
      ...calendarSpecificOptions,
      ...CHART_DESIGN_TOKENS.animation[animationMode],
    }, {
      showToolbox: true,
      enableDataView: false, // Disabled due to overlay issues
      enableMagicType: false, // No chart type switching
        showDataZoom: false,
      showLegend: false,
      onEdit,
      onDelete,
    });
  }, [data, year, title, theme, unit, metric, colorScale, baseOptions, onEdit, onDelete, calendarData]);

  const handleChartReady = (chart: any) => {
    if (chartId) {
      registerChartForActions(chartId, chart);
    }
    if (groupId && chartId) {
      addChartToGroup(chart, groupId);
    }
    if (onChartReady) {
      onChartReady(chart);
    }
  };

  return (
    <EChartsWrapper
      option={option}
      loading={loading}
      error={error}
      height={height}
      onChartReady={handleChartReady}
    />
  );
};

export default EChartsCalendarYearHeatmap;
