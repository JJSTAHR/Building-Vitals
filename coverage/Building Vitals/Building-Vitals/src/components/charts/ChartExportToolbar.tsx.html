import React, { useState } from 'react';
import {
  Box,
  Button,
  ButtonGroup,
  Menu,
  MenuItem,
  ListItemIcon,
  ListItemText,
  Tooltip,
  IconButton,
  Divider,
  CircularProgress,
  Alert,
  Snackbar,
} from '@mui/material';
import {
  GetApp as ExportIcon,
  Image as ImageIcon,
  TableChart as CsvIcon,
  Code as JsonIcon,
  Settings as OptionsIcon,
  CloudDownload as DownloadIcon,
} from '@mui/icons-material';
import { useTheme } from '@mui/material/styles';
import type { ECharts } from 'echarts';
import { exportChartAsImage, exportTimeseriesAsCSV, exportTimeseriesAsJSON, type TimeseriesDataPoint, type ChartExportOptions, type CSVExportOptions } from '../../utils/exportUtils';
import { exportChartWithCalculations } from '../../utils/chartExportEnhanced';
import { Z_INDEX_CONFIG, createToolbarStyles } from '../../utils/toolbarPositioning';

export interface ChartExportData {
  name: string;
  data: Array<[number, number]> | TimeseriesDataPoint[];
  unit?: string;
  markerTags?: string[];
  pointMetadata?: any;
}

interface ChartExportToolbarProps {
  chartInstance?: ECharts;
  chartData?: ChartExportData[];
  chartTitle?: string;
  chartType?: string; // Added to determine which chart type for enhanced export
  position?: 'top-right' | 'top-left' | 'bottom-right' | 'bottom-left';
  size?: 'small' | 'medium';
  showLabels?: boolean;
  enableEnhancedExport?: boolean; // Enable calculated values in export
  onExportStart?: () => void;
  onExportComplete?: (format: string) => void;
  onExportError?: (error: string) => void;
}

const ChartExportToolbar: React.FC<ChartExportToolbarProps> = ({
  chartInstance,
  chartData = [],
  chartTitle = 'chart',
  chartType = 'generic',
  position = 'top-right',
  size = 'small',
  showLabels = false,
  enableEnhancedExport = true,
  onExportStart,
  onExportComplete,
  onExportError,
}) => {
  const theme = useTheme();
  const [anchorEl, setAnchorEl] = useState<null | HTMLElement>(null);
  const [exporting, setExporting] = useState(false);
  const [exportError, setExportError] = useState<string | null>(null);
  const [exportSuccess, setExportSuccess] = useState<string | null>(null);

  const handleExportClick = (event: React.MouseEvent<HTMLElement>) => {
    setAnchorEl(event.currentTarget);
  };

  const handleClose = () => {
    setAnchorEl(null);
  };

  const getPositionStyles = () => {
    const baseStyles = {
      position: 'absolute' as const,
      zIndex: Z_INDEX_CONFIG.CHART_EXPORT_TOOLBAR,
      backdropFilter: 'blur(8px)',
      backgroundColor: theme.palette.mode === 'dark' 
        ? 'rgba(0, 0, 0, 0.8)' 
        : 'rgba(255, 255, 255, 0.9)',
      borderRadius: 1,
      padding: 0.5,
      boxShadow: '0 2px 8px rgba(0,0,0,0.15)',
      border: `1px solid ${theme.palette.divider}`,
    };

    // Position toolbar outside of chart content area to prevent overlap
    switch (position) {
      case 'top-right':
        return { ...baseStyles, top: -50, right: 8 }; // Move above chart area
      case 'top-left':
        return { ...baseStyles, top: -50, left: 8 }; // Move above chart area
      case 'bottom-right':
        return { ...baseStyles, bottom: -50, right: 8 }; // Move below chart area
      case 'bottom-left':
        return { ...baseStyles, bottom: -50, left: 8 }; // Move below chart area
      default:
        return { ...baseStyles, top: -50, right: 8 }; // Default above chart area
    }
  };

  const convertToTimeseriesDataPoints = (data: ChartExportData[]): TimeseriesDataPoint[] => {
    const points: TimeseriesDataPoint[] = [];
    
    data.forEach(series => {
      series.data.forEach(point => {
        if (Array.isArray(point) && point.length >= 2) {
          // Handle [timestamp, value] format
          points.push({
            timestamp: new Date(point[0]),
            value: point[1],
            unit: series.unit,
            pointName: series.name,
          });
        } else if (typeof point === 'object' && 'timestamp' in point && 'value' in point) {
          // Handle TimeseriesDataPoint format
          points.push(point);
        }
      });
    });
    
    return points.sort((a, b) => new Date(a.timestamp).getTime() - new Date(b.timestamp).getTime());
  };

  const handleExport = async (format: 'png' | 'csv' | 'json') => {
    setExporting(true);
    setExportError(null);
    onExportStart?.();

    try {
      const timestamp = new Date().toISOString().slice(0, 19).replace(/[:.]/g, '-');
      const baseFilename = `${chartTitle.replace(/[^a-zA-Z0-9]/g, '_')}_${timestamp}`;

      switch (format) {
        case 'png':
          if (!chartInstance) {
            throw new Error('Chart instance not available for PNG export');
          }
          
          const exportOptions: ChartExportOptions = {
            filename: `${baseFilename}.png`,
            format: 'png',
            backgroundColor: theme.palette.background.paper,
            pixelRatio: 2,
            quality: 1,
          };
          
          await exportChartAsImage(chartInstance, exportOptions);
          setExportSuccess('Chart exported as PNG successfully!');
          break;

        case 'csv':
          // Use enhanced export with calculations if available
          if (enableEnhancedExport && chartInstance) {
            try {
              // Export with calculated values for the specific chart type
              exportChartWithCalculations(chartInstance, chartType, `${baseFilename}.csv`);
              setExportSuccess('Data exported as CSV with calculated values!');
            } catch (enhancedError) {

              // Fallback to basic export
              if (chartData.length === 0) {
                throw new Error('No data available for CSV export');
              }
              const csvData = convertToTimeseriesDataPoints(chartData);
              const csvOptions: CSVExportOptions = {
                filename: `${baseFilename}.csv`,
                includeHeaders: true,
                dateFormat: 'iso',
                precision: 3,
                includeUnits: true,
              };
              exportTimeseriesAsCSV(csvData, csvOptions);
              setExportSuccess('Data exported as CSV successfully!');
            }
          } else {
            // Basic export when enhanced export is not available
            if (chartData.length === 0) {
              throw new Error('No data available for CSV export');
            }
            const csvData = convertToTimeseriesDataPoints(chartData);
            const csvOptions: CSVExportOptions = {
              filename: `${baseFilename}.csv`,
              includeHeaders: true,
              dateFormat: 'iso',
              precision: 3,
              includeUnits: true,
            };
            exportTimeseriesAsCSV(csvData, csvOptions);
            setExportSuccess('Data exported as CSV successfully!');
          }
          break;

        case 'json':
          if (chartData.length === 0) {
            throw new Error('No data available for JSON export');
          }
          
          const jsonData = convertToTimeseriesDataPoints(chartData);
          exportTimeseriesAsJSON(jsonData, `${baseFilename}.json`, true);
          setExportSuccess('Data exported as JSON successfully!');
          break;

        default:
          throw new Error(`Unsupported export format: ${format}`);
      }

      onExportComplete?.(format);
    } catch (error) {
      const errorMessage = error instanceof Error ? error.message : 'Export failed';
      setExportError(errorMessage);
      onExportError?.(errorMessage);
    } finally {
      setExporting(false);
      handleClose();
    }
  };

  const exportOptions = [
    {
      format: 'png' as const,
      icon: <ImageIcon fontSize="small" />,
      label: 'Export as PNG',
      description: 'High-quality image export',
      disabled: !chartInstance,
    },
    {
      format: 'csv' as const,
      icon: <CsvIcon fontSize="small" />,
      label: 'Export as CSV',
      description: 'Raw data in spreadsheet format',
      disabled: chartData.length === 0,
    },
    {
      format: 'json' as const,
      icon: <JsonIcon fontSize="small" />,
      label: 'Export as JSON',
      description: 'Structured data format',
      disabled: chartData.length === 0,
    },
  ];

  return (
    <>
      <Box sx={getPositionStyles()}>
        {showLabels ? (
          <ButtonGroup size={size} variant="outlined">
            <Tooltip title="Export chart and data">
              <Button
                startIcon={exporting ? <CircularProgress size={16} /> : <ExportIcon />}
                disabled={exporting}
                onClick={handleExportClick}
              >
                Export
              </Button>
            </Tooltip>
          </ButtonGroup>
        ) : (
          <Tooltip title="Export options">
            <IconButton
              size={size}
              disabled={exporting}
              onClick={handleExportClick}
              sx={{
                backgroundColor: theme.palette.background.paper,
                '&:hover': {
                  backgroundColor: theme.palette.action.hover,
                },
              }}
            >
              {exporting ? <CircularProgress size={16} /> : <ExportIcon />}
            </IconButton>
          </Tooltip>
        )}

        <Menu
          anchorEl={anchorEl}
          open={Boolean(anchorEl)}
          onClose={handleClose}
          PaperProps={{
            sx: {
              backgroundColor: theme.palette.background.paper,
              backdropFilter: 'blur(10px)',
              border: `1px solid ${theme.palette.divider}`,
              minWidth: 220,
            },
          }}
        >
          <MenuItem disabled sx={{ fontSize: '0.75rem', opacity: 0.7 }}>
            <ListItemText primary="Export Options" />
          </MenuItem>
          <Divider />
          
          {exportOptions.map((option) => (
            <MenuItem
              key={option.format}
              disabled={option.disabled || exporting}
              onClick={() => handleExport(option.format)}
              sx={{ py: 1.5 }}
            >
              <ListItemIcon sx={{ minWidth: 36 }}>
                {option.icon}
              </ListItemIcon>
              <ListItemText
                primary={option.label}
                secondary={option.description}
                primaryTypographyProps={{ fontSize: '0.875rem' }}
                secondaryTypographyProps={{ fontSize: '0.75rem' }}
              />
            </MenuItem>
          ))}
          
          <Divider />
          <MenuItem disabled sx={{ fontSize: '0.75rem', opacity: 0.5 }}>
            <ListItemIcon sx={{ minWidth: 36 }}>
              <OptionsIcon fontSize="small" />
            </ListItemIcon>
            <ListItemText 
              primary={`${chartData.length} series • ${chartData.reduce((sum, s) => sum + s.data.length, 0)} points`}
            />
          </MenuItem>
        </Menu>
      </Box>

      {/* Success/Error Snackbars */}
      <Snackbar
        open={!!exportSuccess}
        autoHideDuration={3000}
        onClose={() => setExportSuccess(null)}
        anchorOrigin={{ vertical: 'bottom', horizontal: 'right' }}
      >
        <Alert onClose={() => setExportSuccess(null)} severity="success" variant="filled">
          {exportSuccess}
        </Alert>
      </Snackbar>

      <Snackbar
        open={!!exportError}
        autoHideDuration={5000}
        onClose={() => setExportError(null)}
        anchorOrigin={{ vertical: 'bottom', horizontal: 'right' }}
      >
        <Alert onClose={() => setExportError(null)} severity="error" variant="filled">
          {exportError}
        </Alert>
      </Snackbar>
    </>
  );
};

export default ChartExportToolbar;
