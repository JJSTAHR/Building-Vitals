import { VAVBoxCard } from './VAVBoxCard';
import type {
  VAVBoxStatus,
  VAVDashboardFilters} from '../../utils/vav/vavPerformanceMetrics';
import {
  generateMockVAVData,
} from '../../utils/vav/vavPerformanceMetrics';

interface VAVDiagnosticsDashboardProps {
  // Real-time data would come from props or hooks
  refreshInterval?: number;
  enableWebSocket?: boolean;
}

export const VAVDiagnosticsDashboard: React.FC<VAVDiagnosticsDashboardProps> = ({
  refreshInterval = 30,
  enableWebSocket = false,
}) => {
  const theme = useTheme();
  const isMobile = useMediaQuery(theme.breakpoints.down('md'));
  const isTablet = useMediaQuery(theme.breakpoints.between('md', 'lg'));

  // State management
  const [vavBoxes, setVavBoxes] = useState<VAVBoxStatus[]>([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);
  const [searchQuery, setSearchQuery] = useState('');
  const [filters, setFilters] = useState<VAVDashboardFilters>({
    floors: [],
    zones: [],
    healthStatus: [],
    sortBy: 'name',
    sortOrder: 'asc',
    showDetailedView: true,
  });
  const [speedDialOpen, setSpeedDialOpen] = useState(false);
  const [mobileFiltersOpen, setMobileFiltersOpen] = useState(false);

  // Load mock data (replace with real data fetching)
  const loadVAVData = useCallback(async () => {
    try {
      setLoading(true);
      setError(null);

      // Simulate API delay
      await new Promise((resolve) => setTimeout(resolve, 1000));

      // Generate mock data (replace with actual API call)
      const mockData = generateMockVAVData(24);
      setVavBoxes(mockData);
    } catch (err) {
      setError(err instanceof Error ? err.message : 'Failed to load VAV data');
    } finally {
      setLoading(false);
    }
  }, []);

  // WebSocket integration for real-time updates
  useEffect(() => {
    if (enableWebSocket) {
      // Mock WebSocket connection - replace with actual WebSocket implementation
      const mockWebSocket = {
        onmessage: null as ((event: { data: string }) => void) | null,
        send: (data: string) => {
          // Mock sending data
          console.log('[WebSocket] Sending:', data);
        },
        close: () => {
          console.log('[WebSocket] Connection closed');
        },
      };

      // Set up message handler
      mockWebSocket.onmessage = (event) => {
        try {
          const update = JSON.parse(event.data);

          if (update.type === 'VAV_UPDATE') {
            setVavBoxes((prevBoxes) =>
              prevBoxes.map((box) =>
                box.id === update.vavId ? { ...box, ...update.data, lastUpdate: Date.now() } : box
              )
            );
          } else if (update.type === 'FAULT_DETECTED') {
            setVavBoxes((prevBoxes) =>
              prevBoxes.map((box) =>
                update.affectedZones.includes(box.zoneName)
                  ? { ...box, faults: [...box.faults, update.fault] }
                  : box
              )
            );
          }
        } catch (error) {
          console.error('[WebSocket] Error parsing message:', error);
        }
      };

      // Simulate periodic updates for demo
      const simulateUpdates = setInterval(() => {
        const randomBoxIndex = Math.floor(Math.random() * vavBoxes.length);
        const randomBox = vavBoxes[randomBoxIndex];

        if (randomBox) {
          const mockUpdate = {
            type: 'VAV_UPDATE',
            vavId: randomBox.id,
            data: {
              currentTemp: randomBox.currentTemp + (Math.random() - 0.5) * 2,
              damperPosition: Math.max(
                0,
                Math.min(100, (randomBox.damperPosition || 50) + (Math.random() - 0.5) * 10)
              ),
              airflow: Math.max(0, (randomBox.airflow || 800) + (Math.random() - 0.5) * 200),
            },
          };

          if (mockWebSocket.onmessage) {
            mockWebSocket.onmessage({ data: JSON.stringify(mockUpdate) });
          }
        }
      }, 5000); // Update every 5 seconds

      return () => {
        clearInterval(simulateUpdates);
        mockWebSocket.close();
      };
    }
  }, [enableWebSocket, vavBoxes]);

  // Auto-refresh effect
  useEffect(() => {
    loadVAVData();

    if (refreshInterval > 0 && !enableWebSocket) {
      const interval = setInterval(loadVAVData, refreshInterval * 1000);
      return () => clearInterval(interval);
    }
  }, [loadVAVData, refreshInterval, enableWebSocket]);

  // Filter and sort VAV boxes
  const filteredAndSortedVAVBoxes = useMemo(() => {
    const filtered = vavBoxes.filter((box) => {
      // Search filter
      if (searchQuery && !box.zoneName.toLowerCase().includes(searchQuery.toLowerCase())) {
        return false;
      }

      // Health status filter
      if (filters.healthStatus.length > 0 && !filters.healthStatus.includes(box.healthStatus)) {
        return false;
      }

      // Floor filter (extract floor from zone name)
      if (filters.floors.length > 0) {
        const floorMatch = box.zoneName.match(/Floor (\d+)/);
        const floor = floorMatch ? `Floor ${floorMatch[1]}` : '';
        if (!filters.floors.includes(floor)) {return false;}
      }

      return true;
    });

    // Sort
    filtered.sort((a, b) => {
      let aValue: any = a.zoneName;
      let bValue: any = b.zoneName;

      switch (filters.sortBy) {
        case 'deviation':
          aValue = a.tempDeviation;
          bValue = b.tempDeviation;
          break;
        case 'damper':
          aValue = a.damperPosition || 0;
          bValue = b.damperPosition || 0;
          break;
        case 'faults':
          aValue = a.faults.length;
          bValue = b.faults.length;
          break;
        case 'energy':
          aValue = a.performance.airflowEfficiency;
          bValue = b.performance.airflowEfficiency;
          break;
        default:
          // Keep string comparison for name
          break;
      }

      if (typeof aValue === 'string') {
        const comparison = aValue.localeCompare(bValue);
        return filters.sortOrder === 'asc' ? comparison : -comparison;
      } else {
        return filters.sortOrder === 'asc' ? aValue - bValue : bValue - aValue;
      }
    });

    return filtered;
  }, [vavBoxes, searchQuery, filters]);

  // Summary statistics
  const summaryStats = useMemo(() => {
    const total = vavBoxes.length;
    const healthy = vavBoxes.filter((box) => box.healthStatus === 'healthy').length;
    const warning = vavBoxes.filter((box) => box.healthStatus === 'warning').length;
    const fault = vavBoxes.filter((box) => box.healthStatus === 'fault').length;
    const unknown = vavBoxes.filter((box) => box.healthStatus === 'unknown').length;
    const totalFaults = vavBoxes.reduce((sum, box) => sum + box.faults.length, 0);

    return { total, healthy, warning, fault, unknown, totalFaults };
  }, [vavBoxes]);

  // Available filters
  const availableFloors = useMemo(() => {
    const floors = new Set<string>();
    vavBoxes.forEach((box) => {
      const floorMatch = box.zoneName.match(/Floor (\d+)/);
      if (floorMatch) {floors.add(`Floor ${floorMatch[1]}`);}
    });
    return Array.from(floors).sort();
  }, [vavBoxes]);

  // Event handlers
  const handleSearchChange = (event: React.ChangeEvent<HTMLInputElement>) => {
    setSearchQuery(event.target.value);
  };

  const handleFilterChange = (key: keyof VAVDashboardFilters, value: any) => {
    setFilters((prev) => ({ ...prev, [key]: value }));
  };

  const handleClearFilters = () => {
    setSearchQuery('');
    setFilters({
      floors: [],
      zones: [],
      healthStatus: [],
      sortBy: 'name',
      sortOrder: 'asc',
      showDetailedView: filters.showDetailedView, // Keep view preference
    });
  };

  const handleExportReport = () => {
    // Generate CSV export data
    const csvData = filteredAndSortedVAVBoxes.map((box) => ({
      'Zone Name': box.zoneName,
      'Health Status': box.healthStatus,
      'Current Temperature (°F)': box.currentTemp || 'N/A',
      'Setpoint (°F)': box.setpoint || 'N/A',
      'Temperature Deviation (°F)': box.tempDeviation,
      'Damper Position (%)': box.damperPosition || 'N/A',
      'Airflow (CFM)': box.airflow || 'N/A',
      'Airflow Efficiency (%)': box.performance.airflowEfficiency.toFixed(1),
      'Control Accuracy (%)': box.performance.tempControlAccuracy.toFixed(1),
      'Damper Reversals/hr': box.performance.damperReversalFrequency.toFixed(1),
      'Active Faults': box.faults.length,
      'Data Quality (%)': (box.performance.dataQuality * 100).toFixed(0),
      'Last Updated': new Date(box.lastUpdate).toLocaleString(),
    }));

    // Convert to CSV
    const headers = Object.keys(csvData[0] || {});
    const csvContent = [
      headers.join(','),
      ...csvData.map((row) =>
        headers.map((header) => `"${row[header as keyof typeof row]}"`).join(',')
      ),
    ].join('\n');

    // Download CSV file
    const blob = new Blob([csvContent], { type: 'text/csv;charset=utf-8;' });
    const link = document.createElement('a');
    const url = URL.createObjectURL(blob);
    link.setAttribute('href', url);
    link.setAttribute(
      'download',
      `vav-diagnostics-report-${new Date().toISOString().split('T')[0]}.csv`
    );
    link.style.visibility = 'hidden';
    document.body.appendChild(link);
    link.click();
    document.body.removeChild(link);
    URL.revokeObjectURL(url);
  };

  const handleVAVBoxClick = (vavBox: VAVBoxStatus) => {
    // TODO: Open detailed view or modal
    console.log('Opening details for:', vavBox.zoneName);
  };

  // Mobile filters component
  const renderFiltersSection = () => {
    const content = (
      <Box sx={{ p: isMobile ? 2 : 3 }}>
        <Box display="flex" alignItems="center" gap={2} mb={2}>
          <FilterIcon color="action" />
          <Typography variant="h6">Filters & Search</Typography>
          <Button
            variant="outlined"
            size="small"
            startIcon={<ClearIcon />}
            onClick={handleClearFilters}
          >
            Clear All
          </Button>
        </Box>

        <Grid container spacing={2}>
          <Grid item xs={12} md={3}>
            <TextField
              fullWidth
              size="small"
              label="Search zones"
              value={searchQuery}
              onChange={handleSearchChange}
              InputProps={{
                startAdornment: <SearchIcon color="action" sx={{ mr: 1 }} />,
              }}
            />
          </Grid>

          <Grid item xs={12} sm={6} md={2}>
            <FormControl fullWidth size="small">
              <InputLabel>Floor</InputLabel>
              <Select
                multiple
                value={filters.floors}
                onChange={(e) => handleFilterChange('floors', e.target.value)}
                renderValue={(selected) => (
                  <Box sx={{ display: 'flex', flexWrap: 'wrap', gap: 0.5 }}>
                    {(selected).map((value) => (
                      <Chip key={value} label={value} size="small" />
                    ))}
                  </Box>
                )}
              >
                {availableFloors.map((floor) => (
                  <MenuItem key={floor} value={floor}>
                    {floor}
                  </MenuItem>
                ))}
              </Select>
            </FormControl>
          </Grid>

          <Grid item xs={12} sm={6} md={2}>
            <FormControl fullWidth size="small">
              <InputLabel>Status</InputLabel>
              <Select
                multiple
                value={filters.healthStatus}
                onChange={(e) => handleFilterChange('healthStatus', e.target.value)}
                renderValue={(selected) => (
                  <Box sx={{ display: 'flex', flexWrap: 'wrap', gap: 0.5 }}>
                    {(selected as string[]).map((value) => (
                      <Chip key={value} label={value} size="small" />
                    ))}
                  </Box>
                )}
              >
                <MenuItem value="healthy">Healthy</MenuItem>
                <MenuItem value="warning">Warning</MenuItem>
                <MenuItem value="fault">Fault</MenuItem>
                <MenuItem value="unknown">Unknown</MenuItem>
              </Select>
            </FormControl>
          </Grid>

          <Grid item xs={12} sm={6} md={2}>
            <FormControl fullWidth size="small">
              <InputLabel>Sort By</InputLabel>
              <Select
                value={filters.sortBy}
                onChange={(e) => handleFilterChange('sortBy', e.target.value)}
              >
                <MenuItem value="name">Zone Name</MenuItem>
                <MenuItem value="deviation">Temperature Deviation</MenuItem>
                <MenuItem value="damper">Damper Position</MenuItem>
                <MenuItem value="faults">Fault Count</MenuItem>
                <MenuItem value="energy">Energy Efficiency</MenuItem>
              </Select>
            </FormControl>
          </Grid>

          <Grid item xs={12} sm={6} md={2}>
            <FormControl fullWidth size="small">
              <InputLabel>Order</InputLabel>
              <Select
                value={filters.sortOrder}
                onChange={(e) => handleFilterChange('sortOrder', e.target.value)}
              >
                <MenuItem value="asc">Ascending</MenuItem>
                <MenuItem value="desc">Descending</MenuItem>
              </Select>
            </FormControl>
          </Grid>
        </Grid>
      </Box>
    );

    if (isMobile) {
      return (
        <>
          <Box display="flex" justifyContent="space-between" alignItems="center" mb={2}>
            <Button
              variant="outlined"
              startIcon={<FilterIcon />}
              onClick={() => setMobileFiltersOpen(true)}
            >
              Filters (
              {
                Object.values(filters).filter((v) => (Array.isArray(v) ? v.length > 0 : false))
                  .length
              }
              )
            </Button>
            <FormControlLabel
              control={
                <Switch
                  checked={filters.showDetailedView}
                  onChange={(e) => handleFilterChange('showDetailedView', e.target.checked)}
                  size="small"
                />
              }
              label={filters.showDetailedView ? 'Detailed' : 'Compact'}
            />
          </Box>

          <Drawer
            anchor="bottom"
            open={mobileFiltersOpen}
            onClose={() => setMobileFiltersOpen(false)}
            PaperProps={{
              sx: {
                borderRadius: '16px 16px 0 0',
                maxHeight: '80vh',
              },
            }}
          >
            {content}
          </Drawer>
        </>
      );
    }

    return <Paper sx={{ mb: 3 }}>{content}</Paper>;
  };

  if (loading && vavBoxes.length === 0) {
    return (
      <Container maxWidth="xl" sx={{ py: 4 }}>
        <LoadingSpinner message="Loading VAV diagnostics..." />
      </Container>
    );
  }

  return (
    <Container maxWidth="xl" sx={{ py: 4 }}>
      {/* Header */}
      <Box
        display="flex"
        justifyContent="space-between"
        alignItems={isMobile ? 'flex-start' : 'center'}
        mb={4}
      >
        <Box>
          <Typography variant={isMobile ? 'h5' : 'h4'} fontWeight="bold" gutterBottom>
            VAV Box Diagnostics
          </Typography>
          {!isMobile && (
            <Typography variant="subtitle1" color="text.secondary">
              Real-time monitoring and fault detection for Variable Air Volume systems
            </Typography>
          )}
        </Box>

        {!isMobile && (
          <Box display="flex" gap={1}>
            <Tooltip title="Refresh data">
              <IconButton onClick={loadVAVData} disabled={loading}>
                <RefreshIcon />
              </IconButton>
            </Tooltip>

            <FormControlLabel
              control={
                <Switch
                  checked={filters.showDetailedView}
                  onChange={(e) => handleFilterChange('showDetailedView', e.target.checked)}
                />
              }
              label={filters.showDetailedView ? 'Detailed' : 'Compact'}
            />
          </Box>
        )}
      </Box>

      {/* Summary Cards */}
      <Grid container spacing={3} mb={4}>
        <Grid item xs={12} sm={6} md={2.4}>
          <Card sx={{ bgcolor: alpha(theme.palette.primary.main, 0.1) }}>
            <CardContent>
              <Typography variant="h4" fontWeight="bold" color="primary">
                {summaryStats.total}
              </Typography>
              <Typography variant="body2" color="text.secondary">
                Total VAV Boxes
              </Typography>
            </CardContent>
          </Card>
        </Grid>

        <Grid item xs={12} sm={6} md={2.4}>
          <Card sx={{ bgcolor: alpha(theme.palette.success.main, 0.1) }}>
            <CardContent>
              <Typography variant="h4" fontWeight="bold" color="success.main">
                {summaryStats.healthy}
              </Typography>
              <Typography variant="body2" color="text.secondary">
                Healthy
              </Typography>
            </CardContent>
          </Card>
        </Grid>

        <Grid item xs={12} sm={6} md={2.4}>
          <Card sx={{ bgcolor: alpha(theme.palette.warning.main, 0.1) }}>
            <CardContent>
              <Typography variant="h4" fontWeight="bold" color="warning.main">
                {summaryStats.warning}
              </Typography>
              <Typography variant="body2" color="text.secondary">
                Warning
              </Typography>
            </CardContent>
          </Card>
        </Grid>

        <Grid item xs={12} sm={6} md={2.4}>
          <Card sx={{ bgcolor: alpha(theme.palette.error.main, 0.1) }}>
            <CardContent>
              <Typography variant="h4" fontWeight="bold" color="error">
                {summaryStats.fault}
              </Typography>
              <Typography variant="body2" color="text.secondary">
                Faults
              </Typography>
            </CardContent>
          </Card>
        </Grid>

        <Grid item xs={12} sm={6} md={2.4}>
          <Card sx={{ bgcolor: alpha(theme.palette.info.main, 0.1) }}>
            <CardContent>
              <Typography variant="h4" fontWeight="bold" color="info.main">
                {summaryStats.totalFaults}
              </Typography>
              <Typography variant="body2" color="text.secondary">
                Total Faults
              </Typography>
            </CardContent>
          </Card>
        </Grid>
      </Grid>

      {/* Filters and Search */}
      {renderFiltersSection()}

      {/* Error Alert */}
      {error && (
        <Alert severity="error" sx={{ mb: 3 }}>
          {error}
        </Alert>
      )}

      {/* Results Info */}
      <Box
        display="flex"
        justifyContent="space-between"
        alignItems="center"
        mb={2}
        flexWrap="wrap"
        gap={1}
      >
        <Box display="flex" alignItems="center" gap={2}>
          <Typography variant="body2" color="text.secondary">
            Showing {filteredAndSortedVAVBoxes.length} of {vavBoxes.length} VAV boxes
          </Typography>
          {enableWebSocket && (
            <Chip
              label="Live Updates"
              color="success"
              size="small"
              sx={{
                animation: 'pulse 2s infinite',
                '@keyframes pulse': {
                  '0%, 100%': { opacity: 1 },
                  '50%': { opacity: 0.7 },
                },
              }}
            />
          )}
        </Box>

        <Box display="flex" alignItems="center" gap={1}>
          {loading && (
            <Typography variant="caption" color="primary">
              Refreshing...
            </Typography>
          )}
          {!isMobile && (
            <Tooltip title="Refresh data">
              <IconButton onClick={loadVAVData} disabled={loading} size="small">
                <RefreshIcon fontSize="small" />
              </IconButton>
            </Tooltip>
          )}
        </Box>
      </Box>

      {/* VAV Boxes Grid */}
      <Grid container spacing={isMobile ? 2 : filters.showDetailedView ? 3 : 2}>
        {filteredAndSortedVAVBoxes.map((vavBox) => {
          // Responsive grid sizing
          let gridProps;
          if (isMobile) {
            gridProps = { xs: 12 };
          } else if (isTablet) {
            gridProps = filters.showDetailedView
              ? { xs: 12, sm: 12, md: 6 }
              : { xs: 12, sm: 6, md: 4 };
          } else {
            gridProps = filters.showDetailedView
              ? { xs: 12, sm: 12, md: 6, lg: 4, xl: 3 }
              : { xs: 12, sm: 6, md: 4, lg: 3, xl: 2 };
          }

          return (
            <Grid item {...gridProps} key={vavBox.id}>
              <VAVBoxCard
                vavBox={vavBox}
                compact={isMobile ? true : !filters.showDetailedView}
                showTrends={!isMobile && filters.showDetailedView}
                onClick={handleVAVBoxClick}
              />
            </Grid>
          );
        })}
      </Grid>

      {/* Empty State */}
      {filteredAndSortedVAVBoxes.length === 0 && !loading && (
        <Paper sx={{ p: 4, textAlign: 'center' }}>
          <Typography variant="h6" color="text.secondary" gutterBottom>
            No VAV boxes found
          </Typography>
          <Typography variant="body2" color="text.secondary">
            Try adjusting your filters or search criteria
          </Typography>
        </Paper>
      )}

      {/* Speed Dial for Actions */}
      <SpeedDial
        ariaLabel="VAV Dashboard Actions"
        sx={{ position: 'fixed', bottom: 24, right: 24 }}
        icon={<SpeedDialIcon />}
        open={speedDialOpen}
        onOpen={() => setSpeedDialOpen(true)}
        onClose={() => setSpeedDialOpen(false)}
      >
        <SpeedDialAction
          icon={<ExportIcon />}
          tooltipTitle="Export Report"
          onClick={handleExportReport}
        />
        <SpeedDialAction
          icon={<SettingsIcon />}
          tooltipTitle="Settings"
          onClick={() => console.log('Settings')}
        />
        <SpeedDialAction
          icon={<FullscreenIcon />}
          tooltipTitle="Fullscreen"
          onClick={() => console.log('Fullscreen')}
        />
      </SpeedDial>
    </Container>
  );
};

export default VAVDiagnosticsDashboard;
