import React, { useState } from 'react';
import {
  Box,
  Container,
  Typography,
  Button,
  Grid,
  Paper,
  Alert,
  Switch,
  FormControlLabel,
  Divider,
} from '@mui/material';
import { Add, PlayArrow, Settings, Visibility } from '@mui/icons-material';
import { glassEffect } from '../../theme/glassmorphism';

// Import all enhanced chart building components
import {
  EnhancedChartWizard,
  DropZoneIndicator,
  ChartPreview,
  PointSelectorLoading,
  ChartBuildingShortcuts,
  ChartPlacementAnimation,
  ChartSlideIn,
  ChartScaleIn,
  ChartBounceIn,
  ChartFadeScaleIn,
  ChartGridAnimation,
  EnhancedNotificationSystem,
  useEnhancedNotifications,
  useChartBuilding,
} from '../charts/wizard';

/**
 * Comprehensive demo showcasing all chart building enhancements
 * This demonstrates the complete UX flow and all available features
 */
const ChartBuildingDemo: React.FC = () => {
  // Notification system
  const { notification, showSuccess, showError, showWarning, showInfo, hideNotification } =
    useEnhancedNotifications();

  // Demo state
  const [wizardOpen, setWizardOpen] = useState(false);
  const [showShortcuts, setShowShortcuts] = useState(false);
  const [demoMode, setDemoMode] = useState<'loading' | 'preview' | 'dropzone' | 'animations'>(
    'dropzone'
  );
  const [animationDemo, setAnimationDemo] = useState(false);
  const [showPreviewDemo, setShowPreviewDemo] = useState(true);

  // Mock data for demonstrations
  const mockSites = [
    { id: 'site1', name: 'Main Building', address: '123 Main St' },
    { id: 'site2', name: 'North Campus', address: '456 North Ave' },
  ];

  const mockPoints = [
    { id: 'temp1', name: 'Zone 1 Temperature', unit: '°F', markerTag: 'temp' },
    { id: 'humidity1', name: 'Zone 1 Humidity', unit: '%RH', markerTag: 'humidity' },
    { id: 'pressure1', name: 'Static Pressure', unit: 'Pa', markerTag: 'pressure' },
  ];

  const mockChartTypes = ['TimeSeries', 'Scatter', 'Heatmap', 'Gauge', 'Psychrometric'];

  // Demo handlers
  const handleCreateChart = (config: any) => {
    console.log('Chart created:', config);
    showSuccess('Chart Created Successfully!', {
      message: `Created ${config.chartType} chart with ${config.selectedPoints?.length || 0} data points`,
      actions: [
        {
          label: 'View Chart',
          action: () => console.log('Navigate to chart'),
          primary: true,
        },
        {
          label: 'Create Another',
          action: () => setWizardOpen(true),
        },
      ],
      metadata: {
        details: [`Site: ${config.siteName}`, `Type: ${config.chartType}`],
      },
    });
    setWizardOpen(false);
  };

  const handleError = () => {
    showError('Validation Failed', {
      message: 'Please fix the following issues before proceeding',
      actions: [
        {
          label: 'Retry',
          action: () => console.log('Retry action'),
          primary: true,
        },
      ],
      metadata: {
        details: ['Missing data points', 'Invalid time range'],
      },
    });
  };

  const handleWarning = () => {
    showWarning('Performance Warning', {
      message: 'Large dataset detected. Consider using data aggregation.',
      duration: 8000,
      metadata: {
        progress: 75,
        details: ['10,000+ data points', 'Slow rendering expected'],
      },
    });
  };

  const triggerAnimationDemo = () => {
    setAnimationDemo(false);
    setTimeout(() => setAnimationDemo(true), 100);
  };

  return (
    <Container maxWidth="xl" sx={{ py: 4 }}>
      {/* Header */}
      <Box sx={{ mb: 4, textAlign: 'center' }}>
        <Typography variant="h3" sx={{ mb: 2, fontWeight: 700, color: 'primary.main' }}>
          Enhanced Chart Building System
        </Typography>
        <Typography variant="h6" color="text.secondary" sx={{ mb: 3 }}>
          Complete UX demonstration with visual feedback, loading states, and smooth animations
        </Typography>

        <Box sx={{ display: 'flex', gap: 2, justifyContent: 'center', flexWrap: 'wrap' }}>
          <Button
            variant="contained"
            size="large"
            startIcon={<Add />}
            onClick={() => setWizardOpen(true)}
            sx={{ minWidth: 200 }}
          >
            Open Enhanced Wizard
          </Button>

          <Button
            variant="outlined"
            startIcon={<Settings />}
            onClick={() => setShowShortcuts(true)}
          >
            Show Shortcuts
          </Button>
        </Box>
      </Box>

      {/* Demo Controls */}
      <Paper sx={{ ...glassEffect.light, p: 3, mb: 4 }}>
        <Typography variant="h6" sx={{ mb: 2 }}>
          Demo Controls
        </Typography>

        <Box sx={{ display: 'flex', gap: 2, mb: 2, flexWrap: 'wrap' }}>
          <Button
            variant={demoMode === 'dropzone' ? 'contained' : 'outlined'}
            onClick={() => setDemoMode('dropzone')}
          >
            Drop Zones
          </Button>
          <Button
            variant={demoMode === 'preview' ? 'contained' : 'outlined'}
            onClick={() => setDemoMode('preview')}
          >
            Chart Preview
          </Button>
          <Button
            variant={demoMode === 'loading' ? 'contained' : 'outlined'}
            onClick={() => setDemoMode('loading')}
          >
            Loading States
          </Button>
          <Button
            variant={demoMode === 'animations' ? 'contained' : 'outlined'}
            onClick={() => setDemoMode('animations')}
          >
            Animations
          </Button>
        </Box>

        <Box sx={{ display: 'flex', gap: 2, flexWrap: 'wrap' }}>
          <Button onClick={() => showSuccess('Demo Success!')}>Test Success</Button>
          <Button onClick={handleError}>Test Error</Button>
          <Button onClick={handleWarning}>Test Warning</Button>
          <Button onClick={() => showInfo('Demo Information', { showProgress: true })}>
            Test Info
          </Button>
        </Box>
      </Paper>

      {/* Demo Content */}
      <Grid container spacing={3}>
        {/* Drop Zone Demo */}
        {demoMode === 'dropzone' && (
          <>
            <Grid item xs={12}>
              <Typography variant="h5" sx={{ mb: 2 }}>
                Drop Zone Indicators
              </Typography>
              <Alert severity="info" sx={{ mb: 3 }}>
                These components provide visual feedback for drag-and-drop interactions and chart
                placement.
              </Alert>
            </Grid>

            <Grid item xs={12} md={4}>
              <DropZoneIndicator
                message="Drag chart here"
                size="large"
                isPlaceholder
                onClick={() => showInfo('Drop zone clicked!')}
              />
            </Grid>

            <Grid item xs={12} md={4}>
              <DropZoneIndicator message="Drop to add chart" size="large" isActive canDrop />
            </Grid>

            <Grid item xs={12} md={4}>
              <DropZoneIndicator
                message="Cannot drop here"
                size="large"
                isDragging
                canDrop={false}
              />
            </Grid>
          </>
        )}

        {/* Preview Demo */}
        {demoMode === 'preview' && (
          <>
            <Grid item xs={12}>
              <Typography variant="h5" sx={{ mb: 2 }}>
                Chart Preview System
              </Typography>
              <Alert severity="info" sx={{ mb: 3 }}>
                Live preview shows users what their chart will look like as they configure it.
              </Alert>

              <FormControlLabel
                control={
                  <Switch
                    checked={showPreviewDemo}
                    onChange={(e) => setShowPreviewDemo(e.target.checked)}
                  />
                }
                label="Show Preview"
                sx={{ mb: 2 }}
              />
            </Grid>

            <Grid item xs={12} md={6}>
              <ChartPreview
                chartType="TimeSeries"
                selectedPoints={mockPoints}
                config={{ showLegend: true, aggregation: 'mean' }}
                siteInfo={{ id: 'site1', name: 'Demo Site' }}
                timeRange="24h"
                show={showPreviewDemo}
                size="large"
                showMetadata
              />
            </Grid>

            <Grid item xs={12} md={6}>
              <ChartPreview
                chartType="Gauge"
                selectedPoints={[mockPoints[0]]}
                config={{ ranges: { normal: { min: 0, max: 70 } } }}
                siteInfo={{ id: 'site1', name: 'Demo Site' }}
                timeRange="1h"
                show={showPreviewDemo}
                size="large"
                showMetadata
              />
            </Grid>
          </>
        )}

        {/* Loading States Demo */}
        {demoMode === 'loading' && (
          <>
            <Grid item xs={12}>
              <Typography variant="h5" sx={{ mb: 2 }}>
                Enhanced Loading States
              </Typography>
              <Alert severity="info" sx={{ mb: 3 }}>
                Comprehensive loading indicators with progress tracking and skeleton screens.
              </Alert>
            </Grid>

            <Grid item xs={12} md={6}>
              <PointSelectorLoading
                stage="fetching"
                progress={{
                  loaded: 250,
                  total: 500,
                  currentOperation: 'Loading temperature sensors...',
                }}
                showDetails
                height={400}
              />
            </Grid>

            <Grid item xs={12} md={6}>
              <PointSelectorLoading
                stage="processing"
                progress={{ loaded: 450, total: 500 }}
                showDetails
                showSkeleton
                height={400}
              />
            </Grid>
          </>
        )}

        {/* Animation Demo */}
        {demoMode === 'animations' && (
          <>
            <Grid item xs={12}>
              <Typography variant="h5" sx={{ mb: 2 }}>
                Chart Placement Animations
              </Typography>
              <Alert severity="info" sx={{ mb: 3 }}>
                Smooth animations for chart addition, removal, and transitions.
              </Alert>

              <Button
                variant="contained"
                startIcon={<PlayArrow />}
                onClick={triggerAnimationDemo}
                sx={{ mb: 3 }}
              >
                Trigger Animation Demo
              </Button>
            </Grid>

            {animationDemo && (
              <ChartGridAnimation
                staggerDelay={150}
                animationType="fadeScale"
                columns={3}
                show={animationDemo}
              >
                {Array.from({ length: 6 }, (_, i) => (
                  <Grid item xs={12} md={4} key={i}>
                    <Paper
                      sx={{
                        ...glassEffect.medium,
                        p: 3,
                        height: 200,
                        display: 'flex',
                        alignItems: 'center',
                        justifyContent: 'center',
                        border: '1px solid rgba(0, 229, 255, 0.3)',
                      }}
                    >
                      <Typography variant="h6" color="primary.main">
                        Chart {i + 1}
                      </Typography>
                    </Paper>
                  </Grid>
                ))}
              </ChartGridAnimation>
            )}

            {!animationDemo && (
              <Grid item xs={12}>
                <Box sx={{ display: 'flex', gap: 2, flexWrap: 'wrap' }}>
                  <ChartSlideIn direction="bottom" show>
                    <Paper sx={{ ...glassEffect.light, p: 2, width: 150, textAlign: 'center' }}>
                      Slide In
                    </Paper>
                  </ChartSlideIn>

                  <ChartScaleIn show>
                    <Paper sx={{ ...glassEffect.light, p: 2, width: 150, textAlign: 'center' }}>
                      Scale In
                    </Paper>
                  </ChartScaleIn>

                  <ChartBounceIn show>
                    <Paper sx={{ ...glassEffect.light, p: 2, width: 150, textAlign: 'center' }}>
                      Bounce In
                    </Paper>
                  </ChartBounceIn>

                  <ChartFadeScaleIn show>
                    <Paper sx={{ ...glassEffect.light, p: 2, width: 150, textAlign: 'center' }}>
                      Fade Scale
                    </Paper>
                  </ChartFadeScaleIn>
                </Box>
              </Grid>
            )}
          </>
        )}
      </Grid>

      {/* Enhanced Chart Wizard */}
      <EnhancedChartWizard
        open={wizardOpen}
        onClose={() => setWizardOpen(false)}
        onComplete={handleCreateChart}
        sites={mockSites}
        points={mockPoints}
        isLoading={{
          sites: false,
          points: false,
        }}
      />

      {/* Keyboard Shortcuts */}
      <ChartBuildingShortcuts
        enabled={wizardOpen}
        showPanel={showShortcuts}
        onHidePanel={() => setShowShortcuts(false)}
        callbacks={{
          showHelp: () => setShowShortcuts(!showShortcuts),
        }}
      />

      {/* Enhanced Notifications */}
      <EnhancedNotificationSystem
        notification={notification}
        onClose={hideNotification}
        position={{ vertical: 'top', horizontal: 'right' }}
        showConfetti
      />

      {/* Implementation Guide */}
      <Paper sx={{ ...glassEffect.light, p: 4, mt: 4 }}>
        <Typography variant="h5" sx={{ mb: 3 }}>
          Implementation Guide
        </Typography>

        <Box sx={{ mb: 3 }}>
          <Typography variant="h6" sx={{ mb: 1 }}>
            Key Features Implemented:
          </Typography>
          <Box component="ul" sx={{ pl: 3 }}>
            <li>Visual drop zone indicators with hover effects and animations</li>
            <li>Live chart preview during configuration</li>
            <li>Comprehensive loading states with progress tracking</li>
            <li>Mobile-responsive wizard with touch interactions</li>
            <li>Multi-step progress indicators with validation</li>
            <li>Enhanced success/error messages with animations</li>
            <li>Keyboard shortcuts for power users</li>
            <li>Smooth chart placement animations</li>
            <li>Centralized chart building state management</li>
          </Box>
        </Box>

        <Divider sx={{ my: 3 }} />

        <Box>
          <Typography variant="h6" sx={{ mb: 1 }}>
            Usage Example:
          </Typography>
          <Paper
            sx={{ p: 2, backgroundColor: 'grey.900', color: 'grey.100', fontFamily: 'monospace' }}
          >
            <pre>{`import { 
  EnhancedChartWizard,
  useChartBuilding,
  useEnhancedNotifications 
} from './components/charts/wizard';

function MyComponent() {
  const { showSuccess } = useEnhancedNotifications();
  
  const handleChartCreated = (config) => {
    showSuccess('Chart created!', {
      actions: [{ label: 'View', action: () => navigate('/chart') }]
    });
  };
  
  return (
    <EnhancedChartWizard
      open={wizardOpen}
      onComplete={handleChartCreated}
      // ... other props
    />
  );
}`}</pre>
          </Paper>
        </Box>
      </Paper>
    </Container>
  );
};

export default ChartBuildingDemo;
