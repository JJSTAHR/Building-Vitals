import React from 'react';
import { Box, Button, Menu, MenuItem, Typography, TextField } from '@mui/material';
import { AdapterDateFns } from '@mui/x-date-pickers/AdapterDateFns';
import { LocalizationProvider } from '@mui/x-date-pickers/LocalizationProvider';
import { DateTimePicker } from '@mui/x-date-pickers/DateTimePicker';
import { subDays, subHours, subMinutes } from 'date-fns';

interface TimeRangePickerProps {
  startTime: Date;
  endTime: Date;
  onTimeRangeChange: (start: Date, end: Date) => void;
}

const TimeRangePicker: React.FC<TimeRangePickerProps> = ({
  startTime,
  endTime,
  onTimeRangeChange,
}) => {
  const [anchorEl, setAnchorEl] = React.useState<null | HTMLElement>(null);
  const open = Boolean(anchorEl);

  const handleClick = (event: React.MouseEvent<HTMLButtonElement>) => {
    setAnchorEl(event.currentTarget);
  };

  const handleClose = () => {
    setAnchorEl(null);
  };

  const handlePresetClick = (preset: string) => {
    const now = new Date();
    let start: Date;
    switch (preset) {
      case 'last15min':
        start = subMinutes(now, 15);
        break;
      case 'lastHour':
        start = subHours(now, 1);
        break;
      case 'last24Hours':
        start = subHours(now, 24);
        break;
      case 'last7Days':
        start = subDays(now, 7);
        break;
      default:
        start = now;
    }
    onTimeRangeChange(start, now);
    handleClose();
  };

  return (
    <LocalizationProvider dateAdapter={AdapterDateFns}>
      <Box sx={{ display: 'flex', alignItems: 'center', gap: 2, mb: 2 }}>
        <Typography variant="subtitle1">Time Range:</Typography>
        <DateTimePicker
          enableAccessibleFieldDOMStructure={false}
          label="Start Time"
          slotProps={{ textField: { size: 'small' } }}
          slots={{ textField: TextField }}
          value={startTime}
          onChange={(newValue) => newValue && onTimeRangeChange(newValue, endTime)}
        />
        <DateTimePicker
          enableAccessibleFieldDOMStructure={false}
          label="End Time"
          slotProps={{ textField: { size: 'small' } }}
          slots={{ textField: TextField }}
          value={endTime}
          onChange={(newValue) => newValue && onTimeRangeChange(startTime, newValue)}
        />
        <Button
          aria-controls={open ? 'basic-menu' : undefined}
          aria-expanded={open ? 'true' : undefined}
          aria-haspopup="true"
          id="basic-button"
          variant="outlined"
          onClick={handleClick}
        >
          Presets
        </Button>
        <Menu
          MenuListProps={{
            'aria-labelledby': 'basic-button',
          }}
          anchorEl={anchorEl}
          id="basic-menu"
          open={open}
          onClose={handleClose}
        >
          <MenuItem onClick={() => handlePresetClick('last15min')}>Last 15 Minutes</MenuItem>
          <MenuItem onClick={() => handlePresetClick('lastHour')}>Last Hour</MenuItem>
          <MenuItem onClick={() => handlePresetClick('last24Hours')}>Last 24 Hours</MenuItem>
          <MenuItem onClick={() => handlePresetClick('last7Days')}>Last 7 Days</MenuItem>
        </Menu>
      </Box>
    </LocalizationProvider>
  );
};

export default TimeRangePicker;
