import React, { useState, _useEffect } from 'react';
import { Box, Typography, IconButton, Menu, MenuItem, Tooltip } from '@mui/material';
import type { Widget } from '../../types/dashboard';
import { debugService } from '../../services/debugService';
import { coreIcons, iconBundles, LazyIcon } from '../../utils/iconLoader';

// Load action icons bundle for this component
const actionIcons = iconBundles.actions();
const navigationIcons = iconBundles.navigation();
const advancedIcons = iconBundles.advanced();

interface ChartWidgetHeaderProps {
  widget: Widget;
  onDelete: (widgetId: string) => void;
  onDuplicate: (widget: Widget) => void;
  onUpdate: (updatedWidget: Widget) => void;
  onExport?: (widget: Widget) => void;
  onDebug?: (widget: Widget) => void;
}

const ChartWidgetHeader: React.FC<ChartWidgetHeaderProps> = ({
  widget,
  onDelete,
  onDuplicate,
  _onUpdate,
  onExport,
  onDebug,
}) => {
  const [anchorEl, setAnchorEl] = useState<null | HTMLElement>(null);
  // const [settingsOpen, setSettingsOpen] = useState(false);

  const handleMenuOpen = (event: React.MouseEvent<HTMLElement>) => {
    setAnchorEl(event.currentTarget);
  };

  const handleMenuClose = () => {
    setAnchorEl(null);
  };

  const handleDelete = () => {
    onDelete(widget.id);
    handleMenuClose();
  };

  const handleDuplicate = () => {
    onDuplicate(widget);
    handleMenuClose();
  };

  const handleSettingsOpen = () => {
    // Settings feature reserved for future implementation
    debugService.info('ChartWidget', 'Widget settings feature coming soon', {
      widgetId: widget.id,
    });
    handleMenuClose();
  };

  const handleExport = () => {
    if (onExport) {
      onExport(widget);
    }
    handleMenuClose();
  };

  const handleDebug = () => {
    if (onDebug) {
      onDebug(widget);
    }
  };

  return (
    <Box
      sx={{
        display: 'flex',
        justifyContent: 'space-between',
        alignItems: 'center',
        p: 1,
        borderBottom: 1,
        borderColor: 'divider',
      }}
    >
      <Tooltip title={widget.title || widget.name}>
        <Typography
          sx={{
            ml: 1,
            overflow: 'hidden',
            textOverflow: 'ellipsis',
            whiteSpace: 'nowrap',
            maxWidth: '70%',
          }}
          variant="h6"
        >
          {widget.title || widget.name}
        </Typography>
      </Tooltip>
      <Box>
        {onDebug && (
          <Tooltip title="Debug Chart Data">
            <IconButton color="primary" size="small" onClick={handleDebug}>
              <LazyIcon icon={advancedIcons.BugReport} />
            </IconButton>
          </Tooltip>
        )}
        <Tooltip title="Widget Options">
          <IconButton size="small" onClick={handleMenuOpen}>
            <LazyIcon icon={navigationIcons.MoreVert} />
          </IconButton>
        </Tooltip>
        <Menu anchorEl={anchorEl} open={Boolean(anchorEl)} onClose={handleMenuClose}>
          <MenuItem onClick={handleSettingsOpen}>
            <LazyIcon icon={coreIcons.Settings} sx={{ mr: 1 }} />
            Settings
          </MenuItem>
          <MenuItem onClick={handleDuplicate}>
            <LazyIcon icon={actionIcons.ContentCopy} sx={{ mr: 1 }} />
            Duplicate
          </MenuItem>
          {onExport && (
            <MenuItem onClick={handleExport}>
              <LazyIcon icon={actionIcons.Download} sx={{ mr: 1 }} />
              Export Data
            </MenuItem>
          )}
          <MenuItem sx={{ color: 'error.main' }} onClick={handleDelete}>
            <LazyIcon icon={coreIcons.Delete} sx={{ mr: 1 }} />
            Delete
          </MenuItem>
        </Menu>
      </Box>
    </Box>
  );
};

export default ChartWidgetHeader;
