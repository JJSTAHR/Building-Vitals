import React, { useState, useEffect } from 'react';
import {
  Dialog,
  DialogTitle,
  DialogContent,
  Stepper,
  Step,
  StepLabel,
  StepContent,
  Button,
  Typography,
  Box,
  Alert,
  Card,
  CardContent,
  Chip,
  Stack,
  IconButton,
  Tooltip,
  List,
  ListItem,
  ListItemIcon,
  ListItemText,
  Fade,
  Paper,
  Divider,
  LinearProgress,
} from '@mui/material';
import {
  CheckCircle as CheckIcon,
  Warning as WarningIcon,
  Info as InfoIcon,
  AccessTime as ClockIcon,
  Language as LanguageIcon,
  Computer as ComputerIcon,
  LocationOn as LocationIcon,
  HelpOutline as HelpIcon,
  NavigateNext as NextIcon,
  NavigateBefore as BackIcon,
} from '@mui/icons-material';
import {
  getBrowserTimeInfo,
  getTimezoneExplanation,
  detectTimezoneIssues,
} from '../../utils/browserTimeDetection';
import { useTimezone } from '../../contexts/TimezoneContext';
import { format } from 'date-fns';
import { getTimezoneDisplay } from '../../utils/timezoneUtils';

interface TimezoneSetupGuideProps {
  open: boolean;
  onClose: () => void;
  onComplete?: () => void;
}

export const TimezoneSetupGuide: React.FC<TimezoneSetupGuideProps> = ({
  open,
  onClose,
  onComplete,
}) => {
  const [activeStep, setActiveStep] = useState(0);
  const [browserInfo, setBrowserInfo] = useState(getBrowserTimeInfo());
  const [issues, setIssues] = useState<string[]>([]);
  const { currentSite, siteTimezone, browserTimezone, isTimezoneMismatch } = useTimezone();

  // Update browser info every second
  useEffect(() => {
    const timer = setInterval(() => {
      const info = getBrowserTimeInfo();
      setBrowserInfo(info);
      setIssues(detectTimezoneIssues(info));
    }, 1000);
    return () => clearInterval(timer);
  }, []);

  const handleNext = () => {
    setActiveStep((prev) => prev + 1);
  };

  const handleBack = () => {
    setActiveStep((prev) => prev - 1);
  };

  const handleComplete = () => {
    onComplete?.();
    onClose();
  };

  const steps = [
    {
      label: 'Understanding Your Time Settings',
      content: (
        <Stack spacing={3}>
          <Typography variant="body2">
            Building Vitals needs to know your computer's time settings to accurately display
            building data.
          </Typography>

          <Card elevation={0} sx={{ bgcolor: 'primary.lighter' }}>
            <CardContent>
              <Stack spacing={2}>
                <Stack direction="row" alignItems="center" spacing={1}>
                  <ComputerIcon color="primary" />
                  <Typography variant="subtitle2">Your Computer's Time</Typography>
                  <Tooltip title="This time comes from your computer's system clock">
                    <HelpIcon fontSize="small" color="action" />
                  </Tooltip>
                </Stack>

                <Box sx={{ pl: 4 }}>
                  <Typography variant="h5" gutterBottom>
                    {format(browserInfo.currentTime, 'h:mm:ss a')}
                  </Typography>
                  <Typography variant="body2" color="text.secondary">
                    {format(browserInfo.currentTime, 'EEEE, MMMM d, yyyy')}
                  </Typography>
                </Box>

                <Divider />

                <List dense>
                  <ListItem>
                    <ListItemIcon>
                      <LocationIcon fontSize="small" />
                    </ListItemIcon>
                    <ListItemText
                      primary="Detected Timezone"
                      secondary={getTimezoneDisplay(browserInfo.timezone)}
                    />
                    <Tooltip title="This is detected from your browser settings">
                      <HelpIcon fontSize="small" color="action" />
                    </Tooltip>
                  </ListItem>

                  <ListItem>
                    <ListItemIcon>
                      <LanguageIcon fontSize="small" />
                    </ListItemIcon>
                    <ListItemText primary="Browser Language" secondary={browserInfo.language} />
                  </ListItem>

                  <ListItem>
                    <ListItemIcon>
                      <ClockIcon fontSize="small" />
                    </ListItemIcon>
                    <ListItemText
                      primary="UTC Offset"
                      secondary={`UTC${browserInfo.timezoneOffset <= 0 ? '+' : '-'}${Math.abs(browserInfo.timezoneOffset) / 60} hours`}
                    />
                  </ListItem>
                </List>
              </Stack>
            </CardContent>
          </Card>

          {issues.length > 0 && (
            <Alert severity="warning" icon={<WarningIcon />}>
              <Typography variant="subtitle2" gutterBottom>
                Potential Issues Detected:
              </Typography>
              {issues.map((issue, index) => (
                <Typography key={index} variant="body2">
                  • {issue}
                </Typography>
              ))}
            </Alert>
          )}

          <Alert severity="info" icon={<InfoIcon />}>
            <Typography variant="body2">
              <strong>Why this matters:</strong> When you view building data, times need to be shown
              correctly for both your location and the building's location.
            </Typography>
          </Alert>
        </Stack>
      ),
    },
    {
      label: 'How Building Timezones Work',
      content: (
        <Stack spacing={3}>
          <Typography variant="body2">
            Buildings may be in different timezones than you. Here's how we handle that:
          </Typography>

          <Paper elevation={0} sx={{ p: 2, bgcolor: 'background.default' }}>
            <Stack spacing={2}>
              <Typography variant="subtitle2" color="primary">
                Example Scenario
              </Typography>
              <Typography variant="body2">
                You're in <Chip label="New York (EST)" size="small" /> viewing a building in{' '}
                <Chip label="Chicago (CST)" size="small" />
              </Typography>

              <Box sx={{ my: 2 }}>
                <LinearProgress
                  variant="determinate"
                  value={100}
                  sx={{ height: 8, borderRadius: 1 }}
                />
              </Box>

              <Stack direction="row" justifyContent="space-between">
                <Box>
                  <Typography variant="caption" color="text.secondary">
                    Your Time
                  </Typography>
                  <Typography variant="body2" fontWeight="medium">
                    3:00 PM EST
                  </Typography>
                </Box>
                <Box textAlign="center">
                  <NextIcon color="action" />
                </Box>
                <Box textAlign="right">
                  <Typography variant="caption" color="text.secondary">
                    Building Time
                  </Typography>
                  <Typography variant="body2" fontWeight="medium">
                    2:00 PM CST
                  </Typography>
                </Box>
              </Stack>
            </Stack>
          </Paper>

          <Stack spacing={2}>
            <Typography variant="subtitle2">Key Principles:</Typography>
            <List>
              <ListItem>
                <ListItemIcon>
                  <CheckIcon color="success" fontSize="small" />
                </ListItemIcon>
                <ListItemText
                  primary="Data is always shown in building time"
                  secondary="When you select '9 AM', it means 9 AM at the building"
                />
              </ListItem>
              <ListItem>
                <ListItemIcon>
                  <CheckIcon color="success" fontSize="small" />
                </ListItemIcon>
                <ListItemText
                  primary="Clear timezone indicators"
                  secondary="You'll always see which timezone is being used"
                />
              </ListItem>
              <ListItem>
                <ListItemIcon>
                  <CheckIcon color="success" fontSize="small" />
                </ListItemIcon>
                <ListItemText
                  primary="Automatic conversion"
                  secondary="We handle all the complex timezone math for you"
                />
              </ListItem>
            </List>
          </Stack>

          {currentSite && isTimezoneMismatch && (
            <Alert severity="info">
              <Typography variant="body2">
                <strong>Current situation:</strong> You're viewing {currentSite.name} which is in{' '}
                <Chip label={getTimezoneDisplay(siteTimezone)} size="small" />, different from your
                timezone <Chip label={getTimezoneDisplay(browserTimezone)} size="small" />
              </Typography>
            </Alert>
          )}
        </Stack>
      ),
    },
    {
      label: 'Timezone Indicators',
      content: (
        <Stack spacing={3}>
          <Typography variant="body2">
            Look for these indicators throughout the application:
          </Typography>

          <Stack spacing={2}>
            {/* Timezone Chip Example */}
            <Card elevation={0} sx={{ bgcolor: 'grey.50' }}>
              <CardContent>
                <Typography variant="subtitle2" gutterBottom>
                  Time Display with Timezone
                </Typography>
                <Box sx={{ mt: 2 }}>
                  <Chip icon={<ClockIcon />} label="2:30 PM CST (UTC-6)" color="warning" />
                  <Typography variant="caption" display="block" mt={1}>
                    Orange indicates timezone differs from yours
                  </Typography>
                </Box>
              </CardContent>
            </Card>

            {/* Warning Message Example */}
            <Card elevation={0} sx={{ bgcolor: 'warning.lighter' }}>
              <CardContent>
                <Stack direction="row" spacing={1} alignItems="flex-start">
                  <WarningIcon fontSize="small" color="warning" />
                  <Box>
                    <Typography variant="subtitle2" color="warning.dark">
                      Timezone Difference
                    </Typography>
                    <Typography variant="caption" color="warning.dark">
                      Time selections will be interpreted in the building's timezone
                      (America/Chicago)
                    </Typography>
                  </Box>
                </Stack>
              </CardContent>
            </Card>

            {/* Tooltip Example */}
            <Card elevation={0} sx={{ bgcolor: 'grey.50' }}>
              <CardContent>
                <Typography variant="subtitle2" gutterBottom>
                  Hover for Details
                </Typography>
                <Stack direction="row" spacing={2} alignItems="center" sx={{ mt: 2 }}>
                  <Typography variant="body2">Building Time: 2:30 PM</Typography>
                  <Tooltip
                    title={
                      <Box>
                        <Typography variant="caption" display="block">
                          Building: Chicago Office
                        </Typography>
                        <Typography variant="caption" display="block">
                          Timezone: Central Time (CST)
                        </Typography>
                        <Typography variant="caption" display="block">
                          Your Time: 3:30 PM EST
                        </Typography>
                      </Box>
                    }
                    arrow
                  >
                    <InfoIcon fontSize="small" color="action" sx={{ cursor: 'help' }} />
                  </Tooltip>
                </Stack>
              </CardContent>
            </Card>
          </Stack>

          <Alert severity="success">
            <Typography variant="body2">
              <strong>Pro tip:</strong> You can click on timezone indicators to see more details or
              adjust settings for specific buildings.
            </Typography>
          </Alert>
        </Stack>
      ),
    },
    {
      label: 'Review & Settings',
      content: (
        <Stack spacing={3}>
          <Typography variant="body2">
            Your timezone settings are configured correctly! Here's a summary:
          </Typography>

          <Card elevation={0} sx={{ bgcolor: 'success.lighter' }}>
            <CardContent>
              <Stack spacing={2}>
                <Stack direction="row" alignItems="center" spacing={1}>
                  <CheckIcon color="success" />
                  <Typography variant="subtitle2" color="success.dark">
                    Setup Complete
                  </Typography>
                </Stack>

                <List dense>
                  <ListItem>
                    <ListItemText
                      primary="Your Timezone"
                      secondary={getTimezoneDisplay(browserTimezone)}
                    />
                  </ListItem>
                  {currentSite && (
                    <ListItem>
                      <ListItemText
                        primary="Current Building"
                        secondary={`${currentSite.name} - ${getTimezoneDisplay(siteTimezone)}`}
                      />
                    </ListItem>
                  )}
                  <ListItem>
                    <ListItemText
                      primary="Detection Method"
                      secondary={
                        browserInfo.detectionMethod === 'automatic'
                          ? 'Automatic (Recommended)'
                          : 'Manual Override'
                      }
                    />
                  </ListItem>
                </List>
              </Stack>
            </CardContent>
          </Card>

          <Typography variant="subtitle2">What's Next?</Typography>
          <List>
            <ListItem>
              <ListItemIcon>
                <InfoIcon fontSize="small" color="primary" />
              </ListItemIcon>
              <ListItemText
                primary="Watch for timezone indicators"
                secondary="Orange badges show when building timezone differs from yours"
              />
            </ListItem>
            <ListItem>
              <ListItemIcon>
                <InfoIcon fontSize="small" color="primary" />
              </ListItemIcon>
              <ListItemText
                primary="Time selections are intuitive"
                secondary="When you pick a time, it applies to the building's local time"
              />
            </ListItem>
            <ListItem>
              <ListItemIcon>
                <InfoIcon fontSize="small" color="primary" />
              </ListItemIcon>
              <ListItemText
                primary="Override if needed"
                secondary="You can manually set building timezones in Settings"
              />
            </ListItem>
          </List>

          <Alert severity="info">
            <Typography variant="body2">
              <strong>Need help?</strong> Click the{' '}
              <Chip icon={<HelpIcon />} label="Help" size="small" /> button on any timezone
              indicator for context-sensitive assistance.
            </Typography>
          </Alert>
        </Stack>
      ),
    },
  ];

  return (
    <Dialog
      open={open}
      onClose={onClose}
      maxWidth="md"
      fullWidth
      PaperProps={{
        sx: { height: '80vh', display: 'flex', flexDirection: 'column' },
      }}
    >
      <DialogTitle>
        <Stack direction="row" alignItems="center" spacing={2}>
          <ClockIcon color="primary" />
          <Typography variant="h6">Timezone Setup Guide</Typography>
        </Stack>
      </DialogTitle>

      <DialogContent sx={{ flex: 1, overflow: 'auto' }}>
        <Stepper activeStep={activeStep} orientation="vertical">
          {steps.map((step, index) => (
            <Step key={step.label}>
              <StepLabel>{step.label}</StepLabel>
              <StepContent>
                <Fade in={activeStep === index} timeout={500}>
                  <Box sx={{ mt: 2, mb: 4 }}>{step.content}</Box>
                </Fade>

                <Box sx={{ mt: 3, display: 'flex', gap: 1 }}>
                  <Button
                    variant="contained"
                    onClick={index === steps.length - 1 ? handleComplete : handleNext}
                    endIcon={index === steps.length - 1 ? <CheckIcon /> : <NextIcon />}
                  >
                    {index === steps.length - 1 ? 'Complete Setup' : 'Continue'}
                  </Button>
                  {index > 0 && (
                    <Button onClick={handleBack} startIcon={<BackIcon />}>
                      Back
                    </Button>
                  )}
                  <Box sx={{ flex: 1 }} />
                  <Button onClick={onClose} color="inherit">
                    Skip Tour
                  </Button>
                </Box>
              </StepContent>
            </Step>
          ))}
        </Stepper>
      </DialogContent>
    </Dialog>
  );
};

export default TimezoneSetupGuide;
