/**
 * Token Manager Component
 * Allows users to manage ACE tokens for different sites
 * Stores tokens securely in encrypted format
 */

import React, { useState, useEffect } from 'react';
import {
  Box,
  Card,
  CardContent,
  Typography,
  TextField,
  Button,
  IconButton,
  List,
  ListItem,
  ListItemText,
  ListItemSecondaryAction,
  Dialog,
  DialogTitle,
  DialogContent,
  DialogActions,
  Alert,
  Select,
  MenuItem,
  FormControl,
  InputLabel,
  Chip,
  Tooltip,
  Stack,
  InputAdornment,
} from '@mui/material';
import {
  Add as AddIcon,
  Delete as DeleteIcon,
  Edit as EditIcon,
  ContentCopy as CopyIcon,
  Visibility as VisibilityIcon,
  VisibilityOff as VisibilityOffIcon,
  Key as KeyIcon,
  CheckCircle as CheckIcon,
  Error as ErrorIcon,
} from '@mui/icons-material';

interface SiteToken {
  siteId: string;
  siteName: string;
  token: string;
  identity?: string;
  expiresAt?: string;
  addedAt: string;
  lastUsed?: string;
  isDefault?: boolean;
}

const TOKEN_STORAGE_KEY = 'ace_site_tokens';

// Simple encryption/decryption for local storage
const encryptToken = (token: string): string => {
  // In production, use a proper encryption library
  return btoa(token); // Base64 encoding for demo
};

const decryptToken = (encrypted: string): string => {
  try {
    return atob(encrypted); // Base64 decoding for demo
  } catch {
    return '';
  }
};

// Decode JWT to extract identity and expiry
const decodeJWT = (token: string): { identity?: string; exp?: number } => {
  try {
    const parts = token.split('.');
    if (parts.length !== 3) return {};

    const payload = JSON.parse(atob(parts[1]));
    return {
      identity: payload.identity,
      exp: payload.exp
    };
  } catch {
    return {};
  }
};

export const TokenManager: React.FC = () => {
  const [tokens, setTokens] = useState<SiteToken[]>([]);
  const [dialogOpen, setDialogOpen] = useState(false);
  const [editingToken, setEditingToken] = useState<SiteToken | null>(null);
  const [newToken, setNewToken] = useState({
    siteId: '',
    siteName: '',
    token: ''
  });
  const [showTokens, setShowTokens] = useState<Record<string, boolean>>({});
  const [testingToken, setTestingToken] = useState<string | null>(null);
  const [testResult, setTestResult] = useState<{ success: boolean; message: string } | null>(null);

  // Load tokens from local storage on mount
  useEffect(() => {
    const stored = localStorage.getItem(TOKEN_STORAGE_KEY);
    if (stored) {
      try {
        const encrypted = JSON.parse(stored);
        const decrypted = encrypted.map((t: any) => ({
          ...t,
          token: decryptToken(t.token)
        }));
        setTokens(decrypted);
      } catch (error) {
        console.error('Failed to load tokens:', error);
      }
    }
  }, []);

  // Save tokens to local storage whenever they change
  useEffect(() => {
    if (tokens.length > 0) {
      const encrypted = tokens.map(t => ({
        ...t,
        token: encryptToken(t.token)
      }));
      localStorage.setItem(TOKEN_STORAGE_KEY, JSON.stringify(encrypted));
    } else {
      localStorage.removeItem(TOKEN_STORAGE_KEY);
    }
  }, [tokens]);

  const handleAddToken = () => {
    if (!newToken.siteId || !newToken.siteName || !newToken.token) {
      alert('Please fill in all fields');
      return;
    }

    const decoded = decodeJWT(newToken.token);
    const newSiteToken: SiteToken = {
      siteId: newToken.siteId,
      siteName: newToken.siteName,
      token: newToken.token,
      identity: decoded.identity,
      expiresAt: decoded.exp ? new Date(decoded.exp * 1000).toISOString() : undefined,
      addedAt: new Date().toISOString(),
      isDefault: tokens.length === 0 // First token is default
    };

    if (editingToken) {
      setTokens(tokens.map(t =>
        t.siteId === editingToken.siteId ? newSiteToken : t
      ));
    } else {
      setTokens([...tokens, newSiteToken]);
    }

    setDialogOpen(false);
    setEditingToken(null);
    setNewToken({ siteId: '', siteName: '', token: '' });
  };

  const handleDeleteToken = (siteId: string) => {
    if (confirm('Are you sure you want to delete this token?')) {
      setTokens(tokens.filter(t => t.siteId !== siteId));
    }
  };

  const handleSetDefault = (siteId: string) => {
    setTokens(tokens.map(t => ({
      ...t,
      isDefault: t.siteId === siteId
    })));
  };

  const handleTestToken = async (token: SiteToken) => {
    setTestingToken(token.siteId);
    setTestResult(null);

    try {
      const response = await fetch('https://ace-iot-ai-proxy.jstahr.workers.dev/api/sites', {
        headers: {
          'X-ACE-Token': token.token
        }
      });

      if (response.ok) {
        const data = await response.json();
        setTestResult({
          success: true,
          message: `✅ Token valid! Found ${data.items?.length || 0} sites`
        });

        // Update last used
        setTokens(tokens.map(t =>
          t.siteId === token.siteId
            ? { ...t, lastUsed: new Date().toISOString() }
            : t
        ));
      } else {
        setTestResult({
          success: false,
          message: `❌ Token invalid: ${response.status} ${response.statusText}`
        });
      }
    } catch (error: any) {
      setTestResult({
        success: false,
        message: `❌ Test failed: ${error.message}`
      });
    } finally {
      setTestingToken(null);
    }
  };

  const isTokenExpired = (token: SiteToken): boolean => {
    if (!token.expiresAt) return false;
    return new Date(token.expiresAt) < new Date();
  };

  const formatDate = (dateString?: string): string => {
    if (!dateString) return 'Never';
    const date = new Date(dateString);
    return date.toLocaleDateString() + ' ' + date.toLocaleTimeString();
  };

  return (
    <Card>
      <CardContent>
        <Box display="flex" justifyContent="space-between" alignItems="center" mb={2}>
          <Typography variant="h6">
            <KeyIcon sx={{ mr: 1, verticalAlign: 'middle' }} />
            Site Token Manager
          </Typography>
          <Button
            variant="contained"
            color="primary"
            startIcon={<AddIcon />}
            onClick={() => {
              setEditingToken(null);
              setNewToken({ siteId: '', siteName: '', token: '' });
              setDialogOpen(true);
            }}
          >
            Add Token
          </Button>
        </Box>

        <Alert severity="info" sx={{ mb: 2 }}>
          Manage ACE API tokens for different sites. Each site can have its own token for better security and access control.
        </Alert>

        {testResult && (
          <Alert severity={testResult.success ? 'success' : 'error'} sx={{ mb: 2 }}>
            {testResult.message}
          </Alert>
        )}

        <List>
          {tokens.length === 0 ? (
            <ListItem>
              <ListItemText
                primary="No tokens configured"
                secondary="Add a token to start accessing site data"
              />
            </ListItem>
          ) : (
            tokens.map((token) => (
              <ListItem key={token.siteId} divider>
                <ListItemText
                  primary={
                    <Box display="flex" alignItems="center" gap={1}>
                      <Typography variant="subtitle1">
                        {token.siteName} ({token.siteId})
                      </Typography>
                      {token.isDefault && (
                        <Chip label="Default" size="small" color="primary" />
                      )}
                      {isTokenExpired(token) && (
                        <Chip label="Expired" size="small" color="error" />
                      )}
                    </Box>
                  }
                  secondary={
                    <Box>
                      <Typography variant="caption" display="block">
                        Identity: {token.identity || 'Unknown'}
                      </Typography>
                      <Typography variant="caption" display="block">
                        Token: {showTokens[token.siteId]
                          ? token.token.substring(0, 50) + '...'
                          : '●●●●●●●●●●●●●●●●'}
                        <IconButton
                          size="small"
                          onClick={() => setShowTokens({ ...showTokens, [token.siteId]: !showTokens[token.siteId] })}
                        >
                          {showTokens[token.siteId] ? <VisibilityOffIcon fontSize="small" /> : <VisibilityIcon fontSize="small" />}
                        </IconButton>
                      </Typography>
                      <Typography variant="caption" display="block" color="text.secondary">
                        Added: {formatDate(token.addedAt)}
                      </Typography>
                      <Typography variant="caption" display="block" color="text.secondary">
                        Expires: {token.expiresAt ? formatDate(token.expiresAt) : 'Never'}
                      </Typography>
                      <Typography variant="caption" display="block" color="text.secondary">
                        Last Used: {formatDate(token.lastUsed)}
                      </Typography>
                    </Box>
                  }
                />
                <ListItemSecondaryAction>
                  <Stack direction="row" spacing={1}>
                    <Tooltip title="Test Token">
                      <IconButton
                        edge="end"
                        onClick={() => handleTestToken(token)}
                        disabled={testingToken === token.siteId}
                      >
                        {isTokenExpired(token) ? <ErrorIcon /> : <CheckIcon />}
                      </IconButton>
                    </Tooltip>
                    {!token.isDefault && (
                      <Tooltip title="Set as Default">
                        <IconButton
                          edge="end"
                          onClick={() => handleSetDefault(token.siteId)}
                        >
                          <KeyIcon />
                        </IconButton>
                      </Tooltip>
                    )}
                    <Tooltip title="Edit">
                      <IconButton
                        edge="end"
                        onClick={() => {
                          setEditingToken(token);
                          setNewToken({
                            siteId: token.siteId,
                            siteName: token.siteName,
                            token: token.token
                          });
                          setDialogOpen(true);
                        }}
                      >
                        <EditIcon />
                      </IconButton>
                    </Tooltip>
                    <Tooltip title="Delete">
                      <IconButton
                        edge="end"
                        onClick={() => handleDeleteToken(token.siteId)}
                      >
                        <DeleteIcon />
                      </IconButton>
                    </Tooltip>
                  </Stack>
                </ListItemSecondaryAction>
              </ListItem>
            ))
          )}
        </List>

        {/* Add/Edit Dialog */}
        <Dialog open={dialogOpen} onClose={() => setDialogOpen(false)} maxWidth="sm" fullWidth>
          <DialogTitle>
            {editingToken ? 'Edit Site Token' : 'Add Site Token'}
          </DialogTitle>
          <DialogContent>
            <Stack spacing={2} sx={{ mt: 1 }}>
              <TextField
                label="Site ID"
                value={newToken.siteId}
                onChange={(e) => setNewToken({ ...newToken, siteId: e.target.value })}
                fullWidth
                helperText="e.g., ses_falls_city or 309"
              />
              <TextField
                label="Site Name"
                value={newToken.siteName}
                onChange={(e) => setNewToken({ ...newToken, siteName: e.target.value })}
                fullWidth
                helperText="e.g., SES Falls City Medical Center"
              />
              <TextField
                label="ACE API Token"
                value={newToken.token}
                onChange={(e) => setNewToken({ ...newToken, token: e.target.value })}
                fullWidth
                multiline
                rows={4}
                helperText="Paste your JWT token from ACE IoT"
              />
            </Stack>
          </DialogContent>
          <DialogActions>
            <Button onClick={() => setDialogOpen(false)}>Cancel</Button>
            <Button onClick={handleAddToken} variant="contained" color="primary">
              {editingToken ? 'Update' : 'Add'}
            </Button>
          </DialogActions>
        </Dialog>
      </CardContent>
    </Card>
  );
};

export default TokenManager;
