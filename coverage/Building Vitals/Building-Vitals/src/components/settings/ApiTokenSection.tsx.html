import React, { useCallback } from 'react';
import {
  Box,
  Typography,
  Paper,
  Button,
  TextField,
  InputAdornment,
  CircularProgress,
  Alert,
  Tooltip,
  IconButton,
} from '@mui/material';
import { HelpOutline, Clear } from '@mui/icons-material';

interface ApiTokenSectionProps {
  tokenInput: string;
  tokenError: string;
  isTokenLoading: boolean;
  isInitialLoading: boolean;
  aceToken: string | null;
  isDirect: boolean;
  onTokenChange: (value: string) => void;
  onTokenSave: () => void;
  onTokenClear: () => void;
}

export const ApiTokenSection: React.FC<ApiTokenSectionProps> = React.memo(
  ({
    tokenInput,
    tokenError,
    isTokenLoading,
    isInitialLoading,
    aceToken,
    isDirect,
    onTokenChange,
    onTokenSave,
    onTokenClear,
  }) => {
    const handleChange = useCallback(
      (e: React.ChangeEvent<HTMLInputElement>) => {
        onTokenChange(e.target.value);
      },
      [onTokenChange]
    );

    return (
      <Paper sx={{ p: 3, mb: 3 }}>
        <Box sx={{ display: 'flex', alignItems: 'center', mb: 2 }}>
          <Typography sx={{ flexGrow: 1 }} variant="h6">
            ACE IoT API Token
          </Typography>
          <Tooltip title="ACE (Automated Control Engine) IoT API requires a JWT token for authentication. Contact your system administrator to obtain a long-lived token.">
            <IconButton size="small">
              <HelpOutline />
            </IconButton>
          </Tooltip>
        </Box>

        {isInitialLoading ? (
          <Box sx={{ display: 'flex', justifyContent: 'center', p: 3 }}>
            <CircularProgress />
          </Box>
        ) : (
          <>
            <Typography paragraph color="text.secondary" variant="body2">
              Enter your ACE IoT API token to connect to the Building Vitals data source.
              {isDirect && (
                <Alert severity="warning" sx={{ mt: 1 }}>
                  <strong>Direct API Mode:</strong> Your token will be sent directly to the ACE IoT
                  API. Use this mode only in secure environments.
                </Alert>
              )}
            </Typography>

            <Box sx={{ mb: 2 }}>
              <TextField
                fullWidth
                InputProps={{
                  endAdornment: tokenInput && (
                    <InputAdornment position="end">
                      <IconButton edge="end" onClick={onTokenClear}>
                        <Clear />
                      </IconButton>
                    </InputAdornment>
                  ),
                }}
                error={!!tokenError}
                helperText={tokenError || 'JWT token format: xxx.yyy.zzz'}
                label="ACE IoT API Token"
                placeholder="Enter your JWT token here"
                type="password"
                value={tokenInput}
                variant="outlined"
                onChange={handleChange}
              />

              <Box sx={{ display: 'flex', justifyContent: 'flex-end', mt: 2 }}>
                <Button
                  color="primary"
                  disabled={isTokenLoading || !tokenInput}
                  variant="contained"
                  onClick={onTokenSave}
                >
                  {isTokenLoading ? <CircularProgress size={24} /> : 'Save Token'}
                </Button>
              </Box>
            </Box>

            {aceToken && (
              <Alert severity="success" sx={{ mb: 2 }}>
                API Token is configured and active
              </Alert>
            )}
          </>
        )}
      </Paper>
    );
  }
);

ApiTokenSection.displayName = 'ApiTokenSection';
